𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲎𑲪𑲱𑱼𑲉𑲍𑲲𑲰𑱺𑲪𑲁𑱱

[𑲍𑲵𑱲𑲬𑲱𑱾𑲚𑲁𑲰𑱺𑲪𑲱𑱲𑲰]

𑱱𑱱 𑱽𑲁𑲎 𑲍𑲊𑲥𑲄𑲙𑲰𑲉 𑱱𑱱

𑱱𑱱 𑲏𑲴𑲶 𑱽𑲁𑲴 𑲀𑲮𑱴𑲅𑱺𑲩𑲳𑲰 𑲏𑲰𑲊𑲩𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲉𑲳𑲰 𑱱𑱱

𑲏𑲰𑲊𑲩𑲰𑲅𑲋𑲴𑱲𑲱𑱺𑲳𑲌𑲥𑲊𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲴 𑱴𑲁𑲠𑲮𑲱𑲰𑲊𑲰𑲉𑲰𑲵 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲉𑲰𑲵 𑲂𑲊𑲩𑲰𑲵 𑲂𑲊𑲁𑲰𑱽𑲴 𑲅𑲩𑲅𑲋𑲴𑱲𑲉𑱺𑲱 𑲍𑲡𑱱 𑱾𑱹𑲢 𑲍𑲒𑱽𑲜𑲮𑲰𑲎, 𑱺𑲰𑲵𑲌𑲢 𑲍𑲥𑲀𑲮𑲰𑲅𑲌𑲲𑲰𑱽𑲩𑲰𑱽 𑱾𑲌𑲩𑱺𑲱 𑲍𑲡𑱱𑱱

𑲏𑲱𑲎 𑲌𑲰𑲊𑲱𑱾𑲲𑱺𑲪 𑲊𑲲𑲰𑱾𑲵 𑲌𑲲𑲰𑱽𑲩𑱺𑲰, 𑲌𑲲𑲰𑱽𑲩𑱺𑲳𑲰𑲅 𑲊𑲲𑲰𑱾𑲁𑱱 𑲊𑲲𑲰𑱾𑲰𑱽𑲝 𑱾𑲪𑲱𑱻𑱲 𑲌𑲲𑲰𑱽𑲩𑱺𑲰, 𑲌𑲲𑲰𑱽𑲩𑱺𑲰𑲉𑲰 𑱽 𑱾𑲪𑲱𑱻𑱴 𑲊𑲲𑲰𑱾𑲁𑱱 𑲉𑱼𑲪𑲲𑲰𑱾𑲵 𑲍𑲰 𑲌𑲲𑲰𑱽𑲩𑱺𑲰, 𑲉𑲰 𑲌𑲲𑲰𑱽𑲩𑱺𑲰 𑱺𑱼𑲪𑲲𑲰𑱾𑲁𑱱𑱱

𑲏𑲳𑲅𑲁𑲳𑲅 𑲅𑲳𑱼𑱽𑲰𑲍𑲵𑲄𑲙𑲰𑲍𑲵𑲍𑲒𑲰𑲊𑲅𑲱𑲄𑲙𑲰𑱽𑲰𑱽𑲱𑱱𑱱

𑲏𑲱𑲎𑲵 𑲌𑲰𑲊𑲱𑱾𑲲𑱺𑲪 𑲍𑲊𑲥𑱼𑲮𑲊𑲡𑲰𑲎 𑲌𑲲𑲰𑱽𑲩𑱺𑲰𑲋𑱲𑲬𑱽𑲰 𑲏𑱽𑲲𑱺𑲞𑱽𑲝𑲰 𑲏𑱽𑲱𑲊𑲲𑱼𑲜𑲮𑲰 𑲏𑲁𑲋𑲰 𑱽 𑲅𑲱𑲁𑲋𑲰 𑱽𑲴𑱽𑲰 𑱽 𑱾𑲊𑲱𑱾𑲲𑲰𑲊𑲝𑲰𑲎𑱱 𑱺𑲍𑲡𑲰𑲂𑲣𑲰𑲊𑲱𑱾𑲲𑱺𑲪 𑲌𑲲𑲰𑱽𑲩𑱺𑲰𑲉𑲰𑲵 𑱽 𑲊𑲲𑲰𑱾𑲁, 𑱽 𑲅𑲳𑱼𑱽𑲰, 𑱽 𑲍𑲵𑲄𑲙𑲰, 𑱽 𑲍𑲵𑲍𑲒𑲰𑲊𑲰𑲎, 𑱽 𑲅𑲱𑲄𑲙𑲰𑱽𑲰𑱽𑲱𑱱 𑱽 𑲂𑱲𑲬𑲲𑲎𑲌𑲪𑲴𑱺𑲪𑱴𑲮𑲪𑲰𑱽𑲄𑲱𑲎𑲥𑲰𑱲𑲰𑲉𑲁𑱽𑲰𑲵𑲍𑲱, 𑱽 𑲊𑲲𑲰𑱾𑲌𑲀𑲜𑱴𑱽𑲜𑲮𑲊𑲍𑲍𑲞𑲪𑲌𑲚𑲅𑲩𑱼𑲮𑲊𑲡𑲰𑲎𑱱 𑱽 𑲂𑱲𑲬𑲲𑲊𑲜𑲮𑲰𑱺𑲲𑲊𑲩𑲰𑲅𑱽𑲝 𑲁𑱽𑲴𑱼𑲮𑲰𑱺𑲲𑲎𑱱𑱱

𑱽 𑲅𑲱𑱼𑲩𑲰 𑱽𑲰𑲅𑲱𑱼𑲩𑲰 𑱽 𑲅𑲱𑱼𑲩𑲰𑱲𑲬𑲉𑲴 𑱽𑲰𑲅𑲱𑱼𑲩𑲰𑱲𑲬𑲉𑲴 𑲉𑲰𑲅𑱽𑲝 𑲄𑲊𑲰𑲁𑲊𑱽𑲵 𑱽 𑲄𑲊𑲰𑲁𑲊𑱽𑱲𑲬𑲉𑲴 𑱽 𑱼𑲲𑲎𑱳𑲍𑲁𑲲𑱼𑲉𑱽𑲱𑲊𑲴𑱼𑲮𑲁𑲰𑲊𑲔𑲰 𑱽 𑲄𑲙𑲰𑱽𑲵 𑱽 𑱾𑲪𑲰𑱾𑲚𑲱𑱺𑲥𑲁𑱱𑱱

𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲍𑲩 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲁𑲰𑲌𑲪𑲱𑱺𑲩 𑲅𑲱𑲎𑲊𑱺𑲱 𑲂𑲱𑱺𑲚𑲰𑲅𑲊𑱽𑲎𑱱 𑲂𑲱𑱺𑲚𑲰𑲅𑲊𑱽𑱽𑲰𑲍𑲚𑲱𑱺𑲥𑲰𑱼𑱺𑲪𑲍𑲚𑲴 𑲅𑲱𑱾𑲊𑲩𑲰𑲍𑲰𑱺𑲱𑱲𑲪𑲰𑱽𑲚𑲴 𑱽𑲱𑲌𑲛𑱽𑲱𑲊𑲥𑲰𑱽𑲎𑱱 𑱺𑲪𑲩𑱼𑲮𑲥𑲅𑲩𑲅𑲍𑲛𑲱𑱺𑲰𑲎 𑲍𑲊𑲥𑲀𑲲𑱼𑲜𑲮𑲰𑲎 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲁𑲰𑲌𑲪𑲱𑱺𑲩 𑲏𑱽𑲲𑱺𑲚𑲊𑲰𑲵 𑲍𑲁𑲩𑱲𑲭𑲵𑲀𑲴𑱼𑲮𑲱𑲁𑲀𑲮𑲱𑲍𑲵𑲀𑲲𑱼𑲜𑲮𑲰𑲎𑱱𑱱

𑱺𑲍𑲡𑲰𑲄𑲤𑲙𑲰𑱺𑲅𑲩𑲎 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲁𑲎𑲰𑲁𑱽𑲚𑲪𑲴 𑲁𑲎𑲰𑲅𑲱𑱼𑲩𑲰𑲁𑱽𑲚𑲪𑲴𑱽𑲲𑱺𑲚𑲊𑲁𑱽𑲚𑲪𑲴𑲍𑲁𑲍𑲁𑲁𑱽𑲚𑲪𑲎 𑲍𑲊𑲥𑱼𑲲𑲎𑱳𑱾𑲪𑲌𑲁𑱽𑲎 𑲍𑱺𑲩𑲁𑲁𑲱𑱻𑲩𑱺𑲥𑲰𑱺 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲉𑲰𑲁𑲲𑱲𑲚𑲴 𑲁𑱽𑲚𑲪𑲎𑱱

𑱺𑱼𑲩𑱻𑲰- 𑱴𑱺𑲳 𑱴𑱺𑲳 𑱾𑲰𑲊𑱴𑱺𑲳 𑱾𑲰𑲊𑲍𑲵𑱴𑱺𑲳 𑲀𑲴𑱼𑲮𑲱 𑲍𑲥𑲰𑲎𑲰𑱱𑱱

𑲏𑲱𑱺𑲱 𑱾𑲪𑲄𑲙𑲰𑱾𑲰𑲊𑲁𑲱𑱺𑲰𑲎𑲪𑲱𑱼𑲉𑲍𑲲𑲰𑱺𑲪𑲵 𑲍𑲁𑲰𑱾𑲚𑲁𑱱𑱱
