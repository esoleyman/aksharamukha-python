10 לִפִּשָׁלָסַנְדַרְשַׁנַפַּרִבַרְטוֹ דַשַׁמַהּ.

אִטִה הִה בִּכְּשַׁבַהּ סַמְבְרֻדַּהּ כֻּמָרַהּ. טַדָה מָנְגַלְיַשַׁטַסַהַסְרַיְהּ לִפִּשָׁלָמֻפַּנִייַטֵה סְמַה דַשַׁבִּרְדָרַכַּסַהַסְרַיְהּ פַּרִבְרֻטַהּ פֻּרַסְכְּרֻטַהּ, דַשַׁבִּשְׁצַ׳ה רַטַסַהַסְרַיְהּ כָדַנִייַבּוֹגַ׳נִייַסְבָדַנִייַפַּרִפּוּרְנַיְרְהִרַנְיַסֻבַרְנַפַּרִפּוּרְנַיְשְׁצַ׳ה. יֵנַה כַּפִּלַבַסְטֻנִה מַהָנַגַרֵה וִיטִצַ׳טְבַרַרַטְיָנְטַרָפַּנַמֻכֵשְׁבַבְּיַבַכִּירְיַטֵה סְמַה אַבִּבִשְׁרָמְיַנְטֵה. אַשְׁטָבִּשְׁצַ׳ה טוּרְיַשַׁטַסַהַסְרַיְהּ פְּרַגֻשְׁיַמָנַיְרְמַהַטָה צַ׳ה פֻּשְׁפַּבַרְשֵׁנָבִּפְּרַבַרְשַׁטָה וִטַרְדִנִרְיוּהַטוֹרַנַגַבָכְּשַׁהַרְמְיַכּוּטָגָרַפְּרָסָדַטַלֵשֻׁה כַּנְיָשַׁטַסַהַסְרָנִה סַרְבָלַנְכָּרַבּוּשִׁטָהּ סְטִטָה אַבּוּוַן. בּוֹדִסַטְּבַם פְּרֵכְּשַׁמָנָהּ כֻּסֻמָנִה צַ׳ה כְּשִׁפַּנְטִה סְמַה. אַשְׁטַב צַ׳ה מַרֻטְכַּנְיָסַהַסְרָנִה וִגַלִטָלַנְכָּרָבַּרַנָלַנְכְּרֻטָנִה רַטְנַבַּדְרַנְכַּרֵנַה גְרֻהִיטָנִה מָרְגַם שׁוֹדַיַנְטְיוֹ בּוֹדִסַטְּבַסְיַה פֻּרַטוֹ גַצַּ׳נְטִה סְמַה. דֵבַנָגַיַכְּשַׁגַנְדַרְבָסֻרַגַרֻדַכִּנַּרַמַהוֹרַגָשְׁצָ׳רְדַכָּיִכָּה גַגַנַטַלָטְפֻּשְׁפַּפַּטַּדָמָנְיַבִּפְּרַלַמְבַּיַנְטִה סְמַה. סַרְבֵה צַ׳ה שָׁכְּיַגַנָהּ שֻׁדּוֹדַנַם רָגָ׳נַם פֻּרַסְכְּרֻטְיַה בּוֹדִסַטְּבַסְיַה פֻּרַטוֹ גַצַּ׳נְטִה סְמַה. אַנֵנַיְבַמְרוּפֵּנַה וְיוּהֵנַה בּוֹדִסַטְּבוֹ לִפִּשָׁלָמֻפַּנִייַטֵה סְמַה..

סַמַנַנְטַרַפְּרַבֵשִׁטַשְׁצַ׳ה בּוֹדִסַטְּבוֹ לִפִּשָׁלָם. אַטַה וִשְׁבָמִטְרוֹ נָמַה דָרַכָּצָ׳רְיוֹ בּוֹדִסַטְּבַסְיַה שְׁרִיַם טֵגַ׳שְׁצָ׳סַהַמָנוֹ דַרַנִטַלֵה נִבִשְׁטוֹ'דוֹמֻכַהּ פְּרַפַּטַטִה סְמַה. טַם טַטָה פְּרַפַּטִטַם דְרֻשְׁטְבָה שֻׁבָּנְגוֹ נָמַה טֻשִׁטַכָּיִכּוֹ דֵבַפֻּטְרוֹ דַכְּשִׁנֵנַה כַּרַטַלֵנַה פַּרִגְרֻהְיוֹטָּפַּיַטִה סְמַה. אֻטָּפְּיַה צַ׳ה גַגַנַטַלַסְטוֹ רָגָ׳נַם שֻׁדּוֹדַנַם טַם צַ׳ה מַהָנְטַם גַ׳נַכָּיַם גָטָבִּרַבְּיַבָּשַׁט—

שָׁסְטְרָנִה יָנִה פְּרַצַ׳לַנְטִה מַנֻשְׁיַלוֹכֵּה
סַנְכְיָה לִפִּשְׁצַ׳ה גַנַנָפִּה צַ׳ה דָטֻטַנְטְרַם.
יֵה שִׁלְפַּיוֹגַה פְּרֻטֻה לַבְכִּכַּה אַפְּרַמֵיָהּ
טֵשְׁבֵשֻׁה שִׁכְּשִׁטֻה פֻּרָה בַּהֻכַּלְפַּכּוֹטְיַהּ..1..

כִּם טוּ גַ׳נַסְיַה אַנֻבַרְטַנַטָם כַּרוֹטִה
לִפִּשָׁלַמָגַטֻה סֻשִׁכְּשִׁטֻה שִׁשְׁיַנָרְטַם.
פַּרִפָּצַ׳נָרְטַה בַּהֻדָרַכַּה אַגְרַיָנֵה
אַנְיָמְשְׁצַ׳ה סַטְּבַנַיֻטָנַמְרֻטֵה וִנֵטֻם..2..

לוֹכּוֹטַּרֵשֻׁה צַ׳טֻסַטְיַפַּטֵה וִדִגְ׳נוֹ
הֵטֻפְּרַטִיטְיַכֻּשַׁלוֹ יַטַה סַמְבַּבַנְטִה.
יַטַה צָ׳נִרוֹדַכְּשַׁיֻה סַמְסְטִטֻה שִׁיטִבָּבַהּ
טַסְמִנְבִדִגְ׳נַה כִּמַטוֹ לִפִּשָׁסְטְרַמָטְרֵה..3..

נֵטַסְיַה אָצַ׳רִיַה אֻטַּרִה וָה טְרִלוֹכֵּה

אִטְיֻכְּטְבָה סַה דֵבַפֻּטְרוֹ בּוֹדִסַטְּבַם דִבְיַיְהּ כֻּסֻמַיְרַבְּיַרְצְ׳יַה טַטְרַיְבָנְטַרְדַדֵה. טַטְרַה דָטְרְיַשְׁצַ׳ה צֵ׳טִיוַרְגָשְׁצַ׳ה סְטָפִּטָה אַבּוּוַן. פַּרִשֵׁשָׁהּ שָׁכְּיָהּ שֻׁדּוֹדַנַפְּרַמֻכָהּ פְּרַכְּרָמַנְטַהּ..

אַטַה בּוֹדִסַטְּבַה אֻרַגַסָרַצַ׳נְדַנַמַיַם לִפִּפַּלַכַּמָדָיַה דִבְיָרְשַׁסֻבַרְנַטִרַכַּם סַמַנְטָנְמַנִרַטְנַפְּרַטְיֻפְּטַם וִשְׁבָמִטְרַמָצָ׳רְיַמֵבַמָהַה-כַּטַמָם מֵה בּוֹ אֻפָּדְיָיַה לִפִּם שִׁכְּשָׁפַּיַסִה. בְּרָהְמִיכַרוֹשְׁטִיפֻּשְׁכַּרַסָרִם אַנְגַלִפִּם וַנְגַלִפִּם מַגַדַלִפִּם מַנְגַלְיַלִפִּם אַנְגֻלִייַלִפִּם שַׁכָּרִלִפִּם בְּרַהְמַבַלִלִפִּם פָּרֻשְׁיַלִפִּם דְרָבִדַלִפִּם כִּרָטַלִפִּם דָכְּשִׁנְיַלִפִּם אֻגְרַלִפִּם סַנְכְיָלִפִּם אַנֻלוֹמַלִפִּם אַבַמוּרְדַלִפִּם דַרַדַלִפִּם כָשְׁיַלִפִּם צִ׳ינַלִפִּם לוּנַלִפִּם הוּנַלִפִּם מַדְיָכְּשַׁרַבִסְטַרַלִפִּם פֻּשְׁפַּלִפִּם דֵבַלִפִּם נָגַלִפִּם יַכְּשַׁלִפִּם גַנְדַרְבַלִפִּם כִּנַּרַלִפִּם מַהוֹרַגַלִפִּם אַסֻרַלִפִּם גַרֻדַלִפִּם מְרֻגַצַ׳כְּרַלִפִּם וָיַסַרֻטַלִפִּם בַּבְמַדֵבַלִפִּם אַנְטַרִיכְּשַׁדֵבַלִפִּם אֻטַּרַכֻּרֻדְבִיפַּלִפִּם אַפַּרַגוֹדָנִילִפִּם פּוּרְבַבִדֵהַלִפִּם אֻטְכְּשֵׁפַּלִפִּם נִכְּשֵׁפַּלִפִּם וִכְּשֵׁפַּלִפִּם פְּרַכְּשֵׁפַּלִפִּם סָגַרַלִפִּם וַגְ׳רַלִפִּם לֵכַפְּרַטִלֵכַלִפִּם אַנֻדְרֻטַלִפִּם שָׁסְטְרָבַרְטָם גַנַנָבַרְטַלִפִּם אֻטְכְּשֵׁפָּבַרְטַלִפִּם נִכְּשֵׁפָּבַרְטַלִפִּם פָּדַלִכִטַלִפִּם דְבִרֻטַּרַפַּדַסַנְדִלִפִּם יָבַדַּשׁוֹטַּרַפַּדַסַנְדִלִפִּם מַדְיָהָרִנִילִפִּם סַרְבַרֻטַסַנְגְרַהַנִילִפִּם וִדְיָנֻלוֹמָבִמִשְׁרִטַלִפִּם רֻשִׁטַפַּסְטַפְּטָם רוֹצַ׳מָנָם דַרַנִיפְּרֵכְּשִׁנִילִפִּם גַגַנַפְּרֵכְּשִׁנִילִפִּם סַרְבַבְשַׁדִנִשְׁיַנְדָם סַרְבַסָרַסַנְגְרַהַנִים סַרְבַבּוּטַרֻטַגְרַהַנִים. אָסָם בּוֹ אֻפָּדְיָיַה צַ׳טֻשַּׁשְׁטִילִפִּינָם כַּטַמָם טְבַם שִׁשְׁיָפַּיִשְׁיַסִה?
