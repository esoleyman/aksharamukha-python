𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖮𑖴𑖟𑖧𑖭𑖳𑖝𑖿𑖨𑖦𑖿𑗂

[𑖭𑖽𑖎𑖿𑖬𑖰𑖢𑖿𑖝𑖦𑖯𑖝𑖴𑖎𑖯]

𑗃 𑖡𑖦𑖾 𑖭𑖨𑖿𑖪𑖕𑖿𑖗𑖯𑖧 𑗃

𑗃 𑖌𑖼 𑖡𑖦𑖺 𑖥𑖐𑖪𑖝𑖿𑖧𑖹 𑖁𑖨𑖿𑖧𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖧𑖹 𑗃

𑖁𑖨𑖿𑖧𑖯𑖪𑖩𑖺𑖎𑖰𑖝𑖸𑖫𑖿𑖪𑖨𑖤𑖺𑖠𑖰𑖭𑖝𑖿𑖝𑖿𑖪𑖺 𑖐𑖦𑖿𑖥𑖱𑖨𑖯𑖧𑖯𑖽 𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖧𑖯𑖽 𑖓𑖨𑖿𑖧𑖯𑖽 𑖓𑖨𑖦𑖯𑖜𑖺 𑖪𑖿𑖧𑖪𑖩𑖺𑖎𑖧𑖝𑖰 𑖭𑖿𑖦𑗂 𑖢𑖗𑖿𑖓 𑖭𑖿𑖎𑖡𑖿𑖠𑖯𑖾, 𑖝𑖯𑖽𑖫𑖿𑖓 𑖭𑖿𑖪𑖥𑖯𑖪𑖫𑖳𑖡𑖿𑖧𑖯𑖡𑖿 𑖢𑖫𑖿𑖧𑖝𑖰 𑖭𑖿𑖦𑗃

𑖂𑖮 𑖫𑖯𑖨𑖰𑖢𑖲𑖝𑖿𑖨 𑖨𑖳𑖢𑖽 𑖫𑖳𑖡𑖿𑖧𑖝𑖯, 𑖫𑖳𑖡𑖿𑖧𑖝𑖹𑖪 𑖨𑖳𑖢𑖦𑖿𑗂 𑖨𑖳𑖢𑖯𑖡𑖿𑖡 𑖢𑖴𑖞𑖎𑖿 𑖫𑖳𑖡𑖿𑖧𑖝𑖯, 𑖫𑖳𑖡𑖿𑖧𑖝𑖯𑖧𑖯 𑖡 𑖢𑖴𑖞𑖐𑖿 𑖨𑖳𑖢𑖦𑖿𑗂 𑖧𑖟𑖿𑖨𑖳𑖢𑖽 𑖭𑖯 𑖫𑖳𑖡𑖿𑖧𑖝𑖯, 𑖧𑖯 𑖫𑖳𑖡𑖿𑖧𑖝𑖯 𑖝𑖟𑖿𑖨𑖳𑖢𑖦𑖿𑗃

𑖊𑖪𑖦𑖸𑖪 𑖪𑖸𑖟𑖡𑖯𑖭𑖽𑖕𑖿𑖗𑖯𑖭𑖽𑖭𑖿𑖎𑖯𑖨𑖪𑖰𑖕𑖿𑖗𑖯𑖡𑖯𑖡𑖰𑗃

𑖂𑖮𑖽 𑖫𑖯𑖨𑖰𑖢𑖲𑖝𑖿𑖨 𑖭𑖨𑖿𑖪𑖠𑖨𑖿𑖦𑖯𑖾 𑖫𑖳𑖡𑖿𑖧𑖝𑖯𑖩𑖎𑖿𑖬𑖜𑖯 𑖀𑖡𑖲𑖝𑖿𑖢𑖡𑖿𑖡𑖯 𑖀𑖡𑖰𑖨𑖲𑖟𑖿𑖠𑖯 𑖀𑖦𑖩𑖯 𑖡 𑖪𑖰𑖦𑖩𑖯 𑖡𑖺𑖡𑖯 𑖡 𑖢𑖨𑖰𑖢𑖳𑖨𑖿𑖜𑖯𑖾𑗂 𑖝𑖭𑖿𑖦𑖯𑖓𑖿𑖔𑖯𑖨𑖰𑖢𑖲𑖝𑖿𑖨 𑖫𑖳𑖡𑖿𑖧𑖝𑖯𑖧𑖯𑖽 𑖡 𑖨𑖳𑖢𑖦𑖿, 𑖡 𑖪𑖸𑖟𑖡𑖯, 𑖡 𑖭𑖽𑖕𑖿𑖗𑖯, 𑖡 𑖭𑖽𑖭𑖿𑖎𑖯𑖨𑖯𑖾, 𑖡 𑖪𑖰𑖕𑖿𑖗𑖯𑖡𑖯𑖡𑖰𑗂 𑖡 𑖓𑖎𑖿𑖬𑖲𑖾𑖫𑖿𑖨𑖺𑖝𑖿𑖨𑖑𑖿𑖨𑖯𑖜𑖕𑖰𑖮𑖿𑖪𑖯𑖎𑖯𑖧𑖦𑖡𑖯𑖽𑖭𑖰, 𑖡 𑖨𑖳𑖢𑖫𑖤𑖿𑖟𑖐𑖡𑖿𑖠𑖨𑖭𑖭𑖿𑖢𑖿𑖨𑖬𑖿𑖘𑖪𑖿𑖧𑖠𑖨𑖿𑖦𑖯𑖾𑗂 𑖡 𑖓𑖎𑖿𑖬𑖲𑖨𑖿𑖠𑖯𑖝𑖲𑖨𑖿𑖧𑖯𑖪𑖡𑖿𑖡 𑖦𑖡𑖺𑖠𑖯𑖝𑖲𑖾𑗃

𑖡 𑖪𑖰𑖟𑖿𑖧𑖯 𑖡𑖯𑖪𑖰𑖟𑖿𑖧𑖯 𑖡 𑖪𑖰𑖟𑖿𑖧𑖯𑖎𑖿𑖬𑖧𑖺 𑖡𑖯𑖪𑖰𑖟𑖿𑖧𑖯𑖎𑖿𑖬𑖧𑖺 𑖧𑖯𑖪𑖡𑖿𑖡 𑖕𑖨𑖯𑖦𑖨𑖜𑖽 𑖡 𑖕𑖨𑖯𑖦𑖨𑖜𑖎𑖿𑖬𑖧𑖺 𑖡 𑖟𑖲𑖾𑖏𑖭𑖦𑖲𑖟𑖧𑖡𑖰𑖨𑖺𑖠𑖦𑖯𑖨𑖿𑖐𑖯 𑖡 𑖕𑖿𑖗𑖯𑖡𑖽 𑖡 𑖢𑖿𑖨𑖯𑖢𑖿𑖝𑖰𑖝𑖿𑖪𑖦𑖿𑗃

𑖤𑖺𑖠𑖰𑖭𑖝𑖿𑖝𑖿𑖪𑖭𑖿𑖧 𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖦𑖯𑖫𑖿𑖨𑖰𑖝𑖿𑖧 𑖪𑖰𑖮𑖨𑖝𑖰 𑖓𑖰𑖝𑖿𑖝𑖯𑖪𑖨𑖜𑖾𑗂 𑖓𑖰𑖝𑖿𑖝𑖯𑖪𑖨𑖜𑖡𑖯𑖭𑖿𑖝𑖰𑖝𑖿𑖪𑖯𑖟𑖝𑖿𑖨𑖭𑖿𑖝𑖺 𑖪𑖰𑖢𑖨𑖿𑖧𑖯𑖭𑖯𑖝𑖰𑖎𑖿𑖨𑖯𑖡𑖿𑖝𑖺 𑖡𑖰𑖬𑖿𑖙𑖡𑖰𑖨𑖿𑖪𑖯𑖜𑖾𑗂 𑖝𑖿𑖨𑖿𑖧𑖠𑖿𑖪𑖪𑖿𑖧𑖪𑖭𑖿𑖞𑖰𑖝𑖯𑖾 𑖭𑖨𑖿𑖪𑖤𑖲𑖟𑖿𑖠𑖯𑖾 𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖦𑖯𑖫𑖿𑖨𑖰𑖝𑖿𑖧 𑖀𑖡𑖲𑖝𑖿𑖝𑖨𑖯𑖽 𑖭𑖦𑖿𑖧𑖎𑖿𑖭𑖽𑖤𑖺𑖠𑖰𑖦𑖥𑖰𑖭𑖽𑖤𑖲𑖟𑖿𑖠𑖯𑖾𑗃

𑖝𑖭𑖿𑖦𑖯𑖕𑖿𑖕𑖿𑖗𑖯𑖝𑖪𑖿𑖧𑖾 𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖦𑖮𑖯𑖦𑖡𑖿𑖝𑖿𑖨𑖺 𑖦𑖮𑖯𑖪𑖰𑖟𑖿𑖧𑖯𑖦𑖡𑖿𑖝𑖿𑖨𑖺𑖡𑖲𑖝𑖿𑖝𑖨𑖦𑖡𑖿𑖝𑖿𑖨𑖺𑖭𑖦𑖭𑖦𑖦𑖡𑖿𑖝𑖿𑖨𑖾 𑖭𑖨𑖿𑖪𑖟𑖲𑖾𑖏𑖢𑖿𑖨𑖫𑖦𑖡𑖾 𑖭𑖝𑖿𑖧𑖦𑖦𑖰𑖞𑖿𑖧𑖝𑖿𑖪𑖯𑖝𑖿 𑖢𑖿𑖨𑖕𑖿𑖗𑖯𑖢𑖯𑖨𑖦𑖰𑖝𑖯𑖧𑖯𑖦𑖲𑖎𑖿𑖝𑖺 𑖦𑖡𑖿𑖝𑖿𑖨𑖾𑗂

𑖝𑖟𑖿𑖧𑖞𑖯- 𑖐𑖝𑖸 𑖐𑖝𑖸 𑖢𑖯𑖨𑖐𑖝𑖸 𑖢𑖯𑖨𑖭𑖽𑖐𑖝𑖸 𑖤𑖺𑖠𑖰 𑖭𑖿𑖪𑖯𑖮𑖯𑗃
