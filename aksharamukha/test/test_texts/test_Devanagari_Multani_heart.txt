𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊦𑊢𑊘𑊡𑊥𑊖𑊢𑊠।

[𑊥𑊚𑊄𑊥𑊛𑊖𑊠𑊖𑊢𑊄]

॥ 𑊚𑊠𑊦 𑊥𑊢𑊤𑊌𑊏𑊡 ॥

॥ 𑊂𑊠 𑊚𑊠 𑊟𑊆𑊤𑊖𑊡 𑊀𑊢𑊡𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊡𑊃 ॥

𑊀𑊢𑊡𑊤𑊣𑊄𑊖𑊥𑊤𑊢𑊝𑊙𑊥𑊖𑊖𑊤 𑊆𑊠𑊟𑊢𑊡𑊚 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊡𑊚 𑊊𑊢𑊡𑊚 𑊊𑊢𑊠𑊕 𑊤𑊡𑊤𑊣𑊄𑊡𑊖 𑊥𑊠। 𑊛𑊏𑊊 𑊥𑊄𑊚𑊙𑊦, 𑊖𑊚𑊥𑊊 𑊥𑊤𑊟𑊤𑊥𑊚𑊡𑊚 𑊛𑊥𑊡𑊖 𑊥𑊠॥

𑊁𑊦 𑊥𑊢𑊛𑊖𑊢 𑊢𑊛𑊚 𑊥𑊚𑊡𑊖, 𑊥𑊚𑊡𑊖𑊤 𑊢𑊛𑊠। 𑊢𑊛𑊚𑊚 𑊛𑊢𑊗𑊄 𑊥𑊚𑊡𑊖, 𑊥𑊚𑊡𑊖𑊡 𑊚 𑊛𑊢𑊗𑊆 𑊢𑊛𑊠। 𑊡𑊘𑊢𑊛𑊚 𑊥𑊀 𑊥𑊚𑊡𑊖, 𑊡𑊀 𑊥𑊚𑊡𑊖 𑊖𑊘𑊢𑊛𑊠॥

𑊃𑊤𑊠𑊤 𑊤𑊘𑊚𑊥𑊚𑊌𑊏𑊥𑊚𑊥𑊄𑊢𑊤𑊌𑊏𑊚𑊚𑊁॥

𑊁𑊦𑊚 𑊥𑊢𑊛𑊖𑊢 𑊥𑊢𑊤𑊙𑊢𑊠𑊦 𑊥𑊚𑊡𑊖𑊣𑊄𑊥𑊕 𑊀𑊚𑊖𑊛𑊚𑊚 𑊀𑊚𑊢𑊘𑊙 𑊀𑊠𑊣 𑊚 𑊤𑊠𑊣 𑊚𑊚 𑊚 𑊛𑊢𑊛𑊢𑊕𑊦। 𑊖𑊥𑊠𑊊𑊋𑊢𑊛𑊖𑊢 𑊥𑊚𑊡𑊖𑊡𑊚 𑊚 𑊢𑊛𑊠, 𑊚 𑊤𑊘𑊚, 𑊚 𑊥𑊚𑊌𑊏, 𑊚 𑊥𑊚𑊥𑊄𑊢𑊦, 𑊚 𑊤𑊌𑊏𑊚𑊚𑊁। 𑊚 𑊊𑊄𑊥𑊦𑊥𑊢𑊖𑊢𑊈𑊢𑊕𑊌𑊦𑊤𑊄𑊡𑊠𑊚𑊚𑊥, 𑊚 𑊢𑊛𑊥𑊝𑊘𑊆𑊚𑊙𑊢𑊥𑊥𑊛𑊢𑊥𑊐𑊤𑊡𑊙𑊢𑊠𑊦। 𑊚 𑊊𑊄𑊥𑊢𑊙𑊖𑊢𑊡𑊤𑊚𑊚 𑊠𑊚𑊙𑊖𑊦॥

𑊚 𑊤𑊘𑊡 𑊚𑊤𑊘𑊡 𑊚 𑊤𑊘𑊡𑊄𑊥𑊡 𑊚𑊤𑊘𑊡𑊄𑊥𑊡 𑊡𑊤𑊚𑊚 𑊌𑊢𑊠𑊢𑊕𑊚 𑊚 𑊌𑊢𑊠𑊢𑊕𑊄𑊥𑊡 𑊚 𑊘𑊦𑊅𑊥𑊠𑊘𑊡𑊚𑊢𑊙𑊠𑊢𑊆 𑊚 𑊌𑊏𑊚𑊚 𑊚 𑊛𑊢𑊛𑊖𑊖𑊤𑊠॥

𑊝𑊙𑊥𑊖𑊖𑊤𑊥𑊡 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊠𑊥𑊢𑊖𑊡 𑊤𑊦𑊢𑊖 𑊊𑊖𑊖𑊤𑊢𑊕𑊦। 𑊊𑊖𑊖𑊤𑊢𑊕𑊚𑊥𑊖𑊖𑊤𑊘𑊖𑊢𑊥𑊖 𑊤𑊛𑊢𑊡𑊥𑊖𑊄𑊢𑊚𑊖 𑊚𑊥𑊑𑊚𑊢𑊤𑊕𑊦। 𑊖𑊢𑊡𑊙𑊤𑊤𑊡𑊤𑊥𑊗𑊖𑊦 𑊥𑊢𑊤𑊝𑊘𑊙𑊦 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊠𑊥𑊢𑊖𑊡 𑊀𑊚𑊖𑊖𑊢𑊚 𑊥𑊠𑊡𑊄𑊥𑊚𑊝𑊙𑊠𑊟𑊥𑊚𑊝𑊘𑊙𑊦॥

𑊖𑊥𑊠𑊌𑊌𑊏𑊖𑊤𑊡𑊦 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊠𑊦𑊠𑊚𑊖𑊢 𑊠𑊦𑊤𑊘𑊡𑊠𑊚𑊖𑊢𑊚𑊖𑊖𑊢𑊠𑊚𑊖𑊢𑊥𑊠𑊥𑊠𑊠𑊚𑊖𑊢𑊦 𑊥𑊢𑊤𑊘𑊦𑊅𑊛𑊢𑊥𑊠𑊚𑊦 𑊥𑊖𑊡𑊠𑊠𑊗𑊡𑊖𑊤𑊖 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊡𑊠𑊄𑊖 𑊠𑊚𑊖𑊢𑊦।

𑊖𑊘𑊡𑊗- 𑊆𑊖 𑊆𑊖 𑊛𑊢𑊆𑊖 𑊛𑊢𑊥𑊚𑊆𑊖 𑊝𑊙 𑊥𑊤𑊦॥

𑊁𑊖 𑊛𑊢𑊌𑊏𑊛𑊢𑊠𑊖𑊦𑊢𑊘𑊡𑊥𑊖𑊢𑊚 𑊥𑊠𑊛𑊖𑊠॥
