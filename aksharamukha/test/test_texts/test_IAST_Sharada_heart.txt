𑆄𑆫𑇀𑆪𑆠𑇀𑆫𑆴𑆫𑆠𑇀𑆤𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆱𑆷𑆠𑇀𑆫𑆩𑇀

(𑆒) 𑆱𑆁𑆱𑇀𑆑𑆸𑆠𑆥𑆳𑆜𑆂

𑆨𑆳𑆫𑆠𑆵𑆪𑆨𑆳𑆰𑆳𑆪𑆳𑆩𑇀-𑆄𑆫𑇀𑆪𑆠𑇀𑆫𑆴𑆫𑆠𑇀𑆤𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆱𑆷𑆠𑇀𑆫𑆩𑇀

𑆨𑆾𑆛𑆨𑆳𑆰𑆳𑆪𑆳𑆩𑇀-𑆦𑆓𑇀𑆱𑇀-𑆥𑆳-𑆢𑆑𑆾𑆤𑇀-𑆩𑇀𑆖𑇀𑆗𑆾𑆓𑇀-𑆓𑇀𑆱𑆶𑆩𑇀-𑆘𑆼𑆱𑇀-𑆱𑆶-𑆝𑇀𑆫𑆤𑇀-𑆥𑆳-𑆩𑇀𑆢𑆾

𑆤𑆩𑆂 𑆱𑆫𑇀𑆮𑆧𑆶𑆢𑇀𑆣𑆧𑆾𑆣𑆴𑆱𑆠𑇀𑆠𑇀𑆮𑆼𑆨𑇀𑆪𑆂

(𑆧𑆶𑆢𑇀𑆣𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆂)

𑆅𑆠𑇀𑆪𑆥𑆴 𑆧𑆶𑆢𑇀𑆣𑆾 𑆨𑆓𑆮𑆳𑆁𑆱𑇀𑆠𑆡𑆳𑆓𑆠𑆾𑇁𑆫𑇀𑆲𑆤𑇀 𑆱𑆩𑇀𑆪𑆑𑇀𑆱𑆁𑆧𑆶𑆢𑇀𑆣𑆾 𑆮𑆴𑆢𑇀𑆪𑆳𑆖𑆫𑆟𑆱𑆩𑇀𑆥𑆤𑇀𑆤𑆂 𑆱𑆶𑆓𑆠𑆾 𑆬𑆾𑆑𑆮𑆴𑆢𑆤𑆶𑆠𑇀𑆠𑆫𑆂 𑆥𑆶𑆫𑆶𑆰𑆢𑆩𑇀𑆪𑆱𑆳𑆫𑆡𑆴𑆂 𑆯𑆳𑆱𑇀𑆠𑆳 𑆢𑆼𑆮𑆩𑆤𑆶𑆰𑇀𑆪𑆳𑆟𑆳𑆁 𑆧𑆶𑆢𑇀𑆣𑆾 𑆨𑆓𑆮𑆳𑆤𑆴𑆠𑆴𑇅

𑆤𑆴𑆰𑇀𑆪𑆤𑇀𑆢𑆂 𑆱 𑆠𑆡𑆳𑆓𑆠𑆂 𑆥𑆶𑆟𑇀𑆪𑆳𑆤𑆳𑆩𑇀, 𑆃𑆮𑆴𑆥𑇀𑆫𑆟𑆳𑆯𑆂 𑆑𑆶𑆯𑆬𑆩𑆷𑆬𑆳𑆤𑆳𑆩𑇀, 𑆃𑆬𑆕𑇀𑆑𑆸𑆠𑆂 𑆑𑇀𑆰𑆳𑆤𑇀𑆠𑇀𑆪𑆳, 𑆄𑆬𑆪𑆂 𑆥𑆶𑆟𑇀𑆪𑆤𑆴𑆣𑆳𑆤𑆳𑆤𑆳𑆩𑇀, 𑆖𑆴𑆠𑇀𑆫𑆴𑆠𑆾𑇁𑆤𑆶𑆮𑇀𑆪𑆚𑇀𑆘𑆤𑆽𑆂 𑆑𑆶𑆱𑆶𑆩𑆴𑆠𑆾 𑆬𑆑𑇀𑆰𑆟𑆽𑆂, 𑆥𑇀𑆫𑆠𑆴𑆫𑆷𑆥𑆾 𑆓𑆾𑆖𑆫𑆼𑆟, 𑆃𑆥𑇀𑆫𑆠𑆴𑆑𑆷𑆬𑆾 𑆢𑆫𑇀𑆯𑆤𑆼𑆤, 𑆃𑆨𑆴𑆫𑆠𑆴𑆂 𑆯𑇀𑆫𑆢𑇀𑆣𑆳(𑆯𑆶𑆢𑇀𑆣𑆳)𑆣𑆴𑆩𑆶𑆑𑇀𑆠𑆳𑆤𑆳𑆩𑇀, 𑆃𑆤𑆨𑆴𑆨𑆷𑆠𑆂 𑆥𑇀𑆫𑆘𑇀𑆚𑆪𑆳, 𑆃𑆤𑆮𑆩𑆫𑇀𑆢𑆤𑆵𑆪𑆾 𑆧𑆬𑆽𑆂, 𑆯𑆳𑆱𑇀𑆠𑆳 𑆱𑆫𑇀𑆮𑆱𑆠𑇀𑆠𑇀𑆮𑆳𑆤𑆳𑆩𑇀, 𑆥𑆴𑆠𑆳 𑆧𑆾𑆣𑆴𑆱𑆠𑇀𑆠𑇀𑆮𑆳𑆤𑆳𑆩𑇀, 𑆫𑆳𑆘𑆳 𑆄𑆫𑇀𑆪𑆥𑆶𑆢𑇀𑆓𑆬𑆳𑆤𑆳𑆩𑇀, 𑆱𑆳𑆫𑇀𑆡𑆮𑆳𑆲𑆂 𑆤𑆴𑆫𑇀𑆮𑆳𑆟𑆤𑆓𑆫𑆱𑆩𑇀𑆥𑇀𑆫𑆱𑇀𑆡𑆴𑆠𑆳𑆤𑆳𑆩𑇀, 𑆃𑆥𑇀𑆫𑆩𑆼𑆪𑆾 𑆘𑇀𑆚𑆳𑆤𑆼𑆤, 𑆃𑆖𑆴𑆤𑇀𑆠𑇀𑆪𑆂 𑆥𑇀𑆫𑆠𑆴𑆨𑆳𑆤𑆼𑆤, 𑆮𑆴𑆯𑆶𑆢𑇀𑆣𑆂 𑆱𑇀𑆮𑆫𑆼𑆟, 𑆄𑆱𑇀𑆮𑆢𑆤𑆵𑆪𑆾 𑆔𑆾𑆰𑆼𑆟, 𑆃𑆱𑆼𑆖𑆤𑆑𑆾 𑆫𑆷𑆥𑆼𑆟, 𑆃𑆥𑇀𑆫𑆠𑆴𑆱𑆩𑆂 𑆑𑆳𑆪𑆼𑆤, 𑆃𑆬𑆴𑆥𑇀𑆠𑆂 𑆑𑆳𑆩𑆽𑆂, 𑆃𑆤𑆶𑆥𑆬𑆴𑆥𑇀𑆠𑆾 𑆫𑆷𑆥𑆽𑆂 𑆃𑆱𑆁𑆱𑆸𑆰𑇀𑆛 𑆄𑆫𑆷𑆥𑇀𑆪𑆽𑆂, 𑆮𑆴𑆥𑇀𑆫𑆩𑆶𑆑𑇀𑆠𑆂 𑆱𑇀𑆑𑆤𑇀𑆣𑆼𑆨𑇀𑆪𑆂, 𑆮𑆴𑆱𑆩𑇀𑆥𑇀𑆫𑆪𑆶𑆑𑇀𑆠𑆾 𑆣𑆳𑆠𑆶𑆨𑆴𑆂, 𑆱𑆁𑆮𑆸𑆠 𑆄𑆪𑆠𑆤𑆽𑆂, 𑆥𑇀𑆫𑆖𑇀𑆗𑆴𑆤𑇀𑆤𑆾 𑆓𑇀𑆫𑆤𑇀𑆡𑆽𑆂, 𑆮𑆴𑆩𑆶𑆑𑇀𑆠𑆂 𑆥𑆫𑆴𑆢𑆳𑆔𑇀𑆤𑆽𑆂, 𑆥𑆫𑆴𑆩𑆶𑆑𑇀𑆠𑆱𑇀𑆠𑆸𑆰𑇀𑆟𑆪𑆳, 𑆏𑆔𑆳𑆢𑆶𑆠𑇀𑆠𑆵𑆫𑇀𑆟𑆂 𑆥𑆫𑆴𑆥𑆷𑆫𑇀𑆟𑆾 𑆘𑇀𑆚𑆳𑆤𑆼𑆤, 𑆥𑇀𑆫𑆠𑆴𑆰𑇀𑆜𑆴𑆠𑆾𑇁𑆠𑆵𑆠𑆳𑆤𑆳𑆓𑆠𑆥𑇀𑆫𑆠𑇀𑆪𑆶𑆠𑇀𑆥𑆤𑇀𑆤𑆳𑆤𑆳𑆁 𑆧𑆶𑆢𑇀𑆣𑆳𑆤𑆳𑆁 𑆨𑆓𑆮𑆠𑆳𑆁 𑆘𑇀𑆚𑆳𑆤𑆼, 𑆃𑆥𑇀𑆫𑆠𑆴𑆰𑇀𑆜𑆴𑆠𑆾 𑆤𑆴𑆫𑇀𑆮𑆳𑆟𑆼, 𑆱𑇀𑆡𑆴𑆠𑆾 𑆨𑆷𑆠𑆑𑆾𑆛𑇀𑆪𑆳𑆩𑇀, 𑆱𑇀𑆡𑆴𑆠𑆂 𑆱𑆫𑇀𑆮𑆱𑆠𑇀𑆠𑇀𑆮𑆳𑆬𑆾𑆑𑆤𑆵𑆪𑆳𑆪𑆳𑆁 𑆨𑆷𑆩𑆿, 𑆱𑆫𑇀𑆮 𑆅𑆩𑆼 𑆠𑆡𑆳𑆓𑆠𑆳𑆤𑆳𑆁 𑆮𑆴𑆯𑆼𑆰𑆠𑆂 𑆱𑆩𑇀𑆪𑆓𑇀 𑆓𑆶𑆟𑆳𑆂𑇅

(𑆣𑆫𑇀𑆩𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆂)

𑆱𑆢𑇀𑆣𑆫𑇀𑆩𑆱𑇀𑆠𑆶 𑆄𑆢𑆿 𑆑𑆬𑇀𑆪𑆳𑆟𑆂, 𑆩𑆣𑇀𑆪𑆼 𑆑𑆬𑇀𑆪𑆳𑆟𑆂, 𑆥𑆫𑇀𑆪𑆮𑆱𑆳𑆤𑆼 𑆑𑆬𑇀𑆪𑆳𑆟𑆂, 𑆱𑇀𑆮𑆫𑇀𑆡𑆂 𑆱𑆶𑆮𑇀𑆪𑆚𑇀𑆘𑆤𑆂, 𑆑𑆼𑆮𑆬𑆂, 𑆥𑆫𑆴𑆥𑆷𑆫𑇀𑆟𑆂, 𑆥𑆫𑆴𑆯𑆶𑆢𑇀𑆣𑆂, 𑆥𑆫𑇀𑆪𑆮𑆢𑆳𑆠𑆂, 𑆱𑇀𑆮𑆳𑆒𑇀𑆪𑆳𑆠𑆂 𑆨𑆓𑆮𑆠𑆾 𑆣𑆫𑇀𑆩𑆂, 𑆱𑆳𑆤𑇀𑆢𑆸𑆰𑇀𑆛𑆴𑆑𑆂, 𑆤𑆴𑆫𑇀𑆘𑇀𑆮𑆫𑆂, 𑆄𑆑𑆳𑆬𑆴𑆑𑆂, 𑆐𑆥𑆤𑆳𑆪𑆴𑆑𑆂, 𑆎𑆲𑆴𑆥𑆯𑇀𑆪𑆴𑆑𑆂, 𑆥𑇀𑆫𑆠𑇀𑆪𑆳𑆠𑇀𑆩𑆮𑆼𑆢𑆤𑆵𑆪𑆾 𑆮𑆴𑆘𑇀𑆚𑆽𑆂, 𑆱𑇀𑆮𑆳𑆒𑇀𑆪𑆳𑆠𑆾 𑆨𑆓𑆮𑆠𑆾 𑆣𑆫𑇀𑆩𑆮𑆴𑆤𑆪𑆂 𑆱𑆶𑆥𑇀𑆫𑆮𑆼𑆢𑆴𑆠𑆂 𑆤𑆽𑆫𑇀𑆪𑆳𑆟𑆴𑆑𑆂, 𑆱𑆁𑆧𑆾𑆣𑆴𑆓𑆳𑆩𑆵, 𑆃𑆨𑆴𑆤𑇀𑆤𑆂 𑆱𑆁𑆱𑇀𑆠𑆷𑆥𑆂, 𑆱𑆥𑇀𑆫𑆠𑆴𑆯𑆫𑆟𑆂, 𑆗𑆴𑆤𑇀𑆤𑆥𑇀𑆬𑆾𑆠𑆴𑆑𑆂𑇅

(𑆱𑆁𑆔𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆂)

𑆱𑆶𑆥𑇀𑆫𑆠𑆴𑆥𑆤𑇀𑆤𑆾 𑆨𑆓𑆮𑆠 𑆄𑆫𑇀𑆪𑆱𑆁𑆔𑆂, 𑆤𑇀𑆪𑆳𑆪𑆥𑇀𑆫𑆠𑆴𑆥𑆤𑇀𑆤𑆂, 𑆉𑆘𑆶𑆥𑇀𑆫𑆠𑆴𑆥𑆤𑇀𑆤𑆂, 𑆱𑆳𑆩𑆵𑆖𑆵𑆥𑇀𑆫𑆠𑆴𑆥𑆤𑇀𑆤𑆂, 𑆃𑆚𑇀𑆘𑆬𑆵𑆑𑆫𑆟𑆵𑆪𑆂, 𑆱𑆳𑆩𑆵𑆖𑆵𑆑𑆫𑆟𑆵𑆪𑆂, 𑆥𑆶𑆟𑇀𑆪𑆯𑇀𑆫𑆵𑆑𑇀𑆰𑆼𑆠𑇀𑆫𑆂, 𑆩𑆲𑆳𑆢𑆑𑇀𑆰𑆴𑆟𑆳𑆥𑆫𑆴𑆯𑆾𑆣𑆑𑆂, 𑆥𑇀𑆫𑆳𑆲𑆮𑆤𑆵𑆪𑆂, 𑆄𑆲𑆮𑆤𑆵𑆪𑆂𑇅

𑇆 𑆄𑆫𑇀𑆪𑆠𑇀𑆫𑆴𑆫𑆠𑇀𑆤𑆳𑆤𑆶𑆱𑇀𑆩𑆸𑆠𑆴𑆱𑆷𑆠𑇀𑆫𑆁 𑆱𑆩𑆳𑆥𑇀𑆠𑆩𑇀𑇆
