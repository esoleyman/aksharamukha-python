𑵡𑶈𑶗𑵬𑵳𑶗𑶈𑶋𑶈𑵳𑶗𑵺𑶊𑵺𑶍𑶉𑶗𑵰𑶗𑶈𑶍𑵳𑶋𑶉𑶎𑵳𑶗𑶈𑵰

(𑵲) 𑶉𑶕𑶉𑶗𑵱𑶗𑶈𑶍𑵳𑶅𑶊𑵾𑶖

𑵯𑶊𑶈𑵳𑶌𑵬𑵯𑶊𑶉𑶊𑵬𑶊𑵰-𑵡𑶈𑶗𑵬𑵳𑶗𑶈𑶋𑶈𑵳𑶗𑵺𑶊𑵺𑶍𑶉𑶗𑵰𑶗𑶈𑶍𑵳𑶋𑶉𑶎𑵳𑶗𑶈𑵰

𑵯𑶓𑵽𑵯𑶊𑶉𑶊𑵬𑶊𑵰-𑶆𑵶𑶗𑶉-𑶅𑶊-𑵸𑵱𑶓𑵺-𑵰𑶗𑵻𑶗𑵼𑶓𑵶-𑵶𑶗𑶉𑶍𑵰-𑶀𑶐𑶉-𑶉𑶍-𑶂𑶗𑶈𑵺-𑶅𑶊-𑵰𑶗𑵸𑶓

𑵺𑵰𑶖 𑶉𑶈𑶗𑵭𑵮𑶍𑵸𑶗𑵹𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶐𑵯𑶗𑵬𑶖

(𑵮𑶍𑵸𑶗𑵹𑶊𑵺𑶍𑶉𑶗𑵰𑶗𑶈𑶍𑵳𑶋𑶖)

𑵢𑵳𑶗𑵬𑶅𑶋 𑵮𑶍𑵸𑶗𑵹𑶓 𑵯𑵶𑵭𑶊𑶕𑶉𑶗𑵳𑵴𑶊𑵶𑵳𑶓𑶈𑶗𑶇𑵺 𑶉𑵰𑶗𑵬𑵱𑶗𑶉𑶕𑵮𑶍𑵸𑶗𑵹𑶓 𑵭𑶋𑵸𑶗𑵬𑶊𑵻𑶈𑵺𑶉𑵰𑶗𑶅𑵺𑶗𑵺𑶖 𑶉𑶍𑵶𑵳𑶓 𑵵𑶓𑵱𑵭𑶋𑵸𑵺𑶍𑵳𑶗𑵳𑶈𑶖 𑶅𑶍𑶈𑶍𑶉𑵸𑵰𑶗𑵬𑶉𑶊𑶈𑵴𑶋𑶖 𑶉𑶊𑶉𑶗𑵳𑶊 𑵸𑶐𑵭𑵰𑵺𑶍𑶉𑶗𑵬𑶊𑵺𑶊𑶕 𑵮𑶍𑵸𑶗𑵹𑶓 𑵯𑵶𑵭𑶊𑵺𑶋𑵳𑶋.

𑵺𑶋𑶉𑶗𑵬𑵺𑶗𑵸𑶖 𑶉 𑵳𑵴𑶊𑵶𑵳𑶖 𑶅𑶍𑵺𑶗𑵬𑶊𑵺𑶊𑵰, 𑵠𑵭𑶋𑶅𑶗𑶈𑵺𑶊𑶉𑶖 𑵱𑶍𑶉𑵵𑵰𑶎𑵵𑶊𑵺𑶊𑵰, 𑵠𑵵𑶄𑶗𑵱𑶗𑶈𑶍𑵳𑶖 𑵱𑶗𑶉𑶊𑵺𑶗𑵳𑶗𑵬𑶊, 𑵡𑵵𑵬𑶖 𑶅𑶍𑵺𑶗𑵬𑵺𑶋𑵹𑶊𑵺𑶊𑵺𑶊𑵰, 𑵻𑶋𑵳𑶗𑶈𑶋𑵳𑶓𑵺𑶍𑵭𑶗𑵬𑵺𑶗𑶀𑵺𑶑𑶖 𑵱𑶍𑶉𑶍𑵰𑶋𑵳𑶓 𑵵𑵱𑶗𑶉𑵺𑶑𑶖, 𑶅𑶗𑶈𑵳𑶋𑶈𑶎𑶅𑶓 𑵶𑶓𑵻𑶈𑶐𑵺, 𑵠𑶅𑶗𑶈𑵳𑶋𑵱𑶎𑵵𑶓 𑵸𑶈𑶗𑶉𑵺𑶐𑵺, 𑵠𑵯𑶋𑶈𑵳𑶋𑶖 𑶉𑶗𑶈𑵸𑶗𑵹𑶊(𑶉𑶍𑵸𑶗𑵹𑶊)𑵹𑶋𑵰𑶍𑵱𑶗𑵳𑶊𑵺𑶊𑵰, 𑵠𑵺𑵯𑶋𑵯𑶎𑵳𑶖 𑶅𑶗𑶈𑶀𑶗𑵺𑵬𑶊, 𑵠𑵺𑵭𑵰𑶈𑶗𑵸𑵺𑶌𑵬𑶓 𑵮𑵵𑶑𑶖, 𑶉𑶊𑶉𑶗𑵳𑶊 𑶉𑶈𑶗𑵭𑶉𑵳𑶗𑵳𑶗𑵭𑶊𑵺𑶊𑵰, 𑶅𑶋𑵳𑶊 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶊𑵺𑶊𑵰, 𑶈𑶊𑶀𑶊 𑵡𑶈𑶗𑵬𑶅𑶍𑵸𑶗𑵶𑵵𑶊𑵺𑶊𑵰, 𑶉𑶊𑶈𑶗𑵴𑵭𑶊𑶇𑶖 𑵺𑶋𑶈𑶗𑵭𑶊𑵺𑵺𑵶𑶈𑶉𑵰𑶗𑶅𑶗𑶈𑶉𑶗𑵴𑶋𑵳𑶊𑵺𑶊𑵰, 𑵠𑶅𑶗𑶈𑵰𑶐𑵬𑶓 𑶀𑶗𑵺𑶊𑵺𑶐𑵺, 𑵠𑵻𑶋𑵺𑶗𑵳𑶗𑵬𑶖 𑶅𑶗𑶈𑵳𑶋𑵯𑶊𑵺𑶐𑵺, 𑵭𑶋𑶉𑶍𑵸𑶗𑵹𑶖 𑶉𑶗𑵭𑶈𑶐𑵺, 𑵡𑶉𑶗𑵭𑵸𑵺𑶌𑵬𑶓 𑵷𑶓𑶉𑶐𑵺, 𑵠𑶉𑶐𑵻𑵺𑵱𑶓 𑶈𑶎𑶅𑶐𑵺, 𑵠𑶅𑶗𑶈𑵳𑶋𑶉𑵰𑶖 𑵱𑶊𑵬𑶐𑵺, 𑵠𑵵𑶋𑶅𑶗𑵳𑶖 𑵱𑶊𑵰𑶑𑶖, 𑵠𑵺𑶍𑶅𑵵𑶋𑶅𑶗𑵳𑶓 𑶈𑶎𑶅𑶑𑶖 𑵠𑶉𑶕𑶉𑶗𑶈𑶍𑶉𑶗𑵽 𑵡𑶈𑶎𑶅𑶗𑵬𑶑𑶖, 𑵭𑶋𑶅𑶗𑶈𑵰𑶍𑵱𑶗𑵳𑶖 𑶉𑶗𑵱𑵺𑶗𑵹𑶐𑵯𑶗𑵬𑶖, 𑵭𑶋𑶉𑵰𑶗𑶅𑶗𑶈𑵬𑶍𑵱𑶗𑵳𑶓 𑵹𑶊𑵳𑶍𑵯𑶋𑶖, 𑶉𑶕𑵭𑶗𑶈𑶍𑵳 𑵡𑵬𑵳𑵺𑶑𑶖, 𑶅𑶗𑶈𑵻𑶗𑵼𑶋𑵺𑶗𑵺𑶓 𑵶𑶗𑶈𑵺𑶗𑵴𑶑𑶖, 𑵭𑶋𑵰𑶍𑵱𑶗𑵳𑶖 𑶅𑶈𑶋𑵸𑶊𑵷𑶗𑵺𑶑𑶖, 𑶅𑶈𑶋𑵰𑶍𑵱𑶗𑵳𑶉𑶗𑵳𑶗𑶈𑶍𑶉𑶗𑵺𑵬𑶊, 𑵪𑵷𑶊𑵸𑶍𑵳𑶗𑵳𑶌𑶈𑶗𑵺𑶖 𑶅𑶈𑶋𑶅𑶎𑶈𑶗𑵺𑶓 𑶀𑶗𑵺𑶊𑵺𑶐𑵺, 𑶅𑶗𑶈𑵳𑶋𑶉𑶗𑵾𑶋𑵳𑶓𑵳𑶌𑵳𑶊𑵺𑶊𑵶𑵳𑶅𑶗𑶈𑵳𑶗𑵬𑶍𑵳𑶗𑶅𑵺𑶗𑵺𑶊𑵺𑶊𑶕 𑵮𑶍𑵸𑶗𑵹𑶊𑵺𑶊𑶕 𑵯𑵶𑵭𑵳𑶊𑶕 𑶀𑶗𑵺𑶊𑵺𑶐, 𑵠𑶅𑶗𑶈𑵳𑶋𑶉𑶗𑵾𑶋𑵳𑶓 𑵺𑶋𑶈𑶗𑵭𑶊𑵺𑶐, 𑶉𑶗𑵴𑶋𑵳𑶓 𑵯𑶎𑵳𑵱𑶓𑵽𑶗𑵬𑶊𑵰, 𑶉𑶗𑵴𑶋𑵳𑶖 𑶉𑶈𑶗𑵭𑶉𑵳𑶗𑵳𑶗𑵭𑶊𑵵𑶓𑵱𑵺𑶌𑵬𑶊𑵬𑶊𑶕 𑵯𑶎𑵰𑶔, 𑶉𑶈𑶗𑵭 𑵢𑵰𑶐 𑵳𑵴𑶊𑵶𑵳𑶊𑵺𑶊𑶕 𑵭𑶋𑶉𑶐𑶉𑵳𑶖 𑶉𑵰𑶗𑵬𑵶 𑵶𑶍𑵺𑶊𑶖.

(𑵹𑶈𑶗𑵰𑶊𑵺𑶍𑶉𑶗𑵰𑶗𑶈𑶍𑵳𑶋𑶖)

𑶉𑵸𑶗𑵹𑶈𑶗𑵰𑶉𑶗𑵳𑶍 𑵡𑵸𑶔 𑵱𑵵𑶗𑵬𑶊𑵺𑶖, 𑵰𑵹𑶗𑵬𑶐 𑵱𑵵𑶗𑵬𑶊𑵺𑶖, 𑶅𑶈𑶗𑵬𑵭𑶉𑶊𑵺𑶐 𑵱𑵵𑶗𑵬𑶊𑵺𑶖, 𑶉𑶗𑵭𑶈𑶗𑵴𑶖 𑶉𑶍𑵭𑶗𑵬𑵺𑶗𑶀𑵺𑶖, 𑵱𑶐𑵭𑵵𑶖, 𑶅𑶈𑶋𑶅𑶎𑶈𑶗𑵺𑶖, 𑶅𑶈𑶋𑶉𑶍𑵸𑶗𑵹𑶖, 𑶅𑶈𑶗𑵬𑵭𑵸𑶊𑵳𑶖, 𑶉𑶗𑵭𑶊𑵲𑶗𑵬𑶊𑵳𑶖 𑵯𑵶𑵭𑵳𑶓 𑵹𑶈𑶗𑵰𑶖, 𑶉𑶊𑵺𑶗𑵸𑶗𑶈𑶍𑶉𑶗𑵽𑶋𑵱𑶖, 𑵺𑶋𑶈𑶗𑶀𑶗𑵭𑶈𑶖, 𑵡𑵱𑶊𑵵𑶋𑵱𑶖, 𑵫𑶅𑵺𑶊𑵬𑶋𑵱𑶖, 𑵨𑶇𑶋𑶅𑶉𑶗𑵬𑶋𑵱𑶖, 𑶅𑶗𑶈𑵳𑶗𑵬𑶊𑵳𑶗𑵰𑵭𑶐𑵸𑵺𑶌𑵬𑶓 𑵭𑶋𑶀𑶗𑵺𑶑𑶖, 𑶉𑶗𑵭𑶊𑵲𑶗𑵬𑶊𑵳𑶓 𑵯𑵶𑵭𑵳𑶓 𑵹𑶈𑶗𑵰𑵭𑶋𑵺𑵬𑶖 𑶉𑶍𑶅𑶗𑶈𑵭𑶐𑵸𑶋𑵳𑶖 𑵺𑶑𑶈𑶗𑵬𑶊𑵺𑶋𑵱𑶖, 𑶉𑶕𑵮𑶓𑵹𑶋𑵶𑶊𑵰𑶌, 𑵠𑵯𑶋𑵺𑶗𑵺𑶖 𑶉𑶕𑶉𑶗𑵳𑶎𑶅𑶖, 𑶉𑶅𑶗𑶈𑵳𑶋𑶉𑶈𑵺𑶖, 𑵼𑶋𑵺𑶗𑵺𑶅𑶗𑵵𑶓𑵳𑶋𑵱𑶖.

(𑶉𑶕𑵷𑶊𑵺𑶍𑶉𑶗𑵰𑶗𑶈𑶍𑵳𑶋𑶖)

𑶉𑶍𑶅𑶗𑶈𑵳𑶋𑶅𑵺𑶗𑵺𑶓 𑵯𑵶𑵭𑵳 𑵡𑶈𑶗𑵬𑶉𑶕𑵷𑶖, 𑵺𑶗𑵬𑶊𑵬𑶅𑶗𑶈𑵳𑶋𑶅𑵺𑶗𑵺𑶖, 𑶈𑶍𑶀𑶍𑶅𑶗𑶈𑵳𑶋𑶅𑵺𑶗𑵺𑶖, 𑶉𑶊𑵰𑶌𑵻𑶌𑶅𑶗𑶈𑵳𑶋𑶅𑵺𑶗𑵺𑶖, 𑵠𑵺𑶗𑶀𑵵𑶌𑵱𑶈𑵺𑶌𑵬𑶖, 𑶉𑶊𑵰𑶌𑵻𑶌𑵱𑶈𑵺𑶌𑵬𑶖, 𑶅𑶍𑵺𑶗𑵬𑶉𑶗𑶈𑶌𑵱𑶗𑶉𑶐𑵳𑶗𑶈𑶖, 𑵰𑶇𑶊𑵸𑵱𑶗𑶉𑶋𑵺𑶊𑶅𑶈𑶋𑶉𑶓𑵹𑵱𑶖, 𑶅𑶗𑶈𑶊𑶇𑵭𑵺𑶌𑵬𑶖, 𑵡𑶇𑵭𑵺𑶌𑵬𑶖.
