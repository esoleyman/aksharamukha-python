𑨀𑨊𑨫𑩇𑨪𑨙𑩇𑨫𑨁𑨫𑨙𑩇𑨝𑨊𑨝𑨂𑨰𑩇𑨢𑩇𑨫𑨉𑨙𑨁𑨰𑨂𑨊𑨙𑩇𑨫𑨢𑨴

(𑨌) 𑨰𑨸𑨰𑩇𑨋𑩇𑨫𑨉𑨙𑨞𑨊𑨕𑨹

𑨡𑨊𑨫𑨙𑨁𑨊𑨪𑨡𑨊𑨯𑨊𑨪𑨊𑨢𑨴-𑨀𑨊𑨫𑩇𑨪𑨙𑩇𑨫𑨁𑨫𑨙𑩇𑨝𑨊𑨝𑨂𑨰𑩇𑨢𑩇𑨫𑨉𑨙𑨁𑨰𑨂𑨊𑨙𑩇𑨫𑨢𑨴

𑨡𑨆𑨔𑨡𑨊𑨯𑨊𑨪𑨊𑨢𑨴-𑨟𑨍𑩇𑨰𑨴-𑨞𑨊-𑨛𑨋𑨆𑨝𑨴-𑨢𑩇𑨣𑩇𑨤𑨆𑨍𑨴-𑨍𑩇𑨰𑨂𑨢𑨴-𑨥𑨄𑨰𑨴-𑨰𑨂-𑨖𑩇𑨫𑨝𑨴-𑨞𑨊-𑨢𑩇𑨛𑨆

𑨝𑨢𑨹 𑨰𑨫𑩇𑨭𑨠𑨂𑨛𑩇𑨜𑨠𑨆𑨜𑨁𑨰𑨙𑩇𑨙𑩇𑨭𑨄𑨡𑩇𑨪𑨹

(𑨠𑨂𑨛𑩇𑨜𑨊𑨝𑨂𑨰𑩇𑨢𑩇𑨫𑨉𑨙𑨁𑨹)

𑨀𑨁𑨙𑩇𑨪𑨞𑨁 𑨠𑨂𑨛𑩇𑨜𑨆 𑨡𑨍𑨭𑨊𑨸𑨰𑩇𑨙𑨚𑨊𑨍𑨙𑨆𑨫𑩇𑨱𑨝𑨴 𑨰𑨢𑩇𑨪𑨋𑩇𑨰𑨸𑨠𑨂𑨛𑩇𑨜𑨆 𑨭𑨁𑨛𑩇𑨪𑨊𑨣𑨫𑨘𑨰𑨢𑩇𑨞𑨝𑩇𑨝𑨹 𑨰𑨂𑨍𑨙𑨆 𑨬𑨆𑨋𑨭𑨁𑨛𑨝𑨂𑨙𑩇𑨙𑨫𑨹 𑨞𑨂𑨫𑨂𑨯𑨛𑨢𑩇𑨪𑨰𑨊𑨫𑨚𑨁𑨹 𑨮𑨊𑨰𑩇𑨙𑨊 𑨛𑨄𑨭𑨢𑨝𑨂𑨯𑩇𑨪𑨊𑨘𑨊𑨸 𑨠𑨂𑨛𑩇𑨜𑨆 𑨡𑨍𑨭𑨊𑨝𑨁𑨙𑨁𑩂

𑨝𑨁𑨯𑩇𑨪𑨝𑩇𑨛𑨹 𑨰 𑨙𑨚𑨊𑨍𑨙𑨹 𑨞𑨂𑨘𑩇𑨪𑨊𑨝𑨊𑨢𑨴, 𑨀𑨭𑨁𑨞𑩇𑨫𑨘𑨊𑨮𑨹 𑨋𑨂𑨮𑨬𑨢𑨂𑨊𑨬𑨊𑨝𑨊𑨢𑨴, 𑨀𑨬𑨏𑩇𑨋𑩇𑨫𑨉𑨙𑨹 𑨲𑨊𑨝𑩇𑨙𑩇𑨪𑨊, 𑨀𑨊𑨬𑨪𑨹 𑨞𑨂𑨘𑩇𑨪𑨝𑨁𑨜𑨊𑨝𑨊𑨝𑨊𑨢𑨴, 𑨣𑨁𑨙𑩇𑨫𑨁𑨙𑨆𑨝𑨂𑨭𑩇𑨪𑨓𑩇𑨥𑨝𑨄𑨊𑨹 𑨋𑨂𑨰𑨂𑨢𑨁𑨙𑨆 𑨬𑨲𑨘𑨄𑨊𑨹, 𑨞𑩇𑨫𑨙𑨁𑨫𑨂𑨊𑨞𑨆 𑨍𑨆𑨣𑨫𑨄𑨘, 𑨀𑨞𑩇𑨫𑨙𑨁𑨋𑨂𑨊𑨬𑨆 𑨛𑨫𑩇𑨮𑨝𑨄𑨝, 𑨀𑨡𑨁𑨫𑨙𑨁𑨹 𑨮𑩇𑨫𑨛𑩇𑨜𑨊(𑨮𑨂𑨛𑩇𑨜𑨊)𑨜𑨁𑨢𑨂𑨋𑩇𑨙𑨊𑨝𑨊𑨢𑨴, 𑨀𑨝𑨡𑨁𑨡𑨂𑨊𑨙𑨹 𑨞𑩇𑨫𑨥𑩇𑨓𑨪𑨊, 𑨀𑨝𑨭𑨢𑨫𑩇𑨛𑨝𑨁𑨊𑨪𑨆 𑨠𑨬𑨄𑨊𑨹, 𑨮𑨊𑨰𑩇𑨙𑨊 𑨰𑨫𑩇𑨭𑨰𑨙𑩇𑨙𑩇𑨭𑨊𑨝𑨊𑨢𑨴, 𑨞𑨁𑨙𑨊 𑨠𑨆𑨜𑨁𑨰𑨙𑩇𑨙𑩇𑨭𑨊𑨝𑨊𑨢𑨴, 𑨫𑨊𑨥𑨊 𑨀𑨊𑨫𑩇𑨪𑨞𑨂𑨛𑩇𑨍𑨬𑨊𑨝𑨊𑨢𑨴, 𑨰𑨊𑨫𑩇𑨚𑨭𑨊𑨱𑨹 𑨝𑨁𑨫𑩇𑨭𑨊𑨘𑨝𑨍𑨫𑨰𑨢𑩇𑨞𑩇𑨫𑨰𑩇𑨚𑨁𑨙𑨊𑨝𑨊𑨢𑨴, 𑨀𑨞𑩇𑨫𑨢𑨄𑨪𑨆 𑨥𑩇𑨓𑨊𑨝𑨄𑨝, 𑨀𑨣𑨁𑨝𑩇𑨙𑩇𑨪𑨹 𑨞𑩇𑨫𑨙𑨁𑨡𑨊𑨝𑨄𑨝, 𑨭𑨁𑨮𑨂𑨛𑩇𑨜𑨹 𑨰𑩇𑨭𑨫𑨄𑨘, 𑨀𑨊𑨰𑩇𑨭𑨛𑨝𑨁𑨊𑨪𑨆 𑨎𑨆𑨯𑨄𑨘, 𑨀𑨰𑨄𑨣𑨝𑨋𑨆 𑨫𑨂𑨊𑨞𑨄𑨘, 𑨀𑨞𑩇𑨫𑨙𑨁𑨰𑨢𑨹 𑨋𑨊𑨪𑨄𑨝, 𑨀𑨬𑨁𑨞𑩇𑨙𑨹 𑨋𑨊𑨢𑨄𑨊𑨹, 𑨀𑨝𑨂𑨞𑨬𑨁𑨞𑩇𑨙𑨆 𑨫𑨂𑨊𑨞𑨄𑨊𑨹 𑨀𑨰𑨸𑨰𑩇𑨫𑨉𑨯𑩇𑨔 𑨀𑨊𑨫𑨂𑨊𑨞𑩇𑨪𑨄𑨊𑨹, 𑨭𑨁𑨞𑩇𑨫𑨢𑨂𑨋𑩇𑨙𑨹 𑨰𑩇𑨋𑨝𑩇𑨜𑨄𑨡𑩇𑨪𑨹, 𑨭𑨁𑨰𑨢𑩇𑨞𑩇𑨫𑨪𑨂𑨋𑩇𑨙𑨆 𑨜𑨊𑨙𑨂𑨡𑨁𑨹, 𑨰𑨸𑨭𑩇𑨫𑨉𑨙 𑨀𑨊𑨪𑨙𑨝𑨄𑨊𑨹, 𑨞𑩇𑨫𑨣𑩇𑨤𑨁𑨝𑩇𑨝𑨆 𑨍𑩇𑨫𑨝𑩇𑨚𑨄𑨊𑨹, 𑨭𑨁𑨢𑨂𑨋𑩇𑨙𑨹 𑨞𑨫𑨁𑨛𑨊𑨎𑩇𑨝𑨄𑨊𑨹, 𑨞𑨫𑨁𑨢𑨂𑨋𑩇𑨙𑨰𑩇𑨙𑩇𑨫𑨉𑨯𑩇𑨘𑨪𑨊, 𑨀𑨆𑨎𑨊𑨛𑨂𑨙𑩇𑨙𑨁𑨊𑨫𑩇𑨘𑨹 𑨞𑨫𑨁𑨞𑨂𑨊𑨫𑩇𑨘𑨆 𑨥𑩇𑨓𑨊𑨝𑨄𑨝, 𑨞𑩇𑨫𑨙𑨁𑨯𑩇𑨕𑨁𑨙𑨆𑨙𑨁𑨊𑨙𑨊𑨝𑨊𑨍𑨙𑨞𑩇𑨫𑨙𑩇𑨪𑨂𑨙𑩇𑨞𑨝𑩇𑨝𑨊𑨝𑨊𑨸 𑨠𑨂𑨛𑩇𑨜𑨊𑨝𑨊𑨸 𑨡𑨍𑨭𑨙𑨊𑨸 𑨥𑩇𑨓𑨊𑨝𑨄, 𑨀𑨞𑩇𑨫𑨙𑨁𑨯𑩇𑨕𑨁𑨙𑨆 𑨝𑨁𑨫𑩇𑨭𑨊𑨘𑨄, 𑨰𑩇𑨚𑨁𑨙𑨆 𑨡𑨂𑨊𑨙𑨋𑨆𑨔𑩇𑨪𑨊𑨢𑨴, 𑨰𑩇𑨚𑨁𑨙𑨹 𑨰𑨫𑩇𑨭𑨰𑨙𑩇𑨙𑩇𑨭𑨊𑨬𑨆𑨋𑨝𑨁𑨊𑨪𑨊𑨪𑨊𑨸 𑨡𑨂𑨊𑨢𑨆𑨊, 𑨰𑨫𑩇𑨭 𑨀𑨁𑨢𑨄 𑨙𑨚𑨊𑨍𑨙𑨊𑨝𑨊𑨸 𑨭𑨁𑨮𑨄𑨯𑨙𑨹 𑨰𑨢𑩇𑨪𑨍𑨴 𑨍𑨂𑨘𑨊𑨹𑩂

(𑨜𑨫𑩇𑨢𑨊𑨝𑨂𑨰𑩇𑨢𑩇𑨫𑨉𑨙𑨁𑨹)

𑨰𑨛𑩇𑨜𑨫𑩇𑨢𑨰𑩇𑨙𑨂 𑨀𑨊𑨛𑨆𑨊 𑨋𑨬𑩇𑨪𑨊𑨘𑨹, 𑨢𑨜𑩇𑨪𑨄 𑨋𑨬𑩇𑨪𑨊𑨘𑨹, 𑨞𑨫𑩇𑨪𑨭𑨰𑨊𑨝𑨄 𑨋𑨬𑩇𑨪𑨊𑨘𑨹, 𑨰𑩇𑨭𑨫𑩇𑨚𑨹 𑨰𑨂𑨭𑩇𑨪𑨓𑩇𑨥𑨝𑨹, 𑨋𑨄𑨭𑨬𑨹, 𑨞𑨫𑨁𑨞𑨂𑨊𑨫𑩇𑨘𑨹, 𑨞𑨫𑨁𑨮𑨂𑨛𑩇𑨜𑨹, 𑨞𑨫𑩇𑨪𑨭𑨛𑨊𑨙𑨹, 𑨰𑩇𑨭𑨊𑨌𑩇𑨪𑨊𑨙𑨹 𑨡𑨍𑨭𑨙𑨆 𑨜𑨫𑩇𑨢𑨹, 𑨰𑨊𑨝𑩇𑨛𑩇𑨫𑨉𑨯𑩇𑨔𑨁𑨋𑨹, 𑨝𑨁𑨫𑩇𑨥𑩇𑨭𑨫𑨹, 𑨀𑨊𑨋𑨊𑨬𑨁𑨋𑨹, 𑨀𑨆𑨊𑨞𑨝𑨊𑨪𑨁𑨋𑨹, 𑨀𑨄𑨊𑨱𑨁𑨞𑨮𑩇𑨪𑨁𑨋𑨹, 𑨞𑩇𑨫𑨙𑩇𑨪𑨊𑨙𑩇𑨢𑨭𑨄𑨛𑨝𑨁𑨊𑨪𑨆 𑨭𑨁𑨥𑩇𑨓𑨄𑨊𑨹, 𑨰𑩇𑨭𑨊𑨌𑩇𑨪𑨊𑨙𑨆 𑨡𑨍𑨭𑨙𑨆 𑨜𑨫𑩇𑨢𑨭𑨁𑨝𑨪𑨹 𑨰𑨂𑨞𑩇𑨫𑨭𑨄𑨛𑨁𑨙𑨹 𑨝𑨄𑨊𑨫𑩇𑨪𑨊𑨘𑨁𑨋𑨹, 𑨰𑨸𑨠𑨆𑨜𑨁𑨍𑨊𑨢𑨁𑨊, 𑨀𑨡𑨁𑨝𑩇𑨝𑨹 𑨰𑨸𑨰𑩇𑨙𑨂𑨊𑨞𑨹, 𑨰𑨞𑩇𑨫𑨙𑨁𑨮𑨫𑨘𑨹, 𑨤𑨁𑨝𑩇𑨝𑨞𑩇𑨬𑨆𑨙𑨁𑨋𑨹𑩂

(𑨰𑨸𑨎𑨊𑨝𑨂𑨰𑩇𑨢𑩇𑨫𑨉𑨙𑨁𑨹)

𑨰𑨂𑨞𑩇𑨫𑨙𑨁𑨞𑨝𑩇𑨝𑨆 𑨡𑨍𑨭𑨙 𑨀𑨊𑨫𑩇𑨪𑨰𑨸𑨎𑨹, 𑨝𑩇𑨪𑨊𑨪𑨞𑩇𑨫𑨙𑨁𑨞𑨝𑩇𑨝𑨹, 𑨫𑨉𑨥𑨂𑨞𑩇𑨫𑨙𑨁𑨞𑨝𑩇𑨝𑨹, 𑨰𑨊𑨢𑨁𑨊𑨣𑨁𑨊𑨞𑩇𑨫𑨙𑨁𑨞𑨝𑩇𑨝𑨹, 𑨀𑨓𑩇𑨥𑨬𑨁𑨊𑨋𑨫𑨘𑨁𑨊𑨪𑨹, 𑨰𑨊𑨢𑨁𑨊𑨣𑨁𑨊𑨋𑨫𑨘𑨁𑨊𑨪𑨹, 𑨞𑨂𑨘𑩇𑨪𑨮𑩇𑨫𑨁𑨊𑨲𑨄𑨙𑩇𑨫𑨹, 𑨢𑨱𑨊𑨛𑨲𑨁𑨘𑨊𑨞𑨫𑨁𑨮𑨆𑨜𑨋𑨹, 𑨞𑩇𑨫𑨊𑨱𑨭𑨝𑨁𑨊𑨪𑨹, 𑨀𑨊𑨱𑨭𑨝𑨁𑨊𑨪𑨹𑩂
