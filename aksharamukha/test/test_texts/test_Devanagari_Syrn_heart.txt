ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܗܪܘܼܕ݁ܲܝܲܣܘܼܬ݁ܪܲܡ.

[ܣܲܡܟ݁ܫܝܼܦ݁ܬ݁ܲܡܵܬ݁ܪܘܼܟ݁ܵ]

.. ܢܲܡܲܗ ܣܲܪܒ݂ܲܓ̰ܢܵܝܲ ..

.. ܘܿܡ ܢܲܡܘܿ ܒ݁ܲܓ݁ܲܒ݂ܲܬ݁ܝܲܝ ܐܵܪܝܲܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܝܲܝ ..

ܐܵܪܝܵܒ݂ܲܠܘܿܟ݁ܝܼܬܹ݁ܫܒ݂ܲܪܲܒ݁ܘܿܕ݁ܝܼܣܲܬ݁ܒ݂ܘܿ ܓ݁ܲܡܒ݁ܝܼܪܵܝܵܡ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܝܵܡ ܟ̰ܲܪܝܵܡ ܟ̰ܲܪܲܡܵܢܘܿ ܒ݂ܝܲܒ݂ܲܠܘܿܟ݁ܲܝܲܬ݁ܝܼ ܣܡܲ. ܦ݁ܲܢܟ̰ܲ ܣܟ݁ܲܢܕ݁ܵܗ, ܬ݁ܵܡܫܟ̰ܲ ܣܒ݂ܲܒ݁ܵܒ݂ܲܫܘܼܢܝܵܢ ܦ݁ܲܫܝܲܬ݁ܝܼ ܣܡܲ..

ܝܼܗܲ ܫܵܪܝܼܦ݁ܘܼܬ݁ܪܲ ܪܘܼܦ݁ܲܡ ܫܘܼܢܝܲܬ݁ܵ, ܫܘܼܢܝܲܬ݁ܲܝܒ݂ܲ ܪܘܼܦ݁ܲܡ. ܪܘܼܦ݁ܵܢܲ ܦ݁ܪܘܼܬ݁ܲܟ݁ ܫܘܼܢܝܲܬ݁ܵ, ܫܘܼܢܝܲܬ݁ܵܝܵ ܢܲ ܦ݁ܪܘܼܬ݁ܲܓ݁ ܪܘܼܦ݁ܲܡ. ܝܲܕ݁ܪܘܼܦ݁ܲܡ ܣܵ ܫܘܼܢܝܲܬ݁ܵ, ܝܵ ܫܘܼܢܝܲܬ݁ܵ ܬ݁ܲܕ݁ܪܘܼܦ݁ܲܡ..

ܐܹܒ݂ܲܡܹܒ݂ܲ ܒ݂ܹܕ݁ܲܢܵܣܲܡܓ̰ܢܵܣܲܡܣܟ݁ܵܪܲܒ݂ܝܼܓ̰ܢܵܢܵܢܝܼ..

ܝܼܗܲܡ ܫܵܪܝܼܦ݁ܘܼܬ݁ܪܲ ܣܲܪܒ݂ܲܕ݁ܲܪܡܵܗ ܫܘܼܢܝܲܬ݁ܵܠܲܟ݁ܫܲܢܵ ܐܲܢܘܼܬ݁ܦ݁ܲܢܵ ܐܲܢܝܼܪܘܼܕ݁ܵ ܐܲܡܲܠܵ ܢܲ ܒ݂ܝܼܡܲܠܵ ܢܘܿܢܵ ܢܲ ܦ݁ܲܪܝܼܦ݁ܘܼܪܢܵܗ. ܬ݁ܲܣܡܵܟ̰ܵܪܝܼܦ݁ܘܼܬ݁ܪܲ ܫܘܼܢܝܲܬ݁ܵܝܵܡ ܢܲ ܪܘܼܦ݁ܲܡ, ܢܲ ܒ݂ܹܕ݁ܲܢܵ, ܢܲ ܣܲܡܓ̰ܢܵ, ܢܲ ܣܲܡܣܟ݁ܵܪܵܗ, ܢܲ ܒ݂ܝܼܓ̰ܢܵܢܵܢܝܼ. ܢܲ ܟ̰ܲܟ݁ܫܘܼܗܫܪܘܿܬ݁ܪܲܓ݁ܪܵܢܲܓ̰ܝܼܗܒ݂ܵܟ݁ܵܝܲܡܲܢܵܡܣܝܼ, ܢܲ ܪܘܼܦ݁ܲܫܲܒ݁ܕ݁ܲܓ݁ܲܢܕ݁ܲܪܲܣܲܣܦ݁ܪܲܫܬ݁ܲܒ݂ܝܲܕ݁ܲܪܡܵܗ. ܢܲ ܟ̰ܲܟ݁ܫܘܼܪܕ݁ܵܬ݁ܘܼܪܝܵܒ݂ܲܢܲ ܡܲܢܘܿܕ݁ܵܬ݁ܘܼܗ..

ܢܲ ܒ݂ܝܼܕ݁ܝܵ ܢܵܒ݂ܝܼܕ݁ܝܵ ܢܲ ܒ݂ܝܼܕ݁ܝܵܟ݁ܫܲܝܘܿ ܢܵܒ݂ܝܼܕ݁ܝܵܟ݁ܫܲܝܘܿ ܝܵܒ݂ܲܢܲ ܓ̰ܲܪܵܡܲܪܲܢܲܡ ܢܲ ܓ̰ܲܪܵܡܲܪܲܢܲܟ݁ܫܲܝܘܿ ܢܲ ܕ݁ܘܼܗܟ݁ܲܣܲܡܘܼܕ݁ܲܝܲܢܝܼܪܘܿܕ݁ܲܡܵܪܓ݁ܵ ܢܲ ܓ̰ܢܵܢܲܡ ܢܲ ܦ݁ܪܵܦ݁ܬ݁ܝܼܬ݁ܒ݂ܲܡ..

ܒ݁ܘܿܕ݁ܝܼܣܲܬ݁ܒ݂ܲܣܝܲ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܡܵܫܪܝܼܬ݁ܝܲ ܒ݂ܝܼܗܲܪܲܬ݁ܝܼ ܟ̰ܝܼܬ݁ܵܒ݂ܲܪܲܢܲܗ. ܟ̰ܝܼܬ݁ܵܒ݂ܲܪܲܢܲܢܵܣܬ݁ܝܼܬ݁ܒ݂ܵܕ݁ܲܬ݁ܪܲܣܬ݁ܘܿ ܒ݂ܝܼܦ݁ܲܪܝܵܣܵܬ݁ܝܼܟ݁ܪܵܢܬ݁ܘܿ ܢܝܼܫܬ݁ܲܢܝܼܪܒ݂ܵܢܲܗ. ܬ݁ܪܝܲܕ݁ܒ݂ܲܒ݂ܝܲܒ݂ܲܣܬ݁ܝܼܬ݁ܵܗ ܣܲܪܒ݂ܲܒ݁ܘܼܕ݁ܵܗ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܡܵܫܪܝܼܬ݁ܝܲ ܐܲܢܘܼܬ݁ܲܪܵܡ ܣܲܡܝܲܟ݁ܣܲܡܒ݁ܘܿܕ݁ܝܼܡܲܒ݁ܝܼܣܲܡܒ݁ܘܼܕ݁ܵܗ..

ܬ݁ܲܣܡܵܓ̰ܢܵܬ݁ܲܒ݂ܝܲܗ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܡܲܗܵܡܲܢܬ݁ܪܘܿ ܡܲܗܵܒ݂ܝܼܕ݁ܝܵܡܲܢܬ݁ܪܘܿܢܘܼܬ݁ܲܪܲܡܲܢܬ݁ܪܘܿܣܲܡܲܣܲܡܲܡܲܢܬ݁ܪܲܗ ܣܲܪܒ݂ܲܕ݁ܘܼܗܟ݁ܲܦ݁ܪܲܫܲܡܲܢܲܗ ܣܲܬ݁ܝܲܡܲܡܝܼܬ݁ܝܲܬ݁ܒ݂ܵܬ݁ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܝܵܡܘܼܟ݁ܬ݁ܘܿ ܡܲܢܬ݁ܪܲܗ.

ܬ݁ܲܕ݁ܝܲܬ݁ܵ- ܓ݁ܲܬܹ݁ ܓ݁ܲܬܹ݁ ܦ݁ܵܪܲܓ݁ܲܬܹ݁ ܦ݁ܵܪܲܣܲܡܓ݁ܲܬܹ݁ ܒ݁ܘܿܕ݁ܝܼ ܣܒ݂ܵܗܵ..

ܝܼܬ݁ܝܼ ܦ݁ܪܲܓ̰ܢܵܦ݁ܵܪܲܡܝܼܬ݁ܵܗܪܘܼܕ݁ܲܝܲܣܘܼܬ݁ܪܲܡ ܣܲܡܵܦ݁ܬ݁ܲܡ..
