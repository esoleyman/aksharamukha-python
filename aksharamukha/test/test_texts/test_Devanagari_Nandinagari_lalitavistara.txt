𑧉𑧉𑧒𑦽𑧊𑧒𑧍𑧠𑦽𑧈𑧟

𑧁𑧆𑧟 𑧍𑧈𑧠𑧊𑧄𑧔𑦿𑧠𑧀𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧚𑧅𑧠𑧇𑧟।

॥ 𑦬𑧞 𑧁𑧆𑧜 𑦿𑧋𑦿𑧒𑦰𑧁𑧞𑦽𑧑𑧂𑧈𑧠𑧇𑧞𑦽𑧉𑧜𑦮𑧀𑧑𑦽𑧔𑧂𑧠𑧈𑦽𑧒𑧌𑧠𑦹𑧒𑦽𑧍𑧈𑧠𑧊𑧄𑧔𑦿𑧠𑧀𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧈𑧠𑧇𑧋𑧠𑧈𑧑𑧊𑦮𑧂𑧠𑧈𑦽𑧠𑧇𑧚𑦮𑧄𑧔𑦿𑧠𑧀𑧚𑧅𑧠𑧇𑧜𑧡𑦽𑧓𑦽𑧑𑧁𑧑𑦰𑦽𑧂𑧠𑧈𑦽𑧠𑧇𑧔𑦽𑧠𑧂𑧁𑧠𑧁𑧚𑧅𑧠𑧇𑧟॥

೧೦ 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧍𑧞𑦿𑧈𑧠𑧋𑧁𑧂𑧈𑧒𑧊𑧈𑧠𑦽𑧜 𑦿𑧋𑧆𑧟।

𑦢𑦽𑧒 𑧎𑧒 𑧅𑧒𑦮𑧠𑧌𑧊𑧟 𑧍𑧞𑧊𑧖𑦿𑧠𑧀𑧟 𑦮𑧔𑧆𑧑𑧈𑧟। 𑦽𑦿𑧑 𑧆𑧑𑧞𑦰𑧉𑧠𑧇𑧋𑦽𑧍𑧎𑧍𑧠𑧈𑧛𑧟 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧔𑧂𑧁𑧓𑧇𑦽𑧚 𑧍𑧠𑧆 𑦿𑧋𑧅𑧒𑧈𑧠𑦿𑧑𑧈𑦮𑧍𑧎𑧍𑧠𑧈𑧛𑧟 𑧂𑧈𑧒𑧊𑧖𑦽𑧟 𑧂𑧔𑧈𑧍𑧠𑦮𑧖𑦽𑧟, 𑦿𑧋𑧅𑧒𑧋𑧠𑦳 𑧈𑦾𑧍𑧎𑧍𑧠𑧈𑧛𑧟 𑦯𑧑𑦿𑧁𑧓𑧇𑧅𑧜𑦵𑧁𑧓𑧇𑧍𑧠𑧊𑧑𑦿𑧁𑧓𑧇𑧂𑧈𑧒𑧂𑧕𑧈𑧠𑦼𑧛𑧈𑧠𑧎𑧒𑧈𑦼𑧠𑧇𑧍𑧔𑧊𑧈𑧠𑦼𑧂𑧈𑧒𑧂𑧕𑧈𑧠𑦼𑧛𑧋𑧠𑦳। 𑧇𑧚𑧁 𑦮𑧂𑧒𑧉𑧊𑧍𑧠𑦽𑧔𑧁𑧒 𑧆𑧎𑧑𑧁𑦰𑧈𑧚 𑧊𑧓𑦾𑧒𑦳𑦽𑧠𑧊𑧈𑧈𑦾𑧠𑧇𑧑𑧞𑦽𑧈𑧑𑧂𑦼𑧆𑧔𑦯𑧚𑧌𑧠𑧊𑧅𑧠𑧇𑧊𑦮𑧓𑧈𑧠𑧇𑦽𑧚 𑧍𑧠𑧆 𑦠𑧅𑧒𑧊𑧒𑧋𑧠𑧈𑧑𑧆𑧠𑧇𑧞𑦽𑧚। 𑦠𑧌𑧠𑦸𑧑𑧅𑧒𑧋𑧠𑦳 𑦽𑧕𑧈𑧠𑧇𑧋𑦽𑧍𑧎𑧍𑧠𑧈𑧛𑧟 𑧂𑧠𑧈𑦱𑧔𑧌𑧠𑧇𑧆𑧑𑦼𑧛𑧈𑧠𑧆𑧎𑦽𑧑 𑦳 𑧂𑧔𑧌𑧠𑧂𑧊𑧈𑧠𑧌𑧚𑦼𑧑𑧅𑧒𑧂𑧠𑧈𑧊𑧈𑧠𑧌𑦽𑧑 𑧊𑧒𑦽𑧈𑧠𑦿𑧒𑧁𑧒𑧈𑧠𑧇𑧕𑧎𑦽𑧜𑧈𑦼𑦰𑧊𑧑𑦮𑧠𑧌𑧎𑧈𑧠𑧆𑧠𑧇𑦮𑧕𑦸𑧑𑦰𑧑𑧈𑧂𑧠𑧈𑧑𑧍𑧑𑦿𑦽𑧉𑧚𑧌𑧔 𑦮𑧁𑧠𑧇𑧑𑧋𑦽𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒 𑧍𑧈𑧠𑧊𑧑𑧉𑧞𑦮𑧑𑧈𑧅𑧕𑧌𑧒𑦽𑧑𑧟 𑧍𑧠𑦾𑧒𑦽𑧑 𑦠𑧅𑧕𑧊𑧁𑧠। 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧞 𑧂𑧠𑧈𑧚𑦮𑧠𑧌𑧆𑧑𑦼𑧑𑧟 𑦮𑧔𑧍𑧔𑧆𑧑𑧁𑧒 𑦳 𑦮𑧠𑧌𑧒𑧂𑧞𑦽𑧒 𑧍𑧠𑧆। 𑦠𑧌𑧠𑦸𑧝 𑦳 𑧆𑧈𑧔𑦽𑧠𑦮𑧁𑧠𑧇𑧑𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒 𑧊𑧒𑦰𑧉𑧒𑦽𑧑𑧉𑧞𑦮𑧑𑧈𑧑𑧅𑧈𑦼𑧑𑧉𑧞𑦮𑧖𑦽𑧑𑧁𑧒 𑧈𑦽𑧠𑧁𑧅𑦿𑧠𑧈𑧞𑦮𑧈𑧚𑦼 𑦰𑧖𑧎𑧓𑦽𑧑𑧁𑧒 𑧆𑧑𑧈𑧠𑦰𑧞 𑧋𑧜𑧀𑧇𑧞𑦽𑧠𑧇𑧜 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧍𑧠𑧇 𑧂𑧔𑧈𑦽𑧜 𑦰𑦳𑧠𑦴𑧞𑦽𑧒 𑧍𑧠𑧆। 𑦿𑧚𑧊𑧁𑧑𑦰𑧇𑦮𑧠𑧌𑦰𑧞𑧀𑧈𑧠𑧊𑧑𑧍𑧔𑧈𑦰𑧈𑧔𑦺𑦮𑧒𑧁𑧠𑧁𑧈𑧆𑧎𑧜𑧈𑦰𑧑𑧋𑧠𑦳𑧑𑧈𑧠𑧀𑦮𑧑𑧇𑧒𑦮𑧑 𑦰𑦰𑧁𑦽𑧉𑧑𑦽𑧠𑧂𑧔𑧌𑧠𑧂𑧂𑦸𑧠𑦸𑦿𑧑𑧆𑧑𑧁𑧠𑧇𑧅𑧒𑧂𑧠𑧈𑧉𑧞𑧄𑧇𑧞𑦽𑧒 𑧍𑧠𑧆। 𑧍𑧈𑧠𑧊𑧚 𑦳 𑧋𑧑𑦮𑧠𑧇𑦰𑦼𑧑𑧟 𑧋𑧔𑦿𑧠𑧀𑧜𑦿𑧁𑧞 𑧈𑧑𑦵𑧑𑧁𑧞 𑧂𑧔𑧈𑧍𑧠𑦮𑧖𑦽𑧠𑧇 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧍𑧠𑧇 𑧂𑧔𑧈𑦽𑧜 𑦰𑦳𑧠𑦴𑧞𑦽𑧒 𑧍𑧠𑧆। 𑦠𑧁𑧚𑧁𑧛𑧊𑧞𑧈𑧕𑧂𑧚𑦼 𑧊𑧠𑧇𑧕𑧎𑧚𑧁 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧜 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧔𑧂𑧁𑧓𑧇𑦽𑧚 𑧍𑧠𑧆॥

𑧍𑧆𑧁𑧞𑦽𑧈𑧂𑧠𑧈𑧊𑧚𑧋𑧒𑦽𑧋𑧠𑦳 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧜 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧠। 𑦠𑦾 𑧊𑧒𑧋𑧠𑧊𑧑𑧆𑧒𑦽𑧠𑧈𑧜 𑧁𑧑𑧆 𑦿𑧑𑧈𑦮𑧑𑦳𑧑𑧈𑧠𑧇𑧜 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧍𑧠𑧇 𑧋𑧠𑧈𑧒𑧇𑧞 𑦽𑧚𑦵𑧋𑧠𑦳𑧑𑧍𑧎𑧆𑧑𑧁𑧜 𑧀𑧈𑦼𑧒𑦽𑧉𑧚 𑧁𑧒𑧊𑧒𑧌𑧠𑦸𑧜𑧡𑧀𑧜𑧆𑧔𑦯𑧟 𑧂𑧠𑧈𑧂𑦽𑦽𑧒 𑧍𑧠𑧆। 𑦽𑧞 𑦽𑦾𑧑 𑧂𑧠𑧈𑧂𑦽𑧒𑦽𑧞 𑦿𑧖𑧌𑧠𑦸𑧠𑧊𑧑 𑧋𑧔𑧅𑧑𑧞𑦰𑧜 𑧁𑧑𑧆 𑦽𑧔𑧌𑧒𑦽𑦮𑧑𑧇𑧒𑦮𑧜 𑦿𑧚𑧊𑧂𑧔𑦽𑧠𑧈𑧜 𑦿𑦮𑧠𑧌𑧒𑦼𑧚𑧁 𑦮𑧈𑦽𑧉𑧚𑧁 𑧂𑧈𑧒𑦰𑧖𑧎𑧠𑧇𑧜𑦽𑧠𑦾𑧑𑧂𑧇𑦽𑧒 𑧍𑧠𑧆। 𑦤𑦽𑧠𑦾𑧑𑧂𑧠𑧇 𑦳 𑦰𑦰𑧁𑦽𑧉𑧍𑧠𑦾𑧜 𑧈𑧑𑦵𑧑𑧁𑧞 𑧋𑧔𑦿𑧠𑧀𑧜𑦿𑧁𑧞 𑦽𑧞 𑦳 𑧆𑧎𑧑𑧞𑦽𑧞 𑦵𑧁𑦮𑧑𑧇𑧞 𑦰𑧑𑦾𑧑𑧅𑧒𑧈𑧅𑧠𑧇𑧅𑧑𑧌𑦽𑧠—

𑧋𑧑𑧍𑧠𑦽𑧠𑧈𑧑𑦼𑧒 𑧇𑧑𑧁𑧒 𑧂𑧠𑧈𑦳𑧉𑧞𑦽𑧒 𑧆𑧁𑧔𑧌𑧠𑧇𑧉𑧜𑦮𑧚
𑧍𑧞𑦯𑧠𑧇𑧑 𑧉𑧒𑧂𑧒𑧋𑧠𑦳 𑦰𑦼𑧁𑧑𑧂𑧒 𑦳 𑧀𑧑𑦽𑧔𑦽𑧞𑦽𑧠𑧈𑧆𑧠।
𑧇𑧚 𑧋𑧒𑧉𑧠𑧂𑧇𑧜𑦰 𑧂𑧖𑦾𑧔 𑧉𑧝𑦮𑧒𑦮 𑦠𑧂𑧠𑧈𑧆𑧚𑧇𑧑𑧟
𑦽𑧚𑧌𑧠𑧊𑧚𑧌𑧔 𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧔 𑧂𑧔𑧈𑧑 𑧄𑧎𑧔𑦮𑧉𑧠𑧂𑦮𑧜𑦸𑧠𑧇𑧟॥೧॥

𑦮𑧒𑧞 𑦽𑧕 𑦵𑧁𑧍𑧠𑧇 𑦠𑧁𑧔𑧊𑧈𑧠𑦽𑧁𑦽𑧑𑧞 𑦮𑧈𑧜𑦽𑧒
𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧆𑧑𑦰𑦽𑧔 𑧍𑧔𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧔 𑧋𑧒𑧌𑧠𑧇𑦼𑧑𑧈𑧠𑦾𑧆𑧠।
𑧂𑧈𑧒𑧂𑧑𑦳𑧁𑧑𑧈𑧠𑦾 𑧄𑧎𑧔𑦿𑧑𑧈𑦮 𑦠𑦰𑧠𑧈𑧇𑧑𑧁𑧚
𑦠𑧁𑧠𑧇𑧑𑧞𑧋𑧠𑦳 𑧍𑦽𑧠𑦽𑧠𑧊𑧁𑧇𑧔𑦽𑧑𑧁𑧆𑧖𑦽𑧚 𑧊𑧒𑧁𑧚𑦽𑧔𑧆𑧠॥೨॥

𑧉𑧜𑦮𑧜𑦽𑧠𑦽𑧈𑧚𑧌𑧔 𑦳𑦽𑧔𑧍𑦽𑧠𑧇𑧂𑦾𑧚 𑧊𑧒𑧀𑧒𑦵𑧠𑦷𑧜
𑧎𑧚𑦽𑧔𑧂𑧠𑧈𑦽𑧓𑦽𑧠𑧇𑦮𑧔𑧋𑧉𑧜 𑧇𑦾 𑧍𑧞𑧅𑧊𑧞𑦽𑧒।
𑧇𑦾 𑦳𑧑𑧁𑧒𑧈𑧜𑧀𑦮𑧠𑧌𑧇𑧔 𑧍𑧞𑧍𑧠𑦾𑧒𑦽𑧔 𑧋𑧓𑦽𑧒𑧅𑧑𑧊𑧟
𑦽𑧍𑧠𑧆𑧒𑧁𑧠𑧊𑧒𑧀𑧒𑦵𑧠𑦷 𑦮𑧒𑧆𑦾𑧜 𑧉𑧒𑧂𑧒𑧋𑧑𑧍𑧠𑦽𑧠𑧈𑧆𑧑𑦽𑧠𑧈𑧚॥೩॥

𑧁𑧚𑦽𑧍𑧠𑧇 𑦡𑦳𑧈𑧒𑧇 𑦤𑦽𑧠𑦽𑧈𑧒 𑧊𑧑 𑦽𑧠𑧈𑧒𑧉𑧜𑦮𑧚
𑧍𑧈𑧠𑧊𑧚𑧌𑧔 𑦿𑧚𑧊𑧆𑧁𑧔𑦵𑧚𑧌𑧠𑧊𑧇𑧆𑧚𑧊 𑦵𑧚𑧌𑧠𑦹𑧟।
𑧁𑧑𑧆𑧑𑧂𑧒 𑦽𑧚𑧌 𑧉𑧒𑧂𑧒𑧁𑧑𑧞 𑧁 𑧎𑧒 𑧊𑧒𑦽𑧠𑦾 𑧇𑧕𑧇𑧞
𑧇𑦽𑧠𑧈𑧚𑧌𑧔 𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧔 𑧂𑧔𑧈𑧑 𑧄𑧎𑧔𑦮𑧉𑧠𑧂𑦮𑧜𑦸𑧠𑧇𑧟॥೪॥

𑧍𑧜 𑦳𑧒𑦽𑧠𑦽𑧀𑧑𑧈 𑦵𑦰𑦽𑧑𑧞 𑧊𑧒𑧊𑧒𑧀𑧑 𑧊𑧒𑦳𑧒𑦽𑧠𑧈𑧑
𑦪𑦮𑦮𑧠𑧌𑦼𑧚𑧁 𑦠𑧇𑧔 𑦵𑧑𑧁𑦽𑧒 𑧋𑧔𑦿𑧠𑧀𑧍𑦽𑧠𑦽𑧠𑧊𑧟।
𑦠𑦿𑧖𑧋𑧠𑧇𑧈𑧕𑧂𑧈𑧎𑧒𑦽𑧍𑧠𑧇 𑦰𑦽𑧒𑧞 𑦳 𑧊𑧚𑦽𑧠𑦽𑧒
𑦮𑧒𑧞 𑧊𑧑 𑧂𑧔𑧁𑧜𑧡𑦾 𑧉𑧒𑧂𑧒𑧁𑧜𑧡𑦮𑧠𑧌𑧈𑦿𑧖𑧋𑧠𑧇𑧈𑧕𑧂𑧑𑧆𑧠॥೫॥

𑦢𑦽𑧠𑧇𑧔𑦮𑧠𑦽𑧠𑧊𑧑 𑧍 𑦿𑧚𑧊𑧂𑧔𑦽𑧠𑧈𑧜 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧞 𑦿𑧒𑧊𑧠𑧇𑧛𑧟 𑦮𑧔𑧍𑧔𑧆𑧛𑧈𑧅𑧠𑧇𑧈𑧠𑦳𑧠𑧇 𑦽𑦽𑧠𑧈𑧛𑧊𑧑𑧞𑦽𑧈𑧠𑦿𑧀𑧚। 𑦽𑦽𑧠𑧈 𑧀𑧑𑦽𑧠𑧈𑧠𑧇𑧋𑧠𑦳 𑦳𑧚𑦸𑧓𑧊𑧈𑧠𑦰𑧑𑧋𑧠𑦳 𑧍𑧠𑦾𑧑𑧂𑧒𑦽𑧑 𑦠𑧅𑧕𑧊𑧁𑧠। 𑧂𑧈𑧒𑧋𑧚𑧌𑧑𑧟 𑧋𑧑𑦮𑧠𑧇𑧑𑧟 𑧋𑧔𑦿𑧠𑧀𑧜𑦿𑧁𑧂𑧠𑧈𑧆𑧔𑦯𑧑𑧟 𑧂𑧠𑧈𑦮𑧠𑧈𑧑𑧆𑧞𑦽𑧟॥

𑦠𑦾 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊 𑦤𑧈𑦰𑧍𑧑𑧈𑦳𑧞𑦿𑧁𑧆𑧇𑧞 𑧉𑧒𑧂𑧒𑧃𑧉𑦮𑧆𑧑𑦿𑧑𑧇 𑦿𑧒𑧊𑧠𑧇𑧑𑧈𑧠𑧌𑧍𑧔𑧊𑧈𑧠𑦼𑦽𑧒𑧈𑦮𑧞 𑧍𑧆𑧞𑦽𑧑𑧁𑧠𑧆𑦼𑧒𑧈𑦽𑧠𑧁𑧂𑧠𑧈𑦽𑧠𑧇𑧔𑧂𑧠𑦽𑧞 𑧊𑧒𑧋𑧠𑧊𑧑𑧆𑧒𑦽𑧠𑧈𑧆𑧑𑦳𑧑𑧈𑧠𑧇𑧆𑧚𑧊𑧆𑧑𑧎-𑦮𑦽𑧆𑧑𑧞 𑧆𑧚 𑧅𑧜 𑦤𑧂𑧑𑧀𑧠𑧇𑧑𑧇 𑧉𑧒𑧂𑧒𑧞 𑧋𑧒𑦮𑧠𑧌𑧑𑧂𑧇𑧍𑧒। 𑧄𑧠𑧈𑧑𑧎𑧠𑧆𑧓𑦯𑧈𑧜𑧌𑧠𑦸𑧓𑧂𑧔𑧌𑧠𑦮𑧈𑧍𑧑𑧈𑧒𑧞 𑦠𑧞𑦰𑧉𑧒𑧂𑧒𑧞 𑧊𑧞𑦰𑧉𑧒𑧂𑧒𑧞 𑧆𑦰𑧀𑧉𑧒𑧂𑧒𑧞 𑧆𑧞𑦰𑧉𑧠𑧇𑧉𑧒𑧂𑧒𑧞 𑦠𑧞𑦰𑧔𑧉𑧓𑧇𑧉𑧒𑧂𑧒𑧞 𑧋𑦮𑧑𑧈𑧒𑧉𑧒𑧂𑧒𑧞 𑧄𑧠𑧈𑧎𑧠𑧆𑧊𑧉𑧒𑧉𑧒𑧂𑧒𑧞 𑧂𑧑𑧈𑧔𑧌𑧠𑧇𑧉𑧒𑧂𑧒𑧞 𑦿𑧠𑧈𑧑𑧊𑧒𑦺𑧉𑧒𑧂𑧒𑧞 𑦮𑧒𑧈𑧑𑦽𑧉𑧒𑧂𑧒𑧞 𑦿𑧑𑦮𑧠𑧌𑧒𑦼𑧠𑧇𑧉𑧒𑧂𑧒𑧞 𑦤𑦰𑧠𑧈𑧉𑧒𑧂𑧒𑧞 𑧍𑧞𑦯𑧠𑧇𑧑𑧉𑧒𑧂𑧒𑧞 𑦠𑧁𑧔𑧉𑧜𑧆𑧉𑧒𑧂𑧒𑧞 𑦠𑧊𑧆𑧕𑧈𑧠𑧀𑧉𑧒𑧂𑧒𑧞 𑦿𑧈𑦿𑧉𑧒𑧂𑧒𑧞 𑦯𑧑𑧌𑧠𑧇𑧉𑧒𑧂𑧒𑧞 𑦳𑧓𑧁𑧉𑧒𑧂𑧒𑧞 𑧉𑧕𑧁𑧉𑧒𑧂𑧒𑧞 𑧎𑧕𑦼𑧉𑧒𑧂𑧒𑧞 𑧆𑧀𑧠𑧇𑧑𑦮𑧠𑧌𑧈𑧊𑧒𑧍𑧠𑦽𑧈𑧉𑧒𑧂𑧒𑧞 𑧂𑧔𑧌𑧠𑧂𑧉𑧒𑧂𑧒𑧞 𑦿𑧚𑧊𑧉𑧒𑧂𑧒𑧞 𑧁𑧑𑦰𑧉𑧒𑧂𑧒𑧞 𑧇𑦮𑧠𑧌𑧉𑧒𑧂𑧒𑧞 𑦰𑧞𑧀𑧈𑧠𑧊𑧉𑧒𑧂𑧒𑧞 𑦮𑧒𑧁𑧠𑧁𑧈𑧉𑧒𑧂𑧒𑧞 𑧆𑧎𑧜𑧈𑦰𑧉𑧒𑧂𑧒𑧞 𑦠𑧍𑧔𑧈𑧉𑧒𑧂𑧒𑧞 𑦰𑧈𑧔𑦺𑧉𑧒𑧂𑧒𑧞 𑧆𑧖𑦰𑦳𑦮𑧠𑧈𑧉𑧒𑧂𑧒𑧞 𑧊𑧑𑧇𑧍𑧈𑧔𑦽𑧉𑧒𑧂𑧒𑧞 𑧅𑧝𑧆𑦿𑧚𑧊𑧉𑧒𑧂𑧒𑧞 𑦠𑧞𑦽𑧈𑧓𑦮𑧠𑧌𑦿𑧚𑧊𑧉𑧒𑧂𑧒𑧞 𑦤𑦽𑧠𑦽𑧈𑦮𑧔𑧈𑧔𑦿𑧠𑧊𑧓𑧂𑧉𑧒𑧂𑧒𑧞 𑦠𑧂𑧈𑦰𑧜𑦺𑧑𑧁𑧓𑧉𑧒𑧂𑧒𑧞 𑧂𑧕𑧈𑧠𑧊𑧊𑧒𑦿𑧚𑧎𑧉𑧒𑧂𑧒𑧞 𑦤𑦽𑧠𑦮𑧠𑧌𑧚𑧂𑧉𑧒𑧂𑧒𑧞 𑧁𑧒𑦮𑧠𑧌𑧚𑧂𑧉𑧒𑧂𑧒𑧞 𑧊𑧒𑦮𑧠𑧌𑧚𑧂𑧉𑧒𑧂𑧒𑧞 𑧂𑧠𑧈𑦮𑧠𑧌𑧚𑧂𑧉𑧒𑧂𑧒𑧞 𑧍𑧑𑦰𑧈𑧉𑧒𑧂𑧒𑧞 𑧊𑦵𑧠𑧈𑧉𑧒𑧂𑧒𑧞 𑧉𑧚𑦯𑧂𑧠𑧈𑦽𑧒𑧉𑧚𑦯𑧉𑧒𑧂𑧒𑧞 𑦠𑧁𑧔𑦿𑧠𑧈𑧔𑦽𑧉𑧒𑧂𑧒𑧞 𑧋𑧑𑧍𑧠𑦽𑧠𑧈𑧑𑧊𑧈𑧠𑦽𑧑𑧞 𑦰𑦼𑧁𑧑𑧊𑧈𑧠𑦽𑧉𑧒𑧂𑧒𑧞 𑦤𑦽𑧠𑦮𑧠𑧌𑧚𑧂𑧑𑧊𑧈𑧠𑦽𑧉𑧒𑧂𑧒𑧞 𑧁𑧒𑦮𑧠𑧌𑧚𑧂𑧑𑧊𑧈𑧠𑦽𑧉𑧒𑧂𑧒𑧞 𑧂𑧑𑦿𑧉𑧒𑦯𑧒𑦽𑧉𑧒𑧂𑧒𑧞 𑦿𑧠𑧊𑧒𑧈𑧔𑦽𑧠𑦽𑧈𑧂𑦿𑧍𑧞𑧀𑧒𑧉𑧒𑧂𑧒𑧞 𑧇𑧑𑧊𑦿𑧠𑦿𑧋𑧜𑦽𑧠𑦽𑧈𑧂𑦿𑧍𑧞𑧀𑧒𑧉𑧒𑧂𑧒𑧞 𑧆𑧀𑧠𑧇𑧑𑧎𑧑𑧈𑧒𑦼𑧓𑧉𑧒𑧂𑧒𑧞 𑧍𑧈𑧠𑧊𑧈𑧔𑦽𑧍𑧞𑦰𑧠𑧈𑧎𑦼𑧓𑧉𑧒𑧂𑧒𑧞 𑧊𑧒𑦿𑧠𑧇𑧑𑧁𑧔𑧉𑧜𑧆𑧑𑧊𑧒𑧆𑧒𑧋𑧠𑧈𑧒𑦽𑧉𑧒𑧂𑧒𑧞 𑦦𑧌𑧒𑦽𑧂𑧍𑧠𑦽𑧂𑧠𑦽𑧑𑧞 𑧈𑧜𑦳𑧆𑧑𑧁𑧑𑧞 𑧀𑧈𑦼𑧓𑧂𑧠𑧈𑧚𑦮𑧠𑧌𑧒𑦼𑧓𑧉𑧒𑧂𑧒𑧞 𑦰𑦰𑧁𑧂𑧠𑧈𑧚𑦮𑧠𑧌𑧒𑦼𑧓𑧉𑧒𑧂𑧒𑧞 𑧍𑧈𑧠𑧊𑧝𑧌𑧀𑧒𑧁𑧒𑧌𑧠𑧇𑧞𑦿𑧑𑧞 𑧍𑧈𑧠𑧊𑧍𑧑𑧈𑧍𑧞𑦰𑧠𑧈𑧎𑦼𑧓𑧞 𑧍𑧈𑧠𑧊𑧅𑧕𑦽𑧈𑧔𑦽𑦰𑧠𑧈𑧎𑦼𑧓𑧆𑧠। 𑦡𑧍𑧑𑧞 𑧅𑧜 𑦤𑧂𑧑𑧀𑧠𑧇𑧑𑧇 𑦳𑦽𑧔𑧌𑧠𑧌𑧌𑧠𑦸𑧓𑧉𑧒𑧂𑧓𑧁𑧑𑧞 𑦮𑦽𑧆𑧑𑧞 𑦽𑧠𑧊𑧞 𑧋𑧒𑧌𑧠𑧇𑧑𑧂𑧇𑧒𑧌𑧠𑧇𑧍𑧒?

𑦠𑦾 𑧊𑧒𑧋𑧠𑧊𑧑𑧆𑧒𑦽𑧠𑧈𑧜 𑦿𑧑𑧈𑦮𑧑𑦳𑧑𑧈𑧠𑧇𑧜 𑧊𑧒𑧍𑧠𑧆𑧒𑦽𑧟 𑧂𑧠𑧈𑧎𑧍𑧒𑦽𑧊𑦿𑧁𑧜 𑧁𑧒𑧎𑦽𑧆𑧑𑧁𑧆𑦿𑦿𑧈𑧠𑧂 𑦢𑧆𑧑𑧞 𑦰𑧑𑦾𑧑𑧆𑧅𑧑𑧌𑦽—

𑦡𑧋𑧠𑦳𑧈𑧠𑧇𑧞 𑧋𑧔𑦿𑧠𑧀𑧍𑦽𑧠𑦽𑧠𑧊𑧍𑧠𑧇 𑧉𑧜𑦮𑧚 𑧉𑧜𑦮𑧑𑧁𑧔𑧊𑧈𑧠𑦽𑧒𑧁𑧜।
𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧟 𑧍𑧈𑧠𑧊𑧋𑧑𑧍𑧠𑦽𑧠𑧈𑧚𑧌𑧔 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧔𑧂𑧑𑦰𑦽𑧟॥೬॥

𑧇𑧚𑧌𑧑𑧆𑧎𑧞 𑧁𑧑𑧆𑧀𑧚𑧇𑧞 𑧉𑧒𑧂𑧓𑧁𑧑𑧞 𑧁 𑧂𑧠𑧈𑦵𑧑𑧁𑧆𑧒।
𑦽𑦽𑧠𑧈𑧛𑧌 𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧟 𑧍𑧞𑦽𑧜 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧔𑧂𑧑𑦰𑦽𑧟॥೭॥

𑧊𑦮𑧠𑦽𑧠𑧈𑧞 𑦳𑧑𑧍𑧠𑧇 𑧁 𑧂𑧋𑧠𑧇𑧑𑧆𑧒 𑧆𑧕𑧈𑧠𑧀𑧑𑧁𑧞 𑦽𑧍𑧠𑧇 𑧁𑧛𑧊 𑦳।
𑧋𑧒𑧌𑧠𑧇𑧇𑧒𑧌𑧠𑧇𑧚 𑦮𑦾𑧞 𑧎𑧠𑧇𑧚𑧁𑧞 𑧉𑧒𑧂𑧒𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧇 𑧂𑧑𑧈𑦰𑧆𑧠॥೮॥

𑦿𑧚𑧊𑦿𑧚𑧊𑧜 𑧎𑧠𑧇𑦽𑧒𑦿𑧚𑧊𑧟 𑧍𑧈𑧠𑧊𑦿𑧚𑧊𑧜𑦽𑧠𑦽𑧆𑧜 𑧊𑧒𑧅𑧔𑧟।
𑦠𑧍𑧆𑧋𑧠𑦳 𑧊𑧒𑧋𑧒𑧌𑧠𑦸𑧋𑧠𑦳 𑧉𑧜𑦮𑧚𑧌𑧠𑧊𑧂𑧠𑧈𑦽𑧒𑧂𑧔𑦿𑧠𑦰𑧉𑧟॥೯॥

𑦠𑧍𑧠𑧇𑧛𑧊 𑦽𑧠𑧊𑧁𑧔𑧅𑧑𑧊𑧚𑧁 𑧂𑧠𑧈𑦵𑧠𑦷𑧜𑧂𑧑𑧇𑧚 𑧊𑧒𑧋𑧚𑧌𑦽𑧟।
𑧋𑧒𑦮𑧠𑧌𑧒𑦽𑧞 𑧋𑧒𑧌𑧠𑧇𑧇𑧒𑧌𑧠𑧇𑧑𑧆𑧒 𑧍𑧈𑧠𑧊𑧉𑧜𑦮𑧂𑧈𑧑𑧇𑦼𑧆𑧠॥೧೦॥

𑦢𑦽𑧒 𑧎𑧒 𑧅𑧒𑦮𑧠𑧌𑧊𑧜 𑦿𑧋 𑦿𑧑𑧈𑦮𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧚𑧁 𑧍𑧑𑧈𑧠𑧀𑧞 𑧉𑧒𑧂𑧒𑧞 𑧋𑧒𑧌𑧠𑧇𑧞𑦽𑧚 𑧍𑧠𑧆। 𑦽𑦽𑧠𑧈 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧀𑧒𑧍𑧠𑦾𑧑𑧁𑧚𑧁 𑦽𑧚𑧌𑧑𑧞 𑦿𑧑𑧈𑦮𑧑𑦼𑧑𑧞 𑧆𑧑𑦽𑧖𑦮𑧑𑧞 𑧊𑧑𑦳𑧇𑦽𑧑𑧞 𑧇𑦿𑧑 𑦠𑦮𑧑𑧈𑧞 𑧂𑧈𑧒𑦮𑧓𑧈𑧠𑦽𑧇𑧞𑦽𑧒 𑧍𑧠𑧆, 𑦽𑦿𑧑 𑦠𑧁𑧒𑦽𑧠𑧇𑧟 𑧍𑧈𑧠𑧊𑧍𑧞𑧍𑧠𑦮𑧑𑧈𑧋𑧄𑧠𑦿𑧜 𑧁𑧒𑧋𑧠𑦳𑧈𑦽𑧒 𑧍𑧠𑧆। 𑦡𑦮𑧑𑧈𑧚 𑧂𑧈𑧒𑦮𑧓𑧈𑧠𑦽𑧠𑧇𑧆𑧑𑧁𑧚 𑦡𑦽𑧠𑧆𑧂𑧈𑧎𑧒𑦽𑧋𑧄𑧠𑦿𑧜 𑧁𑧒𑧋𑧠𑦳𑧈𑦽𑧒 𑧍𑧠𑧆। 𑦢𑦮𑧑𑧈𑧚 𑦢𑧞𑦿𑧠𑧈𑧒𑧇𑧊𑧛𑦮𑧉𑧠𑧇𑧋𑧄𑧠𑦿𑧟। 𑦣𑦮𑧑𑧈𑧚 𑦣𑦽𑧒𑧄𑧎𑧔𑧉𑧞 𑦵𑦰𑦿𑧒𑦽𑧒। 𑦤𑦮𑧑𑧈𑧚 𑦤𑧂𑦿𑧠𑧈𑧊𑧄𑧎𑧔𑧉𑧞 𑦵𑦰𑦿𑧒𑦽𑧒। 𑦥𑦮𑧑𑧈𑧚 𑦥𑧁𑧍𑦽𑧠𑦽𑧠𑧊𑧞 𑦵𑦰𑦿𑧒𑦽𑧒। 𑦪𑦮𑧑𑧈𑧚 𑦪𑧌𑦼𑧑𑧍𑧆𑧔𑦽𑧠𑦾𑧑𑧁𑦿𑧜𑧌𑧋𑧄𑧠𑦿𑧟। 𑦫𑦮𑧑𑧈𑧚 𑦫𑧈𑧠𑧇𑧑𑧂𑦾𑧟 𑧋𑧠𑧈𑧚𑧇𑧑𑧁𑧒𑦽𑧒। 𑦬𑦮𑧑𑧈𑧚 𑦬𑦱𑧜𑦽𑧠𑦽𑧈𑧋𑧄𑧠𑦿𑧟। 𑦭𑦮𑧑𑧈𑧚 𑦭𑧂𑧂𑧑𑦿𑧔𑦮𑧋𑧄𑧠𑦿𑧟। 𑦠𑧞𑦮𑧑𑧈𑧚 𑦠𑧆𑧜𑦱𑧜𑦽𑧠𑧂𑦽𑧠𑦽𑧒𑧋𑧄𑧠𑦿𑧟। 𑦠𑧟𑦮𑧑𑧈𑧚 𑦠𑧍𑧠𑦽𑧞𑦰𑧆𑧁𑧋𑧄𑧠𑦿𑧜 𑧁𑧒𑧋𑧠𑦳𑧈𑦽𑧒 𑧍𑧠𑧆। 𑦮𑦮𑧑𑧈𑧚 𑦮𑧈𑧠𑧆𑧊𑧒𑧂𑧑𑦮𑧑𑧊𑦽𑧑𑧈𑧋𑧄𑧠𑦿𑧟। 𑦯𑦮𑧑𑧈𑧚 𑦯𑧍𑧆𑧍𑧈𑧠𑧊𑧀𑧈𑧠𑧆𑧋𑧄𑧠𑦿𑧟। 𑦰𑦮𑧑𑧈𑧚 𑦰𑧞𑧅𑧓𑧈𑧀𑧈𑧠𑧆𑧂𑧠𑧈𑦽𑧓𑦽𑧠𑧇𑧍𑧆𑧔𑦽𑧠𑧂𑧑𑦿𑧑𑧊𑦽𑧑𑧈𑧋𑧄𑧠𑦿𑧟। 𑦱𑦮𑧑𑧈𑧚 𑦱𑧁𑧂𑦸𑧉𑧑𑧊𑧒𑦿𑧠𑧇𑧑𑧆𑧜𑧎𑧑𑧞𑧀𑦮𑧑𑧈𑧊𑧒𑧀𑧆𑧁𑧋𑧄𑧠𑦿𑧟। 𑦲𑦮𑧑𑧈𑧚𑧡𑦲𑧠𑦰𑧊𑧒𑧋𑧔𑦿𑧠𑧀𑧒𑧋𑧄𑧠𑦿𑧟। 𑦳𑦮𑧑𑧈𑧚 𑦳𑦽𑧔𑧈𑧑𑧈𑧠𑧇𑧍𑦽𑧠𑧇𑧋𑧄𑧠𑦿𑧟। 𑦴𑦮𑧑𑧈𑧚 𑦴𑧞𑦿𑧈𑧑𑦰𑧂𑧠𑧈𑧎𑧑𑦼𑧋𑧄𑧠𑦿𑧟। 𑦵𑦮𑧑𑧈𑧚 𑦵𑧈𑧑𑧆𑧈𑦼𑧍𑧆𑦽𑧒𑦮𑧠𑧈𑧆𑦼𑧋𑧄𑧠𑦿𑧟। 𑦶𑦮𑧑𑧈𑧚 𑦶𑧌𑧀𑧠𑧊𑦵𑧄𑧉𑧁𑧒𑦰𑧠𑧈𑧎𑦼𑧋𑧄𑧠𑦿𑧟। 𑦷𑦮𑧑𑧈𑧚 𑦵𑧠𑦷𑧑𑧂𑧁𑧋𑧄𑧠𑦿𑧟। 𑦸𑦮𑧑𑧈𑧚 𑧂𑦸𑧜𑧂𑦳𑧠𑦴𑧚𑦿𑧁𑧋𑧄𑧠𑦿𑧟। 𑦹𑦮𑧑𑧈𑧚 𑦹𑧂𑧁𑧓𑧇𑧂𑧠𑧈𑧋𑧠𑧁𑧋𑧄𑧠𑦿𑧟। 𑦺𑦮𑧑𑧈𑧚 𑦺𑧆𑧈𑧆𑧑𑧈𑧁𑧒𑦰𑧠𑧈𑧎𑦼𑧋𑧄𑧠𑦿𑧟। 𑦻𑦮𑧑𑧈𑧚 𑧆𑧓𑦻𑧊𑧒𑧌𑧇𑧑 𑦢𑦽𑧒। 𑦼𑦮𑧑𑧈𑧚 𑧈𑧚𑦼𑧔𑦮𑧠𑧉𑧚𑧋𑧑 𑦢𑦽𑧒। 𑦽𑦮𑧑𑧈𑧚 𑦽𑦾𑦽𑧑𑧍𑧞𑧅𑧚𑦿𑧋𑧄𑧠𑦿𑧟। 𑦾𑦮𑧑𑧈𑧚 𑦾𑧑𑧆𑧄𑧉𑧊𑧚𑦰𑧊𑧛𑧋𑧑𑧈𑦿𑧠𑧇𑧋𑧄𑧠𑦿𑧟। 𑦿𑦮𑧑𑧈𑧚 𑦿𑧑𑧁𑦿𑧆𑧍𑧞𑧇𑧆𑧍𑧝𑧈𑧅𑧠𑧇𑧋𑧄𑧠𑦿𑧟। 𑧀𑦮𑧑𑧈𑧚 𑧀𑧁𑧆𑧑𑧈𑧠𑧇𑧑𑦼𑧑𑧞 𑧍𑧂𑧠𑦽𑧊𑧒𑧀𑧆𑧒𑦽𑧒। 𑧁𑦮𑧑𑧈𑧚 𑧁𑧑𑧆𑧈𑧕𑧂𑧂𑧈𑧒𑦵𑧠𑦷𑧑𑧋𑧄𑧠𑦿𑧟। 𑧂𑦮𑧑𑧈𑧚 𑧂𑧈𑧆𑧑𑧈𑧠𑦾𑧋𑧄𑧠𑦿𑧟। 𑧃𑦮𑧑𑧈𑧚 𑧃𑧉𑧂𑧠𑧈𑧑𑧂𑧠𑦽𑧒𑧍𑧑𑦮𑧠𑧌𑧑𑦽𑧠𑦮𑧠𑧈𑧒𑧇𑧑𑧋𑧄𑧠𑦿𑧟। 𑧄𑦮𑧑𑧈𑧚 𑧄𑧞𑧀𑧁𑧆𑧜𑦮𑧠𑧌𑧋𑧄𑧠𑦿𑧟। 𑧅𑦮𑧑𑧈𑧚 𑧅𑧊𑧊𑧒𑧅𑧊𑧋𑧄𑧠𑦿𑧟। 𑧆𑦮𑧑𑧈𑧚 𑧆𑦿𑧆𑧑𑧁𑧜𑧂𑧋𑧆𑧁𑧋𑧄𑧠𑦿𑧟। 𑧇𑦮𑧑𑧈𑧚 𑧇𑦾𑧑𑧊𑦿𑧠𑧀𑧈𑧠𑧆𑧂𑧠𑧈𑦽𑧒𑧊𑧚𑧀𑧋𑧄𑧠𑦿𑧟। 𑧈𑦮𑧑𑧈𑧚 𑧈𑦽𑧠𑧇𑧈𑦽𑧒𑧂𑧈𑧆𑧑𑧈𑧠𑦾𑧈𑦽𑧒𑧋𑧄𑧠𑦿𑧟। 𑧉𑦮𑧑𑧈𑧚 𑧉𑦽𑧑𑦴𑧚𑦿𑧁𑧋𑧄𑧠𑦿𑧟। 𑧊𑦮𑧑𑧈𑧚 𑧊𑧈𑧇𑧑𑧁𑧋𑧄𑧠𑦿𑧟। 𑧋𑦮𑧑𑧈𑧚 𑧋𑧆𑦾𑧊𑧒𑧂𑧋𑧠𑧇𑧁𑧑𑧋𑧄𑧠𑦿𑧟। 𑧌𑦮𑧑𑧈𑧚 𑧌𑦺𑧑𑧇𑦽𑧁𑧁𑧒𑦰𑧠𑧈𑧎𑦼𑧑𑧅𑧒𑦵𑧠𑦷𑦵𑧠𑦷𑧑𑧁𑧑𑧊𑧑𑧂𑧠𑦽𑧒𑧋𑧄𑧠𑦿𑧟। 𑧍𑦮𑧑𑧈𑧚 𑧍𑧈𑧠𑧊𑦵𑧠𑦷𑦵𑧠𑦷𑧑𑧁𑧑𑧅𑧒𑧍𑧞𑧄𑧜𑧀𑧁𑧋𑧄𑧠𑦿𑧟। 𑧎𑦮𑧑𑧈𑧚 𑧎𑦽𑦮𑧠𑧉𑧚𑧋𑧊𑧒𑧈𑧑𑦰𑧋𑧄𑧠𑦿𑧟। 𑦮𑧠𑧌𑦮𑧑𑧈𑧚 𑧂𑧈𑧒𑦮𑧓𑧈𑧠𑦽𑧠𑧇𑧆𑧑𑧁𑧚 𑦮𑧠𑧌𑦼𑧂𑧈𑧠𑧇𑧞𑦽𑧑𑧅𑧒𑧉𑧑𑧂𑧠𑧇𑧍𑧈𑧠𑧊𑧀𑧈𑧠𑧆𑧋𑧄𑧠𑦿𑧜 𑧁𑧒𑧋𑧠𑦳𑧈𑦽𑧒 𑧍𑧠𑧆॥

𑦢𑦽𑧒 𑧎𑧒 𑧅𑧒𑦮𑧠𑧌𑧊𑧍𑧠𑦽𑧚𑧌𑧑𑧞 𑦿𑧑𑧈𑦮𑧑𑦼𑧑𑧞 𑧆𑧑𑦽𑧖𑦮𑧑𑧞 𑧊𑧑𑦳𑧇𑦽𑧑𑧞 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧁𑧔𑧅𑧑𑧊𑧚𑧁𑧛𑧊 𑧂𑧠𑧈𑧆𑧔𑦯𑧑𑧁𑧠𑧇𑧍𑧞𑦯𑧠𑧇𑧚𑧇𑧑𑧁𑧒 𑧀𑧈𑧠𑧆𑧆𑧔𑦯𑧋𑦽𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒 𑧁𑧒𑧋𑧠𑦳𑧈𑧞𑦽𑧒 𑧍𑧠𑧆॥

𑦽𑦿𑧑𑧁𑧔𑧂𑧕𑧈𑧠𑧊𑧚𑦼 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧚𑧁 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧍𑧠𑦾𑧒𑦽𑧚𑧁 𑦿𑧠𑧊𑧑𑦽𑧠𑧈𑧒𑧞𑧋𑦿𑧠𑦿𑧑𑧈𑦮𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒 𑧂𑧈𑧒𑧂𑧑𑦳𑧒𑦽𑧑𑧁𑧠𑧇𑧅𑧕𑧊𑧁𑧠। 𑦠𑧁𑧔𑦽𑧠𑦽𑧈𑧑𑧇𑧑𑧞 𑧍𑧆𑧠𑧇𑦮𑧠𑧍𑧞𑧄𑧜𑧀𑧝 𑦳𑧒𑦽𑧠𑦽𑧑𑧁𑧠𑧇𑧔𑦽𑧠𑧂𑧑𑦿𑧒𑦽𑧑𑧁𑧒 𑦿𑧠𑧊𑧑𑦽𑧠𑧈𑧒𑧞𑧋𑦿𑧠𑦿𑧑𑧈𑧒𑦮𑧑𑧍𑧎𑧍𑧠𑧈𑧑𑦼𑧒। 𑦠𑧇𑧞 𑧎𑧚𑦽𑧔𑧈𑧇𑧞 𑧂𑧠𑧈𑦽𑧠𑧇𑧇𑧜 𑧇𑦳𑧠𑦴𑧒𑦮𑧠𑧌𑧒𑦽𑧜𑧡𑧂𑧒 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧜 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧆𑧔𑧂𑧑𑦰𑦳𑧠𑦴𑦽𑧒 𑧍𑧠𑧆॥

॥ 𑦢𑦽𑧒 𑧋𑧠𑧈𑧓𑧉𑧉𑧒𑦽𑧊𑧒𑧍𑧠𑦽𑧈𑧚 𑧉𑧒𑧂𑧒𑧋𑧑𑧉𑧑𑧍𑧞𑦿𑧈𑧠𑧋𑧁𑧂𑧈𑧒𑧊𑧈𑧠𑦽𑧜 𑧁𑧑𑧆 𑦿𑧋𑧆𑧜𑧡𑧀𑧠𑧇𑧑𑧇𑧟॥
