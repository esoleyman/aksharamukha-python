ܐܵܪܝܲܬ݁ܪܝܼܪܲܬ݁ܢܵܢܘܼܣܡܪܘܼܬ݁ܝܼܣܘܼܬ݁ܪܲܡ

(ܟ݁ܲ) ܣܲܡܣܟ݁ܪܘܼܬ݁ܲܦ݁ܵܬ݁ܲܗ

ܒ݁ܵܪܲܬ݁ܝܼܝܲܒ݁ܵܫܵܝܵܡ-ܐܵܪܝܲܬ݁ܪܝܼܪܲܬ݁ܢܵܢܘܼܣܡܪܘܼܬ݁ܝܼܣܘܼܬ݁ܪܲܡ

ܒ݁ܘܿܬ݁ܲܒ݁ܵܫܵܝܵܡ-ܦ݁ܲܓ݁ܣ-ܦ݁ܵ-ܕ݁ܲܟ݁ܘܿܢ-ܡܟ̰ܘܿܓ݁-ܓ݁ܣܘܼܡ-ܓ̰ܹܣ-ܣܘܼ-ܕ݁ܪܲܢ-ܦ݁ܵ-ܡܕ݁ܘܿ

ܢܲܡܲܗ ܣܲܪܒ݂ܲܒ݁ܘܼܕ݁ܲܒ݁ܘܿܕ݁ܝܼܣܲܬ݁ܒ݂ܹܒ݁ܝܲܗ

(ܒ݁ܘܼܕ݁ܵܢܘܼܣܡܪܘܼܬ݁ܝܼܗ)

ܝܼܬ݁ܝܲܦ݁ܝܼ ܒ݁ܘܼܕ݁ܘܿ ܒ݁ܲܓ݁ܲܒ݂ܵܡܣܬ݁ܲܬ݁ܵܓ݁ܲܬ݁ܘܿܪܗܲܢ ܣܲܡܝܲܟ݁ܣܲܡܒ݁ܘܼܕ݁ܘܿ ܒ݂ܝܼܕ݁ܝܵܟ̰ܲܪܲܢܲܣܲܡܦ݁ܲܢܲܗ ܣܘܼܓ݁ܲܬ݁ܘܿ ܠܘܿܟ݁ܲܒ݂ܝܼܕ݁ܲܢܘܼܬ݁ܲܪܲܗ ܦ݁ܘܼܪܘܼܫܲܕ݁ܲܡܝܲܣܵܪܲܬ݁ܝܼܗ ܫܵܣܬ݁ܵ ܕܹ݁ܒ݂ܲܡܲܢܘܼܫܝܵܢܵܡ ܒ݁ܘܼܕ݁ܘܿ ܒ݁ܲܓ݁ܲܒ݂ܵܢܝܼܬ݁ܝܼ.

ܢܝܼܫܝܲܢܕ݁ܲܗ ܣܲ ܬ݁ܲܬ݁ܵܓ݁ܲܬ݁ܲܗ ܦ݁ܘܼܢܝܵܢܵܡ, ܐܲܒ݂ܝܼܦ݁ܪܲܢܵܫܲܗ ܟ݁ܘܼܫܲܠܲܡܘܼܠܵܢܵܡ, ܐܲܠܲܢܟ݁ܪܘܼܬ݁ܲܗ ܟ݁ܫܵܢܬ݁ܝܵ, ܐܵܠܲܝܲܗ ܦ݁ܘܼܢܝܲܢܝܼܕ݁ܵܢܵܢܵܡ, ܟ̰ܝܼܬ݁ܪܝܼܬ݁ܘܿܢܘܼܒ݂ܝܲܢܓ̰ܲܢܲܝܗ ܟ݁ܘܼܣܘܼܡܝܼܬ݁ܘܿ ܠܲܟ݁ܫܲܢܲܝܗ, ܦ݁ܪܲܬ݁ܝܼܪܘܼܦ݁ܘܿ ܓ݁ܘܿܟ̰ܲܪܹܢܲ, ܐܲܦ݁ܪܲܬ݁ܝܼܟ݁ܘܼܠܘܿ ܕ݁ܲܪܫܲܢܹܢܲ, ܐܲܒ݁ܝܼܪܲܬ݁ܝܼܗ ܫܪܲܕ݁ܵ(ܫܘܼܕ݁ܵ)ܕ݁ܝܼܡܘܼܟ݁ܬ݁ܵܢܵܡ, ܐܲܢܲܒ݁ܝܼܒ݁ܘܼܬ݁ܲܗ ܦ݁ܪܲܓ̰ܢܲܝܵ, ܐܲܢܲܒ݂ܲܡܲܪܕ݁ܲܢܝܼܝܘܿ ܒ݁ܲܠܲܝܗ, ܫܵܣܬ݁ܵ ܣܲܪܒ݂ܲܣܲܬ݁ܒ݂ܵܢܵܡ, ܦ݁ܝܼܬ݁ܵ ܒ݁ܘܿܕ݁ܝܼܣܲܬ݁ܒ݂ܵܢܵܡ, ܪܵܓ̰ܵ ܐܵܪܝܲܦ݁ܘܼܕ݁ܓ݁ܲܠܵܢܵܡ, ܣܵܪܬ݁ܲܒ݂ܵܗܲܗ ܢܝܼܪܒ݂ܵܢܲܢܲܓ݁ܲܪܲܣܲܡܦ݁ܪܲܣܬ݁ܝܼܬ݁ܵܢܵܡ, ܐܲܦ݁ܪܲܡܹܝܘܿ ܓ̰ܢܵܢܹܢܲ, ܐܲܟ̰ܝܼܢܬ݁ܝܲܗ ܦ݁ܪܲܬ݁ܝܼܒ݁ܵܢܹܢܲ, ܒ݂ܝܼܫܘܼܕ݁ܲܗ ܣܒ݂ܲܪܹܢܲ, ܐܵܣܒ݂ܲܕ݁ܲܢܝܼܝܘܿ ܓ݁ܘܿܫܹܢܲ, ܐܲܣܹܟ̰ܲܢܲܟ݁ܘܿ ܪܘܼܦܹ݁ܢܲ, ܐܲܦ݁ܪܲܬ݁ܝܼܣܲܡܲܗ ܟ݁ܵܝܹܢܲ, ܐܲܠܝܼܦ݁ܬ݁ܲܗ ܟ݁ܵܡܲܝܗ, ܐܲܢܘܼܦ݁ܲܠܝܼܦ݁ܬ݁ܘܿ ܪܘܼܦ݁ܲܝܗ ܐܲܣܲܡܣܪܘܼܫܬ݁ܲ ܐܵܪܘܼܦ݁ܝܲܝܗ, ܒ݂ܝܼܦ݁ܪܲܡܘܼܟ݁ܬ݁ܲܗ ܣܟ݁ܲܢܕܹ݁ܒ݁ܝܲܗ, ܒ݂ܝܼܣܲܡܦ݁ܪܲܝܘܼܟ݁ܬ݁ܘܿ ܕ݁ܵܬ݁ܘܼܒ݁ܝܼܗ, ܣܲܡܒ݂ܪܘܼܬ݁ܲ ܐܵܝܲܬ݁ܲܢܲܝܗ, ܦ݁ܪܲܟ̰ܝܼܢܘܿ ܓ݁ܪܲܢܬ݁ܲܝܗ, ܒ݂ܝܼܡܘܼܟ݁ܬ݁ܲܗ ܦ݁ܲܪܝܼܕ݁ܵܓ݁ܢܲܝܗ, ܦ݁ܲܪܝܼܡܘܼܟ݁ܬ݁ܲܣܬ݁ܪܘܼܫܢܲܝܵ, ܘܿܓ݁ܵܕ݁ܘܼܬ݁ܝܼܪܢܲܗ ܦ݁ܲܪܝܼܦ݁ܘܼܪܢܘܿ ܓ̰ܢܵܢܹܢܲ, ܦ݁ܪܲܬ݁ܝܼܫܬ݁ܝܼܬ݁ܘܿܬ݁ܝܼܬ݁ܵܢܵܓ݁ܲܬ݁ܲܦ݁ܪܲܬ݁ܝܘܼܬ݁ܦ݁ܲܢܵܢܵܡ ܒ݁ܘܼܕ݁ܵܢܵܡ ܒ݁ܲܓ݁ܲܒ݂ܲܬ݁ܵܡ ܓ̰ܢܵܢܹ, ܐܲܦ݁ܪܲܬ݁ܝܼܫܬ݁ܝܼܬ݁ܘܿ ܢܝܼܪܒ݂ܵܢܹ, ܣܬ݁ܝܼܬ݁ܘܿ ܒ݁ܘܼܬ݁ܲܟ݁ܘܿܬ݁ܝܵܡ, ܣܬ݁ܝܼܬ݁ܲܗ ܣܲܪܒ݂ܲܣܲܬ݁ܒ݂ܵܠܘܿܟ݁ܲܢܝܼܝܵܝܵܡ ܒ݁ܘܼܡܲܘ, ܣܲܪܒ݂ܲ ܝܼܡܹ ܬ݁ܲܬ݁ܵܓ݁ܲܬ݁ܵܢܵܡ ܒ݂ܝܼܫܹܫܲܬ݁ܲܗ ܣܲܡܝܲܓ݁ ܓ݁ܘܼܢܵܗ.

(ܕ݁ܲܪܡܵܢܘܼܣܡܪܘܼܬ݁ܝܼܗ)

ܣܲܕ݁ܲܪܡܲܣܬ݁ܘܼ ܐܵܕ݁ܲܘ ܟ݁ܲܠܝܵܢܲܗ, ܡܲܕ݁ܝܹ ܟ݁ܲܠܝܵܢܲܗ, ܦ݁ܲܪܝܲܒ݂ܲܣܵܢܹ ܟ݁ܲܠܝܵܢܲܗ, ܣܒ݂ܲܪܬ݁ܲܗ ܣܘܼܒ݂ܝܲܢܓ̰ܲܢܲܗ, ܟܹ݁ܒ݂ܲܠܲܗ, ܦ݁ܲܪܝܼܦ݁ܘܼܪܢܲܗ, ܦ݁ܲܪܝܼܫܘܼܕ݁ܲܗ, ܦ݁ܲܪܝܲܒ݂ܲܕ݁ܵܬ݁ܲܗ, ܣܒ݂ܵܟ݁ܝܵܬ݁ܲܗ ܒ݁ܲܓ݁ܲܒ݂ܲܬ݁ܘܿ ܕ݁ܲܪܡܲܗ, ܣܵܢܕ݁ܪܘܼܫܬ݁ܝܼܟ݁ܲܗ, ܢܝܼܪܓ̰ܒ݂ܲܪܲܗ, ܐܵܟ݁ܵܠܝܼܟ݁ܲܗ, ܐܲܘܦ݁ܲܢܵܝܝܼܟ݁ܲܗ, ܐܲܝܗܝܼܦ݁ܲܫܝܝܼܟ݁ܲܗ, ܦ݁ܪܲܬ݁ܝܵܬ݁ܡܲܒ݂ܹܕ݁ܲܢܝܼܝܘܿ ܒ݂ܝܼܓ̰ܢܲܝܗ, ܣܒ݂ܵܟ݁ܝܵܬ݁ܘܿ ܒ݁ܲܓ݁ܲܒ݂ܲܬ݁ܘܿ ܕ݁ܲܪܡܲܒ݂ܝܼܢܲܝܲܗ ܣܘܼܦ݁ܪܲܒ݂ܹܕ݁ܝܼܬ݁ܲܗ ܢܲܝܪܝܵܢܝܼܟ݁ܲܗ, ܣܲܡܒ݁ܘܿܕ݁ܝܼܓ݁ܵܡܝܼ, ܐܲܒ݁ܝܼܢܲܗ ܣܲܡܣܬ݁ܘܼܦ݁ܲܗ, ܣܲܦ݁ܪܲܬ݁ܝܼܫܲܪܲܢܲܗ, ܟ̰ܝܼܢܲܦ݁ܠܘܿܬ݁ܝܼܟ݁ܲܗ.

(ܣܲܡܓ݁ܵܢܘܼܣܡܪܘܼܬ݁ܝܼܗ)

ܣܘܼܦ݁ܪܲܬ݁ܝܼܦ݁ܲܢܘܿ ܒ݁ܲܓ݁ܲܒ݂ܲܬ݁ܲ ܐܵܪܝܲܣܲܡܓ݁ܲܗ, ܢܝܵܝܲܦ݁ܪܲܬ݁ܝܼܦ݁ܲܢܲܗ, ܪܘܼܓ̰ܘܼܦ݁ܪܲܬ݁ܝܼܦ݁ܲܢܲܗ, ܣܵܡܝܼܟ̰ܝܼܦ݁ܪܲܬ݁ܝܼܦ݁ܲܢܲܗ, ܐܲܢܓ̰ܲܠܝܼܟ݁ܲܪܲܢܝܼܝܲܗ, ܣܵܡܝܼܟ̰ܝܼܟ݁ܲܪܲܢܝܼܝܲܗ, ܦ݁ܘܼܢܝܲܫܪܝܼܟ݁ܫܹܬ݁ܪܲܗ, ܡܲܗܵܕ݁ܲܟ݁ܫܝܼܢܵܦ݁ܲܪܝܼܫܘܿܕ݁ܲܟ݁ܲܗ, ܦ݁ܪܵܗܲܒ݂ܲܢܝܼܝܲܗ, ܐܵܗܲܒ݂ܲܢܝܼܝܲܗ.
