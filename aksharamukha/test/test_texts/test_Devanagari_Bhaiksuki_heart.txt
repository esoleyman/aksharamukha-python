𑰢𑰿𑰨𑰕𑰿𑰗𑰯𑰢𑰯𑰨𑰦𑰰𑰝𑰯𑰮𑰴𑰟𑰧𑰭𑰳𑰝𑰿𑰨𑰦𑰿𑱁

[𑰭𑰽𑰎𑰿𑰬𑰰𑰢𑰿𑰝𑰦𑰯𑰝𑰴𑰎𑰯]

𑱂𑱃𑰡𑰦𑰾𑱃𑰭𑰨𑰿𑰪𑰕𑰿𑰗𑰯𑰧𑱃𑱂

𑱂𑱃𑰌𑰼𑱃𑰡𑰦𑰺𑱃𑰥𑰐𑰪𑰝𑰿𑰧𑰹𑱃𑰁𑰨𑰿𑰧𑰢𑰿𑰨𑰕𑰿𑰗𑰯𑰢𑰯𑰨𑰦𑰰𑰝𑰯𑰧𑰹𑱃𑱂

𑰁𑰨𑰿𑰧𑰯𑰪𑰩𑰺𑰎𑰰𑰝𑰸𑰫𑰿𑰪𑰨𑰤𑰺𑰠𑰰𑰭𑰝𑰿𑰝𑰿𑰪𑰺𑱃𑰐𑰦𑰿𑰥𑰱𑰨𑰯𑰧𑰯𑰽𑱃𑰢𑰿𑰨𑰕𑰿𑰗𑰯𑰢𑰯𑰨𑰦𑰰𑰝𑰯𑰧𑰯𑰽𑱃𑰓𑰨𑰿𑰧𑰯𑰽𑱃𑰓𑰨𑰦𑰯𑰜𑰺𑱃𑰪𑰿𑰧𑰪𑰩𑰺𑰎𑰧𑰝𑰰𑱃𑰭𑰿𑰦𑱁𑱃𑰢𑰗𑰿𑰓𑱃𑰭𑰿𑰎𑰡𑰿𑰠𑰯𑰾,𑱃𑰝𑰯𑰽𑰫𑰿𑰓𑱃𑰭𑰿𑰪𑰥𑰯𑰪𑰫𑰳𑰡𑰿𑰧𑰯𑰡𑰿𑱃𑰢𑰫𑰿𑰧𑰝𑰰𑱃𑰭𑰿𑰦𑱂

𑰂𑰮𑱃𑰫𑰯𑰨𑰰𑰢𑰲𑰝𑰿𑰨𑱃𑰨𑰳𑰢𑰽𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯,𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰹𑰪𑱃𑰨𑰳𑰢𑰦𑰿𑱁𑱃𑰨𑰳𑰢𑰯𑰡𑰿𑰡𑱃𑰢𑰴𑰞𑰎𑰿𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯,𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯𑰧𑰯𑱃𑰡𑱃𑰢𑰴𑰞𑰐𑰿𑱃𑰨𑰳𑰢𑰦𑰿𑱁𑱃𑰧𑰟𑰿𑰨𑰳𑰢𑰽𑱃𑰭𑰯𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯,𑱃𑰧𑰯𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯𑱃𑰝𑰟𑰿𑰨𑰳𑰢𑰦𑰿𑱂

𑰊𑰪𑰦𑰸𑰪𑱃𑰪𑰸𑰟𑰡𑰯𑰭𑰽𑰕𑰿𑰗𑰯𑰭𑰽𑰭𑰿𑰎𑰯𑰨𑰪𑰰𑰕𑰿𑰗𑰯𑰡𑰯𑰡𑰰𑱂

𑰂𑰮𑰽𑱃𑰫𑰯𑰨𑰰𑰢𑰲𑰝𑰿𑰨𑱃𑰭𑰨𑰿𑰪𑰠𑰨𑰿𑰦𑰯𑰾𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯𑰩𑰎𑰿𑰬𑰜𑰯𑱃𑰀𑰡𑰲𑰝𑰿𑰢𑰡𑰿𑰡𑰯𑱃𑰀𑰡𑰰𑰨𑰲𑰟𑰿𑰠𑰯𑱃𑰀𑰦𑰩𑰯𑱃𑰡𑱃𑰪𑰰𑰦𑰩𑰯𑱃𑰡𑰺𑰡𑰯𑱃𑰡𑱃𑰢𑰨𑰰𑰢𑰳𑰨𑰿𑰜𑰯𑰾𑱁𑱃𑰝𑰭𑰿𑰦𑰯𑰓𑰿𑰔𑰯𑰨𑰰𑰢𑰲𑰝𑰿𑰨𑱃𑰫𑰳𑰡𑰿𑰧𑰝𑰯𑰧𑰯𑰽𑱃𑰡𑱃𑰨𑰳𑰢𑰦𑰿,𑱃𑰡𑱃𑰪𑰸𑰟𑰡𑰯,𑱃𑰡𑱃𑰭𑰽𑰕𑰿𑰗𑰯,𑱃𑰡𑱃𑰭𑰽𑰭𑰿𑰎𑰯𑰨𑰯𑰾,𑱃𑰡𑱃𑰪𑰰𑰕𑰿𑰗𑰯𑰡𑰯𑰡𑰰𑱁𑱃𑰡𑱃𑰓𑰎𑰿𑰬𑰲𑰾𑰫𑰿𑰨𑰺𑰝𑰿𑰨𑰑𑰿𑰨𑰯𑰜𑰕𑰰𑰮𑰿𑰪𑰯𑰎𑰯𑰧𑰦𑰡𑰯𑰽𑰭𑰰,𑱃𑰡𑱃𑰨𑰳𑰢𑰫𑰤𑰿𑰟𑰐𑰡𑰿𑰠𑰨𑰭𑰭𑰿𑰢𑰿𑰨𑰬𑰿𑰘𑰪𑰿𑰧𑰠𑰨𑰿𑰦𑰯𑰾𑱁𑱃𑰡𑱃𑰓𑰎𑰿𑰬𑰲𑰨𑰿𑰠𑰯𑰝𑰲𑰨𑰿𑰧𑰯𑰪𑰡𑰿𑰡𑱃𑰦𑰡𑰺𑰠𑰯𑰝𑰲𑰾𑱂

𑰡𑱃𑰪𑰰𑰟𑰿𑰧𑰯𑱃𑰡𑰯𑰪𑰰𑰟𑰿𑰧𑰯𑱃𑰡𑱃𑰪𑰰𑰟𑰿𑰧𑰯𑰎𑰿𑰬𑰧𑰺𑱃𑰡𑰯𑰪𑰰𑰟𑰿𑰧𑰯𑰎𑰿𑰬𑰧𑰺𑱃𑰧𑰯𑰪𑰡𑰿𑰡𑱃𑰕𑰨𑰯𑰦𑰨𑰜𑰽𑱃𑰡𑱃𑰕𑰨𑰯𑰦𑰨𑰜𑰎𑰿𑰬𑰧𑰺𑱃𑰡𑱃𑰟𑰲𑰾𑰏𑰭𑰦𑰲𑰟𑰧𑰡𑰰𑰨𑰺𑰠𑰦𑰯𑰨𑰿𑰐𑰯𑱃𑰡𑱃𑰕𑰿𑰗𑰯𑰡𑰽𑱃𑰡𑱃𑰢𑰿𑰨𑰯𑰢𑰿𑰝𑰰𑰝𑰿𑰪𑰦𑰿𑱂

𑰤𑰺𑰠𑰰𑰭𑰝𑰿𑰝𑰿𑰪𑰭𑰿𑰧𑱃𑰢𑰿𑰨𑰕𑰿𑰗𑰯𑰢𑰯𑰨𑰦𑰰𑰝𑰯𑰦𑰯𑰫𑰿𑰨𑰰𑰝𑰿𑰧𑱃𑰪𑰰𑰮𑰨𑰝𑰰𑱃𑰓𑰰𑰝𑰿𑰝𑰯𑰪𑰨𑰜𑰾𑱁𑱃𑰓𑰰𑰝𑰿𑰝𑰯𑰪𑰨𑰜𑰡𑰯𑰭𑰿𑰝𑰰𑰝𑰿𑰪𑰯𑰟𑰝𑰿𑰨𑰭𑰿𑰝𑰺𑱃𑰪𑰰𑰢𑰨𑰿𑰧𑰯𑰭𑰯𑰝𑰰𑰎𑰿𑰨𑰯𑰡𑰿𑰝𑰺𑱃𑰡𑰰𑰬𑰿𑰙𑰡𑰰𑰨𑰿𑰪𑰯𑰜𑰾𑱁𑱃𑰝𑰿𑰨𑰿𑰧𑰠𑰿𑰪𑰪𑰿𑰧𑰪𑰭𑰿𑰞𑰰𑰝𑰯𑰾𑱃𑰭𑰨𑰿𑰪𑰤𑰲𑰟𑰿𑰠𑰯𑰾𑱃𑰢𑰿𑰨𑰕𑰿𑰗𑰯𑰢𑰯𑰨𑰦𑰰𑰝𑰯𑰦𑰯𑰫𑰿𑰨𑰰𑰝𑰿𑰧𑱃𑰀𑰡𑰲𑰝𑰿𑰝𑰨𑰯𑰽𑱃𑰭𑰦𑰿𑰧𑰎𑰿𑰭𑰽𑰤𑰺𑰠𑰰𑰦𑰥𑰰𑰭𑰽𑰤𑰲𑰟𑰿𑰠𑰯𑰾𑱂
