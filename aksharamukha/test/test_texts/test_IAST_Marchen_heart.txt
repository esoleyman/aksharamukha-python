𑲏𑲰𑲊𑲩𑱺𑲪𑲱𑲊𑱺𑲝𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲍𑲲𑲰𑱺𑲪𑲁

(𑱳) 𑲍𑲵𑲍𑲒𑲪𑲱𑱺𑱾𑲰𑱻𑲎

𑲀𑲮𑲰𑲊𑱺𑲱𑲰𑲉𑲀𑲮𑲰𑲌𑲰𑲉𑲰𑲁-𑲏𑲰𑲊𑲩𑱺𑲪𑲱𑲊𑱺𑲝𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲍𑲲𑲰𑱺𑲪𑲁

𑲀𑲮𑲴𑱺𑲀𑲮𑲰𑲌𑲰𑲉𑲰𑲁-𑱿𑱴𑲭-𑱾𑲰-𑱼𑱲𑲴𑱽-𑲁𑲢𑲣𑲴𑱴-𑱴𑲭𑲲𑲁-𑲄𑲳𑲍-𑲍𑲲-𑱼𑲪𑱽-𑱾𑲰-𑲁𑲜𑲴

𑱽𑲁𑲎 𑲍𑲊𑲥𑲀𑲲𑱼𑲜𑲮𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲳𑲀𑲮𑲩𑲎

(𑲀𑲲𑱼𑲜𑲮𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲎)

𑲏𑲱𑱺𑲩𑱾𑲱 𑲀𑲲𑱼𑲜𑲮𑲴 𑲀𑲮𑱴𑲅𑲰𑲵𑲍𑲚𑱻𑲰𑱴𑱺𑲴𑲊𑲮𑱽 𑲍𑲁𑲩𑱲𑲭𑲵𑲀𑲲𑱼𑲜𑲮𑲴 𑲅𑲱𑱼𑲩𑲰𑲂𑲊𑱽𑲍𑲁𑲞𑱽𑲝𑲎 𑲍𑲲𑱴𑱺𑲴 𑲋𑲴𑱲𑲅𑲱𑱼𑱽𑲲𑱺𑲚𑲊𑲎 𑱾𑲲𑲊𑲲𑲌𑱼𑲁𑲩𑲍𑲰𑲊𑱻𑲱𑲎 𑲌𑲰𑲍𑲚𑲰 𑱼𑲳𑲅𑲁𑱽𑲲𑲌𑲩𑲰𑱽𑲰𑲵 𑲀𑲲𑱼𑲜𑲮𑲴 𑲀𑲮𑱴𑲅𑲰𑱽𑲱𑱺𑲱𑱱

𑱽𑲱𑲌𑲩𑱽𑲜𑲎 𑲍 𑱺𑱻𑲰𑱴𑱺𑲎 𑱾𑲲𑱽𑲩𑲰𑱽𑲰𑲁, 𑲏𑲅𑲱𑱾𑲪𑱽𑲰𑲌𑲎 𑱲𑲲𑲌𑲋𑲁𑲲𑲰𑲋𑲰𑱽𑲰𑲁, 𑲏𑲋𑱵𑲒𑲪𑲱𑱺𑲎 𑱲𑲬𑲰𑱽𑲚𑲩𑲰, 𑲏𑲰𑲋𑲉𑲎 𑱾𑲲𑱽𑲩𑱽𑲱𑱼𑲮𑲰𑱽𑲰𑱽𑲰𑲁, 𑲂𑲱𑱺𑲪𑲱𑱺𑲴𑱽𑲲𑲅𑲩𑱹𑲤𑱽𑲳𑲰𑲎 𑱲𑲲𑲍𑲲𑲁𑲱𑱺𑲴 𑲋𑱲𑲬𑱽𑲳𑲰𑲎, 𑱾𑲪𑱺𑲱𑲊𑲲𑲰𑱾𑲴 𑱴𑲴𑲂𑲊𑲳𑱽, 𑲏𑱾𑲪𑱺𑲱𑱲𑲲𑲰𑲋𑲴 𑱼𑲊𑲬𑱽𑲳𑱽, 𑲏𑲀𑲮𑲱𑲊𑱺𑲱𑲎 𑲌𑲪𑱼𑲜𑲮𑲰(𑲌𑲲𑱼𑲜𑲮𑲰)𑱼𑲮𑲱𑲁𑲲𑱲𑲚𑲰𑱽𑲰𑲁, 𑲏𑱽𑲀𑲮𑲱𑲀𑲮𑲲𑲰𑱺𑲎 𑱾𑲪𑲄𑲙𑲉𑲰, 𑲏𑱽𑲅𑲁𑲊𑲜𑱽𑲱𑲰𑲉𑲴 𑲀𑲋𑲳𑲰𑲎, 𑲌𑲰𑲍𑲚𑲰 𑲍𑲊𑲥𑲍𑱺𑲚𑲥𑲰𑱽𑲰𑲁, 𑱾𑲱𑱺𑲰 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲰𑱽𑲰𑲁, 𑲊𑲰𑲄𑲰 𑲏𑲰𑲊𑲩𑱾𑲲𑱼𑲔𑲋𑲰𑱽𑲰𑲁, 𑲍𑲰𑲊𑲛𑲅𑲰𑲎𑲎 𑱽𑲱𑲊𑲥𑲰𑱽𑱽𑱴𑲊𑲍𑲁𑲞𑲪𑲍𑲛𑲱𑱺𑲰𑱽𑲰𑲁, 𑲏𑱾𑲪𑲁𑲳𑲉𑲴 𑲄𑲙𑲰𑱽𑲳𑱽, 𑲏𑲂𑲱𑱽𑲚𑲩𑲎 𑱾𑲪𑱺𑲱𑲀𑲮𑲰𑱽𑲳𑱽, 𑲅𑲱𑲌𑲲𑱼𑲜𑲮𑲎 𑲍𑲥𑲊𑲳𑱽, 𑲏𑲰𑲍𑲥𑱼𑱽𑲱𑲰𑲉𑲴 𑱴𑲮𑲴𑲌𑲳𑱽, 𑲏𑲍𑲳𑲂𑱽𑱲𑲴 𑲊𑲲𑲰𑱾𑲳𑱽, 𑲏𑱾𑲪𑱺𑲱𑲍𑲁𑲎 𑱲𑲰𑲉𑲳𑱽, 𑲏𑲋𑲱𑱾𑲚𑲎 𑱲𑲰𑲁𑲳𑲰𑲎, 𑲏𑱽𑲲𑱾𑲋𑲱𑱾𑲚𑲴 𑲊𑲲𑲰𑱾𑲳𑲰𑲎 𑲏𑲍𑲵𑲍𑲪𑲱𑲌𑲚 𑲏𑲰𑲊𑲲𑲰𑱾𑲩𑲳𑲰𑲎, 𑲅𑲱𑱾𑲪𑲁𑲲𑱲𑲚𑲎 𑲍𑲒𑱽𑲜𑲮𑲳𑲀𑲮𑲩𑲎, 𑲅𑲱𑲍𑲁𑲞𑲪𑲉𑲲𑱲𑲚𑲴 𑱼𑲮𑲰𑱺𑲲𑲀𑲮𑲱𑲎, 𑲍𑲵𑲅𑲪𑲱𑱺 𑲏𑲰𑲉𑱺𑱽𑲳𑲰𑲎, 𑱾𑲪𑲂𑲣𑲱𑱽𑲝𑲴 𑱴𑲪𑱽𑲛𑲳𑲰𑲎, 𑲅𑲱𑲁𑲲𑱲𑲚𑲎 𑱾𑲊𑲱𑱼𑲰𑱴𑲮𑲝𑲳𑲰𑲎, 𑱾𑲊𑲱𑲁𑲲𑱲𑲚𑲍𑲚𑲪𑲱𑲌𑲝𑲉𑲰, 𑲏𑲴𑱴𑲮𑲰𑱼𑲲𑱺𑲚𑲱𑲰𑲊𑲝𑲎 𑱾𑲊𑲱𑱾𑲲𑲰𑲊𑲝𑲴 𑲄𑲙𑲰𑱽𑲳𑱽, 𑱾𑲪𑱺𑲱𑲌𑲛𑲱𑱺𑲴𑱺𑲱𑲰𑱺𑲰𑱽𑲰𑱴𑱺𑱾𑲪𑱺𑲩𑲲𑱺𑲞𑱽𑲝𑲰𑱽𑲰𑲵 𑲀𑲲𑱼𑲜𑲮𑲰𑱽𑲰𑲵 𑲀𑲮𑱴𑲅𑱺𑲰𑲵 𑲄𑲙𑲰𑱽𑲳, 𑲏𑱾𑲪𑱺𑲱𑲌𑲛𑲱𑱺𑲴 𑱽𑲱𑲊𑲥𑲰𑱽𑲳, 𑲍𑲛𑲱𑱺𑲴 𑲀𑲮𑲲𑲰𑱺𑱲𑲴𑱺𑲩𑲰𑲁, 𑲍𑲛𑲱𑱺𑲎 𑲍𑲊𑲥𑲍𑱺𑲚𑲥𑲰𑲋𑲴𑱲𑱽𑲱𑲰𑲉𑲰𑲉𑲰𑲵 𑲀𑲮𑲲𑲰𑲁𑲰𑲴, 𑲍𑲊𑲥 𑲏𑲱𑲁𑲳 𑱺𑱻𑲰𑱴𑱺𑲰𑱽𑲰𑲵 𑲅𑲱𑲌𑲳𑲌𑱺𑲎 𑲍𑲁𑲩𑱴 𑱴𑲲𑱽𑲰𑲎𑱱

(𑱼𑲮𑲊𑲡𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲎)

𑲍𑱼𑲜𑲮𑲊𑲡𑲍𑲚𑲲 𑲏𑲰𑱼𑲰𑲴 𑱲𑲋𑲩𑲰𑱽𑲎, 𑲁𑱼𑲮𑲩𑲳 𑱲𑲋𑲩𑲰𑱽𑲎, 𑱾𑲊𑲩𑲅𑲍𑲰𑱽𑲳 𑱲𑲋𑲩𑲰𑱽𑲎, 𑲍𑲥𑲊𑲛𑲎 𑲍𑲲𑲅𑲩𑱹𑲤𑱽𑲎, 𑱲𑲳𑲅𑲋𑲎, 𑱾𑲊𑲱𑱾𑲲𑲰𑲊𑲝𑲎, 𑱾𑲊𑲱𑲌𑲲𑱼𑲜𑲮𑲎, 𑱾𑲊𑲩𑲅𑱼𑲰𑱺𑲎, 𑲍𑲥𑲰𑱳𑲩𑲰𑱺𑲎 𑲀𑲮𑱴𑲅𑱺𑲴 𑱼𑲮𑲊𑲡𑲎, 𑲍𑲰𑱽𑲜𑲪𑲱𑲌𑲚𑲱𑱲𑲎, 𑱽𑲱𑲊𑲤𑲥𑲊𑲎, 𑲏𑲰𑱲𑲰𑲋𑲱𑱲𑲎, 𑲏𑲰𑲴𑱾𑱽𑲰𑲉𑲱𑱲𑲎, 𑲏𑲳𑲰𑲎𑲱𑱾𑲌𑲩𑲱𑱲𑲎, 𑱾𑲪𑱺𑲩𑲰𑱺𑲡𑲅𑲳𑱼𑱽𑲱𑲰𑲉𑲴 𑲅𑲱𑲄𑲙𑲳𑲰𑲎, 𑲍𑲥𑲰𑱳𑲩𑲰𑱺𑲴 𑲀𑲮𑱴𑲅𑱺𑲴 𑱼𑲮𑲊𑲡𑲅𑲱𑱽𑲉𑲎 𑲍𑲲𑱾𑲪𑲅𑲳𑱼𑲱𑱺𑲎 𑱽𑲳𑲰𑲊𑲩𑲰𑱽𑲱𑱲𑲎, 𑲍𑲵𑲀𑲴𑱼𑲮𑲱𑱴𑲰𑲁𑲱𑲰, 𑲏𑲀𑲮𑲱𑱽𑲝𑲎 𑲍𑲵𑲍𑲚𑲲𑲰𑱾𑲎, 𑲍𑱾𑲪𑱺𑲱𑲌𑲊𑱽𑲎, 𑲃𑲱𑱽𑲝𑱾𑲫𑲴𑱺𑲱𑱲𑲎𑱱

(𑲍𑲵𑱴𑲮𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲎)

𑲍𑲲𑱾𑲪𑱺𑲱𑱾𑱽𑲝𑲴 𑲀𑲮𑱴𑲅𑱺 𑲏𑲰𑲊𑲩𑲍𑲵𑱴𑲮𑲎, 𑱽𑲩𑲰𑲉𑱾𑲪𑱺𑲱𑱾𑱽𑲝𑲎, 𑲊𑲱𑲄𑲲𑱾𑲪𑱺𑲱𑱾𑱽𑲝𑲎, 𑲍𑲰𑲁𑲱𑲰𑲂𑲱𑲰𑱾𑲪𑱺𑲱𑱾𑱽𑲝𑲎, 𑲏𑱹𑲤𑲋𑲱𑲰𑱲𑲊𑱽𑲱𑲰𑲉𑲎, 𑲍𑲰𑲁𑲱𑲰𑲂𑲱𑲰𑱲𑲊𑱽𑲱𑲰𑲉𑲎, 𑱾𑲲𑱽𑲩𑲌𑲪𑲱𑲰𑱲𑲬𑲳𑱺𑲪𑲎, 𑲁𑲎𑲰𑱼𑱲𑲬𑲱𑱽𑲰𑱾𑲊𑲱𑲌𑲴𑱼𑲮𑱲𑲎, 𑱾𑲪𑲰𑲎𑲅𑱽𑲱𑲰𑲉𑲎, 𑲏𑲰𑲎𑲅𑱽𑲱𑲰𑲉𑲎𑱱

𑱱𑱱 𑲏𑲰𑲊𑲩𑱺𑲪𑲱𑲊𑱺𑲝𑲰𑱽𑲲𑲍𑲡𑲪𑲱𑱺𑲱𑲍𑲲𑲰𑱺𑲪𑲵 𑲍𑲁𑲰𑱾𑲚𑲁𑱱𑱱
