𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧎𑧖𑦿𑧇𑧍𑧕𑦽𑧠𑧈𑧆𑧠।

[𑧍𑧞𑦮𑧠𑧌𑧒𑧂𑧠𑦽𑧆𑧑𑦽𑧖𑦮𑧑]

॥ 𑧁𑧆𑧟 𑧍𑧈𑧠𑧊𑦵𑧠𑦷𑧑𑧇 ॥

॥ 𑦬𑧞 𑧁𑧆𑧜 𑧅𑦰𑧊𑦽𑧠𑧇𑧛 𑦡𑧈𑧠𑧇𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧇𑧛 ॥

𑦡𑧈𑧠𑧇𑧑𑧊𑧉𑧜𑦮𑧒𑦽𑧚𑧋𑧠𑧊𑧈𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧜 𑦰𑧞𑧅𑧓𑧈𑧑𑧇𑧑𑧞 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧇𑧑𑧞 𑦳𑧈𑧠𑧇𑧑𑧞 𑦳𑧈𑧆𑧑𑦼𑧜 𑧊𑧠𑧇𑧊𑧉𑧜𑦮𑧇𑦽𑧒 𑧍𑧠𑧆। 𑧂𑧞𑦳 𑧍𑧠𑦮𑧞𑧀𑧑𑧟, 𑦽𑧑𑧞𑧋𑧠𑦳 𑧍𑧠𑧊𑧅𑧑𑧊𑧋𑧕𑧁𑧠𑧇𑧑𑧁𑧠 𑧂𑧋𑧠𑧇𑦽𑧒 𑧍𑧠𑧆॥

𑦢𑧎 𑧋𑧑𑧈𑧒𑧂𑧔𑦽𑧠𑧈 𑧈𑧕𑧂𑧞 𑧋𑧕𑧁𑧠𑧇𑦽𑧑, 𑧋𑧕𑧁𑧠𑧇𑦽𑧛𑧊 𑧈𑧕𑧂𑧆𑧠। 𑧈𑧕𑧂𑧑𑧁𑧠𑧁 𑧂𑧖𑦾𑦮𑧠 𑧋𑧕𑧁𑧠𑧇𑦽𑧑, 𑧋𑧕𑧁𑧠𑧇𑦽𑧑𑧇𑧑 𑧁 𑧂𑧖𑦾𑦰𑧠 𑧈𑧕𑧂𑧆𑧠। 𑧇𑦿𑧠𑧈𑧕𑧂𑧞 𑧍𑧑 𑧋𑧕𑧁𑧠𑧇𑦽𑧑, 𑧇𑧑 𑧋𑧕𑧁𑧠𑧇𑦽𑧑 𑦽𑦿𑧠𑧈𑧕𑧂𑧆𑧠॥

𑦪𑧊𑧆𑧚𑧊 𑧊𑧚𑦿𑧁𑧑𑧍𑧞𑦵𑧠𑦷𑧑𑧍𑧞𑧍𑧠𑦮𑧑𑧈𑧊𑧒𑦵𑧠𑦷𑧑𑧁𑧑𑧁𑧒॥

𑦢𑧎𑧞 𑧋𑧑𑧈𑧒𑧂𑧔𑦽𑧠𑧈 𑧍𑧈𑧠𑧊𑧀𑧈𑧠𑧆𑧑𑧟 𑧋𑧕𑧁𑧠𑧇𑦽𑧑𑧉𑦮𑧠𑧌𑦼𑧑 𑦠𑧁𑧔𑦽𑧠𑧂𑧁𑧠𑧁𑧑 𑦠𑧁𑧒𑧈𑧔𑦿𑧠𑧀𑧑 𑦠𑧆𑧉𑧑 𑧁 𑧊𑧒𑧆𑧉𑧑 𑧁𑧜𑧁𑧑 𑧁 𑧂𑧈𑧒𑧂𑧕𑧈𑧠𑦼𑧑𑧟। 𑦽𑧍𑧠𑧆𑧑𑦳𑧠𑦴𑧑𑧈𑧒𑧂𑧔𑦽𑧠𑧈 𑧋𑧕𑧁𑧠𑧇𑦽𑧑𑧇𑧑𑧞 𑧁 𑧈𑧕𑧂𑧆𑧠, 𑧁 𑧊𑧚𑦿𑧁𑧑, 𑧁 𑧍𑧞𑦵𑧠𑦷𑧑, 𑧁 𑧍𑧞𑧍𑧠𑦮𑧑𑧈𑧑𑧟, 𑧁 𑧊𑧒𑦵𑧠𑦷𑧑𑧁𑧑𑧁𑧒। 𑧁 𑦳𑦮𑧠𑧌𑧔𑧟𑧋𑧠𑧈𑧜𑦽𑧠𑧈𑦱𑧠𑧈𑧑𑦼𑦵𑧒𑧎𑧠𑧊𑧑𑦮𑧑𑧇𑧆𑧁𑧑𑧞𑧍𑧒, 𑧁 𑧈𑧕𑧂𑧋𑧄𑧠𑦿𑦰𑧞𑧀𑧈𑧍𑧍𑧠𑧂𑧠𑧈𑧌𑧠𑦸𑧊𑧠𑧇𑧀𑧈𑧠𑧆𑧑𑧟। 𑧁 𑦳𑦮𑧠𑧌𑧔𑧈𑧠𑧀𑧑𑦽𑧔𑧈𑧠𑧇𑧑𑧊𑧁𑧠𑧁 𑧆𑧁𑧜𑧀𑧑𑦽𑧔𑧟॥

𑧁 𑧊𑧒𑦿𑧠𑧇𑧑 𑧁𑧑𑧊𑧒𑦿𑧠𑧇𑧑 𑧁 𑧊𑧒𑦿𑧠𑧇𑧑𑦮𑧠𑧌𑧇𑧜 𑧁𑧑𑧊𑧒𑦿𑧠𑧇𑧑𑦮𑧠𑧌𑧇𑧜 𑧇𑧑𑧊𑧁𑧠𑧁 𑦵𑧈𑧑𑧆𑧈𑦼𑧞 𑧁 𑦵𑧈𑧑𑧆𑧈𑦼𑦮𑧠𑧌𑧇𑧜 𑧁 𑦿𑧔𑧟𑦯𑧍𑧆𑧔𑦿𑧇𑧁𑧒𑧈𑧜𑧀𑧆𑧑𑧈𑧠𑦰𑧑 𑧁 𑦵𑧠𑦷𑧑𑧁𑧞 𑧁 𑧂𑧠𑧈𑧑𑧂𑧠𑦽𑧒𑦽𑧠𑧊𑧆𑧠॥

𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧍𑧠𑧇 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧆𑧑𑧋𑧠𑧈𑧒𑦽𑧠𑧇 𑧊𑧒𑧎𑧈𑦽𑧒 𑦳𑧒𑦽𑧠𑦽𑧑𑧊𑧈𑦼𑧟। 𑦳𑧒𑦽𑧠𑦽𑧑𑧊𑧈𑦼𑧁𑧑𑧍𑧠𑦽𑧒𑦽𑧠𑧊𑧑𑦿𑦽𑧠𑧈𑧍𑧠𑦽𑧜 𑧊𑧒𑧂𑧈𑧠𑧇𑧑𑧍𑧑𑦽𑧒𑦮𑧠𑧈𑧑𑧞𑦽𑧜 𑧁𑧒𑧌𑧠𑦹𑧁𑧒𑧈𑧠𑧊𑧑𑦼𑧟। 𑦽𑧠𑧈𑧠𑧇𑧀𑧠𑧊𑧊𑧠𑧇𑧊𑧍𑧠𑦾𑧒𑦽𑧑𑧟 𑧍𑧈𑧠𑧊𑧄𑧔𑦿𑧠𑧀𑧑𑧟 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧆𑧑𑧋𑧠𑧈𑧒𑦽𑧠𑧇 𑦠𑧁𑧔𑦽𑧠𑦽𑧈𑧑𑧞 𑧍𑧆𑧠𑧇𑦮𑧠𑧍𑧞𑧄𑧜𑧀𑧒𑧆𑧅𑧒𑧍𑧞𑧄𑧔𑦿𑧠𑧀𑧑𑧟॥

𑦽𑧍𑧠𑧆𑧑𑦵𑧠𑦵𑧠𑦷𑧑𑦽𑧊𑧠𑧇𑧟 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧆𑧎𑧑𑧆𑧞𑦽𑧠𑧈𑧜 𑧆𑧎𑧑𑧊𑧒𑦿𑧠𑧇𑧑𑧆𑧞𑦽𑧠𑧈𑧜𑧡𑧁𑧔𑦽𑧠𑦽𑧈𑧆𑧞𑦽𑧠𑧈𑧜𑧡𑧍𑧆𑧍𑧆𑧆𑧞𑦽𑧠𑧈𑧟 𑧍𑧈𑧠𑧊𑦿𑧔𑧟𑦯𑧂𑧠𑧈𑧋𑧆𑧁𑧟 𑧍𑦽𑧠𑧇𑧆𑧆𑧒𑦾𑧠𑧇𑦽𑧠𑧊𑧑𑦽𑧠 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧇𑧑𑧆𑧔𑦮𑧠𑦽𑧜 𑧆𑧞𑦽𑧠𑧈𑧟।

𑦽𑦿𑧠𑧇𑦾𑧑- 𑦰𑦽𑧚 𑦰𑦽𑧚 𑧂𑧑𑧈𑦰𑦽𑧚 𑧂𑧑𑧈𑧍𑧞𑦰𑦽𑧚 𑧄𑧜𑧀𑧒 𑧍𑧠𑧊𑧑𑧎𑧑॥

𑦢𑦽𑧒 𑧂𑧠𑧈𑦵𑧠𑦷𑧑𑧂𑧑𑧈𑧆𑧒𑦽𑧑𑧎𑧖𑦿𑧇𑧍𑧕𑦽𑧠𑧈𑧞 𑧍𑧆𑧑𑧂𑧠𑦽𑧆𑧠॥
