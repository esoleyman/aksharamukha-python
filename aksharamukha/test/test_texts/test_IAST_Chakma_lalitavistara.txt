𑄣𑄧𑄣𑄨𑄖𑄧𑄤𑄨𑄥𑄴𑄖𑄧𑄢𑄧𑄂

𑄚𑄧𑄟𑄧𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄝𑄪𑄘𑄴𑄙𑄧𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄬𑄞𑄳𑄠𑄧𑄂𑅁

𑅂 𑄃𑄮𑄀 𑄚𑄧𑄟𑄮 𑄘𑄧𑄥𑄧𑄘𑄨𑄉𑄧𑄚𑄧𑄚𑄴𑄖𑄛𑄧𑄢𑄳𑄠𑄧𑄚𑄴𑄖𑄧𑄣𑄮𑄇𑄧𑄙𑄖𑄪𑄛𑄳𑄢𑄧𑄖𑄨𑄥𑄴𑄒𑄨𑄖𑄧𑄥𑄧𑄢𑄳𑄤𑄧𑄝𑄪𑄘𑄴𑄙𑄧𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄢𑄳𑄠𑄧𑄥𑄳𑄢𑄤𑄧𑄇𑄧𑄛𑄳𑄢𑄧𑄖𑄳𑄠𑄬𑄇𑄧𑄝𑄪𑄘𑄴𑄙𑄬𑄞𑄳𑄠𑄮'𑄖𑄩𑄖𑄚𑄉𑄧𑄖𑄧𑄛𑄳𑄢𑄧𑄖𑄳𑄠𑄪𑄖𑄴𑄛𑄧𑄚𑄴𑄬𑄞𑄳𑄠𑄧𑄂𑅂

𑄷𑄶 𑄣𑄨𑄛𑄨𑄥𑄣𑄥𑄧𑄁𑄘𑄧𑄢𑄴𑄥𑄧𑄚𑄧𑄛𑄧𑄢𑄨𑄤𑄧𑄢𑄴𑄖𑄮 𑄘𑄧𑄥𑄧𑄟𑄧𑄂𑅁

𑄃𑄨𑄖𑄨 𑄦𑄨 𑄞𑄨𑄇𑄴𑄥𑄧𑄤𑄧𑄂 𑄥𑄧𑄁𑄤𑄳𑄢𑄨𑄘𑄴𑄙𑄧𑄂 𑄇𑄪𑄟𑄢𑄧𑄂𑅁 𑄖𑄧𑄘 𑄟𑄋𑄴𑄉𑄧𑄣𑄳𑄠𑄧𑄥𑄧𑄖𑄧𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄰𑄂 𑄣𑄨𑄛𑄨𑄥𑄣𑄟𑄪𑄛𑄧𑄚𑄩𑄠𑄧𑄖𑄬 𑄥𑄴𑄟𑄧 𑄘𑄧𑄥𑄧𑄞𑄨𑄢𑄴𑄘𑄢𑄧𑄇𑄧𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄰𑄂 𑄛𑄧𑄢𑄨𑄤𑄳𑄢𑄨𑄖𑄧𑄂 𑄛𑄪𑄢𑄧𑄥𑄴𑄇𑄳𑄢𑄨𑄖𑄧𑄂, 𑄘𑄧𑄥𑄧𑄞𑄨𑄥𑄴𑄌𑄧 𑄢𑄧𑄗𑄧𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄰𑄂 𑄈𑄘𑄧𑄚𑄩𑄠𑄧𑄞𑄮𑄎𑄧𑄚𑄩𑄠𑄧𑄥𑄳𑄤𑄘𑄧𑄚𑄩𑄠𑄧𑄛𑄧𑄢𑄨𑄛𑄫𑄢𑄴𑄕𑄰𑄢𑄴𑄦𑄨𑄢𑄧𑄕𑄳𑄠𑄧𑄥𑄪𑄤𑄧𑄢𑄴𑄕𑄧𑄛𑄧𑄢𑄨𑄛𑄫𑄢𑄴𑄕𑄰𑄥𑄴𑄌𑄧𑅁 𑄡𑄬𑄚𑄧 𑄇𑄧𑄛𑄨𑄣𑄧𑄤𑄧𑄥𑄴𑄖𑄪𑄚𑄨 𑄟𑄧𑄦𑄚𑄧𑄉𑄧𑄢𑄬 𑄤𑄩𑄗𑄨𑄌𑄧𑄖𑄳𑄤𑄧𑄢𑄧𑄢𑄧𑄗𑄳𑄠𑄚𑄴𑄖𑄧𑄢𑄛𑄧𑄕𑄧𑄟𑄪𑄈𑄬𑄥𑄳𑄤𑄧𑄞𑄳𑄠𑄧𑄤𑄧𑄇𑄩𑄢𑄳𑄠𑄧𑄖𑄬 𑄥𑄴𑄟𑄧 𑄃𑄧𑄞𑄨𑄤𑄨𑄥𑄳𑄢𑄟𑄳𑄠𑄧𑄚𑄴𑄖𑄬𑅁 𑄃𑄧𑄥𑄴𑄑𑄞𑄨𑄥𑄴𑄌𑄧 𑄖𑄫𑄢𑄳𑄠𑄧𑄥𑄧𑄖𑄧𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄰𑄂 𑄛𑄳𑄢𑄧𑄊𑄪𑄥𑄳𑄠𑄧𑄟𑄕𑄰𑄢𑄴𑄟𑄧𑄦𑄧𑄖 𑄌𑄧 𑄛𑄪𑄥𑄴𑄛𑄧𑄤𑄧𑄢𑄴𑄥𑄬𑄕𑄞𑄨𑄛𑄳𑄢𑄧𑄤𑄧𑄢𑄴𑄥𑄧𑄖 𑄤𑄨𑄖𑄧𑄢𑄴𑄘𑄨𑄚𑄨𑄢𑄳𑄠𑄫𑄦𑄧𑄖𑄮𑄢𑄧𑄕𑄧𑄉𑄧𑄤𑄇𑄴𑄥𑄧𑄦𑄧𑄢𑄴𑄟𑄳𑄠𑄧𑄇𑄫𑄑𑄉𑄢𑄧𑄛𑄳𑄢𑄥𑄘𑄧𑄖𑄧𑄣𑄬𑄥𑄪 𑄇𑄧𑄚𑄳𑄠𑄥𑄧𑄖𑄧𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄕𑄨 𑄥𑄧𑄢𑄳𑄤𑄣𑄧𑄁𑄇𑄢𑄧𑄞𑄫𑄥𑄨𑄖𑄂 𑄥𑄴𑄗𑄨𑄖 𑄃𑄧𑄞𑄫𑄤𑄧𑄚𑄴𑅁 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄧𑄁 𑄛𑄳𑄢𑄬𑄇𑄴𑄥𑄧𑄟𑄕𑄂 𑄇𑄪𑄥𑄪𑄟𑄚𑄨 𑄌𑄧 𑄇𑄴𑄥𑄨𑄛𑄧𑄚𑄴𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄃𑄧𑄥𑄴𑄑𑄯 𑄌𑄧 𑄟𑄧𑄢𑄪𑄖𑄴𑄇𑄧𑄚𑄳𑄠𑄥𑄧𑄦𑄧𑄥𑄳𑄢𑄕𑄨 𑄤𑄨𑄉𑄧𑄣𑄨𑄖𑄣𑄧𑄁𑄇𑄢𑄞𑄧𑄢𑄧𑄕𑄣𑄧𑄁𑄇𑄳𑄢𑄨𑄖𑄚𑄨 𑄢𑄧𑄖𑄳𑄚𑄧𑄞𑄧𑄘𑄳𑄢𑄧𑄁𑄇𑄧𑄢𑄬𑄕𑄧 𑄉𑄳𑄢𑄨𑄦𑄩𑄖𑄚𑄨 𑄟𑄢𑄴𑄉𑄧𑄁 𑄥𑄮𑄙𑄧𑄠𑄧𑄚𑄴𑄖𑄳𑄠𑄮 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄧𑄥𑄳𑄠𑄧 𑄛𑄪𑄢𑄧𑄖𑄮 𑄉𑄧𑄌𑄴𑄍𑄧𑄚𑄴𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄘𑄬𑄤𑄧𑄚𑄉𑄧𑄠𑄧𑄇𑄴𑄥𑄧𑄉𑄧𑄚𑄴𑄙𑄧𑄢𑄳𑄤𑄥𑄪𑄢𑄧𑄉𑄧𑄢𑄪𑄓𑄧𑄇𑄨𑄚𑄴𑄧𑄢𑄧𑄟𑄧𑄦𑄮𑄢𑄧𑄉𑄥𑄴𑄌𑄢𑄴𑄙𑄧𑄇𑄠𑄨𑄇 𑄉𑄧𑄉𑄧𑄚𑄧𑄖𑄧𑄣𑄖𑄴𑄛𑄪𑄥𑄴𑄛𑄧𑄛𑄧𑄑𑄴𑄧𑄘𑄟𑄚𑄳𑄠𑄧𑄞𑄨𑄛𑄳𑄢𑄧𑄣𑄧𑄟𑄴𑄝𑄧𑄠𑄧𑄚𑄴𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄥𑄧𑄢𑄳𑄤𑄬 𑄌𑄧 𑄥𑄇𑄳𑄠𑄧𑄉𑄧𑄕𑄂 𑄥𑄪𑄘𑄴𑄙𑄮𑄘𑄧𑄚𑄧𑄁 𑄢𑄎𑄚𑄧𑄁 𑄛𑄪𑄢𑄧𑄥𑄴𑄇𑄳𑄢𑄨𑄖𑄳𑄠𑄧 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄧𑄥𑄳𑄠𑄧 𑄛𑄪𑄢𑄧𑄖𑄮 𑄉𑄧𑄌𑄴𑄍𑄧𑄚𑄴𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄃𑄧𑄚𑄬𑄚𑄰𑄤𑄧𑄁𑄢𑄫𑄛𑄬𑄕𑄧 𑄤𑄳𑄠𑄫𑄦𑄬𑄚𑄧 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄮 𑄣𑄨𑄛𑄨𑄥𑄣𑄟𑄪𑄛𑄧𑄚𑄩𑄠𑄧𑄖𑄬 𑄥𑄴𑄟𑄧𑅂

𑄥𑄧𑄟𑄧𑄚𑄧𑄚𑄴𑄖𑄧𑄢𑄧𑄛𑄳𑄢𑄧𑄤𑄬𑄥𑄨𑄖𑄧𑄥𑄴𑄌𑄧 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄮 𑄣𑄨𑄛𑄨𑄥𑄣𑄟𑄴𑅁 𑄃𑄧𑄗𑄧 𑄤𑄨𑄥𑄳𑄤𑄟𑄨𑄖𑄳𑄢𑄮 𑄚𑄟𑄧 𑄘𑄢𑄧𑄇𑄌𑄢𑄳𑄠𑄮 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄧𑄥𑄳𑄠𑄧 𑄥𑄳𑄢𑄨𑄠𑄧𑄁 𑄖𑄬𑄎𑄧𑄥𑄴𑄌𑄥𑄧𑄦𑄧𑄟𑄚𑄮 𑄙𑄧𑄢𑄧𑄕𑄨𑄖𑄧𑄣𑄬 𑄚𑄨𑄤𑄨𑄥𑄴𑄑𑄮'𑄙𑄮𑄟𑄪𑄈𑄧𑄂 𑄛𑄳𑄢𑄧𑄛𑄧𑄖𑄧𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄖𑄧𑄁 𑄖𑄧𑄗 𑄛𑄳𑄢𑄧𑄛𑄧𑄖𑄨𑄖𑄧𑄁 𑄘𑄳𑄢𑄨𑄥𑄴𑄑𑄳𑄤 𑄥𑄪𑄞𑄋𑄴𑄉𑄮 𑄚𑄟𑄧 𑄖𑄪𑄥𑄨𑄖𑄧𑄇𑄠𑄨𑄇𑄮 𑄘𑄬𑄤𑄧𑄛𑄪𑄖𑄳𑄢𑄮 𑄘𑄧𑄇𑄴𑄥𑄨𑄕𑄬𑄚𑄧 𑄇𑄧𑄢𑄧𑄖𑄧𑄣𑄬𑄚𑄧 𑄛𑄧𑄢𑄨𑄉𑄳𑄢𑄨𑄦𑄳𑄠𑄮𑄖𑄴𑄗𑄛𑄧𑄠𑄧𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄃𑄪𑄖𑄴𑄗𑄛𑄳𑄠𑄧 𑄌𑄧 𑄉𑄧𑄉𑄧𑄚𑄧𑄖𑄧𑄣𑄧𑄥𑄴𑄗𑄮 𑄢𑄎𑄚𑄧𑄁 𑄥𑄪𑄘𑄴𑄙𑄮𑄘𑄧𑄚𑄧𑄁 𑄖𑄧𑄁 𑄌𑄧 𑄟𑄧𑄦𑄚𑄴𑄖𑄧𑄁 𑄎𑄧𑄚𑄧𑄇𑄠𑄧𑄁 𑄉𑄗𑄞𑄨𑄢𑄧𑄞𑄳𑄠𑄧𑄞𑄥𑄧𑄖𑄴—
