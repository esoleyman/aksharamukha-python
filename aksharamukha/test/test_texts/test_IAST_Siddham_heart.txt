𑖁𑖨𑖿𑖧𑖝𑖿𑖨𑖰𑖨𑖝𑖿𑖡𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖭𑖳𑖝𑖿𑖨𑖦𑖿

(𑖏) 𑖭𑖽𑖭𑖿𑖎𑖴𑖝𑖢𑖯𑖙𑖾

𑖥𑖯𑖨𑖝𑖱𑖧𑖥𑖯𑖬𑖯𑖧𑖯𑖦𑖿-𑖁𑖨𑖿𑖧𑖝𑖿𑖨𑖰𑖨𑖝𑖿𑖡𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖭𑖳𑖝𑖿𑖨𑖦𑖿

𑖥𑖺𑖘𑖥𑖯𑖬𑖯𑖧𑖯𑖦𑖿-𑖣𑖐𑖿𑖭𑖿-𑖢𑖯-𑖟𑖎𑖺𑖡𑖿-𑖦𑖿𑖓𑖿𑖔𑖺𑖐𑖿-𑖐𑖿𑖭𑖲𑖦𑖿-𑖕𑖸𑖭𑖿-𑖭𑖲-𑖚𑖿𑖨𑖡𑖿-𑖢𑖯-𑖦𑖿𑖟𑖺

𑖡𑖦𑖾 𑖭𑖨𑖿𑖪𑖤𑖲𑖟𑖿𑖠𑖤𑖺𑖠𑖰𑖭𑖝𑖿𑖝𑖿𑖪𑖸𑖥𑖿𑖧𑖾

(𑖤𑖲𑖟𑖿𑖠𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖾)

𑖂𑖝𑖿𑖧𑖢𑖰 𑖤𑖲𑖟𑖿𑖠𑖺 𑖥𑖐𑖪𑖯𑖽𑖭𑖿𑖝𑖞𑖯𑖐𑖝𑖺𑖨𑖿𑖮𑖡𑖿 𑖭𑖦𑖿𑖧𑖎𑖿𑖭𑖽𑖤𑖲𑖟𑖿𑖠𑖺 𑖪𑖰𑖟𑖿𑖧𑖯𑖓𑖨𑖜𑖭𑖦𑖿𑖢𑖡𑖿𑖡𑖾 𑖭𑖲𑖐𑖝𑖺 𑖩𑖺𑖎𑖪𑖰𑖟𑖡𑖲𑖝𑖿𑖝𑖨𑖾 𑖢𑖲𑖨𑖲𑖬𑖟𑖦𑖿𑖧𑖭𑖯𑖨𑖞𑖰𑖾 𑖫𑖯𑖭𑖿𑖝𑖯 𑖟𑖸𑖪𑖦𑖡𑖲𑖬𑖿𑖧𑖯𑖜𑖯𑖽 𑖤𑖲𑖟𑖿𑖠𑖺 𑖥𑖐𑖪𑖯𑖡𑖰𑖝𑖰𑗂

𑖡𑖰𑖬𑖿𑖧𑖡𑖿𑖟𑖾 𑖭 𑖝𑖞𑖯𑖐𑖝𑖾 𑖢𑖲𑖜𑖿𑖧𑖯𑖡𑖯𑖦𑖿, 𑖀𑖪𑖰𑖢𑖿𑖨𑖜𑖯𑖫𑖾 𑖎𑖲𑖫𑖩𑖦𑖳𑖩𑖯𑖡𑖯𑖦𑖿, 𑖀𑖩𑖒𑖿𑖎𑖴𑖝𑖾 𑖎𑖿𑖬𑖯𑖡𑖿𑖝𑖿𑖧𑖯, 𑖁𑖩𑖧𑖾 𑖢𑖲𑖜𑖿𑖧𑖡𑖰𑖠𑖯𑖡𑖯𑖡𑖯𑖦𑖿, 𑖓𑖰𑖝𑖿𑖨𑖰𑖝𑖺𑖡𑖲𑖪𑖿𑖧𑖗𑖿𑖕𑖡𑖹𑖾 𑖎𑖲𑖭𑖲𑖦𑖰𑖝𑖺 𑖩𑖎𑖿𑖬𑖜𑖹𑖾, 𑖢𑖿𑖨𑖝𑖰𑖨𑖳𑖢𑖺 𑖐𑖺𑖓𑖨𑖸𑖜, 𑖀𑖢𑖿𑖨𑖝𑖰𑖎𑖳𑖩𑖺 𑖟𑖨𑖿𑖫𑖡𑖸𑖡, 𑖀𑖥𑖰𑖨𑖝𑖰𑖾 𑖫𑖿𑖨𑖟𑖿𑖠𑖯(𑖫𑖲𑖟𑖿𑖠𑖯)𑖠𑖰𑖦𑖲𑖎𑖿𑖝𑖯𑖡𑖯𑖦𑖿, 𑖀𑖡𑖥𑖰𑖥𑖳𑖝𑖾 𑖢𑖿𑖨𑖕𑖿𑖗𑖧𑖯, 𑖀𑖡𑖪𑖦𑖨𑖿𑖟𑖡𑖱𑖧𑖺 𑖤𑖩𑖹𑖾, 𑖫𑖯𑖭𑖿𑖝𑖯 𑖭𑖨𑖿𑖪𑖭𑖝𑖿𑖝𑖿𑖪𑖯𑖡𑖯𑖦𑖿, 𑖢𑖰𑖝𑖯 𑖤𑖺𑖠𑖰𑖭𑖝𑖿𑖝𑖿𑖪𑖯𑖡𑖯𑖦𑖿, 𑖨𑖯𑖕𑖯 𑖁𑖨𑖿𑖧𑖢𑖲𑖟𑖿𑖐𑖩𑖯𑖡𑖯𑖦𑖿, 𑖭𑖯𑖨𑖿𑖞𑖪𑖯𑖮𑖾 𑖡𑖰𑖨𑖿𑖪𑖯𑖜𑖡𑖐𑖨𑖭𑖦𑖿𑖢𑖿𑖨𑖭𑖿𑖞𑖰𑖝𑖯𑖡𑖯𑖦𑖿, 𑖀𑖢𑖿𑖨𑖦𑖸𑖧𑖺 𑖕𑖿𑖗𑖯𑖡𑖸𑖡, 𑖀𑖓𑖰𑖡𑖿𑖝𑖿𑖧𑖾 𑖢𑖿𑖨𑖝𑖰𑖥𑖯𑖡𑖸𑖡, 𑖪𑖰𑖫𑖲𑖟𑖿𑖠𑖾 𑖭𑖿𑖪𑖨𑖸𑖜, 𑖁𑖭𑖿𑖪𑖟𑖡𑖱𑖧𑖺 𑖑𑖺𑖬𑖸𑖜, 𑖀𑖭𑖸𑖓𑖡𑖎𑖺 𑖨𑖳𑖢𑖸𑖜, 𑖀𑖢𑖿𑖨𑖝𑖰𑖭𑖦𑖾 𑖎𑖯𑖧𑖸𑖡, 𑖀𑖩𑖰𑖢𑖿𑖝𑖾 𑖎𑖯𑖦𑖹𑖾, 𑖀𑖡𑖲𑖢𑖩𑖰𑖢𑖿𑖝𑖺 𑖨𑖳𑖢𑖹𑖾 𑖀𑖭𑖽𑖭𑖴𑖬𑖿𑖘 𑖁𑖨𑖳𑖢𑖿𑖧𑖹𑖾, 𑖪𑖰𑖢𑖿𑖨𑖦𑖲𑖎𑖿𑖝𑖾 𑖭𑖿𑖎𑖡𑖿𑖠𑖸𑖥𑖿𑖧𑖾, 𑖪𑖰𑖭𑖦𑖿𑖢𑖿𑖨𑖧𑖲𑖎𑖿𑖝𑖺 𑖠𑖯𑖝𑖲𑖥𑖰𑖾, 𑖭𑖽𑖪𑖴𑖝 𑖁𑖧𑖝𑖡𑖹𑖾, 𑖢𑖿𑖨𑖓𑖿𑖔𑖰𑖡𑖿𑖡𑖺 𑖐𑖿𑖨𑖡𑖿𑖞𑖹𑖾, 𑖪𑖰𑖦𑖲𑖎𑖿𑖝𑖾 𑖢𑖨𑖰𑖟𑖯𑖑𑖿𑖡𑖹𑖾, 𑖢𑖨𑖰𑖦𑖲𑖎𑖿𑖝𑖭𑖿𑖝𑖴𑖬𑖿𑖜𑖧𑖯, 𑖌𑖑𑖯𑖟𑖲𑖝𑖿𑖝𑖱𑖨𑖿𑖜𑖾 𑖢𑖨𑖰𑖢𑖳𑖨𑖿𑖜𑖺 𑖕𑖿𑖗𑖯𑖡𑖸𑖡, 𑖢𑖿𑖨𑖝𑖰𑖬𑖿𑖙𑖰𑖝𑖺𑖝𑖱𑖝𑖯𑖡𑖯𑖐𑖝𑖢𑖿𑖨𑖝𑖿𑖧𑖲𑖝𑖿𑖢𑖡𑖿𑖡𑖯𑖡𑖯𑖽 𑖤𑖲𑖟𑖿𑖠𑖯𑖡𑖯𑖽 𑖥𑖐𑖪𑖝𑖯𑖽 𑖕𑖿𑖗𑖯𑖡𑖸, 𑖀𑖢𑖿𑖨𑖝𑖰𑖬𑖿𑖙𑖰𑖝𑖺 𑖡𑖰𑖨𑖿𑖪𑖯𑖜𑖸, 𑖭𑖿𑖞𑖰𑖝𑖺 𑖥𑖳𑖝𑖎𑖺𑖘𑖿𑖧𑖯𑖦𑖿, 𑖭𑖿𑖞𑖰𑖝𑖾 𑖭𑖨𑖿𑖪𑖭𑖝𑖿𑖝𑖿𑖪𑖯𑖩𑖺𑖎𑖡𑖱𑖧𑖯𑖧𑖯𑖽 𑖥𑖳𑖦𑖻, 𑖭𑖨𑖿𑖪 𑖂𑖦𑖸 𑖝𑖞𑖯𑖐𑖝𑖯𑖡𑖯𑖽 𑖪𑖰𑖫𑖸𑖬𑖝𑖾 𑖭𑖦𑖿𑖧𑖐𑖿 𑖐𑖲𑖜𑖯𑖾𑗂

(𑖠𑖨𑖿𑖦𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖾)

𑖭𑖟𑖿𑖠𑖨𑖿𑖦𑖭𑖿𑖝𑖲 𑖁𑖟𑖻 𑖎𑖩𑖿𑖧𑖯𑖜𑖾, 𑖦𑖠𑖿𑖧𑖸 𑖎𑖩𑖿𑖧𑖯𑖜𑖾, 𑖢𑖨𑖿𑖧𑖪𑖭𑖯𑖡𑖸 𑖎𑖩𑖿𑖧𑖯𑖜𑖾, 𑖭𑖿𑖪𑖨𑖿𑖞𑖾 𑖭𑖲𑖪𑖿𑖧𑖗𑖿𑖕𑖡𑖾, 𑖎𑖸𑖪𑖩𑖾, 𑖢𑖨𑖰𑖢𑖳𑖨𑖿𑖜𑖾, 𑖢𑖨𑖰𑖫𑖲𑖟𑖿𑖠𑖾, 𑖢𑖨𑖿𑖧𑖪𑖟𑖯𑖝𑖾, 𑖭𑖿𑖪𑖯𑖏𑖿𑖧𑖯𑖝𑖾 𑖥𑖐𑖪𑖝𑖺 𑖠𑖨𑖿𑖦𑖾, 𑖭𑖯𑖡𑖿𑖟𑖴𑖬𑖿𑖘𑖰𑖎𑖾, 𑖡𑖰𑖨𑖿𑖕𑖿𑖪𑖨𑖾, 𑖁𑖎𑖯𑖩𑖰𑖎𑖾, 𑖍𑖢𑖡𑖯𑖧𑖰𑖎𑖾, 𑖋𑖮𑖰𑖢𑖫𑖿𑖧𑖰𑖎𑖾, 𑖢𑖿𑖨𑖝𑖿𑖧𑖯𑖝𑖿𑖦𑖪𑖸𑖟𑖡𑖱𑖧𑖺 𑖪𑖰𑖕𑖿𑖗𑖹𑖾, 𑖭𑖿𑖪𑖯𑖏𑖿𑖧𑖯𑖝𑖺 𑖥𑖐𑖪𑖝𑖺 𑖠𑖨𑖿𑖦𑖪𑖰𑖡𑖧𑖾 𑖭𑖲𑖢𑖿𑖨𑖪𑖸𑖟𑖰𑖝𑖾 𑖡𑖹𑖨𑖿𑖧𑖯𑖜𑖰𑖎𑖾, 𑖭𑖽𑖤𑖺𑖠𑖰𑖐𑖯𑖦𑖱, 𑖀𑖥𑖰𑖡𑖿𑖡𑖾 𑖭𑖽𑖭𑖿𑖝𑖳𑖢𑖾, 𑖭𑖢𑖿𑖨𑖝𑖰𑖫𑖨𑖜𑖾, 𑖔𑖰𑖡𑖿𑖡𑖢𑖿𑖩𑖺𑖝𑖰𑖎𑖾𑗂

(𑖭𑖽𑖑𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖾)

𑖭𑖲𑖢𑖿𑖨𑖝𑖰𑖢𑖡𑖿𑖡𑖺 𑖥𑖐𑖪𑖝 𑖁𑖨𑖿𑖧𑖭𑖽𑖑𑖾, 𑖡𑖿𑖧𑖯𑖧𑖢𑖿𑖨𑖝𑖰𑖢𑖡𑖿𑖡𑖾, 𑖆𑖕𑖲𑖢𑖿𑖨𑖝𑖰𑖢𑖡𑖿𑖡𑖾, 𑖭𑖯𑖦𑖱𑖓𑖱𑖢𑖿𑖨𑖝𑖰𑖢𑖡𑖿𑖡𑖾, 𑖀𑖗𑖿𑖕𑖩𑖱𑖎𑖨𑖜𑖱𑖧𑖾, 𑖭𑖯𑖦𑖱𑖓𑖱𑖎𑖨𑖜𑖱𑖧𑖾, 𑖢𑖲𑖜𑖿𑖧𑖫𑖿𑖨𑖱𑖎𑖿𑖬𑖸𑖝𑖿𑖨𑖾, 𑖦𑖮𑖯𑖟𑖎𑖿𑖬𑖰𑖜𑖯𑖢𑖨𑖰𑖫𑖺𑖠𑖎𑖾, 𑖢𑖿𑖨𑖯𑖮𑖪𑖡𑖱𑖧𑖾, 𑖁𑖮𑖪𑖡𑖱𑖧𑖾𑗂

𑗃 𑖁𑖨𑖿𑖧𑖝𑖿𑖨𑖰𑖨𑖝𑖿𑖡𑖯𑖡𑖲𑖭𑖿𑖦𑖴𑖝𑖰𑖭𑖳𑖝𑖿𑖨𑖽 𑖭𑖦𑖯𑖢𑖿𑖝𑖦𑖿𑗃
