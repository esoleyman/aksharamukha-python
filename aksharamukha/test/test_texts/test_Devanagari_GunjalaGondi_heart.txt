𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑶇𑶗𑶈𑶍𑵸𑵬𑶉𑶎𑵳𑶗𑶈𑵰.

[𑶉𑶕𑵱𑶗𑶉𑶋𑶅𑶗𑵳𑵰𑶊𑵳𑶗𑶈𑶍𑵱𑶊]

.. 𑵺𑵰𑶖 𑶉𑶈𑶗𑵭𑶀𑶗𑵺𑶊𑵬 ..

.. 𑶘 𑵺𑵰𑶓 𑵯𑵶𑵭𑵳𑶗𑵬𑶑 𑵡𑶈𑶗𑵬𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵬𑶑 ..

𑵡𑶈𑶗𑵬𑶊𑵭𑵵𑶓𑵱𑶋𑵳𑶐𑶉𑶗𑵭𑶈𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶓 𑵶𑵰𑶗𑵯𑶌𑶈𑶊𑵬𑶊𑶕 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵬𑶊𑶕 𑵻𑶈𑶗𑵬𑶊𑶕 𑵻𑶈𑵰𑶊𑵺𑶓 𑵭𑶗𑵬𑵭𑵵𑶓𑵱𑵬𑵳𑶋 𑶉𑶗𑵰. 𑶅𑵺𑶗𑵻 𑶉𑶗𑵱𑵺𑶗𑵹𑶊𑶖, 𑵳𑶊𑶕𑶉𑶗𑵻 𑶉𑶗𑵭𑵯𑶊𑵭𑶉𑶎𑵺𑶗𑵬𑶊𑵺 𑶅𑶉𑶗𑵬𑵳𑶋 𑶉𑶗𑵰..

𑵢𑶇 𑶉𑶊𑶈𑶋𑶅𑶍𑵳𑶗𑶈 𑶈𑶎𑶅𑶕 𑶉𑶎𑵺𑶗𑵬𑵳𑶊, 𑶉𑶎𑵺𑶗𑵬𑵳𑶑𑵭 𑶈𑶎𑶅𑵰. 𑶈𑶎𑶅𑶊𑵺𑶗𑵺 𑶅𑶗𑶈𑶍𑵴𑵱 𑶉𑶎𑵺𑶗𑵬𑵳𑶊, 𑶉𑶎𑵺𑶗𑵬𑵳𑶊𑵬𑶊 𑵺 𑶅𑶗𑶈𑶍𑵴𑵶 𑶈𑶎𑶅𑵰. 𑵬𑵸𑶗𑶈𑶎𑶅𑶕 𑶉𑶊 𑶉𑶎𑵺𑶗𑵬𑵳𑶊, 𑵬𑶊 𑶉𑶎𑵺𑶗𑵬𑵳𑶊 𑵳𑵸𑶗𑶈𑶎𑶅𑵰..

𑵧𑵭𑵰𑶐𑵭 𑵭𑶐𑵸𑵺𑶊𑶉𑶕𑶀𑶗𑵺𑶊𑶉𑶕𑶉𑶗𑵱𑶊𑶈𑵭𑶋𑶀𑶗𑵺𑶊𑵺𑶊𑵺𑶋..

𑵢𑶇𑶕 𑶉𑶊𑶈𑶋𑶅𑶍𑵳𑶗𑶈 𑶉𑶈𑶗𑵭𑵹𑶈𑶗𑵰𑶊𑶖 𑶉𑶎𑵺𑶗𑵬𑵳𑶊𑵵𑵱𑶗𑶉𑵺𑶊 𑵠𑵺𑶍𑵳𑶗𑶅𑵺𑶗𑵺𑶊 𑵠𑵺𑶋𑶈𑶍𑵸𑶗𑵹𑶊 𑵠𑵰𑵵𑶊 𑵺 𑵭𑶋𑵰𑵵𑶊 𑵺𑶓𑵺𑶊 𑵺 𑶅𑶈𑶋𑶅𑶎𑶈𑶗𑵺𑶊𑶖. 𑵳𑶉𑶗𑵰𑶊𑵻𑶗𑵼𑶊𑶈𑶋𑶅𑶍𑵳𑶗𑶈 𑶉𑶎𑵺𑶗𑵬𑵳𑶊𑵬𑶊𑶕 𑵺 𑶈𑶎𑶅𑵰, 𑵺 𑵭𑶐𑵸𑵺𑶊, 𑵺 𑶉𑶕𑶀𑶗𑵺𑶊, 𑵺 𑶉𑶕𑶉𑶗𑵱𑶊𑶈𑶊𑶖, 𑵺 𑵭𑶋𑶀𑶗𑵺𑶊𑵺𑶊𑵺𑶋. 𑵺 𑵻𑵱𑶗𑶉𑶍𑶖𑶉𑶗𑶈𑶓𑵳𑶗𑶈𑵷𑶗𑶈𑶊𑵺𑶀𑶋𑶇𑶗𑵭𑶊𑵱𑶊𑵬𑵰𑵺𑶊𑶕𑶉𑶋, 𑵺 𑶈𑶎𑶅𑶉𑵮𑶗𑵸𑵶𑵺𑶗𑵹𑶈𑶉𑶉𑶗𑶅𑶗𑶈𑶉𑶗𑵽𑵭𑶗𑵬𑵹𑶈𑶗𑵰𑶊𑶖. 𑵺 𑵻𑵱𑶗𑶉𑶍𑶈𑶗𑵹𑶊𑵳𑶍𑶈𑶗𑵬𑶊𑵭𑵺𑶗𑵺 𑵰𑵺𑶓𑵹𑶊𑵳𑶍𑶖..

𑵺 𑵭𑶋𑵸𑶗𑵬𑶊 𑵺𑶊𑵭𑶋𑵸𑶗𑵬𑶊 𑵺 𑵭𑶋𑵸𑶗𑵬𑶊𑵱𑶗𑶉𑵬𑶓 𑵺𑶊𑵭𑶋𑵸𑶗𑵬𑶊𑵱𑶗𑶉𑵬𑶓 𑵬𑶊𑵭𑵺𑶗𑵺 𑶀𑶈𑶊𑵰𑶈𑵺𑶕 𑵺 𑶀𑶈𑶊𑵰𑶈𑵺𑵱𑶗𑶉𑵬𑶓 𑵺 𑵸𑶍𑶖𑵲𑶉𑵰𑶍𑵸𑵬𑵺𑶋𑶈𑶓𑵹𑵰𑶊𑶈𑶗𑵶𑶊 𑵺 𑶀𑶗𑵺𑶊𑵺𑶕 𑵺 𑶅𑶗𑶈𑶊𑶅𑶗𑵳𑶋𑵳𑶗𑵭𑵰..

𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶉𑶗𑵬 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵰𑶊𑶉𑶗𑶈𑶋𑵳𑶗𑵬 𑵭𑶋𑶇𑶈𑵳𑶋 𑵻𑶋𑵳𑶗𑵳𑶊𑵭𑶈𑵺𑶖. 𑵻𑶋𑵳𑶗𑵳𑶊𑵭𑶈𑵺𑵺𑶊𑶉𑶗𑵳𑶋𑵳𑶗𑵭𑶊𑵸𑵳𑶗𑶈𑶉𑶗𑵳𑶓 𑵭𑶋𑶅𑶈𑶗𑵬𑶊𑶉𑶊𑵳𑶋𑵱𑶗𑶈𑶊𑵺𑶗𑵳𑶓 𑵺𑶋𑶉𑶗𑵾𑵺𑶋𑶈𑶗𑵭𑶊𑵺𑶖. 𑵳𑶗𑶈𑶗𑵬𑵹𑶗𑵭𑵭𑶗𑵬𑵭𑶉𑶗𑵴𑶋𑵳𑶊𑶖 𑶉𑶈𑶗𑵭𑵮𑶍𑵸𑶗𑵹𑶊𑶖 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵰𑶊𑶉𑶗𑶈𑶋𑵳𑶗𑵬 𑵠𑵺𑶍𑵳𑶗𑵳𑶈𑶊𑶕 𑶉𑵰𑶗𑵬𑵱𑶗𑶉𑶕𑵮𑶓𑵹𑶋𑵰𑵯𑶋𑶉𑶕𑵮𑶍𑵸𑶗𑵹𑶊𑶖..

𑵳𑶉𑶗𑵰𑶊𑶀𑶗𑶀𑶗𑵺𑶊𑵳𑵭𑶗𑵬𑶖 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵰𑶇𑶊𑵰𑵺𑶗𑵳𑶗𑶈𑶓 𑵰𑶇𑶊𑵭𑶋𑵸𑶗𑵬𑶊𑵰𑵺𑶗𑵳𑶗𑶈𑶓𑵺𑶍𑵳𑶗𑵳𑶈𑵰𑵺𑶗𑵳𑶗𑶈𑶓𑶉𑵰𑶉𑵰𑵰𑵺𑶗𑵳𑶗𑶈𑶖 𑶉𑶈𑶗𑵭𑵸𑶍𑶖𑵲𑶅𑶗𑶈𑶉𑵰𑵺𑶖 𑶉𑵳𑶗𑵬𑵰𑵰𑶋𑵴𑶗𑵬𑵳𑶗𑵭𑶊𑵳 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑵬𑶊𑵰𑶍𑵱𑶗𑵳𑶓 𑵰𑵺𑶗𑵳𑶗𑶈𑶖.

𑵳𑵸𑶗𑵬𑵴𑶊- 𑵶𑵳𑶐 𑵶𑵳𑶐 𑶅𑶊𑶈𑵶𑵳𑶐 𑶅𑶊𑶈𑶉𑶕𑵶𑵳𑶐 𑵮𑶓𑵹𑶋 𑶉𑶗𑵭𑶊𑶇𑶊..

𑵢𑵳𑶋 𑶅𑶗𑶈𑶀𑶗𑵺𑶊𑶅𑶊𑶈𑵰𑶋𑵳𑶊𑶇𑶗𑶈𑶍𑵸𑵬𑶉𑶎𑵳𑶗𑶈𑶕 𑶉𑵰𑶊𑶅𑶗𑵳𑵰..
