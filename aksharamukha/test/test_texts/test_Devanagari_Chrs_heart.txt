𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾵𐿂𐾴𐾺𐾿𐾶𐿄𐿂𐾽.

[𐾿𐾽𐾻𐿃𐿁𐿄𐾽𐾰𐿄𐿂𐾻𐾰]

.. 𐾾𐾽𐾵 𐾿𐿂𐾲𐾳𐾾𐾰𐾺 ..

.. 𐾰𐾶𐾽 𐾾𐾽𐾶 𐾲𐾳𐾲𐿄𐾺𐾺 𐾰𐿂𐾺𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾺𐾺 ..

𐾰𐿂𐾺𐾰𐾲𐾼𐾶𐾻𐿄𐾺𐿃𐾲𐿂𐾲𐾶𐾴𐾿𐿄𐾲𐾶 𐾳𐾽𐾲𐾺𐿂𐾰𐾺𐾰𐾽 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾺𐾰𐾽 𐿄𐿃𐿂𐾺𐾰𐾽 𐿄𐿃𐿂𐾽𐾰𐾾𐾶 𐾲𐾺𐾲𐾼𐾶𐾻𐾺𐿄 𐾿𐾽. 𐿁𐾾𐿄𐿃 𐾿𐾻𐾾𐾴𐾰𐾵, 𐿄𐾰𐾽𐿃𐿄𐿃 𐾿𐾲𐾲𐾰𐾲𐿃𐾶𐾾𐾺𐾰𐾾 𐿁𐿃𐾺𐿄 𐾿𐾽..

𐾰𐾵 𐿃𐾰𐿂𐿁𐿄𐿂 𐿂𐾶𐿁𐾽 𐿃𐾶𐾾𐾺𐿄𐾰, 𐿃𐾶𐾾𐾺𐿄𐾺𐾲 𐿂𐾶𐿁𐾽. 𐿂𐾶𐿁𐾰𐾾 𐿁𐿂𐿄𐾻 𐿃𐾶𐾾𐾺𐿄𐾰, 𐿃𐾶𐾾𐾺𐿄𐾰𐾺𐾰 𐾾 𐿁𐿂𐿄𐾳 𐿂𐾶𐿁𐾽. 𐾺𐾴𐿂𐾶𐿁𐾽 𐾿𐾰 𐿃𐾶𐾾𐾺𐿄𐾰, 𐾺𐾰 𐿃𐾶𐾾𐾺𐿄𐾰 𐿄𐾴𐿂𐾶𐿁𐾽..

𐾰𐾺𐾲𐾽𐾺𐾲 𐾲𐾺𐾴𐾾𐾰𐾿𐾽𐾳𐾾𐾰𐾿𐾽𐾿𐾻𐾰𐿂𐾲𐾳𐾾𐾰𐾾𐾰𐾾..

𐾰𐾵𐾽 𐿃𐾰𐿂𐿁𐿄𐿂 𐾿𐿂𐾲𐾴𐿂𐾽𐾰𐾵 𐿃𐾶𐾾𐾺𐿄𐾰𐾼𐾻𐿃𐾾𐾰 𐾰𐾾𐿄𐿁𐾾𐾰 𐾰𐾾𐿂𐾴𐾰 𐾰𐾽𐾼𐾰 𐾾 𐾲𐾽𐾼𐾰 𐾾𐾶𐾾𐾰 𐾾 𐿁𐿂𐿁𐾶𐿂𐾾𐾰𐾵. 𐿄𐾿𐾽𐾰𐿄𐿃𐾰𐿂𐿁𐿄𐿂 𐿃𐾶𐾾𐾺𐿄𐾰𐾺𐾰𐾽 𐾾 𐿂𐾶𐿁𐾽, 𐾾 𐾲𐾺𐾴𐾾𐾰, 𐾾 𐾿𐾽𐾳𐾾𐾰, 𐾾 𐾿𐾽𐾿𐾻𐾰𐿂𐾰𐾵, 𐾾 𐾲𐾳𐾾𐾰𐾾𐾰𐾾. 𐾾 𐿄𐿃𐾻𐿃𐾵𐿃𐿂𐾶𐿄𐿂𐾳𐿂𐾰𐾾𐾳𐾵𐾲𐾰𐾻𐾰𐾺𐾽𐾾𐾰𐾽𐾿, 𐾾 𐿂𐾶𐿁𐿃𐾲𐾴𐾳𐾾𐾴𐿂𐾿𐾿𐿁𐿂𐿃𐿄𐾲𐾺𐾴𐿂𐾽𐾰𐾵. 𐾾 𐿄𐿃𐾻𐿃𐿂𐾴𐾰𐿄𐿂𐾺𐾰𐾲𐾾 𐾽𐾾𐾶𐾴𐾰𐿄𐾵..

𐾾 𐾲𐾴𐾺𐾰 𐾾𐾰𐾲𐾴𐾺𐾰 𐾾 𐾲𐾴𐾺𐾰𐾻𐿃𐾺𐾶 𐾾𐾰𐾲𐾴𐾺𐾰𐾻𐿃𐾺𐾶 𐾺𐾰𐾲𐾾 𐾳𐿂𐾰𐾽𐿂𐾾𐾽 𐾾 𐾳𐿂𐾰𐾽𐿂𐾾𐾻𐿃𐾺𐾶 𐾾 𐾴𐾵𐾻𐾿𐾽𐾴𐾺𐾾𐿂𐾶𐾴𐾽𐾰𐿂𐾳𐾰 𐾾 𐾳𐾾𐾰𐾾𐾽 𐾾 𐿁𐿂𐾰𐿁𐿄𐿄𐾲𐾽..

𐾲𐾶𐾴𐾿𐿄𐾲𐾿𐾺 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾽𐾰𐿃𐿂𐿄𐾺 𐾲𐾵𐿂𐿄 𐿄𐿃𐿄𐾰𐾲𐿂𐾾𐾵. 𐿄𐿃𐿄𐾰𐾲𐿂𐾾𐾾𐾰𐾿𐿄𐿄𐾲𐾰𐾴𐿄𐿂𐾿𐿄𐾶 𐾲𐿁𐿂𐾺𐾰𐾿𐾰𐿄𐾻𐿂𐾰𐾾𐿄𐾶 𐾾𐿃𐿄𐾾𐿂𐾲𐾰𐾾𐾵. 𐿄𐿂𐾺𐾴𐾲𐾲𐾺𐾲𐾿𐿄𐿄𐾰𐾵 𐾿𐿂𐾲𐾲𐾴𐾰𐾵 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾽𐾰𐿃𐿂𐿄𐾺 𐾰𐾾𐿄𐿂𐾰𐾽 𐾿𐾽𐾺𐾻𐾿𐾽𐾲𐾶𐾴𐾽𐾲𐾿𐾽𐾲𐾴𐾰𐾵..

𐿄𐾿𐾽𐾰𐾳𐾾𐾰𐿄𐾲𐾺𐾵 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾽𐾵𐾰𐾽𐾾𐿄𐿂𐾶 𐾽𐾵𐾰𐾲𐾴𐾺𐾰𐾽𐾾𐿄𐿂𐾶𐾾𐿄𐿂𐾽𐾾𐿄𐿂𐾶𐾿𐾽𐾿𐾽𐾽𐾾𐿄𐿂𐾵 𐾿𐿂𐾲𐾴𐾵𐾻𐿁𐿂𐿃𐾽𐾾𐾵 𐾿𐿄𐾺𐾽𐾽𐿄𐾺𐿄𐾲𐾰𐿄 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾺𐾰𐾽𐾻𐿄𐾶 𐾽𐾾𐿄𐿂𐾵.

𐿄𐾴𐾺𐿄𐾰- 𐾳𐿄𐾺 𐾳𐿄𐾺 𐿁𐾰𐿂𐾳𐿄𐾺 𐿁𐾰𐿂𐾿𐾽𐾳𐿄𐾺 𐾲𐾶𐾴 𐾿𐾲𐾰𐾵𐾰..

𐾰𐿄 𐿁𐿂𐾳𐾾𐾰𐿁𐾰𐿂𐾽𐿄𐾰𐾵𐿂𐾴𐾺𐾿𐾶𐿄𐿂𐾽 𐾿𐾽𐾰𐿁𐿄𐾽..
