𑲋𑲋𑲱𑱺𑲅𑲱𑲍𑲚𑲊𑲎

𑱽𑲁𑲎 𑲍𑲊𑲥𑲀𑲲𑱼𑲜𑲮𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲳𑲀𑲮𑲩𑲎𑱱

𑱱𑱱 𑲏𑲴𑲶 𑱽𑲁𑲴 𑱼𑲌𑱼𑲱𑱴𑱽𑱽𑲚𑲰𑱾𑲊𑲩𑱽𑲚𑲋𑲴𑱲𑱼𑲮𑲰𑱺𑲲𑱾𑲪𑱺𑲱𑲌𑲛𑲱𑱺𑲍𑲊𑲥𑲀𑲲𑱼𑲜𑲮𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲰𑲊𑲩𑲌𑲪𑲰𑲅𑱲𑱾𑲪𑱺𑲩𑲳𑱲𑲀𑲲𑱼𑲜𑲮𑲳𑲀𑲮𑲩𑲴𑱺𑲱𑲰𑱺𑲰𑱽𑲰𑱴𑱺𑱾𑲪𑱺𑲩𑲲𑱺𑲞𑱽𑲝𑲳𑲀𑲮𑲩𑲎𑱱𑱱

༡༠ 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲍𑲵𑱼𑲊𑲬𑱽𑱾𑲊𑲱𑲅𑲊𑲚𑲴 𑱼𑲌𑲁𑲎𑱱

𑲏𑲱𑱺𑲱 𑲎𑲱 𑲀𑲮𑲱𑱲𑲬𑲅𑲎 𑲍𑲵𑲅𑲪𑲱𑱼𑲜𑲮𑲎 𑱲𑲲𑲁𑲰𑲊𑲎𑱱 𑱺𑱼𑲰 𑲁𑲰𑱵𑲔𑲋𑲩𑲌𑱺𑲍𑲎𑲍𑲪𑲳𑲰𑲎 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲁𑲲𑱾𑱽𑲱𑲰𑲉𑱺𑲳 𑲍𑲡 𑱼𑲌𑲀𑲮𑲱𑲊𑲜𑲰𑲊𑱲𑲍𑲎𑲍𑲪𑲳𑲰𑲎 𑱾𑲊𑲱𑲅𑲪𑲱𑱺𑲎 𑱾𑲲𑲊𑲍𑲒𑲪𑲱𑱺𑲎, 𑱼𑲌𑲀𑲮𑲱𑲌𑲢 𑲊𑱻𑲍𑲎𑲍𑲪𑲳𑲰𑲎 𑱳𑲰𑱼𑱽𑲱𑲰𑲉𑲀𑲮𑲴𑲄𑱽𑲱𑲰𑲉𑲍𑲥𑲰𑱼𑱽𑲱𑲰𑲉𑱾𑲊𑲱𑱾𑲲𑲰𑲊𑲝𑲳𑲰𑲊𑲮𑲱𑲊𑱽𑲩𑲍𑲲𑲅𑲊𑲝𑱾𑲊𑲱𑱾𑲲𑲰𑲊𑲝𑲳𑲰𑲌𑲢𑱱 𑲉𑲳𑱽 𑱲𑱾𑲱𑲋𑲅𑲍𑲚𑲲𑱽𑲱 𑲁𑲎𑲰𑱽𑱴𑲊𑲳 𑲅𑲱𑲰𑱻𑲱𑲂𑱺𑲥𑲊𑲊𑱻𑲩𑲰𑱽𑲚𑲊𑲰𑱾𑱽𑲁𑲲𑱳𑲳𑲌𑲥𑲀𑲮𑲩𑲅𑱲𑲱𑲰𑲊𑲩𑱺𑲳 𑲍𑲡 𑲏𑲀𑲮𑲱𑲅𑲱𑲌𑲪𑲰𑲁𑲩𑱽𑲚𑲳𑱱 𑲏𑲌𑲚𑲰𑲀𑲮𑲱𑲌𑲢 𑱺𑲲𑲰𑲊𑲩𑲌𑱺𑲍𑲎𑲍𑲪𑲳𑲰𑲎 𑱾𑲪𑱴𑲮𑲲𑲌𑲩𑲁𑲰𑱽𑲳𑲰𑲊𑲡𑲎𑱺𑲰 𑲂 𑱾𑲲𑲌𑲞𑲅𑲊𑲬𑲳𑱽𑲰𑲀𑲮𑲱𑱾𑲪𑲅𑲊𑲬𑱺𑲰 𑲅𑲱𑱺𑲊𑲜𑲱𑱽𑲱𑲊𑲩𑲲𑲰𑲎𑱺𑲴𑲊𑱽𑱴𑲅𑲰𑱲𑲬𑲎𑲊𑲡𑲩𑱲𑲲𑲰𑱺𑲰𑱴𑲰𑲊𑱾𑲪𑲰𑲍𑲰𑱼𑱺𑲋𑲳𑲌𑲲 𑱲𑱽𑲩𑲰𑲌𑱺𑲍𑲎𑲍𑲪𑲰𑱽𑲱 𑲍𑲊𑲥𑲰𑲋𑲵𑱲𑲰𑲊𑲀𑲮𑲲𑲰𑲌𑲱𑱺𑲰𑲎 𑲍𑲛𑲱𑱺𑲰 𑲏𑲀𑲮𑲲𑲰𑲅𑱽𑱱 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲵 𑱾𑲪𑲳𑱲𑲬𑲁𑲰𑱽𑲰𑲎 𑱲𑲲𑲍𑲲𑲁𑲰𑱽𑲱 𑲂 𑱲𑲬𑲱𑱾𑱽𑲚𑲱 𑲍𑲡𑱱 𑲏𑲌𑲚𑲰𑲴 𑲂 𑲁𑲊𑲲𑱺𑲒𑱽𑲩𑲰𑲍𑲎𑲍𑲪𑲰𑱽𑲱 𑲅𑲱𑱴𑲋𑲱𑱺𑲰𑲋𑲵𑱲𑲰𑲊𑲰𑲀𑲮𑲊𑱽𑲰𑲋𑲵𑱲𑲪𑲱𑱺𑲰𑱽𑲱 𑲊𑱺𑲝𑲀𑲮𑱼𑲪𑲵𑱲𑲊𑲳𑱽 𑱴𑲪𑲱𑲎𑲱𑲰𑱺𑲰𑱽𑲱 𑲁𑲰𑲊𑲔𑲵 𑲌𑲴𑱼𑲮𑲉𑱽𑲚𑲩𑲴 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲍𑲩 𑱾𑲲𑲊𑱺𑲴 𑱴𑲂𑲣𑱽𑲚𑲱 𑲍𑲡𑱱 𑱼𑲳𑲅𑱽𑲰𑱴𑲉𑱲𑲬𑱴𑱽𑲜𑲮𑲊𑲥𑲰𑲍𑲲𑲊𑱴𑲊𑲲𑱼𑱲𑲱𑱽𑲝𑲊𑲁𑲎𑲴𑲊𑱴𑲰𑲌𑲢𑲰𑲊𑲜𑲮𑱲𑲰𑲉𑲱𑱲𑲰 𑱴𑱴𑱽𑱺𑲋𑲰𑱺𑲞𑲲𑲌𑲞𑱾𑱺𑲚𑱼𑲰𑲁𑲰𑱽𑲩𑲀𑲮𑲱𑱾𑲪𑲋𑲁𑲠𑲉𑱽𑲚𑲱 𑲍𑲡𑱱 𑲍𑲊𑲥𑲳 𑲂 𑲌𑲰𑱲𑲩𑱴𑱽𑲰𑲎 𑲌𑲲𑱼𑲜𑲮𑲴𑱼𑱽𑲵 𑲊𑲰𑲄𑲰𑱽𑲵 𑱾𑲲𑲊𑲍𑲒𑲪𑲱𑱺𑲩 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲍𑲩 𑱾𑲲𑲊𑱺𑲴 𑱴𑲂𑲣𑱽𑲚𑲱 𑲍𑲡𑱱 𑲏𑱽𑲳𑱽𑲳𑲰𑲅𑲵𑲊𑲲𑲰𑱾𑲳𑱽 𑲅𑲩𑲲𑲰𑲎𑲳𑱽 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲴 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲁𑲲𑱾𑱽𑲱𑲰𑲉𑱺𑲳 𑲍𑲡𑱱𑱱

𑲍𑲁𑱽𑱽𑲚𑲊𑱾𑲪𑲅𑲳𑲌𑲱𑱺𑲌𑲢 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲴 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲁𑱱 𑲏𑱻 𑲅𑲱𑲌𑲥𑲰𑲁𑲱𑱺𑲪𑲴 𑱽𑲰𑲁 𑱼𑲰𑲊𑱲𑲰𑲂𑲰𑲊𑲩𑲴 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲍𑲩 𑲌𑲪𑲱𑲉𑲵 𑱺𑲳𑲄𑲌𑲢𑲰𑲍𑲎𑲁𑲰𑱽𑲴 𑱼𑲮𑲊𑱽𑲱𑱺𑲋𑲳 𑱽𑲱𑲅𑲱𑲌𑲚𑲴𑱼𑲮𑲴𑲁𑲲𑱳𑲎 𑱾𑲪𑱾𑱺𑱺𑲱 𑲍𑲡𑱱 𑱺𑲵 𑱺𑱻𑲰 𑱾𑲪𑱾𑱺𑲱𑱺𑲵 𑱼𑲪𑲱𑲌𑲚𑲥𑲰 𑲌𑲲𑲀𑲮𑲰𑱵𑲔𑲴 𑱽𑲰𑲁 𑱺𑲲𑲌𑲱𑱺𑱲𑲰𑲉𑲱𑱲𑲴 𑱼𑲳𑲅𑱾𑲲𑱺𑲪𑲴 𑱼𑱲𑲬𑲱𑱽𑲳𑱽 𑱲𑲊𑱺𑲋𑲳𑱽 𑱾𑲊𑲱𑱴𑲪𑲱𑲎𑲩𑲴𑱺𑲛𑲰𑱾𑲉𑱺𑲱 𑲍𑲡𑱱 𑲏𑲲𑱺𑲛𑲰𑱾𑲩 𑲂 𑱴𑱴𑱽𑱺𑲋𑲍𑲛𑲴 𑲊𑲰𑲄𑲰𑱽𑲵 𑲌𑲲𑱼𑲜𑲮𑲴𑱼𑱽𑲵 𑱺𑲵 𑲂 𑲁𑲎𑲰𑱽𑲚𑲵 𑲄𑱽𑱲𑲰𑲉𑲵 𑱴𑲰𑱻𑲰𑲀𑲮𑲱𑲊𑲀𑲮𑲩𑲀𑲮𑲰𑲌𑱺—

𑲌𑲰𑲍𑲚𑲪𑲰𑱽𑲱 𑲉𑲰𑱽𑲱 𑱾𑲪𑲂𑲋𑱽𑲚𑲱 𑲁𑱽𑲲𑲌𑲩𑲋𑲴𑱲𑲳
𑲍𑲵𑱳𑲩𑲰 𑲋𑲱𑱾𑲱𑲌𑲢 𑱴𑱽𑱽𑲰𑱾𑲱 𑲂 𑱼𑲮𑲰𑱺𑲲𑱺𑱽𑲚𑲪𑲁𑱱
𑲉𑲳 𑲌𑲱𑲋𑲞𑲉𑲴𑱴 𑱾𑲪𑲱𑱻𑲲 𑲋𑲰𑲴𑱲𑲱𑱲 𑲏𑱾𑲪𑲁𑲳𑲉𑲰𑲎
𑱺𑲳𑲌𑲥𑲳𑲌𑲲 𑲌𑲱𑱲𑲬𑲱𑱺𑲲 𑱾𑲲𑲊𑲰 𑲀𑲎𑲲𑱲𑲋𑲞𑱲𑲴𑱺𑲩𑲎𑱱𑱱༡𑱱𑱱

𑱲𑲱𑲵 𑱺𑲲𑲰 𑲄𑱽𑲍𑲩 𑲏𑱽𑲲𑲅𑲊𑲚𑱽𑱺𑲰𑲵 𑱲𑲊𑲴𑱺𑲱
𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲁𑲰𑱴𑱺𑲲 𑲍𑲲𑲌𑲱𑱲𑲬𑲱𑱺𑲲 𑲌𑲱𑲌𑲩𑱽𑲰𑲊𑲛𑲁𑱱
𑱾𑲊𑲱𑱾𑲰𑲂𑱽𑲰𑲊𑲛 𑲀𑲎𑲲𑱼𑲰𑲊𑱲 𑲏𑱴𑲪𑲉𑲰𑱽𑲳
𑲏𑱽𑲩𑲰𑲵𑲌𑲢 𑲍𑱺𑲚𑲥𑱽𑲉𑲲𑱺𑲰𑱽𑲁𑲪𑲱𑱺𑲳 𑲅𑲱𑱽𑲳𑱺𑲲𑲁𑱱𑱱༢𑱱𑱱

𑲋𑲴𑱲𑲴𑱺𑲚𑲊𑲳𑲌𑲲 𑲂𑱺𑲲𑲍𑱺𑲩𑱾𑱻𑲳 𑲅𑲱𑱼𑲮𑲱𑲄𑲙𑲴
𑲎𑲳𑱺𑲲𑱾𑲪𑱺𑲱𑲰𑱺𑲩𑱲𑲲𑲌𑲋𑲴 𑲉𑱻 𑲍𑲵𑲀𑲮𑲅𑱽𑲚𑲱𑱱
𑲉𑱻 𑲂𑲰𑱽𑲱𑲊𑲴𑱼𑲮𑱲𑲬𑲉𑲲 𑲍𑲵𑲍𑲛𑲱𑱺𑲲 𑲌𑲱𑲰𑱺𑲱𑲀𑲮𑲰𑲅𑲎
𑱺𑲍𑲡𑲱𑱽𑲥𑲱𑱼𑲮𑲱𑲄𑲙 𑱲𑲱𑲁𑱻𑲴 𑲋𑲱𑱾𑲱𑲌𑲰𑲍𑲚𑲪𑲁𑲰𑱺𑲪𑲳𑱱𑱱༣𑱱𑱱

𑱽𑲳𑱺𑲍𑲩 𑲏𑲰𑲂𑲊𑲱𑲉 𑲏𑲲𑱺𑲚𑲊𑲱 𑲅𑲰 𑱺𑲪𑲱𑲋𑲴𑱲𑲳
𑲍𑲊𑲥𑲳𑲌𑲲 𑱼𑲳𑲅𑲁𑱽𑲲𑲄𑲳𑲌𑲥𑲉𑲁𑲳𑲅 𑲄𑲳𑲌𑲛𑲎𑱱
𑱽𑲰𑲁𑲰𑱾𑲱 𑱺𑲳𑲌 𑲋𑲱𑱾𑲱𑱽𑲰𑲵 𑱽 𑲎𑲱 𑲅𑲱𑱺𑲛 𑲉𑲲𑲰𑲉𑲵
𑲉𑱺𑲪𑲳𑲌𑲲 𑲌𑲱𑱲𑲬𑲱𑱺𑲲 𑱾𑲲𑲊𑲰 𑲀𑲎𑲲𑱲𑲋𑲞𑱲𑲴𑱺𑲩𑲎𑱱𑱱༤𑱱𑱱

𑲍𑲴 𑲂𑲱𑱺𑲚𑱼𑲮𑲰𑲊 𑲄𑱴𑱺𑲰𑲵 𑲅𑲱𑲅𑲱𑱼𑲮𑲰 𑲅𑲱𑲂𑲱𑱺𑲪𑲰
𑲏𑲳𑱲𑱲𑲬𑱽𑲳𑱽 𑲏𑲉𑲲 𑲄𑲰𑱽𑱺𑲱 𑲌𑲲𑱼𑲜𑲮𑲍𑱺𑲚𑲥𑲎𑱱
𑲏𑱼𑲪𑲱𑲌𑲩𑲊𑲲𑲰𑱾𑲊𑲎𑲱𑱺𑲍𑲩 𑱴𑱺𑲱𑲵 𑲂 𑲅𑲳𑱺𑲚𑲱
𑱲𑲱𑲵 𑲅𑲰 𑱾𑲲𑱽𑲴𑱻 𑲋𑲱𑱾𑲱𑱽𑲴𑱲𑲬𑲊𑱼𑲪𑲱𑲌𑲩𑲊𑲲𑲰𑱾𑲰𑲁𑱱𑱱༥𑱱𑱱

𑲏𑲱𑱺𑲩𑲲𑱲𑲚𑲥𑲰 𑲍 𑱼𑲳𑲅𑱾𑲲𑱺𑲪𑲴 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥𑲵 𑱼𑲱𑲅𑲩𑲳𑲰𑲎 𑱲𑲲𑲍𑲲𑲁𑲳𑲰𑲊𑲀𑲮𑲩𑲊𑲢𑲩 𑱺𑱺𑲪𑲳𑲰𑲅𑲰𑱽𑲚𑲊𑲜𑱼𑲮𑲳𑱱 𑱺𑱺𑲪 𑱼𑲮𑲰𑱺𑲪𑲩𑲌𑲢 𑲂𑲳𑱺𑲱𑲰𑲅𑲊𑲔𑲰𑲌𑲢 𑲍𑲛𑲰𑱾𑲱𑱺𑲰 𑲏𑲀𑲮𑲲𑲰𑲅𑱽𑱱 𑱾𑲊𑲱𑲌𑲳𑲌𑲰𑲎 𑲌𑲰𑱲𑲩𑲰𑲎 𑲌𑲲𑱼𑲜𑲮𑲴𑱼𑱽𑱾𑲪𑲁𑲲𑱳𑲰𑲎 𑱾𑲪𑱲𑲪𑲰𑲁𑱽𑲚𑲎𑱱𑱱

𑲏𑱻 𑲀𑲴𑱼𑲮𑲱𑲍𑱺𑲚𑲥 𑲏𑲲𑲊𑱴𑲍𑲰𑲊𑲂𑱽𑲜𑱽𑲁𑲉𑲵 𑲋𑲱𑱾𑲱𑱿𑲋𑱲𑲁𑲰𑱼𑲰𑲉 𑱼𑲱𑲅𑲩𑲰𑲊𑲬𑲍𑲲𑲅𑲊𑲝𑱺𑲱𑲊𑱲𑲵 𑲍𑲁𑱽𑲚𑲰𑱽𑲡𑱽𑲱𑲊𑱺𑲝𑱾𑲪𑱺𑲩𑲲𑱾𑲚𑲵 𑲅𑲱𑲌𑲥𑲰𑲁𑲱𑱺𑲪𑲁𑲰𑲂𑲰𑲊𑲩𑲁𑲳𑲅𑲁𑲰𑲎-𑱲𑱺𑲁𑲰𑲵 𑲁𑲳 𑲀𑲮𑲴 𑲏𑲲𑱾𑲰𑱼𑲮𑲩𑲰𑲉 𑲋𑲱𑱾𑲱𑲵 𑲌𑲱𑱲𑲬𑲰𑱾𑲉𑲍𑲱𑱱 𑲀𑲪𑲰𑲎𑲡𑲱𑲰𑱳𑲊𑲴𑲌𑲚𑲱𑲰𑱾𑲲𑲌𑲒𑲊𑲍𑲰𑲊𑲱𑲵 𑲏𑱵𑲔𑲋𑲱𑱾𑲱𑲵 𑲅𑱵𑲔𑲋𑲱𑱾𑲱𑲵 𑲁𑱴𑱼𑲮𑲋𑲱𑱾𑲱𑲵 𑲁𑱵𑲔𑲋𑲩𑲋𑲱𑱾𑲱𑲵 𑲏𑱵𑲔𑲲𑲋𑲱𑲰𑲉𑲋𑲱𑱾𑲱𑲵 𑲌𑱲𑲰𑲊𑲱𑲋𑲱𑱾𑲱𑲵 𑲀𑲪𑲎𑲡𑲅𑲋𑲱𑲋𑲱𑱾𑲱𑲵 𑱾𑲰𑲊𑲲𑲌𑲩𑲋𑲱𑱾𑲱𑲵 𑱼𑲪𑲰𑲅𑲱𑱼𑲋𑲱𑱾𑲱𑲵 𑱲𑲱𑲊𑲰𑱺𑲋𑲱𑱾𑲱𑲵 𑱼𑲰𑱲𑲬𑲱𑱽𑲩𑲋𑲱𑱾𑲱𑲵 𑲏𑲲𑱴𑲪𑲋𑲱𑱾𑲱𑲵 𑲍𑲵𑱳𑲩𑲰𑲋𑲱𑱾𑲱𑲵 𑲏𑱽𑲲𑲋𑲴𑲁𑲋𑲱𑱾𑲱𑲵 𑲏𑲅𑲁𑲲𑲰𑲊𑲜𑲮𑲋𑲱𑱾𑲱𑲵 𑱼𑲊𑱼𑲋𑲱𑱾𑲱𑲵 𑱳𑲰𑲌𑲩𑲋𑲱𑱾𑲱𑲵 𑲂𑲱𑲰𑱽𑲋𑲱𑱾𑲱𑲵 𑲋𑲲𑲰𑱽𑲋𑲱𑱾𑲱𑲵 𑲎𑲲𑲰𑱽𑲋𑲱𑱾𑲱𑲵 𑲁𑱼𑲮𑲩𑲰𑱲𑲬𑲊𑲅𑲱𑲍𑲚𑲊𑲋𑲱𑱾𑲱𑲵 𑱾𑲲𑲌𑲞𑲋𑲱𑱾𑲱𑲵 𑱼𑲳𑲅𑲋𑲱𑱾𑲱𑲵 𑱽𑲰𑱴𑲋𑲱𑱾𑲱𑲵 𑲉𑱲𑲬𑲋𑲱𑱾𑲱𑲵 𑱴𑱽𑲜𑲮𑲊𑲥𑲋𑲱𑱾𑲱𑲵 𑱲𑲱𑱽𑲝𑲊𑲋𑲱𑱾𑲱𑲵 𑲁𑲎𑲴𑲊𑱴𑲋𑲱𑱾𑲱𑲵 𑲏𑲍𑲲𑲊𑲋𑲱𑱾𑲱𑲵 𑱴𑲊𑲲𑱼𑲋𑲱𑱾𑲱𑲵 𑲁𑲪𑲱𑱴𑲂𑱲𑲪𑲋𑲱𑱾𑲱𑲵 𑲅𑲰𑲉𑲍𑲊𑲲𑱺𑲋𑲱𑱾𑲱𑲵 𑲀𑲮𑲰𑲴𑲁𑱼𑲳𑲅𑲋𑲱𑱾𑲱𑲵 𑲏𑱽𑲚𑲊𑲱𑲰𑱲𑲬𑱼𑲳𑲅𑲋𑲱𑱾𑲱𑲵 𑲏𑲲𑱺𑲚𑲊𑱲𑲲𑲊𑲲𑱼𑲥𑲱𑲰𑱾𑲋𑲱𑱾𑲱𑲵 𑲏𑱾𑲊𑱴𑲴𑱼𑲰𑱽𑲱𑲰𑲋𑲱𑱾𑲱𑲵 𑱾𑲲𑲰𑲊𑲥𑲅𑲱𑱼𑲳𑲎𑲋𑲱𑱾𑲱𑲵 𑲏𑲲𑱺𑲒𑲬𑲳𑱾𑲋𑲱𑱾𑲱𑲵 𑱽𑲱𑱲𑲬𑲳𑱾𑲋𑲱𑱾𑲱𑲵 𑲅𑲱𑱲𑲬𑲳𑱾𑲋𑲱𑱾𑲱𑲵 𑱾𑲪𑱲𑲬𑲳𑱾𑲋𑲱𑱾𑲱𑲵 𑲍𑲰𑱴𑲊𑲋𑲱𑱾𑲱𑲵 𑲅𑲄𑲪𑲋𑲱𑱾𑲱𑲵 𑲋𑲳𑱳𑱾𑲪𑱺𑲱𑲋𑲳𑱳𑲋𑲱𑱾𑲱𑲵 𑲏𑱽𑲲𑱼𑲪𑲲𑱺𑲋𑲱𑱾𑲱𑲵 𑲌𑲰𑲍𑲚𑲪𑲰𑲅𑲊𑲚𑲰𑲵 𑱴𑱽𑱽𑲰𑲅𑲊𑲚𑲋𑲱𑱾𑲱𑲵 𑲏𑲲𑱺𑲒𑲬𑲳𑱾𑲰𑲅𑲊𑲚𑲋𑲱𑱾𑲱𑲵 𑱽𑲱𑱲𑲬𑲳𑱾𑲰𑲅𑲊𑲚𑲋𑲱𑱾𑲱𑲵 𑱾𑲰𑱼𑲋𑲱𑱳𑲱𑱺𑲋𑲱𑱾𑲱𑲵 𑱼𑲥𑲱𑲊𑲲𑱺𑲚𑲊𑱾𑱼𑲍𑲵𑱼𑲮𑲱𑲋𑲱𑱾𑲱𑲵 𑲉𑲰𑲅𑱼𑲜𑲌𑲴𑱺𑲚𑲊𑱾𑱼𑲍𑲵𑱼𑲮𑲱𑲋𑲱𑱾𑲱𑲵 𑲁𑱼𑲮𑲩𑲰𑲎𑲰𑲊𑲱𑱽𑲱𑲰𑲋𑲱𑱾𑲱𑲵 𑲍𑲊𑲥𑲊𑲲𑱺𑲍𑲵𑱴𑲪𑲎𑱽𑲱𑲰𑲋𑲱𑱾𑲱𑲵 𑲅𑲱𑱼𑲩𑲰𑱽𑲲𑲋𑲴𑲁𑲰𑲅𑲱𑲁𑲱𑲌𑲪𑲱𑱺𑲋𑲱𑱾𑲱𑲵 𑲊𑲱𑲌𑲱𑱺𑱾𑲍𑲚𑱾𑲚𑲰𑲵 𑲊𑲴𑲂𑲁𑲰𑱽𑲰𑲵 𑱼𑲮𑲊𑱽𑲱𑲰𑱾𑲪𑲳𑱲𑲬𑲱𑱽𑲱𑲰𑲋𑲱𑱾𑲱𑲵 𑱴𑱴𑱽𑱾𑲪𑲳𑱲𑲬𑲱𑱽𑲱𑲰𑲋𑲱𑱾𑲱𑲵 𑲍𑲊𑲥𑲰𑲴𑲌𑱼𑲮𑲱𑱽𑲱𑲌𑲩𑱽𑲜𑲰𑲵 𑲍𑲊𑲥𑲍𑲰𑲊𑲍𑲵𑱴𑲪𑲎𑱽𑲱𑲰𑲵 𑲍𑲊𑲥𑲀𑲮𑲲𑲰𑱺𑲊𑲲𑱺𑱴𑲪𑲎𑱽𑲱𑲰𑲁𑱱 𑲏𑲰𑲍𑲰𑲵 𑲀𑲮𑲴 𑲏𑲲𑱾𑲰𑱼𑲮𑲩𑲰𑲉 𑲂𑱺𑲲𑲌𑲬𑲌𑲚𑲱𑲰𑲋𑲱𑱾𑲱𑲰𑱽𑲰𑲵 𑱲𑱺𑲁𑲰𑲵 𑱺𑲥𑲵 𑲌𑲱𑲌𑲩𑲰𑱾𑲉𑲱𑲌𑲩𑲍𑲱?

𑲏𑱻 𑲅𑲱𑲌𑲥𑲰𑲁𑲱𑱺𑲪𑲴 𑱼𑲰𑲊𑱲𑲰𑲂𑲰𑲊𑲩𑲴 𑲅𑲱𑲍𑲡𑲱𑱺𑲎 𑱾𑲪𑲎𑲍𑲱𑱺𑲅𑱼𑱽𑲴 𑱽𑲱𑲎𑱺𑲁𑲰𑱽𑲁𑱼𑱼𑲊𑲞 𑲏𑲱𑲁𑲰𑲵 𑱴𑲰𑱻𑲰𑲁𑲀𑲮𑲰𑲌𑱺—

𑲏𑲰𑲌𑲢𑲊𑲩𑲵 𑲌𑲲𑱼𑲜𑲮𑲍𑱺𑲚𑲥𑲍𑲩 𑲋𑲴𑱲𑲳 𑲋𑲴𑱲𑲰𑱽𑲲𑲅𑲊𑲚𑲱𑱽𑲴𑱱
𑲌𑲱𑱲𑲬𑲱𑱺𑲎 𑲍𑲊𑲥𑲌𑲰𑲍𑲚𑲪𑲳𑲌𑲲 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲁𑲲𑱾𑲰𑱴𑱺𑲎𑱱𑱱༦𑱱𑱱

𑲉𑲳𑲌𑲰𑲁𑲎𑲵 𑱽𑲰𑲁𑱼𑲮𑲳𑲉𑲵 𑲋𑲱𑱾𑲱𑲰𑱽𑲰𑲵 𑱽 𑱾𑲪𑲄𑲰𑱽𑲁𑲱𑱱
𑱺𑱺𑲪𑲳𑲰𑲌 𑲌𑲱𑱲𑲬𑲱𑱺𑲎 𑲍𑱽𑲚𑲴 𑲋𑲱𑱾𑲱𑲌𑲰𑲋𑲰𑲁𑲲𑱾𑲰𑱴𑱺𑲎𑱱𑱱༧𑱱𑱱

𑲅𑱲𑲚𑲪𑲵 𑲂𑲰𑲍𑲩 𑱽 𑱾𑲌𑲩𑲰𑲁𑲱 𑲁𑲲𑲰𑲊𑲜𑲮𑲰𑱽𑲵 𑱺𑲍𑲩 𑱽𑲳𑲰𑲅 𑲂𑱱
𑲌𑲱𑲌𑲩𑲉𑲱𑲌𑲩𑲳 𑱲𑱻𑲵 𑲎𑲩𑲳𑱽𑲵 𑲋𑲱𑱾𑲱𑱾𑲪𑲄𑲙𑲰𑲉 𑱾𑲰𑲊𑱴𑲁𑱱𑱱༨𑱱𑱱

𑱼𑲳𑲅𑱼𑲳𑲅𑲴 𑲎𑲩𑱺𑲱𑱼𑲳𑲅𑲎 𑲍𑲊𑲥𑱼𑲳𑲅𑲴𑱺𑲚𑲁𑲴 𑲅𑲱𑲀𑲮𑲲𑲎𑱱
𑲏𑲍𑲁𑲌𑲢 𑲅𑲱𑲌𑲱𑲌𑲚𑲌𑲢 𑲋𑲴𑱲𑲳𑲌𑲥𑱾𑲪𑱺𑲱𑱾𑲲𑱼𑲔𑲋𑲎𑱱𑱱༩𑱱𑱱

𑲏𑲍𑲩𑲳𑲰𑲅 𑱺𑲥𑱽𑲲𑲀𑲮𑲰𑲅𑲳𑱽 𑱾𑲪𑲄𑲙𑲴𑱾𑲰𑲉𑲳 𑲅𑲱𑲌𑲳𑲌𑱺𑲎𑱱
