𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄦𑄳𑄢𑄨𑄘𑄧𑄠𑄧𑄥𑄫𑄖𑄳𑄢𑄧𑄟𑄴𑅁

[𑄥𑄧𑄁𑄇𑄴𑄥𑄨𑄛𑄴𑄖𑄧𑄟𑄖𑄳𑄢𑄨𑄇]

𑅂 𑄚𑄧𑄟𑄧𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄎𑄴𑄐𑄠𑄧 𑅂

𑅂 𑄃𑄮𑄀 𑄚𑄧𑄟𑄮 𑄞𑄧𑄉𑄧𑄤𑄧𑄖𑄳𑄠𑄰 𑄃𑄢𑄳𑄠𑄧𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄠𑄰 𑅂

𑄃𑄢𑄳𑄠𑄤𑄧𑄣𑄮𑄇𑄨𑄖𑄬𑄥𑄳𑄤𑄧𑄢𑄧𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄮 𑄉𑄧𑄟𑄴𑄞𑄩𑄢𑄠𑄁 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄠𑄁 𑄌𑄧𑄢𑄳𑄠𑄁 𑄌𑄧𑄢𑄧𑄟𑄕𑄮 𑄤𑄳𑄠𑄧𑄤𑄧𑄣𑄮𑄇𑄧𑄠𑄧𑄖𑄨 𑄥𑄴𑄟𑄧𑅁 𑄛𑄧𑄐𑄴𑄌𑄧 𑄥𑄴𑄇𑄧𑄚𑄴𑄙𑄂, 𑄖𑄁𑄥𑄴𑄌𑄧 𑄥𑄳𑄤𑄧𑄞𑄤𑄧𑄥𑄫𑄚𑄳𑄠𑄚𑄴 𑄛𑄧𑄥𑄳𑄠𑄧𑄖𑄨 𑄥𑄴𑄟𑄧𑅂

𑄃𑄨𑄦𑄧 𑄥𑄢𑄨𑄛𑄪𑄖𑄳𑄢𑄧 𑄢𑄫𑄛𑄧𑄁 𑄥𑄫𑄚𑄳𑄠𑄧𑄖, 𑄥𑄫𑄚𑄳𑄠𑄧𑄖𑄰𑄤𑄧 𑄢𑄫𑄛𑄧𑄟𑄴𑅁 𑄢𑄫𑄛𑄚𑄴𑄧 𑄛𑄳𑄢𑄨𑄗𑄧𑄇𑄴 𑄥𑄫𑄚𑄳𑄠𑄧𑄖, 𑄥𑄫𑄚𑄳𑄠𑄧𑄖𑄠 𑄚𑄧 𑄛𑄳𑄢𑄨𑄗𑄧𑄉𑄴 𑄢𑄫𑄛𑄧𑄟𑄴𑅁 𑄡𑄧𑄘𑄳𑄢𑄫𑄛𑄧𑄁 𑄥 𑄥𑄫𑄚𑄳𑄠𑄧𑄖, 𑄡 𑄥𑄫𑄚𑄳𑄠𑄧𑄖 𑄖𑄧𑄘𑄳𑄢𑄫𑄛𑄧𑄟𑄴𑅂

𑄃𑄬𑄤𑄧𑄟𑄬𑄤𑄧 𑄤𑄬𑄘𑄧𑄚𑄥𑄧𑄁𑄎𑄴𑄐𑄥𑄧𑄁𑄥𑄴𑄇𑄢𑄧𑄤𑄨𑄎𑄴𑄐𑄚𑄚𑄨𑅂

𑄃𑄨𑄦𑄧𑄁 𑄥𑄢𑄨𑄛𑄪𑄖𑄳𑄢𑄧 𑄥𑄧𑄢𑄳𑄤𑄧𑄙𑄧𑄢𑄴𑄟𑄂 𑄥𑄫𑄚𑄳𑄠𑄧𑄖𑄣𑄧𑄇𑄴𑄥𑄧𑄕 𑄃𑄧𑄚𑄪𑄖𑄴𑄛𑄧𑄚𑄳𑄚 𑄃𑄧𑄚𑄨𑄢𑄪𑄘𑄴𑄙 𑄃𑄧𑄟𑄧𑄣 𑄚𑄧 𑄤𑄨𑄟𑄧𑄣 𑄚𑄮𑄚 𑄚𑄧 𑄛𑄧𑄢𑄨𑄛𑄫𑄢𑄴𑄕𑄂𑅁 𑄖𑄧𑄥𑄴𑄟𑄌𑄴𑄍𑄢𑄨𑄛𑄪𑄖𑄳𑄢𑄧 𑄥𑄫𑄚𑄳𑄠𑄧𑄖𑄠𑄁 𑄚𑄧 𑄢𑄫𑄛𑄧𑄟𑄴, 𑄚𑄧 𑄤𑄬𑄘𑄧𑄚, 𑄚𑄧 𑄥𑄧𑄁𑄎𑄴𑄐, 𑄚𑄧 𑄥𑄧𑄁𑄥𑄴𑄇𑄢𑄂, 𑄚𑄧 𑄤𑄨𑄎𑄴𑄐𑄚𑄚𑄨𑅁 𑄚𑄧 𑄌𑄧𑄇𑄴𑄥𑄪𑄂𑄥𑄳𑄢𑄮𑄖𑄳𑄢𑄧𑄊𑄳𑄢𑄕𑄧𑄎𑄨𑄦𑄳𑄤𑄇𑄠𑄧𑄟𑄧𑄚𑄁𑄥𑄨, 𑄚𑄧 𑄢𑄫𑄛𑄧𑄥𑄧𑄝𑄴𑄘𑄧𑄉𑄧𑄚𑄴𑄙𑄧𑄢𑄧𑄥𑄧𑄥𑄴𑄛𑄳𑄢𑄧𑄥𑄴𑄑𑄧𑄤𑄳𑄠𑄧𑄙𑄧𑄢𑄴𑄟𑄂𑅁 𑄚𑄧 𑄌𑄧𑄇𑄴𑄥𑄪𑄢𑄴𑄙𑄖𑄪𑄢𑄳𑄠𑄤𑄧𑄚𑄴𑄧 𑄟𑄧𑄚𑄮𑄙𑄖𑄪𑄂𑅂

𑄚𑄧 𑄤𑄨𑄘𑄳𑄠 𑄚𑄤𑄨𑄘𑄳𑄠 𑄚𑄧 𑄤𑄨𑄘𑄳𑄠𑄇𑄴𑄥𑄧𑄠𑄮 𑄚𑄤𑄨𑄘𑄳𑄠𑄇𑄴𑄥𑄧𑄠𑄮 𑄡𑄤𑄧𑄚𑄴𑄧 𑄎𑄧𑄢𑄟𑄧𑄢𑄧𑄕𑄧𑄁 𑄚𑄧 𑄎𑄧𑄢𑄟𑄧𑄢𑄧𑄕𑄧𑄇𑄴𑄥𑄧𑄠𑄮 𑄚𑄧 𑄘𑄪𑄂𑄈𑄧𑄥𑄧𑄟𑄪𑄘𑄧𑄠𑄧𑄚𑄨𑄢𑄮𑄙𑄧𑄟𑄢𑄴𑄉 𑄚𑄧 𑄎𑄴𑄐𑄚𑄧𑄁 𑄚𑄧 𑄛𑄳𑄢𑄛𑄴𑄖𑄨𑄖𑄳𑄤𑄧𑄟𑄴𑅂

𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄧𑄥𑄳𑄠𑄧 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄟𑄥𑄳𑄢𑄨𑄖𑄳𑄠𑄧 𑄤𑄨𑄦𑄧𑄢𑄧𑄖𑄨 𑄌𑄨𑄖𑄴𑄖𑄤𑄧𑄢𑄧𑄕𑄧𑄂𑅁 𑄌𑄨𑄖𑄴𑄖𑄤𑄧𑄢𑄧𑄕𑄧𑄚𑄥𑄴𑄖𑄨𑄖𑄳𑄤𑄘𑄧𑄖𑄳𑄢𑄧𑄥𑄴𑄖𑄮 𑄤𑄨𑄛𑄧𑄢𑄳𑄠𑄥𑄖𑄨𑄇𑄳𑄢𑄚𑄴𑄖𑄮 𑄚𑄨𑄥𑄴𑄒𑄧𑄚𑄨𑄢𑄳𑄤𑄕𑄧𑄂𑅁 𑄖𑄳𑄢𑄳𑄠𑄧𑄙𑄳𑄤𑄧𑄤𑄳𑄠𑄧𑄤𑄧𑄥𑄴𑄗𑄨𑄖𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄝𑄪𑄘𑄴𑄙𑄂 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄟𑄥𑄳𑄢𑄨𑄖𑄳𑄠𑄧 𑄃𑄧𑄚𑄪𑄖𑄴𑄧𑄢𑄁 𑄥𑄧𑄟𑄳𑄠𑄧𑄇𑄴𑄥𑄧𑄁𑄝𑄮𑄙𑄨𑄟𑄧𑄞𑄨𑄥𑄧𑄁𑄝𑄪𑄘𑄴𑄙𑄂𑅂

𑄖𑄧𑄥𑄴𑄟𑄎𑄴𑄎𑄴𑄐𑄖𑄧𑄤𑄳𑄠𑄧𑄂 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄟𑄧𑄦𑄟𑄧𑄚𑄴𑄖𑄳𑄢𑄮 𑄟𑄧𑄦𑄤𑄨𑄘𑄳𑄠𑄟𑄧𑄚𑄴𑄖𑄳𑄢𑄮'𑄚𑄪𑄖𑄴𑄧𑄢𑄧𑄟𑄧𑄚𑄴𑄖𑄳𑄢𑄮'𑄥𑄧𑄟𑄧𑄥𑄧𑄟𑄧𑄟𑄧𑄚𑄴𑄖𑄳𑄢𑄧𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄘𑄪𑄂𑄈𑄧𑄛𑄳𑄢𑄧𑄥𑄧𑄟𑄧𑄚𑄧𑄂 𑄥𑄧𑄖𑄳𑄠𑄧𑄟𑄧𑄟𑄨𑄗𑄳𑄠𑄧𑄖𑄳𑄤𑄖𑄴 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄛𑄢𑄧𑄟𑄨𑄖𑄠𑄟𑄪𑄇𑄴𑄖𑄮 𑄟𑄧𑄚𑄴𑄖𑄳𑄢𑄧𑄂𑅁
