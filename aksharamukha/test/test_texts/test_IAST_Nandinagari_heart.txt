𑦡𑧈𑧠𑧇𑦽𑧠𑧈𑧒𑧈𑦽𑧠𑧁𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧍𑧕𑦽𑧠𑧈𑧆𑧠

(𑦯) 𑧍𑧞𑧍𑧠𑦮𑧖𑦽𑧂𑧑𑦹𑧟

𑧅𑧑𑧈𑦽𑧓𑧇𑧅𑧑𑧌𑧑𑧇𑧑𑧆𑧠-𑦡𑧈𑧠𑧇𑦽𑧠𑧈𑧒𑧈𑦽𑧠𑧁𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧍𑧕𑦽𑧠𑧈𑧆𑧠

𑧅𑧜𑦸𑧅𑧑𑧌𑧑𑧇𑧑𑧆𑧠-𑧃𑦰𑧠𑧍𑧠-𑧂𑧑-𑦿𑦮𑧜𑧁𑧠-𑧆𑧠𑦳𑧠𑦴𑧜𑦰𑧠-𑦰𑧠𑧍𑧔𑧆𑧠-𑦵𑧚𑧍𑧠-𑧍𑧔-𑦺𑧠𑧈𑧁𑧠-𑧂𑧑-𑧆𑧠𑦿𑧜

𑧁𑧆𑧟 𑧍𑧈𑧠𑧊𑧄𑧔𑦿𑧠𑧀𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧚𑧅𑧠𑧇𑧟

(𑧄𑧔𑦿𑧠𑧀𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧟)

𑦢𑦽𑧠𑧇𑧂𑧒 𑧄𑧔𑦿𑧠𑧀𑧜 𑧅𑦰𑧊𑧑𑧞𑧍𑧠𑦽𑦾𑧑𑦰𑦽𑧜𑧡𑧈𑧠𑧎𑧁𑧠 𑧍𑧆𑧠𑧇𑦮𑧠𑧍𑧞𑧄𑧔𑦿𑧠𑧀𑧜 𑧊𑧒𑦿𑧠𑧇𑧑𑦳𑧈𑦼𑧍𑧞𑧂𑧁𑧠𑧁𑧟 𑧍𑧔𑦰𑦽𑧜 𑧉𑧜𑦮𑧊𑧒𑦿𑧁𑧔𑦽𑧠𑦽𑧈𑧟 𑧂𑧔𑧈𑧔𑧌𑦿𑧆𑧠𑧇𑧍𑧑𑧈𑦾𑧒𑧟 𑧋𑧑𑧍𑧠𑦽𑧑 𑦿𑧚𑧊𑧆𑧁𑧔𑧌𑧠𑧇𑧑𑦼𑧑𑧞 𑧄𑧔𑦿𑧠𑧀𑧜 𑧅𑦰𑧊𑧑𑧁𑧒𑦽𑧒।

𑧁𑧒𑧌𑧠𑧇𑧞𑦿𑧟 𑧍 𑦽𑦾𑧑𑦰𑦽𑧟 𑧂𑧔𑦼𑧠𑧇𑧑𑧁𑧑𑧆𑧠, 𑦠𑧊𑧒𑧂𑧠𑧈𑦼𑧑𑧋𑧟 𑦮𑧔𑧋𑧉𑧆𑧕𑧉𑧑𑧁𑧑𑧆𑧠, 𑦠𑧉𑧞𑦮𑧖𑦽𑧟 𑦮𑧠𑧌𑧑𑧞𑦽𑧠𑧇𑧑, 𑦡𑧉𑧇𑧟 𑧂𑧔𑦼𑧠𑧇𑧁𑧒𑧀𑧑𑧁𑧑𑧁𑧑𑧆𑧠, 𑦳𑧒𑦽𑧠𑧈𑧒𑦽𑧜𑧡𑧁𑧔𑧊𑧠𑧇𑧞𑦵𑧁𑧛𑧟 𑦮𑧔𑧍𑧔𑧆𑧒𑦽𑧜 𑧉𑦮𑧠𑧌𑦼𑧛𑧟, 𑧂𑧠𑧈𑦽𑧒𑧈𑧕𑧂𑧜 𑦰𑧜𑦳𑧈𑧚𑦼, 𑦠𑧂𑧠𑧈𑦽𑧒𑦮𑧕𑧉𑧜 𑦿𑧈𑧠𑧋𑧁𑧚𑧁, 𑦠𑧅𑧒𑧈𑦽𑧒𑧟 𑧋𑧠𑧈𑦿𑧠𑧀𑧑(𑧋𑧔𑦿𑧠𑧀𑧑)𑧀𑧒𑧆𑧔𑦮𑧠𑦽𑧑𑧁𑧑𑧆𑧠, 𑦠𑧁𑧅𑧒𑧅𑧕𑦽𑧟 𑧂𑧠𑧈𑦵𑧠𑦷𑧇𑧑, 𑦠𑧁𑧊𑧆𑧈𑧠𑦿𑧁𑧓𑧇𑧜 𑧄𑧉𑧛𑧟, 𑧋𑧑𑧍𑧠𑦽𑧑 𑧍𑧈𑧠𑧊𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧁𑧑𑧆𑧠, 𑧂𑧒𑦽𑧑 𑧄𑧜𑧀𑧒𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧁𑧑𑧆𑧠, 𑧈𑧑𑦵𑧑 𑦡𑧈𑧠𑧇𑧂𑧔𑦿𑧠𑦰𑧉𑧑𑧁𑧑𑧆𑧠, 𑧍𑧑𑧈𑧠𑦾𑧊𑧑𑧎𑧟 𑧁𑧒𑧈𑧠𑧊𑧑𑦼𑧁𑦰𑧈𑧍𑧞𑧂𑧠𑧈𑧍𑧠𑦾𑧒𑦽𑧑𑧁𑧑𑧆𑧠, 𑦠𑧂𑧠𑧈𑧆𑧚𑧇𑧜 𑦵𑧠𑦷𑧑𑧁𑧚𑧁, 𑦠𑦳𑧒𑧞𑦽𑧠𑧇𑧟 𑧂𑧠𑧈𑦽𑧒𑧅𑧑𑧁𑧚𑧁, 𑧊𑧒𑧋𑧔𑦿𑧠𑧀𑧟 𑧍𑧠𑧊𑧈𑧚𑦼, 𑦡𑧍𑧠𑧊𑦿𑧁𑧓𑧇𑧜 𑦱𑧜𑧌𑧚𑦼, 𑦠𑧍𑧚𑦳𑧁𑦮𑧜 𑧈𑧕𑧂𑧚𑦼, 𑦠𑧂𑧠𑧈𑦽𑧒𑧍𑧆𑧟 𑦮𑧑𑧇𑧚𑧁, 𑦠𑧉𑧒𑧂𑧠𑦽𑧟 𑦮𑧑𑧆𑧛𑧟, 𑦠𑧁𑧔𑧂𑧉𑧒𑧂𑧠𑦽𑧜 𑧈𑧕𑧂𑧛𑧟 𑦠𑧍𑧞𑧍𑧖𑧌𑧠𑦸 𑦡𑧈𑧕𑧂𑧠𑧇𑧛𑧟, 𑧊𑧒𑧂𑧠𑧈𑧆𑧔𑦮𑧠𑦽𑧟 𑧍𑧠𑦮𑧞𑧀𑧚𑧅𑧠𑧇𑧟, 𑧊𑧒𑧍𑧞𑧂𑧠𑧈𑧇𑧔𑦮𑧠𑦽𑧜 𑧀𑧑𑦽𑧔𑧅𑧒𑧟, 𑧍𑧞𑧊𑧖𑦽 𑦡𑧇𑦽𑧁𑧛𑧟, 𑧂𑧠𑧈𑦳𑧠𑦴𑧒𑧁𑧠𑧁𑧜 𑦰𑧠𑧈𑧞𑦾𑧛𑧟, 𑧊𑧒𑧆𑧔𑦮𑧠𑦽𑧟 𑧂𑧈𑧒𑦿𑧑𑦱𑧠𑧁𑧛𑧟, 𑧂𑧈𑧒𑧆𑧔𑦮𑧠𑦽𑧍𑧠𑦽𑧖𑧌𑧠𑦼𑧇𑧑, 𑦬𑦱𑧑𑦿𑧔𑦽𑧠𑦽𑧓𑧈𑧠𑦼𑧟 𑧂𑧈𑧒𑧂𑧕𑧈𑧠𑦼𑧜 𑦵𑧠𑦷𑧑𑧁𑧚𑧁, 𑧂𑧠𑧈𑦽𑧒𑧌𑧠𑦹𑧒𑦽𑧜𑧡𑦽𑧓𑦽𑧑𑧁𑧑𑦰𑦽𑧂𑧠𑧈𑦽𑧠𑧇𑧔𑦽𑧠𑧂𑧁𑧠𑧁𑧑𑧁𑧑𑧞 𑧄𑧔𑦿𑧠𑧀𑧑𑧁𑧑𑧞 𑧅𑦰𑧊𑦽𑧑𑧞 𑦵𑧠𑦷𑧑𑧁𑧚, 𑦠𑧂𑧠𑧈𑦽𑧒𑧌𑧠𑦹𑧒𑦽𑧜 𑧁𑧒𑧈𑧠𑧊𑧑𑦼𑧚, 𑧍𑧠𑦾𑧒𑦽𑧜 𑧅𑧕𑦽𑦮𑧜𑦸𑧠𑧇𑧑𑧆𑧠, 𑧍𑧠𑦾𑧒𑦽𑧟 𑧍𑧈𑧠𑧊𑧍𑦽𑧠𑦽𑧠𑧊𑧑𑧉𑧜𑦮𑧁𑧓𑧇𑧑𑧇𑧑𑧞 𑧅𑧕𑧆𑧝, 𑧍𑧈𑧠𑧊 𑦢𑧆𑧚 𑦽𑦾𑧑𑦰𑦽𑧑𑧁𑧑𑧞 𑧊𑧒𑧋𑧚𑧌𑦽𑧟 𑧍𑧆𑧠𑧇𑦰𑧠 𑦰𑧔𑦼𑧑𑧟।

(𑧀𑧈𑧠𑧆𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧟)

𑧍𑦿𑧠𑧀𑧈𑧠𑧆𑧍𑧠𑦽𑧔 𑦡𑦿𑧝 𑦮𑧉𑧠𑧇𑧑𑦼𑧟, 𑧆𑧀𑧠𑧇𑧚 𑦮𑧉𑧠𑧇𑧑𑦼𑧟, 𑧂𑧈𑧠𑧇𑧊𑧍𑧑𑧁𑧚 𑦮𑧉𑧠𑧇𑧑𑦼𑧟, 𑧍𑧠𑧊𑧈𑧠𑦾𑧟 𑧍𑧔𑧊𑧠𑧇𑧞𑦵𑧁𑧟, 𑦮𑧚𑧊𑧉𑧟, 𑧂𑧈𑧒𑧂𑧕𑧈𑧠𑦼𑧟, 𑧂𑧈𑧒𑧋𑧔𑦿𑧠𑧀𑧟, 𑧂𑧈𑧠𑧇𑧊𑦿𑧑𑦽𑧟, 𑧍𑧠𑧊𑧑𑦯𑧠𑧇𑧑𑦽𑧟 𑧅𑦰𑧊𑦽𑧜 𑧀𑧈𑧠𑧆𑧟, 𑧍𑧑𑧞𑦿𑧖𑧌𑧠𑦸𑧒𑦮𑧟, 𑧁𑧒𑧈𑧠𑦵𑧠𑧊𑧈𑧟, 𑦡𑦮𑧑𑧉𑧒𑦮𑧟, 𑦭𑧂𑧁𑧑𑧇𑧒𑦮𑧟, 𑦫𑧎𑧒𑧂𑧋𑧠𑧇𑧒𑦮𑧟, 𑧂𑧠𑧈𑦽𑧠𑧇𑧑𑦽𑧠𑧆𑧊𑧚𑦿𑧁𑧓𑧇𑧜 𑧊𑧒𑦵𑧠𑦷𑧛𑧟, 𑧍𑧠𑧊𑧑𑦯𑧠𑧇𑧑𑦽𑧜 𑧅𑦰𑧊𑦽𑧜 𑧀𑧈𑧠𑧆𑧊𑧒𑧁𑧇𑧟 𑧍𑧔𑧂𑧠𑧈𑧊𑧚𑦿𑧒𑦽𑧟 𑧁𑧛𑧈𑧠𑧇𑧑𑦼𑧒𑦮𑧟, 𑧍𑧞𑧄𑧜𑧀𑧒𑦰𑧑𑧆𑧓, 𑦠𑧅𑧒𑧁𑧠𑧁𑧟 𑧍𑧞𑧍𑧠𑦽𑧕𑧂𑧟, 𑧍𑧂𑧠𑧈𑦽𑧒𑧋𑧈𑦼𑧟, 𑦴𑧒𑧁𑧠𑧁𑧂𑧠𑧉𑧜𑦽𑧒𑦮𑧟।

(𑧍𑧞𑦱𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧟)

𑧍𑧔𑧂𑧠𑧈𑦽𑧒𑧂𑧁𑧠𑧁𑧜 𑧅𑦰𑧊𑦽 𑦡𑧈𑧠𑧇𑧍𑧞𑦱𑧟, 𑧁𑧠𑧇𑧑𑧇𑧂𑧠𑧈𑦽𑧒𑧂𑧁𑧠𑧁𑧟, 𑦦𑦵𑧔𑧂𑧠𑧈𑦽𑧒𑧂𑧁𑧠𑧁𑧟, 𑧍𑧑𑧆𑧓𑦳𑧓𑧂𑧠𑧈𑦽𑧒𑧂𑧁𑧠𑧁𑧟, 𑦠𑧞𑦵𑧉𑧓𑦮𑧈𑦼𑧓𑧇𑧟, 𑧍𑧑𑧆𑧓𑦳𑧓𑦮𑧈𑦼𑧓𑧇𑧟, 𑧂𑧔𑦼𑧠𑧇𑧋𑧠𑧈𑧓𑦮𑧠𑧌𑧚𑦽𑧠𑧈𑧟, 𑧆𑧎𑧑𑦿𑦮𑧠𑧌𑧒𑦼𑧑𑧂𑧈𑧒𑧋𑧜𑧀𑦮𑧟, 𑧂𑧠𑧈𑧑𑧎𑧊𑧁𑧓𑧇𑧟, 𑦡𑧎𑧊𑧁𑧓𑧇𑧟।

॥ 𑦡𑧈𑧠𑧇𑦽𑧠𑧈𑧒𑧈𑦽𑧠𑧁𑧑𑧁𑧔𑧍𑧠𑧆𑧖𑦽𑧒𑧍𑧕𑦽𑧠𑧈𑧞 𑧍𑧆𑧑𑧂𑧠𑦽𑧆𑧠॥
