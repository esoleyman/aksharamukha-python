𑣚𑣚𑣂𑣕𑣟𑣂𑣞𑣕𑣜𑣄

𑣓𑣖𑣄 𑣞𑣜𑣟𑣗𑣃𑣔𑣔𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣈𑣗𑣅𑣄.

.. 𑣉𑣀 𑣓𑣖𑣉 𑣔𑣞𑣔𑣂𑣋𑣓𑣓𑣕𑣁𑣘𑣜𑣅𑣓𑣕𑣚𑣉𑣌𑣔𑣁𑣕𑣃𑣘𑣜𑣕𑣂𑣞𑣒𑣂𑣕𑣞𑣜𑣟𑣗𑣃𑣔𑣔𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣁𑣜𑣅𑣞𑣜𑣁𑣟𑣌𑣘𑣜𑣕𑣅𑣈𑣌𑣗𑣃𑣔𑣔𑣈𑣗𑣅𑣉'𑣕𑣆𑣕𑣁𑣓𑣁𑣋𑣕𑣘𑣜𑣕𑣅𑣃𑣕𑣘𑣓𑣓𑣈𑣗𑣅𑣄..

𑣡𑣠 𑣚𑣂𑣘𑣂𑣞𑣁𑣚𑣁𑣞𑣀𑣔𑣜𑣞𑣓𑣘𑣜𑣂𑣟𑣜𑣕𑣉 𑣔𑣞𑣖𑣄.

𑣂𑣕𑣂 𑣙𑣂 𑣗𑣂𑣌𑣞𑣟𑣄 𑣞𑣀𑣟𑣜𑣃𑣔𑣔𑣄 𑣌𑣃𑣖𑣁𑣜𑣄. 𑣕𑣔𑣁 𑣖𑣁𑣊𑣋𑣚𑣅𑣞𑣕𑣞𑣙𑣞𑣜𑣆𑣄 𑣚𑣂𑣘𑣂𑣞𑣁𑣚𑣁𑣖𑣃𑣘𑣓𑣆𑣅𑣕𑣈 𑣞𑣖 𑣔𑣞𑣗𑣂𑣜𑣔𑣁𑣜𑣌𑣞𑣙𑣞𑣜𑣆𑣄 𑣘𑣜𑣂𑣟𑣜𑣃𑣕𑣄 𑣘𑣃𑣜𑣞𑣌𑣜𑣃𑣕𑣄, 𑣔𑣞𑣗𑣂𑣞𑣏 𑣜𑣕𑣞𑣙𑣞𑣜𑣆𑣄 𑣌𑣁𑣔𑣓𑣆𑣅𑣗𑣉𑣎𑣓𑣆𑣅𑣞𑣟𑣁𑣔𑣓𑣆𑣅𑣘𑣜𑣂𑣘𑣇𑣜𑣐𑣆𑣜𑣙𑣂𑣜𑣐𑣅𑣞𑣃𑣟𑣜𑣐𑣘𑣜𑣂𑣘𑣇𑣜𑣐𑣆𑣞𑣏. 𑣅𑣈𑣓 𑣌𑣘𑣂𑣚𑣟𑣞𑣕𑣃𑣓𑣂 𑣖𑣙𑣁𑣓𑣋𑣜𑣈 𑣟𑣆𑣕𑣂𑣏𑣕𑣟𑣜𑣜𑣕𑣅𑣁𑣓𑣕𑣜𑣁𑣘𑣐𑣖𑣃𑣌𑣈𑣞𑣟𑣗𑣅𑣟𑣌𑣆𑣜𑣅𑣕𑣈 𑣞𑣖 𑣁𑣗𑣂𑣟𑣂𑣞𑣜𑣁𑣖𑣅𑣓𑣕𑣈. 𑣁𑣞𑣒𑣁𑣗𑣂𑣞𑣏 𑣕𑣇𑣜𑣅𑣞𑣕𑣞𑣙𑣞𑣜𑣆𑣄 𑣘𑣜𑣋𑣃𑣞𑣅𑣖𑣁𑣐𑣆𑣜𑣖𑣙𑣕𑣁 𑣏 𑣘𑣃𑣞𑣘𑣟𑣜𑣞𑣈𑣐𑣁𑣗𑣂𑣘𑣜𑣟𑣜𑣞𑣕𑣁 𑣟𑣂𑣕𑣜𑣔𑣂𑣓𑣂𑣜𑣅𑣇𑣙𑣕𑣉𑣜𑣐𑣋𑣟𑣁𑣌𑣞𑣙𑣜𑣖𑣅𑣌𑣇𑣒𑣁𑣋𑣁𑣜𑣘𑣜𑣁𑣞𑣁𑣔𑣕𑣚𑣈𑣞𑣃 𑣌𑣓𑣅𑣁𑣞𑣕𑣞𑣙𑣞𑣜𑣁𑣐𑣂 𑣞𑣜𑣟𑣁𑣚𑣀𑣌𑣁𑣜𑣗𑣇𑣞𑣂𑣕𑣁𑣄 𑣞𑣕𑣂𑣕𑣁 𑣁𑣗𑣇𑣟𑣓. 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣀 𑣘𑣜𑣈𑣌𑣞𑣖𑣁𑣐𑣁𑣄 𑣌𑣃𑣞𑣃𑣖𑣁𑣓𑣂 𑣏 𑣌𑣞𑣂𑣘𑣓𑣕𑣂 𑣞𑣖. 𑣁𑣞𑣒𑣇 𑣏 𑣖𑣜𑣃𑣕𑣌𑣓𑣅𑣁𑣞𑣙𑣞𑣜𑣁𑣐𑣂 𑣟𑣂𑣋𑣚𑣂𑣕𑣁𑣚𑣀𑣌𑣁𑣜𑣁𑣗𑣜𑣐𑣁𑣚𑣀𑣌𑣜𑣃𑣕𑣁𑣓𑣂 𑣜𑣕𑣓𑣗𑣔𑣜𑣀𑣌𑣜𑣈𑣐 𑣋𑣜𑣃𑣙𑣆𑣕𑣁𑣓𑣂 𑣖𑣁𑣜𑣋𑣀 𑣞𑣉𑣔𑣅𑣓𑣕𑣅𑣉 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣞𑣅 𑣘𑣃𑣜𑣕𑣉 𑣋𑣏𑣏𑣓𑣕𑣂 𑣞𑣖. 𑣔𑣈𑣟𑣓𑣁𑣋𑣅𑣌𑣞𑣋𑣓𑣔𑣜𑣟𑣁𑣞𑣃𑣜𑣋𑣜𑣃𑣑𑣌𑣂𑣓𑣓𑣜𑣖𑣙𑣉𑣜𑣋𑣁𑣞𑣏𑣁𑣜𑣔𑣌𑣁𑣅𑣂𑣌𑣁 𑣋𑣋𑣓𑣕𑣚𑣁𑣕𑣘𑣃𑣞𑣘𑣘𑣒𑣒𑣔𑣁𑣖𑣁𑣓𑣅𑣗𑣂𑣘𑣜𑣚𑣖𑣗𑣅𑣓𑣕𑣂 𑣞𑣖. 𑣞𑣜𑣟𑣈 𑣏 𑣞𑣁𑣌𑣅𑣋𑣐𑣁𑣄 𑣞𑣃𑣔𑣔𑣉𑣔𑣓𑣀 𑣜𑣁𑣎𑣁𑣓𑣀 𑣘𑣃𑣜𑣞𑣌𑣜𑣃𑣕𑣅 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣞𑣅 𑣘𑣃𑣜𑣕𑣉 𑣋𑣏𑣏𑣓𑣕𑣂 𑣞𑣖. 𑣁𑣓𑣈𑣓𑣆𑣟𑣀𑣜𑣇𑣘𑣈𑣐 𑣟𑣅𑣇𑣙𑣈𑣓 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣉 𑣚𑣂𑣘𑣂𑣞𑣁𑣚𑣁𑣖𑣃𑣘𑣓𑣆𑣅𑣕𑣈 𑣞𑣖..

𑣞𑣖𑣓𑣓𑣕𑣜𑣘𑣜𑣟𑣈𑣞𑣂𑣕𑣞𑣏 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣉 𑣚𑣂𑣘𑣂𑣞𑣁𑣚𑣁𑣖. 𑣁𑣕 𑣟𑣂𑣞𑣟𑣁𑣖𑣂𑣕𑣜𑣉 𑣓𑣁𑣖 𑣔𑣁𑣜𑣌𑣁𑣏𑣁𑣜𑣅𑣉 𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣞𑣅 𑣞𑣜𑣂𑣅𑣀 𑣕𑣈𑣎𑣞𑣏𑣁𑣞𑣙𑣖𑣁𑣓𑣉 𑣔𑣜𑣐𑣂𑣕𑣚𑣈 𑣓𑣂𑣟𑣂𑣞𑣒𑣉'𑣔𑣉𑣖𑣃𑣌𑣄 𑣘𑣜𑣘𑣕𑣕𑣂 𑣞𑣖. 𑣕𑣀 𑣕𑣕𑣁 𑣘𑣜𑣘𑣕𑣂𑣕𑣀 𑣔𑣜𑣃𑣞𑣒𑣟𑣁 𑣞𑣃𑣗𑣁𑣊𑣋𑣉 𑣓𑣁𑣖 𑣕𑣃𑣞𑣂𑣕𑣌𑣁𑣅𑣂𑣌𑣉 𑣔𑣈𑣟𑣘𑣃𑣕𑣜𑣉 𑣔𑣌𑣞𑣂𑣐𑣈𑣓 𑣌𑣜𑣕𑣚𑣈𑣓 𑣘𑣜𑣂𑣋𑣜𑣃𑣙𑣅𑣉𑣕𑣕𑣁𑣘𑣅𑣕𑣂 𑣞𑣖. 𑣃𑣕𑣕𑣁𑣘𑣅 𑣏 𑣋𑣋𑣓𑣕𑣚𑣞𑣕𑣉 𑣜𑣁𑣎𑣁𑣓𑣀 𑣞𑣃𑣔𑣔𑣉𑣔𑣓𑣀 𑣕𑣀 𑣏 𑣖𑣙𑣁𑣓𑣕𑣀 𑣎𑣓𑣌𑣁𑣅𑣀 𑣋𑣁𑣕𑣁𑣗𑣂𑣜𑣗𑣅𑣗𑣁𑣞𑣕—

𑣞𑣁𑣞𑣕𑣜𑣁𑣐𑣂 𑣅𑣁𑣓𑣂 𑣘𑣜𑣏𑣚𑣓𑣕𑣂 𑣖𑣓𑣃𑣞𑣅𑣚𑣉𑣌𑣈
𑣞𑣀𑣌𑣅𑣁 𑣚𑣂𑣘𑣂𑣞𑣏 𑣋𑣐𑣓𑣁𑣘𑣂 𑣏 𑣔𑣁𑣕𑣃𑣕𑣓𑣕𑣜𑣖.
𑣅𑣈 𑣞𑣂𑣚𑣘𑣅𑣉𑣋 𑣘𑣜𑣃𑣕𑣃 𑣚𑣇𑣌𑣂𑣌 𑣁𑣘𑣜𑣖𑣈𑣅𑣁𑣄
𑣕𑣈𑣞𑣟𑣈𑣞𑣃 𑣞𑣂𑣌𑣞𑣂𑣕𑣃 𑣘𑣃𑣜𑣁 𑣗𑣙𑣃𑣌𑣚𑣘𑣌𑣉𑣒𑣅𑣄..𑣡..

𑣌𑣂𑣀 𑣕𑣇 𑣎𑣓𑣞𑣅 𑣁𑣓𑣃𑣟𑣜𑣕𑣓𑣕𑣁𑣀 𑣌𑣜𑣉𑣕𑣂
𑣚𑣂𑣘𑣂𑣞𑣁𑣚𑣖𑣁𑣋𑣕𑣃 𑣞𑣃𑣞𑣂𑣌𑣞𑣂𑣕𑣃 𑣞𑣂𑣞𑣅𑣐𑣁𑣜𑣕𑣖.
𑣘𑣜𑣂𑣘𑣁𑣏𑣓𑣁𑣜𑣕 𑣗𑣙𑣃𑣔𑣁𑣜𑣌 𑣁𑣋𑣜𑣅𑣁𑣓𑣈
𑣁𑣓𑣅𑣁𑣀𑣞𑣏 𑣞𑣕𑣕𑣟𑣓𑣅𑣃𑣕𑣁𑣓𑣖𑣜𑣃𑣕𑣈 𑣟𑣂𑣓𑣈𑣕𑣃𑣖..𑣢..

𑣚𑣉𑣌𑣉𑣕𑣕𑣜𑣈𑣞𑣃 𑣏𑣕𑣃𑣞𑣕𑣅𑣘𑣕𑣈 𑣟𑣂𑣔𑣂𑣎𑣍𑣉
𑣙𑣈𑣕𑣃𑣘𑣜𑣕𑣆𑣕𑣅𑣌𑣃𑣞𑣚𑣉 𑣅𑣕 𑣞𑣀𑣗𑣟𑣓𑣕𑣂.
𑣅𑣕 𑣏𑣁𑣓𑣂𑣜𑣉𑣔𑣌𑣞𑣅𑣃 𑣞𑣀𑣞𑣕𑣂𑣕𑣃 𑣞𑣆𑣕𑣂𑣗𑣁𑣟𑣄
𑣕𑣞𑣖𑣂𑣓𑣟𑣂𑣔𑣂𑣎𑣍 𑣌𑣂𑣖𑣕𑣉 𑣚𑣂𑣘𑣂𑣞𑣁𑣞𑣕𑣜𑣖𑣁𑣕𑣜𑣈..𑣣..

𑣓𑣈𑣕𑣞𑣅 𑣁𑣙𑣏𑣜𑣂𑣅 𑣃𑣕𑣕𑣜𑣂 𑣟𑣁 𑣕𑣜𑣂𑣚𑣉𑣌𑣈
𑣞𑣜𑣟𑣈𑣞𑣃 𑣔𑣈𑣟𑣖𑣓𑣃𑣎𑣈𑣞𑣟𑣅𑣖𑣈𑣟 𑣎𑣈𑣞𑣒𑣄.
𑣓𑣁𑣖𑣁𑣘𑣂 𑣕𑣈𑣞 𑣚𑣂𑣘𑣂𑣓𑣁𑣀 𑣓 𑣙𑣂 𑣟𑣂𑣕𑣕 𑣅𑣇𑣅𑣀
𑣅𑣕𑣜𑣈𑣞𑣃 𑣞𑣂𑣌𑣞𑣂𑣕𑣃 𑣘𑣃𑣜𑣁 𑣗𑣙𑣃𑣌𑣚𑣘𑣌𑣉𑣒𑣅𑣄..𑣤..

𑣞𑣉 𑣏𑣂𑣕𑣕𑣔𑣁𑣜 𑣎𑣋𑣕𑣁𑣀 𑣟𑣂𑣟𑣂𑣔𑣁 𑣟𑣂𑣏𑣂𑣕𑣜𑣁
𑣈𑣌𑣌𑣞𑣐𑣈𑣓 𑣁𑣅𑣃 𑣎𑣁𑣓𑣕𑣂 𑣞𑣃𑣔𑣔𑣞𑣕𑣕𑣟𑣄.
𑣁𑣔𑣜𑣃𑣞𑣅𑣜𑣇𑣘𑣜𑣙𑣂𑣕𑣞𑣅 𑣋𑣕𑣂𑣀 𑣏 𑣟𑣈𑣕𑣕𑣂
