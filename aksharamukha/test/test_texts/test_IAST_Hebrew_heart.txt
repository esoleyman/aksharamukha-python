אָרְיַטְרִרַטְנָנֻסְמְרֻטִסוּטְרַם

(כַה) סַמְסְכְּרֻטַפָּטַהּ

בָּרַטִייַבָּשָׁיָם-אָרְיַטְרִרַטְנָנֻסְמְרֻטִסוּטְרַם

בּוֹטַבָּשָׁיָם-פַּגְס-פָּה-דַכּוֹן-מְצַּ׳וֹג-גְסֻם-גֵ׳ס-סֻה-דְרַן-פָּה-מְדוֹ

נַמַהּ סַרְבַבֻּדַּבּוֹדִסַטְּבֵבְּיַהּ

(בֻּדָּנֻסְמְרֻטִהּ)

אִטְיַפִּה בֻּדּוֹ בַּגַבָמְסְטַטָגַטוֹ'רְהַן סַמְיַכְּסַמְבֻּדּוֹ וִדְיָצַ׳רַנַסַמְפַּנַּהּ סֻגַטוֹ לוֹכַּבִדַנֻטַּרַהּ פֻּרֻשַׁדַמְיַסָרַטִהּ שָׁסְטָה דֵבַמַנֻשְׁיָנָם בֻּדּוֹ בַּגַבָנִטִה.

נִשְׁיַנְדַהּ סַה טַטָגַטַהּ פֻּנְיָנָם, אַבִפְּרַנָשַׁהּ כֻּשַׁלַמוּלָנָם, אַלַנְכְּרֻטַהּ כְּשָׁנְטְיָה, אָלַיַהּ פֻּנְיַנִדָנָנָם, צִ׳טְרִטוֹ'נֻבְיַנְגַ׳נַיְהּ כֻּסֻמִטוֹ לַכְּשַׁנַיְהּ, פְּרַטִרוּפּוֹ גוֹצַ׳רֵנַה, אַפְּרַטִכּוּלוֹ דַרְשַׁנֵנַה, אַבִּרַטִהּ שְׁרַדָּה(שֻׁדָּה)דִמֻכְּטָנָם, אַנַבִּבּוּטַהּ פְּרַגְ׳נַיָה, אַנַבַמַרְדַנִייוֹ בַּלַיְהּ, שָׁסְטָה סַרְבַסַטְּבָנָם, פִּטָה בּוֹדִסַטְּבָנָם, רָגָ׳ה אָרְיַפֻּדְגַלָנָם, סָרְטַבָהַהּ נִרְבָנַנַגַרַסַמְפְּרַסְטִטָנָם, אַפְּרַמֵיוֹ גְ׳נָנֵנַה, אַצִ׳נְטְיַהּ פְּרַטִבָּנֵנַה, וִשֻׁדַּהּ סְבַרֵנַה, אָסְבַדַנִייוֹ גוֹשֵׁנַה, אַסֵצַ׳נַכּוֹ רוּפֵּנַה, אַפְּרַטִסַמַהּ כָּיֵנַה, אַלִפְּטַהּ כָּמַיְהּ, אַנֻפַּלִפְּטוֹ רוּפַּיְהּ אַסַמְסְרֻשְׁטַה אָרוּפְּיַיְהּ, וִפְּרַמֻכְּטַהּ סְכַּנְדֵבְּיַהּ, וִסַמְפְּרַיֻכְּטוֹ דָטֻבִּהּ, סַמְבְרֻטַה אָיַטַנַיְהּ, פְּרַצִּ׳נּוֹ גְרַנְטַיְהּ, וִמֻכְּטַהּ פַּרִדָגְנַיְהּ, פַּרִמֻכְּטַסְטְרֻשְׁנַיָה, אוֹגָדֻטִּירְנַהּ פַּרִפּוּרְנוֹ גְ׳נָנֵנַה, פְּרַטִשְׁטִטוֹ'טִיטָנָגַטַפְּרַטְיֻטְפַּנָּנָם בֻּדָּנָם בַּגַבַטָם גְ׳נָנֵה, אַפְּרַטִשְׁטִטוֹ נִרְבָנֵה, סְטִטוֹ בּוּטַכּוֹטְיָם, סְטִטַהּ סַרְבַסַטְּבָלוֹכַּנִייָיָם בּוּמַב, סַרְבַה אִמֵה טַטָגַטָנָם וִשֵׁשַׁטַהּ סַמְיַג גֻנָהּ.

(דַרְמָנֻסְמְרֻטִהּ)

סַדַּרְמַסְטֻה אָדַב כַּלְיָנַהּ, מַדְיֵה כַּלְיָנַהּ, פַּרְיַבַסָנֵה כַּלְיָנַהּ, סְבַרְטַהּ סֻבְיַנְגַ׳נַהּ, כֵּבַלַהּ, פַּרִפּוּרְנַהּ, פַּרִשֻׁדַּהּ, פַּרְיַבַדָטַהּ, סְבָכְיָטַהּ בַּגַבַטוֹ דַרְמַהּ, סָנְדְרֻשְׁטִכַּהּ, נִרְגְ׳וַרַהּ, אָכָּלִכַּהּ, אַבְפַּנָיִכַּהּ, אַיְהִפַּשְׁיִכַּהּ, פְּרַטְיָטְמַבֵדַנִייוֹ וִגְ׳נַיְהּ, סְבָכְיָטוֹ בַּגַבַטוֹ דַרְמַבִנַיַהּ סֻפְּרַבֵדִטַהּ נַיְרְיָנִכַּהּ, סַמְבּוֹדִגָמִי, אַבִּנַּהּ סַמְסְטוּפַּהּ, סַפְּרַטִשַׁרַנַהּ, צִ׳נַּפְּלוֹטִכַּהּ.

(סַנְגָנֻסְמְרֻטִהּ)

סֻפְּרַטִפַּנּוֹ בַּגַבַטַה אָרְיַסַנְגַהּ, נְיָיַפְּרַטִפַּנַּהּ, רֻגֻ׳פְּרַטִפַּנַּהּ, סָמִיצִ׳יפְּרַטִפַּנַּהּ, אַנְגַ׳לִיכַּרַנִייַהּ, סָמִיצִ׳יכַּרַנִייַהּ, פֻּנְיַשְׁרִיכְּשֵׁטְרַהּ, מַהָדַכְּשִׁנָפַּרִשׁוֹדַכַּהּ, פְּרָהַבַנִייַהּ, אָהַבַנִייַהּ.
