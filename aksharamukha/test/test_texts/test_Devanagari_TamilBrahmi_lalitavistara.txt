10 𑀮𑀺𑀧𑀺𑀰𑀸𑀮𑀸𑀲𑀫𑁆𑀢𑀭𑁆𑀰𑀦𑀧𑀭𑀺𑀯𑀭𑁆𑀢𑁄 𑀢𑀰𑀫𑀂.

𑀇𑀢𑀺 𑀳𑀺 𑀧𑀺𑀓𑁆𑀱𑀯𑀂 𑀲𑀫𑁆𑀯𑁆𑀭𑀼𑀢𑁆𑀥𑀂 𑀓𑀼𑀫𑀸𑀭𑀂. 𑀢𑀢𑀸 𑀫𑀸𑀗𑁆𑀓𑀮𑁆𑀬𑀰𑀢𑀲𑀳𑀲𑁆𑀭𑁃𑀂 𑀮𑀺𑀧𑀺𑀰𑀸𑀮𑀸𑀫𑀼𑀧𑀦𑀻𑀬𑀢𑁂 𑀲𑁆𑀫 𑀢𑀰𑀧𑀺𑀭𑁆𑀢𑀸𑀭𑀓𑀲𑀳𑀲𑁆𑀭𑁃𑀂 𑀧𑀭𑀺𑀯𑁆𑀭𑀼𑀢𑀂 𑀧𑀼𑀭𑀲𑁆𑀓𑁆𑀭𑀼𑀢𑀂, 𑀢𑀰𑀧𑀺𑀰𑁆𑀘 𑀭𑀢𑀲𑀳𑀲𑁆𑀭𑁃𑀂 𑀓𑀸𑀢𑀦𑀻𑀬𑀧𑁄𑀚𑀦𑀻𑀬𑀲𑁆𑀯𑀸𑀢𑀦𑀻𑀬𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑁃𑀭𑁆𑀳𑀺𑀭𑀡𑁆𑀬𑀲𑀼𑀯𑀭𑁆𑀡𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑁃𑀰𑁆𑀘. 𑀬𑁂𑀦 𑀓𑀧𑀺𑀮𑀯𑀲𑁆𑀢𑀼𑀦𑀺 𑀫𑀳𑀸𑀦𑀓𑀭𑁂 𑀯𑀻𑀢𑀺𑀘𑀢𑁆𑀯𑀭𑀭𑀢𑁆𑀬𑀸𑀦𑁆𑀢𑀭𑀸𑀧𑀡𑀫𑀼𑀓𑁂𑀱𑁆𑀯𑀧𑁆𑀬𑀯𑀓𑀻𑀭𑁆𑀬𑀢𑁂 𑀲𑁆𑀫 𑀅𑀧𑀺𑀯𑀺𑀰𑁆𑀭𑀸𑀫𑁆𑀬𑀦𑁆𑀢𑁂. 𑀅𑀱𑁆𑀝𑀸𑀧𑀺𑀰𑁆𑀘 𑀢𑀽𑀭𑁆𑀬𑀰𑀢𑀲𑀳𑀲𑁆𑀭𑁃𑀂 𑀧𑁆𑀭𑀓𑀼𑀱𑁆𑀬𑀫𑀸𑀡𑁃𑀭𑁆𑀫𑀳𑀢𑀸 𑀘 𑀧𑀼𑀱𑁆𑀧𑀯𑀭𑁆𑀱𑁂𑀡𑀸𑀧𑀺𑀧𑁆𑀭𑀯𑀭𑁆𑀱𑀢𑀸 𑀯𑀺𑀢𑀭𑁆𑀢𑀺𑀦𑀺𑀭𑁆𑀬𑀽𑀳𑀢𑁄𑀭𑀡𑀓𑀯𑀸𑀓𑁆𑀱𑀳𑀭𑁆𑀫𑁆𑀬𑀓𑀽𑀝𑀸𑀓𑀸𑀭𑀧𑁆𑀭𑀸𑀲𑀸𑀢𑀢𑀮𑁂𑀱𑀼 𑀓𑀦𑁆𑀬𑀸𑀰𑀢𑀲𑀳𑀲𑁆𑀭𑀸𑀡𑀺 𑀲𑀭𑁆𑀯𑀸𑀮𑀫𑁆𑀓𑀸𑀭𑀧𑀽𑀱𑀺𑀢𑀸𑀂 𑀲𑁆𑀢𑀺𑀢𑀸 𑀅𑀧𑀽𑀯𑀦𑁆. 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀫𑁆 𑀧𑁆𑀭𑁂𑀓𑁆𑀱𑀫𑀸𑀡𑀸𑀂 𑀓𑀼𑀲𑀼𑀫𑀸𑀦𑀺 𑀘 𑀓𑁆𑀱𑀺𑀧𑀦𑁆𑀢𑀺 𑀲𑁆𑀫. 𑀅𑀱𑁆𑀝𑁅 𑀘 𑀫𑀭𑀼𑀢𑁆𑀓𑀦𑁆𑀬𑀸𑀲𑀳𑀲𑁆𑀭𑀸𑀡𑀺 𑀯𑀺𑀓𑀮𑀺𑀢𑀸𑀮𑀫𑁆𑀓𑀸𑀭𑀸𑀧𑀭𑀡𑀸𑀮𑀫𑁆𑀓𑁆𑀭𑀼𑀢𑀸𑀦𑀺 𑀭𑀢𑁆𑀦𑀧𑀢𑁆𑀭𑀫𑁆𑀓𑀭𑁂𑀡 𑀓𑁆𑀭𑀼𑀳𑀻𑀢𑀸𑀦𑀺 𑀫𑀸𑀭𑁆𑀓𑀫𑁆 𑀰𑁄𑀥𑀬𑀦𑁆𑀢𑁆𑀬𑁄 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀲𑁆𑀬 𑀧𑀼𑀭𑀢𑁄 𑀓𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑁆𑀫. 𑀢𑁂𑀯𑀦𑀸𑀓𑀬𑀓𑁆𑀱𑀓𑀦𑁆𑀥𑀭𑁆𑀯𑀸𑀲𑀼𑀭𑀓𑀭𑀼𑀝𑀓𑀺𑀦𑁆𑀦𑀭𑀫𑀳𑁄𑀭𑀓𑀸𑀰𑁆𑀘𑀸𑀭𑁆𑀥𑀓𑀸𑀬𑀺𑀓𑀸 𑀓𑀓𑀦𑀢𑀮𑀸𑀢𑁆𑀧𑀼𑀱𑁆𑀧𑀧𑀝𑁆𑀝𑀢𑀸𑀫𑀸𑀦𑁆𑀬𑀧𑀺𑀧𑁆𑀭𑀮𑀫𑁆𑀧𑀬𑀦𑁆𑀢𑀺 𑀲𑁆𑀫. 𑀲𑀭𑁆𑀯𑁂 𑀘 𑀰𑀸𑀓𑁆𑀬𑀓𑀡𑀸𑀂 𑀰𑀼𑀢𑁆𑀥𑁄𑀢𑀦𑀫𑁆 𑀭𑀸𑀚𑀸𑀦𑀫𑁆 𑀧𑀼𑀭𑀲𑁆𑀓𑁆𑀭𑀼𑀢𑁆𑀬 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀲𑁆𑀬 𑀧𑀼𑀭𑀢𑁄 𑀓𑀘𑁆𑀘𑀦𑁆𑀢𑀺 𑀲𑁆𑀫. 𑀅𑀦𑁂𑀦𑁃𑀯𑀫𑁆𑀭𑀽𑀧𑁂𑀡 𑀯𑁆𑀬𑀽𑀳𑁂𑀦 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑁄 𑀮𑀺𑀧𑀺𑀰𑀸𑀮𑀸𑀫𑀼𑀧𑀦𑀻𑀬𑀢𑁂 𑀲𑁆𑀫..

𑀲𑀫𑀦𑀦𑁆𑀢𑀭𑀧𑁆𑀭𑀯𑁂𑀰𑀺𑀢𑀰𑁆𑀘 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑁄 𑀮𑀺𑀧𑀺𑀰𑀸𑀮𑀸𑀫𑁆. 𑀅𑀢 𑀯𑀺𑀰𑁆𑀯𑀸𑀫𑀺𑀢𑁆𑀭𑁄 𑀦𑀸𑀫 𑀢𑀸𑀭𑀓𑀸𑀘𑀸𑀭𑁆𑀬𑁄 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀲𑁆𑀬 𑀰𑁆𑀭𑀺𑀬𑀫𑁆 𑀢𑁂𑀚𑀰𑁆𑀘𑀸𑀲𑀳𑀫𑀸𑀦𑁄 𑀥𑀭𑀡𑀺𑀢𑀮𑁂 𑀦𑀺𑀯𑀺𑀱𑁆𑀝𑁄'𑀥𑁄𑀫𑀼𑀓𑀂 𑀧𑁆𑀭𑀧𑀢𑀢𑀺 𑀲𑁆𑀫. 𑀢𑀫𑁆 𑀢𑀢𑀸 𑀧𑁆𑀭𑀧𑀢𑀺𑀢𑀫𑁆 𑀢𑁆𑀭𑀼𑀱𑁆𑀝𑁆𑀯𑀸 𑀰𑀼𑀧𑀸𑀗𑁆𑀓𑁄 𑀦𑀸𑀫 𑀢𑀼𑀱𑀺𑀢𑀓𑀸𑀬𑀺𑀓𑁄 𑀢𑁂𑀯𑀧𑀼𑀢𑁆𑀭𑁄 𑀢𑀓𑁆𑀱𑀺𑀡𑁂𑀦 𑀓𑀭𑀢𑀮𑁂𑀦 𑀧𑀭𑀺𑀓𑁆𑀭𑀼𑀳𑁆𑀬𑁄𑀢𑁆𑀢𑀸𑀧𑀬𑀢𑀺 𑀲𑁆𑀫. 𑀉𑀢𑁆𑀢𑀸𑀧𑁆𑀬 𑀘 𑀓𑀓𑀦𑀢𑀮𑀲𑁆𑀢𑁄 𑀭𑀸𑀚𑀸𑀦𑀫𑁆 𑀰𑀼𑀢𑁆𑀥𑁄𑀢𑀦𑀫𑁆 𑀢𑀫𑁆 𑀘 𑀫𑀳𑀸𑀦𑁆𑀢𑀫𑁆 𑀚𑀦𑀓𑀸𑀬𑀫𑁆 𑀓𑀸𑀢𑀸𑀧𑀺𑀭𑀧𑁆𑀬𑀧𑀸𑀱𑀢𑁆—

𑀰𑀸𑀲𑁆𑀢𑁆𑀭𑀸𑀡𑀺 𑀬𑀸𑀦𑀺 𑀧𑁆𑀭𑀘𑀮𑀦𑁆𑀢𑀺 𑀫𑀦𑀼𑀱𑁆𑀬𑀮𑁄𑀓𑁂
𑀲𑀫𑁆𑀓𑁆𑀬𑀸 𑀮𑀺𑀧𑀺𑀰𑁆𑀘 𑀓𑀡𑀦𑀸𑀧𑀺 𑀘 𑀥𑀸𑀢𑀼𑀢𑀦𑁆𑀢𑁆𑀭𑀫𑁆.
𑀬𑁂 𑀰𑀺𑀮𑁆𑀧𑀬𑁄𑀓 𑀧𑁆𑀭𑀼𑀢𑀼 𑀮𑁅𑀓𑀺𑀓 𑀅𑀧𑁆𑀭𑀫𑁂𑀬𑀸𑀂
