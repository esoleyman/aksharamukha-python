ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܗܪܽܕ݁ܰܝܰܣܽܬ݁ܪܰܡ.

[ܣܰܡܟ݁ܫܺܦ݁ܬ݁ܰܡܳܬ݁ܪܽܟ݁ܳ]

.. ܢܰܡܰܗ ܣܰܪܒ݂ܰܓ̰ܢܳܝܰ ..

.. ܐّܡ ܢܰܡّ ܒ݁ܰܓ݁ܰܒ݂ܰܬ݁ܝܰܝ ܐܳܪܝܰܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܝܰܝ ..

ܐܳܪܝܳܒ݂ܰܠّܟ݁ܺܬ݁ܶܫܒ݂ܰܪܰܒّ݁ܕ݁ܺܣܰܬ݁ܒّ݂ ܓ݁ܰܡܒ݁ܺܪܳܝܳܡ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܝܳܡ ܟ̰ܰܪܝܳܡ ܟ̰ܰܪܰܡܳܢّ ܒ݂ܝܰܒ݂ܰܠّܟ݁ܰܝܰܬ݁ܺ ܣܡܰ. ܦ݁ܰܢܟ̰ܰ ܣܟ݁ܰܢܕ݁ܳܗ, ܬ݁ܳܡܫܟ̰ܰ ܣܒ݂ܰܒ݁ܳܒ݂ܰܫܽܢܝܳܢ ܦ݁ܰܫܝܰܬ݁ܺ ܣܡܰ..

ܐܺܗܰ ܫܳܪܺܦ݁ܽܬ݁ܪܰ ܪܽܦ݁ܰܡ ܫܽܢܝܰܬ݁ܳ, ܫܽܢܝܰܬ݁ܰܝܒ݂ܰ ܪܽܦ݁ܰܡ. ܪܽܦ݁ܳܢܰ ܦ݁ܪܽܬ݁ܰܟ݁ ܫܽܢܝܰܬ݁ܳ, ܫܽܢܝܰܬ݁ܳܝܳ ܢܰ ܦ݁ܪܽܬ݁ܰܓ݁ ܪܽܦ݁ܰܡ. ܝܰܕ݁ܪܽܦ݁ܰܡ ܣܳ ܫܽܢܝܰܬ݁ܳ, ܝܳ ܫܽܢܝܰܬ݁ܳ ܬ݁ܰܕ݁ܪܽܦ݁ܰܡ..

ܐܶܒ݂ܰܡܶܒ݂ܰ ܒ݂ܶܕ݁ܰܢܳܣܰܡܓ̰ܢܳܣܰܡܣܟ݁ܳܪܰܒ݂ܺܓ̰ܢܳܢܳܢܺ..

ܐܺܗܰܡ ܫܳܪܺܦ݁ܽܬ݁ܪܰ ܣܰܪܒ݂ܰܕ݁ܰܪܡܳܗ ܫܽܢܝܰܬ݁ܳܠܰܟ݁ܫܰܢܳ ܐܰܢܽܬ݁ܦ݁ܰܢܳ ܐܰܢܺܪܽܕ݁ܳ ܐܰܡܰܠܳ ܢܰ ܒ݂ܺܡܰܠܳ ܢّܢܳ ܢܰ ܦ݁ܰܪܺܦ݁ܽܪܢܳܗ. ܬ݁ܰܣܡܳܟ̰ܳܪܺܦ݁ܽܬ݁ܪܰ ܫܽܢܝܰܬ݁ܳܝܳܡ ܢܰ ܪܽܦ݁ܰܡ, ܢܰ ܒ݂ܶܕ݁ܰܢܳ, ܢܰ ܣܰܡܓ̰ܢܳ, ܢܰ ܣܰܡܣܟ݁ܳܪܳܗ, ܢܰ ܒ݂ܺܓ̰ܢܳܢܳܢܺ. ܢܰ ܟ̰ܰܟ݁ܫܽܗܫܪّܬ݁ܪܰܓ݁ܪܳܢܰܓ̰ܺܗܒ݂ܳܟ݁ܳܝܰܡܰܢܳܡܣܺ, ܢܰ ܪܽܦ݁ܰܫܰܒ݁ܕ݁ܰܓ݁ܰܢܕ݁ܰܪܰܣܰܣܦ݁ܪܰܫܬ݁ܰܒ݂ܝܰܕ݁ܰܪܡܳܗ. ܢܰ ܟ̰ܰܟ݁ܫܽܪܕ݁ܳܬ݁ܽܪܝܳܒ݂ܰܢܰ ܡܰܢّܕ݁ܳܬ݁ܽܗ..

ܢܰ ܒ݂ܺܕ݁ܝܳ ܢܳܒ݂ܺܕ݁ܝܳ ܢܰ ܒ݂ܺܕ݁ܝܳܟ݁ܫܰܝّ ܢܳܒ݂ܺܕ݁ܝܳܟ݁ܫܰܝّ ܝܳܒ݂ܰܢܰ ܓ̰ܰܪܳܡܰܪܰܢܰܡ ܢܰ ܓ̰ܰܪܳܡܰܪܰܢܰܟ݁ܫܰܝّ ܢܰ ܕ݁ܽܗܟ݁ܰܣܰܡܽܕ݁ܰܝܰܢܺܪّܕ݁ܰܡܳܪܓ݁ܳ ܢܰ ܓ̰ܢܳܢܰܡ ܢܰ ܦ݁ܪܳܦ݁ܬ݁ܺܬ݁ܒ݂ܰܡ..

ܒّ݁ܕ݁ܺܣܰܬ݁ܒ݂ܰܣܝܰ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܡܳܫܪܺܬ݁ܝܰ ܒ݂ܺܗܰܪܰܬ݁ܺ ܟ̰ܺܬ݁ܳܒ݂ܰܪܰܢܰܗ. ܟ̰ܺܬ݁ܳܒ݂ܰܪܰܢܰܢܳܣܬ݁ܺܬ݁ܒ݂ܳܕ݁ܰܬ݁ܪܰܣܬّ݁ ܒ݂ܺܦ݁ܰܪܝܳܣܳܬ݁ܺܟ݁ܪܳܢܬّ݁ ܢܺܫܬ݁ܰܢܺܪܒ݂ܳܢܰܗ. ܬ݁ܪܝܰܕ݁ܒ݂ܰܒ݂ܝܰܒ݂ܰܣܬ݁ܺܬ݁ܳܗ ܣܰܪܒ݂ܰܒ݁ܽܕ݁ܳܗ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܡܳܫܪܺܬ݁ܝܰ ܐܰܢܽܬ݁ܰܪܳܡ ܣܰܡܝܰܟ݁ܣܰܡܒّ݁ܕ݁ܺܡܰܒ݁ܺܣܰܡܒ݁ܽܕ݁ܳܗ..

ܬ݁ܰܣܡܳܓ̰ܢܳܬ݁ܰܒ݂ܝܰܗ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܡܰܗܳܡܰܢܬ݁ܪّ ܡܰܗܳܒ݂ܺܕ݁ܝܳܡܰܢܬ݁ܪّܢܽܬ݁ܰܪܰܡܰܢܬ݁ܪّܣܰܡܰܣܰܡܰܡܰܢܬ݁ܪܰܗ ܣܰܪܒ݂ܰܕ݁ܽܗܟ݁ܰܦ݁ܪܰܫܰܡܰܢܰܗ ܣܰܬ݁ܝܰܡܰܡܺܬ݁ܝܰܬ݁ܒ݂ܳܬ݁ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܝܳܡܽܟ݁ܬّ݁ ܡܰܢܬ݁ܪܰܗ.

ܬ݁ܰܕ݁ܝܰܬ݁ܳ- ܓ݁ܰܬ݁ܶ ܓ݁ܰܬ݁ܶ ܦ݁ܳܪܰܓ݁ܰܬ݁ܶ ܦ݁ܳܪܰܣܰܡܓ݁ܰܬ݁ܶ ܒّ݁ܕ݁ܺ ܣܒ݂ܳܗܳ..

ܐܺܬ݁ܺ ܦ݁ܪܰܓ̰ܢܳܦ݁ܳܪܰܡܺܬ݁ܳܗܪܽܕ݁ܰܝܰܣܽܬ݁ܪܰܡ ܣܰܡܳܦ݁ܬ݁ܰܡ..
