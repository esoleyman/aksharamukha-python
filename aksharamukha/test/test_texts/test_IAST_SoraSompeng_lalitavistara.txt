𑃘𑃘𑃤𑃑𑃚𑃤𑃐𑃑𑃝𑃞

𑃙𑃖𑃞 𑃐𑃝𑃚𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃣𑃒𑃠𑃜𑃞.

.. 𑃦𑃖 𑃙𑃖𑃦 𑃔𑃨𑃐𑃠𑃔𑃨𑃤𑃕𑃙𑃙𑃑𑃢𑃛𑃝𑃜𑃙𑃑𑃘𑃦𑃟𑃔𑃨𑃠𑃢𑃑𑃥𑃛𑃝𑃑𑃤𑃐𑃨𑃑𑃨𑃠𑃤𑃑𑃐𑃝𑃚𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃢𑃝𑃜𑃐𑃠𑃝𑃢𑃚𑃟𑃛𑃝𑃑𑃜𑃣𑃟𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃣𑃒𑃠𑃜𑃦'𑃑𑃤𑃑𑃢𑃙𑃢𑃕𑃑𑃛𑃝𑃑𑃜𑃥𑃑𑃛𑃙𑃙𑃣𑃒𑃠𑃜𑃞..

𑃱𑃰 𑃘𑃤𑃛𑃤𑃐𑃠𑃢𑃘𑃢𑃐𑃖𑃔𑃨𑃝𑃐𑃠𑃙𑃛𑃝𑃤𑃚𑃝𑃑𑃦 𑃔𑃨𑃐𑃠𑃖𑃞.

𑃤𑃑𑃤 𑃞𑃤 𑃒𑃠𑃤𑃟𑃐𑃨𑃚𑃞 𑃐𑃖𑃚𑃝𑃤𑃔𑃨𑃔𑃨𑃠𑃞 𑃟𑃥𑃖𑃢𑃝𑃞. 𑃑𑃔𑃨𑃢 𑃖𑃢𑃗𑃕𑃘𑃜𑃐𑃠𑃑𑃐𑃞𑃐𑃝𑃧𑃞 𑃘𑃤𑃛𑃤𑃐𑃠𑃢𑃘𑃢𑃖𑃥𑃛𑃙𑃤𑃜𑃑𑃣 𑃐𑃖 𑃔𑃨𑃐𑃠𑃒𑃠𑃤𑃝𑃔𑃨𑃢𑃝𑃟𑃐𑃞𑃐𑃝𑃧𑃞 𑃛𑃝𑃤𑃚𑃝𑃤𑃑𑃞 𑃛𑃥𑃝𑃐𑃟𑃝𑃤𑃑𑃞, 𑃔𑃨𑃐𑃠𑃒𑃠𑃤𑃐𑃠𑃓 𑃝𑃑𑃠𑃐𑃞𑃐𑃝𑃧𑃞 𑃟𑃠𑃢𑃔𑃨𑃙𑃤𑃜𑃒𑃠𑃦𑃠𑃙𑃤𑃜𑃐𑃚𑃢𑃔𑃨𑃙𑃤𑃜𑃛𑃝𑃤𑃛𑃥𑃝𑃙𑃨𑃧𑃝𑃞𑃤𑃝𑃙𑃨𑃜𑃐𑃥𑃚𑃝𑃙𑃨𑃛𑃝𑃤𑃛𑃥𑃝𑃙𑃨𑃧𑃐𑃠𑃓. 𑃜𑃣𑃙 𑃟𑃛𑃤𑃘𑃚𑃐𑃑𑃥𑃙𑃤 𑃖𑃞𑃢𑃙𑃕𑃝𑃣 𑃚𑃤𑃑𑃠𑃤𑃓𑃑𑃚𑃝𑃝𑃑𑃠𑃜𑃢𑃙𑃑𑃝𑃢𑃛𑃙𑃨𑃖𑃥𑃟𑃠𑃣𑃐𑃨𑃚𑃒𑃠𑃜𑃚𑃟𑃤𑃝𑃜𑃑𑃣 𑃐𑃖 𑃦𑃨𑃒𑃠𑃤𑃚𑃤𑃐𑃠𑃝𑃢𑃖𑃜𑃙𑃑𑃣. 𑃦𑃨𑃐𑃨𑃑𑃨𑃢𑃒𑃠𑃤𑃐𑃠𑃓 𑃑𑃥𑃝𑃜𑃐𑃠𑃑𑃐𑃞𑃐𑃝𑃧𑃞 𑃛𑃝𑃕𑃠𑃥𑃐𑃨𑃜𑃖𑃢𑃙𑃨𑃧𑃝𑃖𑃞𑃑𑃢 𑃓 𑃛𑃥𑃐𑃨𑃛𑃚𑃝𑃐𑃨𑃣𑃙𑃨𑃢𑃒𑃠𑃤𑃛𑃝𑃚𑃝𑃐𑃨𑃑𑃢 𑃚𑃤𑃑𑃝𑃔𑃨𑃤𑃙𑃤𑃝𑃜𑃥𑃞𑃑𑃦𑃝𑃙𑃨𑃕𑃚𑃢𑃟𑃐𑃨𑃞𑃝𑃖𑃜𑃟𑃥𑃑𑃨𑃢𑃕𑃢𑃝𑃛𑃝𑃢𑃐𑃢𑃔𑃨𑃑𑃘𑃣𑃐𑃨𑃥 𑃟𑃙𑃜𑃢𑃐𑃠𑃑𑃐𑃞𑃐𑃝𑃢𑃙𑃨𑃤 𑃐𑃝𑃚𑃢𑃘𑃖𑃟𑃢𑃝𑃒𑃠𑃥𑃐𑃨𑃤𑃑𑃢𑃞 𑃐𑃑𑃠𑃤𑃑𑃢 𑃦𑃨𑃒𑃠𑃥𑃚𑃙. 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃖 𑃛𑃝𑃣𑃟𑃐𑃨𑃖𑃢𑃙𑃨𑃢𑃞 𑃟𑃥𑃐𑃥𑃖𑃢𑃙𑃤 𑃓 𑃟𑃐𑃨𑃤𑃛𑃙𑃑𑃤 𑃐𑃖. 𑃦𑃨𑃐𑃨𑃑𑃨𑃦𑃥 𑃓 𑃖𑃝𑃥𑃑𑃟𑃙𑃜𑃢𑃐𑃞𑃐𑃝𑃢𑃙𑃨𑃤 𑃚𑃤𑃕𑃘𑃤𑃑𑃢𑃘𑃖𑃟𑃢𑃝𑃢𑃒𑃠𑃝𑃙𑃨𑃢𑃘𑃖𑃟𑃝𑃤𑃑𑃢𑃙𑃤 𑃝𑃑𑃙𑃒𑃠𑃔𑃨𑃝𑃖𑃟𑃝𑃣𑃙𑃨 𑃕𑃝𑃤𑃞𑃤𑃑𑃢𑃙𑃤 𑃖𑃢𑃝𑃕𑃖 𑃐𑃠𑃦𑃔𑃨𑃠𑃜𑃙𑃑𑃜𑃦 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃐𑃜 𑃛𑃥𑃝𑃑𑃦 𑃕𑃓𑃓𑃠𑃙𑃑𑃤 𑃐𑃖. 𑃔𑃨𑃣𑃚𑃙𑃢𑃕𑃜𑃟𑃐𑃨𑃕𑃙𑃔𑃨𑃠𑃝𑃚𑃢𑃐𑃥𑃝𑃕𑃝𑃥𑃔𑃟𑃤𑃙𑃙𑃝𑃖𑃞𑃦𑃝𑃕𑃢𑃐𑃠𑃓𑃢𑃝𑃔𑃨𑃠𑃟𑃢𑃜𑃤𑃟𑃢 𑃕𑃕𑃙𑃑𑃘𑃢𑃑𑃛𑃥𑃐𑃨𑃛𑃛𑃑𑃨𑃑𑃨𑃔𑃨𑃢𑃖𑃢𑃙𑃜𑃒𑃠𑃤𑃛𑃝𑃘𑃖𑃒𑃜𑃙𑃑𑃤 𑃐𑃖. 𑃐𑃝𑃚𑃣 𑃓 𑃐𑃠𑃢𑃟𑃜𑃕𑃙𑃨𑃢𑃞 𑃐𑃠𑃥𑃔𑃨𑃔𑃨𑃠𑃦𑃔𑃨𑃙𑃖 𑃝𑃢𑃠𑃢𑃙𑃖 𑃛𑃥𑃝𑃐𑃟𑃝𑃤𑃑𑃜 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃐𑃜 𑃛𑃥𑃝𑃑𑃦 𑃕𑃓𑃓𑃠𑃙𑃑𑃤 𑃐𑃖. 𑃦𑃨𑃙𑃣𑃙𑃧𑃚𑃖𑃝𑃥𑃛𑃣𑃙𑃨 𑃚𑃜𑃥𑃞𑃣𑃙 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃦 𑃘𑃤𑃛𑃤𑃐𑃠𑃢𑃘𑃢𑃖𑃥𑃛𑃙𑃤𑃜𑃑𑃣 𑃐𑃖..

𑃐𑃖𑃙𑃙𑃑𑃝𑃛𑃝𑃚𑃣𑃐𑃠𑃤𑃑𑃐𑃠𑃓 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃦 𑃘𑃤𑃛𑃤𑃐𑃠𑃢𑃘𑃢𑃖. 𑃦𑃨𑃑𑃠 𑃚𑃤𑃐𑃠𑃚𑃢𑃖𑃤𑃑𑃝𑃦 𑃙𑃢𑃖 𑃔𑃨𑃢𑃝𑃟𑃢𑃓𑃢𑃝𑃜𑃦 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃐𑃜 𑃐𑃠𑃝𑃤𑃜𑃖 𑃑𑃣𑃠𑃐𑃠𑃓𑃢𑃐𑃞𑃖𑃢𑃙𑃦 𑃔𑃨𑃠𑃝𑃙𑃨𑃤𑃑𑃘𑃣 𑃙𑃤𑃚𑃤𑃐𑃨𑃑𑃨𑃦'𑃔𑃨𑃠𑃦𑃖𑃥𑃟𑃠𑃞 𑃛𑃝𑃛𑃑𑃑𑃤 𑃐𑃖. 𑃑𑃖 𑃑𑃑𑃠𑃢 𑃛𑃝𑃛𑃑𑃤𑃑𑃖 𑃔𑃨𑃝𑃤𑃐𑃨𑃑𑃨𑃚𑃢 𑃐𑃠𑃥𑃒𑃠𑃢𑃗𑃕𑃦 𑃙𑃢𑃖 𑃑𑃥𑃐𑃨𑃤𑃑𑃟𑃢𑃜𑃤𑃟𑃦 𑃔𑃨𑃣𑃚𑃛𑃥𑃑𑃝𑃦 𑃔𑃨𑃟𑃐𑃨𑃤𑃙𑃨𑃣𑃙 𑃟𑃝𑃑𑃘𑃣𑃙 𑃛𑃝𑃤𑃕𑃝𑃤𑃞𑃜𑃦𑃑𑃑𑃠𑃢𑃛𑃜𑃑𑃤 𑃐𑃖. 𑃥𑃑𑃑𑃠𑃢𑃛𑃜 𑃓 𑃕𑃕𑃙𑃑𑃘𑃐𑃑𑃠𑃦 𑃝𑃢𑃠𑃢𑃙𑃖 𑃐𑃠𑃥𑃔𑃨𑃔𑃨𑃠𑃦𑃔𑃨𑃙𑃖 𑃑𑃖 𑃓 𑃖𑃞𑃢𑃙𑃑𑃖 𑃠𑃙𑃟𑃢𑃜𑃖 𑃕𑃢𑃑𑃠𑃢𑃒𑃠𑃤𑃝𑃒𑃠𑃜𑃒𑃠𑃢𑃐𑃨𑃑—

𑃐𑃠𑃢𑃐𑃑𑃝𑃢𑃙𑃨𑃤 𑃜𑃢𑃙𑃤 𑃛𑃝𑃓𑃘𑃙𑃑𑃤 𑃖𑃙𑃥𑃐𑃨𑃜𑃘𑃦𑃟𑃣
𑃐𑃖𑃟𑃠𑃜𑃢 𑃘𑃤𑃛𑃤𑃐𑃠𑃓 𑃕𑃙𑃨𑃙𑃢𑃛𑃤 𑃓 𑃔𑃨𑃠𑃢𑃑𑃥𑃑𑃙𑃑𑃝𑃖.
𑃜𑃣 𑃐𑃠𑃤𑃘𑃛𑃜𑃦𑃕 𑃛𑃝𑃤𑃑𑃠𑃥 𑃘𑃦𑃥𑃟𑃤𑃟 𑃦𑃨𑃛𑃝𑃖𑃣𑃜𑃢𑃞
