𑊀𑊢𑊡𑊖𑊢𑊢𑊖𑊚𑊚𑊥𑊠𑊢𑊖𑊥𑊖𑊢𑊠

(𑊅) 𑊥𑊚𑊥𑊄𑊢𑊖𑊛𑊑𑊦

𑊟𑊢𑊖𑊡𑊟𑊥𑊡𑊠-𑊀𑊢𑊡𑊖𑊢𑊢𑊖𑊚𑊚𑊥𑊠𑊢𑊖𑊥𑊖𑊢𑊠

𑊟𑊐𑊟𑊥𑊡𑊠-𑊜𑊆𑊥-𑊛𑊀-𑊘𑊄𑊚-𑊠𑊊𑊋𑊆-𑊆𑊥𑊠-𑊌𑊥-𑊥𑊂-𑊒𑊢𑊚-𑊛𑊀-𑊠𑊘

𑊚𑊠𑊦 𑊥𑊢𑊤𑊝𑊘𑊙𑊝𑊙𑊥𑊖𑊖𑊤𑊟𑊡𑊦

(𑊝𑊘𑊙𑊚𑊥𑊠𑊢𑊖𑊦)

𑊁𑊖𑊡𑊛 𑊝𑊘𑊙 𑊟𑊆𑊤𑊚𑊥𑊖𑊗𑊆𑊖𑊢𑊦𑊚 𑊥𑊠𑊡𑊄𑊥𑊚𑊝𑊘𑊙 𑊤𑊘𑊡𑊊𑊢𑊕𑊥𑊠𑊛𑊚𑊚𑊦 𑊥𑊆𑊖 𑊣𑊄𑊤𑊘𑊚𑊖𑊖𑊢𑊦 𑊛𑊢𑊥𑊘𑊠𑊡𑊥𑊢𑊗𑊦 𑊥𑊥𑊖 𑊘𑊤𑊠𑊚𑊥𑊡𑊕𑊚 𑊝𑊘𑊙 𑊟𑊆𑊤𑊚𑊖𑊁।

𑊚𑊥𑊡𑊚𑊘𑊦 𑊥 𑊖𑊗𑊆𑊖𑊦 𑊛𑊕𑊡𑊚𑊠, 𑊀𑊤𑊛𑊢𑊕𑊥𑊦 𑊄𑊥𑊣𑊠𑊣𑊚𑊠, 𑊀𑊣𑊚𑊄𑊢𑊖𑊦 𑊄𑊥𑊚𑊖𑊡, 𑊀𑊣𑊡𑊦 𑊛𑊕𑊡𑊚𑊙𑊚𑊚𑊠, 𑊊𑊖𑊢𑊖𑊚𑊤𑊡𑊏𑊌𑊚𑊦 𑊄𑊥𑊠𑊖 𑊣𑊄𑊥𑊕𑊦, 𑊛𑊢𑊖𑊢𑊛𑊂 𑊆𑊊𑊢𑊕, 𑊀𑊛𑊢𑊖𑊄𑊣𑊂 𑊘𑊢𑊥𑊚𑊚, 𑊀𑊟𑊢𑊖𑊦 𑊥𑊢𑊘𑊙(𑊥𑊘𑊙)𑊙𑊠𑊄𑊖𑊚𑊠, 𑊀𑊚𑊟𑊟𑊖𑊦 𑊛𑊢𑊌𑊏𑊡, 𑊀𑊚𑊤𑊠𑊢𑊘𑊚𑊡 𑊝𑊣𑊦, 𑊥𑊥𑊖 𑊥𑊢𑊤𑊥𑊖𑊖𑊤𑊚𑊠, 𑊛𑊖 𑊝𑊙𑊥𑊖𑊖𑊤𑊚𑊠, 𑊢𑊌 𑊀𑊢𑊡𑊛𑊘𑊆𑊣𑊚𑊠, 𑊥𑊢𑊗𑊤𑊦𑊦 𑊚𑊢𑊤𑊕𑊚𑊆𑊢𑊥𑊠𑊛𑊢𑊥𑊗𑊖𑊚𑊠, 𑊀𑊛𑊢𑊠𑊡 𑊌𑊏𑊚𑊚, 𑊀𑊊𑊚𑊖𑊡𑊦 𑊛𑊢𑊖𑊟𑊚𑊚, 𑊤𑊥𑊘𑊙𑊦 𑊥𑊤𑊢𑊕, 𑊀𑊥𑊤𑊘𑊚𑊡 𑊈𑊥𑊕, 𑊀𑊥𑊊𑊚𑊄 𑊢𑊛𑊕, 𑊀𑊛𑊢𑊖𑊥𑊠𑊦 𑊄𑊡𑊚, 𑊀𑊣𑊛𑊖𑊦 𑊄𑊠𑊦, 𑊀𑊚𑊛𑊣𑊛𑊖 𑊢𑊛𑊦 𑊀𑊥𑊚𑊥𑊢𑊥𑊐 𑊀𑊢𑊛𑊡𑊦, 𑊤𑊛𑊢𑊠𑊄𑊖𑊦 𑊥𑊄𑊚𑊙𑊟𑊡𑊦, 𑊤𑊥𑊠𑊛𑊢𑊡𑊄𑊖 𑊙𑊖𑊟𑊦, 𑊥𑊚𑊤𑊢𑊖 𑊀𑊡𑊖𑊚𑊦, 𑊛𑊢𑊊𑊋𑊚𑊚 𑊆𑊢𑊚𑊗𑊦, 𑊤𑊠𑊄𑊖𑊦 𑊛𑊢𑊘𑊈𑊚𑊦, 𑊛𑊢𑊠𑊄𑊖𑊥𑊖𑊢𑊥𑊕𑊡, 𑊂𑊈𑊘𑊖𑊖𑊢𑊕𑊦 𑊛𑊢𑊛𑊢𑊕 𑊌𑊏𑊚𑊚, 𑊛𑊢𑊖𑊥𑊑𑊖𑊖𑊖𑊚𑊆𑊖𑊛𑊢𑊖𑊡𑊖𑊛𑊚𑊚𑊚𑊚 𑊝𑊘𑊙𑊚𑊚 𑊟𑊆𑊤𑊖𑊚 𑊌𑊏𑊚, 𑊀𑊛𑊢𑊖𑊥𑊑𑊖 𑊚𑊢𑊤𑊕, 𑊥𑊗𑊖 𑊟𑊖𑊄𑊐𑊡𑊠, 𑊥𑊗𑊖𑊦 𑊥𑊢𑊤𑊥𑊖𑊖𑊤𑊣𑊄𑊚𑊡𑊡𑊚 𑊟𑊠, 𑊥𑊢𑊤 𑊁𑊠 𑊖𑊗𑊆𑊖𑊚𑊚 𑊤𑊥𑊥𑊖𑊦 𑊥𑊠𑊡𑊆 𑊆𑊕𑊦।

(𑊙𑊢𑊠𑊚𑊥𑊠𑊢𑊖𑊦)

𑊥𑊘𑊙𑊢𑊠𑊥𑊖 𑊀𑊘 𑊄𑊣𑊡𑊕𑊦, 𑊠𑊙𑊡 𑊄𑊣𑊡𑊕𑊦, 𑊛𑊢𑊡𑊤𑊥𑊚 𑊄𑊣𑊡𑊕𑊦, 𑊥𑊤𑊢𑊗𑊦 𑊥𑊤𑊡𑊏𑊌𑊚𑊦, 𑊄𑊤𑊣𑊦, 𑊛𑊢𑊛𑊢𑊕𑊦, 𑊛𑊢𑊥𑊘𑊙𑊦, 𑊛𑊢𑊡𑊤𑊘𑊖𑊦, 𑊥𑊤𑊅𑊡𑊖𑊦 𑊟𑊆𑊤𑊖 𑊙𑊢𑊠𑊦, 𑊥𑊚𑊘𑊢𑊥𑊐𑊄𑊦, 𑊚𑊢𑊌𑊤𑊢𑊦, 𑊀𑊄𑊣𑊄𑊦, 𑊂𑊛𑊚𑊡𑊄𑊦, 𑊃𑊦𑊛𑊥𑊡𑊄𑊦, 𑊛𑊢𑊖𑊡𑊖𑊠𑊤𑊘𑊚𑊡 𑊤𑊌𑊏𑊦, 𑊥𑊤𑊅𑊡𑊖 𑊟𑊆𑊤𑊖 𑊙𑊢𑊠𑊤𑊚𑊡𑊦 𑊥𑊛𑊢𑊤𑊘𑊖𑊦 𑊚𑊢𑊡𑊕𑊄𑊦, 𑊥𑊚𑊝𑊙𑊆𑊠, 𑊀𑊟𑊚𑊚𑊦 𑊥𑊚𑊥𑊖𑊛𑊦, 𑊥𑊛𑊢𑊖𑊥𑊢𑊕𑊦, 𑊋𑊚𑊚𑊛𑊣𑊖𑊄𑊦।

(𑊥𑊚𑊈𑊚𑊥𑊠𑊢𑊖𑊦)

𑊥𑊛𑊢𑊖𑊛𑊚𑊚 𑊟𑊆𑊤𑊖 𑊀𑊢𑊡𑊥𑊚𑊈𑊦, 𑊚𑊡𑊡𑊛𑊢𑊖𑊛𑊚𑊚𑊦, 𑊢𑊌𑊛𑊢𑊖𑊛𑊚𑊚𑊦, 𑊥𑊠𑊊𑊛𑊢𑊖𑊛𑊚𑊚𑊦, 𑊀𑊏𑊌𑊣𑊄𑊢𑊕𑊡𑊦, 𑊥𑊠𑊊𑊄𑊢𑊕𑊡𑊦, 𑊛𑊕𑊡𑊥𑊢𑊄𑊥𑊖𑊢𑊦, 𑊠𑊦𑊘𑊄𑊥𑊕𑊛𑊢𑊥𑊙𑊄𑊦, 𑊛𑊢𑊦𑊤𑊚𑊡𑊦, 𑊀𑊦𑊤𑊚𑊡𑊦।
