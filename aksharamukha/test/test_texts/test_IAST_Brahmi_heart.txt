𑀆𑀭𑁆𑀬𑀢𑁆𑀭𑀺𑀭𑀢𑁆𑀦𑀸𑀦𑀼𑀲𑁆𑀫𑀾𑀢𑀺𑀲𑀽𑀢𑁆𑀭𑀫𑁆

(𑀔) 𑀲𑀁𑀲𑁆𑀓𑀾𑀢𑀧𑀸𑀞𑀂

𑀪𑀸𑀭𑀢𑀻𑀬𑀪𑀸𑀱𑀸𑀬𑀸𑀫𑁆-𑀆𑀭𑁆𑀬𑀢𑁆𑀭𑀺𑀭𑀢𑁆𑀦𑀸𑀦𑀼𑀲𑁆𑀫𑀾𑀢𑀺𑀲𑀽𑀢𑁆𑀭𑀫𑁆

𑀪𑁄𑀝𑀪𑀸𑀱𑀸𑀬𑀸𑀫𑁆-𑀨𑀕𑁆𑀲𑁆-𑀧𑀸-𑀤𑀓𑁄𑀦𑁆-𑀫𑁆𑀘𑁆𑀙𑁄𑀕𑁆-𑀕𑁆𑀲𑀼𑀫𑁆-𑀚𑁂𑀲𑁆-𑀲𑀼-𑀟𑁆𑀭𑀦𑁆-𑀧𑀸-𑀫𑁆𑀤𑁄

𑀦𑀫𑀂 𑀲𑀭𑁆𑀯𑀩𑀼𑀤𑁆𑀥𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑁂𑀪𑁆𑀬𑀂

(𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀫𑀾𑀢𑀺𑀂)

𑀇𑀢𑁆𑀬𑀧𑀺 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀁𑀲𑁆𑀢𑀣𑀸𑀕𑀢𑁄𑀭𑁆𑀳𑀦𑁆 𑀲𑀫𑁆𑀬𑀓𑁆𑀲𑀁𑀩𑀼𑀤𑁆𑀥𑁄 𑀯𑀺𑀤𑁆𑀬𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀂 𑀲𑀼𑀕𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀤𑀦𑀼𑀢𑁆𑀢𑀭𑀂 𑀧𑀼𑀭𑀼𑀱𑀤𑀫𑁆𑀬𑀲𑀸𑀭𑀣𑀺𑀂 𑀰𑀸𑀲𑁆𑀢𑀸 𑀤𑁂𑀯𑀫𑀦𑀼𑀱𑁆𑀬𑀸𑀡𑀸𑀁 𑀩𑀼𑀤𑁆𑀥𑁄 𑀪𑀕𑀯𑀸𑀦𑀺𑀢𑀺𑁇

𑀦𑀺𑀱𑁆𑀬𑀦𑁆𑀤𑀂 𑀲 𑀢𑀣𑀸𑀕𑀢𑀂 𑀧𑀼𑀡𑁆𑀬𑀸𑀦𑀸𑀫𑁆, 𑀅𑀯𑀺𑀧𑁆𑀭𑀡𑀸𑀰𑀂 𑀓𑀼𑀰𑀮𑀫𑀽𑀮𑀸𑀦𑀸𑀫𑁆, 𑀅𑀮𑀗𑁆𑀓𑀾𑀢𑀂 𑀓𑁆𑀱𑀸𑀦𑁆𑀢𑁆𑀬𑀸, 𑀆𑀮𑀬𑀂 𑀧𑀼𑀡𑁆𑀬𑀦𑀺𑀥𑀸𑀦𑀸𑀦𑀸𑀫𑁆, 𑀘𑀺𑀢𑁆𑀭𑀺𑀢𑁄𑀦𑀼𑀯𑁆𑀬𑀜𑁆𑀚𑀦𑁃𑀂 𑀓𑀼𑀲𑀼𑀫𑀺𑀢𑁄 𑀮𑀓𑁆𑀱𑀡𑁃𑀂, 𑀧𑁆𑀭𑀢𑀺𑀭𑀽𑀧𑁄 𑀕𑁄𑀘𑀭𑁂𑀡, 𑀅𑀧𑁆𑀭𑀢𑀺𑀓𑀽𑀮𑁄 𑀤𑀭𑁆𑀰𑀦𑁂𑀦, 𑀅𑀪𑀺𑀭𑀢𑀺𑀂 𑀰𑁆𑀭𑀤𑁆𑀥𑀸(𑀰𑀼𑀤𑁆𑀥𑀸)𑀥𑀺𑀫𑀼𑀓𑁆𑀢𑀸𑀦𑀸𑀫𑁆, 𑀅𑀦𑀪𑀺𑀪𑀽𑀢𑀂 𑀧𑁆𑀭𑀚𑁆𑀜𑀬𑀸, 𑀅𑀦𑀯𑀫𑀭𑁆𑀤𑀦𑀻𑀬𑁄 𑀩𑀮𑁃𑀂, 𑀰𑀸𑀲𑁆𑀢𑀸 𑀲𑀭𑁆𑀯𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀦𑀸𑀫𑁆, 𑀧𑀺𑀢𑀸 𑀩𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀦𑀸𑀫𑁆, 𑀭𑀸𑀚𑀸 𑀆𑀭𑁆𑀬𑀧𑀼𑀤𑁆𑀕𑀮𑀸𑀦𑀸𑀫𑁆, 𑀲𑀸𑀭𑁆𑀣𑀯𑀸𑀳𑀂 𑀦𑀺𑀭𑁆𑀯𑀸𑀡𑀦𑀕𑀭𑀲𑀫𑁆𑀧𑁆𑀭𑀲𑁆𑀣𑀺𑀢𑀸𑀦𑀸𑀫𑁆, 𑀅𑀧𑁆𑀭𑀫𑁂𑀬𑁄 𑀚𑁆𑀜𑀸𑀦𑁂𑀦, 𑀅𑀘𑀺𑀦𑁆𑀢𑁆𑀬𑀂 𑀧𑁆𑀭𑀢𑀺𑀪𑀸𑀦𑁂𑀦, 𑀯𑀺𑀰𑀼𑀤𑁆𑀥𑀂 𑀲𑁆𑀯𑀭𑁂𑀡, 𑀆𑀲𑁆𑀯𑀤𑀦𑀻𑀬𑁄 𑀖𑁄𑀱𑁂𑀡, 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀭𑀽𑀧𑁂𑀡, 𑀅𑀧𑁆𑀭𑀢𑀺𑀲𑀫𑀂 𑀓𑀸𑀬𑁂𑀦, 𑀅𑀮𑀺𑀧𑁆𑀢𑀂 𑀓𑀸𑀫𑁃𑀂, 𑀅𑀦𑀼𑀧𑀮𑀺𑀧𑁆𑀢𑁄 𑀭𑀽𑀧𑁃𑀂 𑀅𑀲𑀁𑀲𑀾𑀱𑁆𑀝 𑀆𑀭𑀽𑀧𑁆𑀬𑁃𑀂, 𑀯𑀺𑀧𑁆𑀭𑀫𑀼𑀓𑁆𑀢𑀂 𑀲𑁆𑀓𑀦𑁆𑀥𑁂𑀪𑁆𑀬𑀂, 𑀯𑀺𑀲𑀫𑁆𑀧𑁆𑀭𑀬𑀼𑀓𑁆𑀢𑁄 𑀥𑀸𑀢𑀼𑀪𑀺𑀂, 𑀲𑀁𑀯𑀾𑀢 𑀆𑀬𑀢𑀦𑁃𑀂, 𑀧𑁆𑀭𑀘𑁆𑀙𑀺𑀦𑁆𑀦𑁄 𑀕𑁆𑀭𑀦𑁆𑀣𑁃𑀂, 𑀯𑀺𑀫𑀼𑀓𑁆𑀢𑀂 𑀧𑀭𑀺𑀤𑀸𑀖𑁆𑀦𑁃𑀂, 𑀧𑀭𑀺𑀫𑀼𑀓𑁆𑀢𑀲𑁆𑀢𑀾𑀱𑁆𑀡𑀬𑀸, 𑀑𑀖𑀸𑀤𑀼𑀢𑁆𑀢𑀻𑀭𑁆𑀡𑀂 𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑁄 𑀚𑁆𑀜𑀸𑀦𑁂𑀦, 𑀧𑁆𑀭𑀢𑀺𑀱𑁆𑀞𑀺𑀢𑁄𑀢𑀻𑀢𑀸𑀦𑀸𑀕𑀢𑀧𑁆𑀭𑀢𑁆𑀬𑀼𑀢𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀁 𑀩𑀼𑀤𑁆𑀥𑀸𑀦𑀸𑀁 𑀪𑀕𑀯𑀢𑀸𑀁 𑀚𑁆𑀜𑀸𑀦𑁂, 𑀅𑀧𑁆𑀭𑀢𑀺𑀱𑁆𑀞𑀺𑀢𑁄 𑀦𑀺𑀭𑁆𑀯𑀸𑀡𑁂, 𑀲𑁆𑀣𑀺𑀢𑁄 𑀪𑀽𑀢𑀓𑁄𑀝𑁆𑀬𑀸𑀫𑁆, 𑀲𑁆𑀣𑀺𑀢𑀂 𑀲𑀭𑁆𑀯𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀮𑁄𑀓𑀦𑀻𑀬𑀸𑀬𑀸𑀁 𑀪𑀽𑀫𑁅, 𑀲𑀭𑁆𑀯 𑀇𑀫𑁂 𑀢𑀣𑀸𑀕𑀢𑀸𑀦𑀸𑀁 𑀯𑀺𑀰𑁂𑀱𑀢𑀂 𑀲𑀫𑁆𑀬𑀕𑁆 𑀕𑀼𑀡𑀸𑀂𑁇

(𑀥𑀭𑁆𑀫𑀸𑀦𑀼𑀲𑁆𑀫𑀾𑀢𑀺𑀂)

𑀲𑀤𑁆𑀥𑀭𑁆𑀫𑀲𑁆𑀢𑀼 𑀆𑀤𑁅 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀫𑀥𑁆𑀬𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀧𑀭𑁆𑀬𑀯𑀲𑀸𑀦𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀲𑁆𑀯𑀭𑁆𑀣𑀂 𑀲𑀼𑀯𑁆𑀬𑀜𑁆𑀚𑀦𑀂, 𑀓𑁂𑀯𑀮𑀂, 𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑀂, 𑀧𑀭𑀺𑀰𑀼𑀤𑁆𑀥𑀂, 𑀧𑀭𑁆𑀬𑀯𑀤𑀸𑀢𑀂, 𑀲𑁆𑀯𑀸𑀔𑁆𑀬𑀸𑀢𑀂 𑀪𑀕𑀯𑀢𑁄 𑀥𑀭𑁆𑀫𑀂, 𑀲𑀸𑀦𑁆𑀤𑀾𑀱𑁆𑀝𑀺𑀓𑀂, 𑀦𑀺𑀭𑁆𑀚𑁆𑀯𑀭𑀂, 𑀆𑀓𑀸𑀮𑀺𑀓𑀂, 𑀒𑀧𑀦𑀸𑀬𑀺𑀓𑀂, 𑀐𑀳𑀺𑀧𑀰𑁆𑀬𑀺𑀓𑀂, 𑀧𑁆𑀭𑀢𑁆𑀬𑀸𑀢𑁆𑀫𑀯𑁂𑀤𑀦𑀻𑀬𑁄 𑀯𑀺𑀚𑁆𑀜𑁃𑀂, 𑀲𑁆𑀯𑀸𑀔𑁆𑀬𑀸𑀢𑁄 𑀪𑀕𑀯𑀢𑁄 𑀥𑀭𑁆𑀫𑀯𑀺𑀦𑀬𑀂 𑀲𑀼𑀧𑁆𑀭𑀯𑁂𑀤𑀺𑀢𑀂 𑀦𑁃𑀭𑁆𑀬𑀸𑀡𑀺𑀓𑀂, 𑀲𑀁𑀩𑁄𑀥𑀺𑀕𑀸𑀫𑀻, 𑀅𑀪𑀺𑀦𑁆𑀦𑀂 𑀲𑀁𑀲𑁆𑀢𑀽𑀧𑀂, 𑀲𑀧𑁆𑀭𑀢𑀺𑀰𑀭𑀡𑀂, 𑀙𑀺𑀦𑁆𑀦𑀧𑁆𑀮𑁄𑀢𑀺𑀓𑀂𑁇

(𑀲𑀁𑀖𑀸𑀦𑀼𑀲𑁆𑀫𑀾𑀢𑀺𑀂)

𑀲𑀼𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑁄 𑀪𑀕𑀯𑀢 𑀆𑀭𑁆𑀬𑀲𑀁𑀖𑀂, 𑀦𑁆𑀬𑀸𑀬𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀋𑀚𑀼𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀲𑀸𑀫𑀻𑀘𑀻𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀅𑀜𑁆𑀚𑀮𑀻𑀓𑀭𑀡𑀻𑀬𑀂, 𑀲𑀸𑀫𑀻𑀘𑀻𑀓𑀭𑀡𑀻𑀬𑀂, 𑀧𑀼𑀡𑁆𑀬𑀰𑁆𑀭𑀻𑀓𑁆𑀱𑁂𑀢𑁆𑀭𑀂, 𑀫𑀳𑀸𑀤𑀓𑁆𑀱𑀺𑀡𑀸𑀧𑀭𑀺𑀰𑁄𑀥𑀓𑀂, 𑀧𑁆𑀭𑀸𑀳𑀯𑀦𑀻𑀬𑀂, 𑀆𑀳𑀯𑀦𑀻𑀬𑀂𑁇
