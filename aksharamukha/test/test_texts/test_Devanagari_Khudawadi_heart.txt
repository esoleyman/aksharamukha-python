𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋞𑋪𑋙𑋡𑋏𑋘𑋝𑋤𑋍𑋪𑋙𑋗𑋪।

[𑋝𑋟𑊺𑋪𑋜𑋩𑋡𑋒𑋪𑋍𑋗𑋠𑋍𑋪𑋙𑋡𑊺𑋠]

॥ 𑋑𑋗𑋞𑋪 𑋝𑋙𑋪𑋛𑋂𑋪𑋅𑋠𑋘 ॥

॥ 𑊸𑋟 𑋑𑋗𑋧 𑋖𑊼𑋛𑋍𑋪𑋘𑋦 𑊱𑋙𑋪𑋘𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋘𑋦 ॥

𑊱𑋙𑋪𑋘𑋠𑋛𑋚𑋧𑊺𑋡𑋍𑋥𑋜𑋪𑋛𑋙𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋧 𑊼𑋗𑋪𑋖𑋢𑋙𑋠𑋘𑋠𑋟 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋘𑋠𑋟 𑋀𑋙𑋪𑋘𑋠𑋟 𑋀𑋙𑋗𑋠𑋌𑋧 𑋛𑋪𑋘𑋛𑋚𑋧𑊺𑋘𑋍𑋡 𑋝𑋪𑋗। 𑋒𑋅𑋪𑋀 𑋝𑋪𑊺𑋑𑋪𑋐𑋠𑋞𑋪, 𑋍𑋠𑋟𑋜𑋪𑋀 𑋝𑋪𑋛𑋖𑋠𑋛𑋜𑋤𑋑𑋪𑋘𑋠𑋑𑋪 𑋒𑋜𑋪𑋘𑋍𑋡 𑋝𑋪𑋗॥

𑊲𑋞 𑋜𑋠𑋙𑋡𑋒𑋣𑋍𑋪𑋙 𑋙𑋤𑋒𑋟 𑋜𑋤𑋑𑋪𑋘𑋍𑋠, 𑋜𑋤𑋑𑋪𑋘𑋍𑋦𑋛 𑋙𑋤𑋒𑋗𑋪। 𑋙𑋤𑋒𑋠𑋑𑋪𑋑 𑋒𑋪𑋙𑋡𑋎𑊺𑋪 𑋜𑋤𑋑𑋪𑋘𑋍𑋠, 𑋜𑋤𑋑𑋪𑋘𑋍𑋠𑋘𑋠 𑋑 𑋒𑋪𑋙𑋡𑋎𑊼𑋪 𑋙𑋤𑋒𑋗𑋪। 𑋘𑋏𑋪𑋙𑋤𑋒𑋟 𑋝𑋠 𑋜𑋤𑋑𑋪𑋘𑋍𑋠, 𑋘𑋠 𑋜𑋤𑋑𑋪𑋘𑋍𑋠 𑋍𑋏𑋪𑋙𑋤𑋒𑋗𑋪॥

𑊶𑋛𑋗𑋥𑋛 𑋛𑋥𑋏𑋑𑋠𑋝𑋟𑋂𑋪𑋅𑋠𑋝𑋟𑋝𑋪𑊺𑋠𑋙𑋛𑋡𑋂𑋪𑋅𑋠𑋑𑋠𑋑𑋡॥

𑊲𑋞𑋟 𑋜𑋠𑋙𑋡𑋒𑋣𑋍𑋪𑋙 𑋝𑋙𑋪𑋛𑋐𑋙𑋪𑋗𑋠𑋞𑋪 𑋜𑋤𑋑𑋪𑋘𑋍𑋠𑋚𑊺𑋪𑋜𑋩𑋌𑋠 𑊰𑋑𑋣𑋍𑋪𑋒𑋑𑋪𑋑𑋠 𑊰𑋑𑋡𑋙𑋣𑋏𑋪𑋐𑋠 𑊰𑋗𑋚𑋠 𑋑 𑋛𑋡𑋗𑋚𑋠 𑋑𑋧𑋑𑋠 𑋑 𑋒𑋙𑋡𑋒𑋤𑋙𑋪𑋌𑋠𑋞𑋪। 𑋍𑋝𑋪𑋗𑋠𑋀𑋪𑋁𑋠𑋙𑋡𑋒𑋣𑋍𑋪𑋙 𑋜𑋤𑋑𑋪𑋘𑋍𑋠𑋘𑋠𑋟 𑋑 𑋙𑋤𑋒𑋗𑋪, 𑋑 𑋛𑋥𑋏𑋑𑋠, 𑋑 𑋝𑋟𑋂𑋪𑋅𑋠, 𑋑 𑋝𑋟𑋝𑋪𑊺𑋠𑋙𑋠𑋞𑋪, 𑋑 𑋛𑋡𑋂𑋪𑋅𑋠𑋑𑋠𑋑𑋡। 𑋑 𑋀𑊺𑋪𑋜𑋩𑋣𑋞𑋪𑋜𑋪𑋙𑋧𑋍𑋪𑋙𑊾𑋪𑋙𑋠𑋌𑋂𑋡𑋞𑋪𑋛𑋠𑊺𑋠𑋘𑋗𑋑𑋠𑋟𑋝𑋡, 𑋑 𑋙𑋤𑋒𑋜𑋔𑋪𑋏𑊼𑋑𑋪𑋐𑋙𑋝𑋝𑋪𑋒𑋪𑋙𑋜𑋩𑋪𑋆𑋛𑋪𑋘𑋐𑋙𑋪𑋗𑋠𑋞𑋪। 𑋑 𑋀𑊺𑋪𑋜𑋩𑋣𑋙𑋪𑋐𑋠𑋍𑋣𑋙𑋪𑋘𑋠𑋛𑋑𑋪𑋑 𑋗𑋑𑋧𑋐𑋠𑋍𑋣𑋞𑋪॥

𑋑 𑋛𑋡𑋏𑋪𑋘𑋠 𑋑𑋠𑋛𑋡𑋏𑋪𑋘𑋠 𑋑 𑋛𑋡𑋏𑋪𑋘𑋠𑊺𑋪𑋜𑋩𑋘𑋧 𑋑𑋠𑋛𑋡𑋏𑋪𑋘𑋠𑊺𑋪𑋜𑋩𑋘𑋧 𑋘𑋠𑋛𑋑𑋪𑋑 𑋂𑋙𑋠𑋗𑋙𑋌𑋟 𑋑 𑋂𑋙𑋠𑋗𑋙𑋌𑊺𑋪𑋜𑋩𑋘𑋧 𑋑 𑋏𑋣𑋞𑋪𑊻𑋝𑋗𑋣𑋏𑋘𑋑𑋡𑋙𑋧𑋐𑋗𑋠𑋙𑋪𑊼𑋠 𑋑 𑋂𑋪𑋅𑋠𑋑𑋟 𑋑 𑋒𑋪𑋙𑋠𑋒𑋪𑋍𑋡𑋍𑋪𑋛𑋗𑋪॥

𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋝𑋪𑋘 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋗𑋠𑋜𑋪𑋙𑋡𑋍𑋪𑋘 𑋛𑋡𑋞𑋙𑋍𑋡 𑋀𑋡𑋍𑋪𑋍𑋠𑋛𑋙𑋌𑋞𑋪। 𑋀𑋡𑋍𑋪𑋍𑋠𑋛𑋙𑋌𑋑𑋠𑋝𑋪𑋍𑋡𑋍𑋪𑋛𑋠𑋏𑋍𑋪𑋙𑋝𑋪𑋍𑋧 𑋛𑋡𑋒𑋙𑋪𑋘𑋠𑋝𑋠𑋍𑋡𑊺𑋪𑋙𑋠𑋑𑋪𑋍𑋧 𑋑𑋡𑋜𑋩𑋪𑋇𑋑𑋡𑋙𑋪𑋛𑋠𑋌𑋞𑋪। 𑋍𑋪𑋙𑋪𑋘𑋐𑋪𑋛𑋛𑋪𑋘𑋛𑋝𑋪𑋎𑋡𑋍𑋠𑋞𑋪 𑋝𑋙𑋪𑋛𑋔𑋣𑋏𑋪𑋐𑋠𑋞𑋪 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋗𑋠𑋜𑋪𑋙𑋡𑋍𑋪𑋘 𑊰𑋑𑋣𑋍𑋪𑋍𑋙𑋠𑋟 𑋝𑋗𑋪𑋘𑊺𑋪𑋝𑋟𑋔𑋧𑋐𑋡𑋗𑋖𑋡𑋝𑋟𑋔𑋣𑋏𑋪𑋐𑋠𑋞𑋪॥

𑋍𑋝𑋪𑋗𑋠𑋂𑋪𑋂𑋪𑋅𑋠𑋍𑋛𑋪𑋘𑋞𑋪 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋗𑋞𑋠𑋗𑋑𑋪𑋍𑋪𑋙𑋧 𑋗𑋞𑋠𑋛𑋡𑋏𑋪𑋘𑋠𑋗𑋑𑋪𑋍𑋪𑋙𑋧ऽ𑋑𑋣𑋍𑋪𑋍𑋙𑋗𑋑𑋪𑋍𑋪𑋙𑋧ऽ𑋝𑋗𑋝𑋗𑋗𑋑𑋪𑋍𑋪𑋙𑋞𑋪 𑋝𑋙𑋪𑋛𑋏𑋣𑋞𑋪𑊻𑋒𑋪𑋙𑋜𑋗𑋑𑋞𑋪 𑋝𑋍𑋪𑋘𑋗𑋗𑋡𑋎𑋪𑋘𑋍𑋪𑋛𑋠𑋍𑋪 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋘𑋠𑋗𑋣𑊺𑋪𑋍𑋧 𑋗𑋑𑋪𑋍𑋪𑋙𑋞𑋪।

𑋍𑋏𑋪𑋘𑋎𑋠- 𑊼𑋍𑋥 𑊼𑋍𑋥 𑋒𑋠𑋙𑊼𑋍𑋥 𑋒𑋠𑋙𑋝𑋟𑊼𑋍𑋥 𑋔𑋧𑋐𑋡 𑋝𑋪𑋛𑋠𑋞𑋠॥

𑊲𑋍𑋡 𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋒𑋠𑋙𑋗𑋡𑋍𑋠𑋞𑋪𑋙𑋡𑋏𑋘𑋝𑋤𑋍𑋪𑋙𑋟 𑋝𑋗𑋠𑋒𑋪𑋍𑋗𑋪॥
