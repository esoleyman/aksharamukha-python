𑌲𑌲𑌿𑌤𑌵𑌿𑌸𑍍𑌤𑌰𑌃

𑌨𑌮𑌃 𑌸𑌰𑍍𑌵𑌬𑍁𑌦𑍍𑌧𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍇𑌭𑍍𑌯𑌃।

॥ 𑍐 𑌨𑌮𑍋 𑌦𑌶𑌦𑌿𑌗𑌨𑌨𑍍𑌤𑌾𑌪𑌰𑍍𑌯𑌨𑍍𑌤𑌲𑍋𑌕𑌧𑌾𑌤𑍁𑌪𑍍𑌰𑌤𑌿𑌷𑍍𑌠𑌿𑌤𑌸𑌰𑍍𑌵𑌬𑍁𑌦𑍍𑌧𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌾𑌰𑍍𑌯𑌶𑍍𑌰𑌾𑌵𑌕𑌪𑍍𑌰𑌤𑍍𑌯𑍇𑌕𑌬𑍁𑌦𑍍𑌧𑍇𑌭𑍍𑌯𑍋𑌽𑌤𑍀𑌤𑌾𑌨𑌾𑌗𑌤𑌪𑍍𑌰𑌤𑍍𑌯𑍁𑌤𑍍𑌪𑌨𑍍𑌨𑍇𑌭𑍍𑌯𑌃॥

௧௦ 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌸𑌂𑌦𑌰𑍍𑌶𑌨𑌪𑌰𑌿𑌵𑌰𑍍𑌤𑍋 𑌦𑌶𑌮𑌃।

𑌇𑌤𑌿 𑌹𑌿 𑌭𑌿𑌕𑍍𑌷𑌵𑌃 𑌸𑌂𑌵𑍃𑌦𑍍𑌧𑌃 𑌕𑍁𑌮𑌾𑌰𑌃। 𑌤𑌦𑌾 𑌮𑌾𑌙𑍍𑌗𑌲𑍍𑌯𑌶𑌤𑌸𑌹𑌸𑍍𑌰𑍈𑌃 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍁𑌪𑌨𑍀𑌯𑌤𑍇 𑌸𑍍𑌮 𑌦𑌶𑌭𑌿𑌰𑍍𑌦𑌾𑌰𑌕𑌸𑌹𑌸𑍍𑌰𑍈𑌃 𑌪𑌰𑌿𑌵𑍃𑌤𑌃 𑌪𑍁𑌰𑌸𑍍𑌕𑍃𑌤𑌃, 𑌦𑌶𑌭𑌿𑌶𑍍𑌚 𑌰𑌥𑌸𑌹𑌸𑍍𑌰𑍈𑌃 𑌖𑌾𑌦𑌨𑍀𑌯𑌭𑍋𑌜𑌨𑍀𑌯𑌸𑍍𑌵𑌾𑌦𑌨𑍀𑌯𑌪𑌰𑌿𑌪𑍂𑌰𑍍𑌣𑍈𑌰𑍍𑌹𑌿𑌰𑌣𑍍𑌯𑌸𑍁𑌵𑌰𑍍𑌣𑌪𑌰𑌿𑌪𑍂𑌰𑍍𑌣𑍈𑌶𑍍𑌚। 𑌯𑍇𑌨 𑌕𑌪𑌿𑌲𑌵𑌸𑍍𑌤𑍁𑌨𑌿 𑌮𑌹𑌾𑌨𑌗𑌰𑍇 𑌵𑍀𑌥𑌿𑌚𑌤𑍍𑌵𑌰𑌰𑌥𑍍𑌯𑌾𑌨𑍍𑌤𑌰𑌾𑌪𑌣𑌮𑍁𑌖𑍇𑌷𑍍𑌵𑌭𑍍𑌯𑌵𑌕𑍀𑌰𑍍𑌯𑌤𑍇 𑌸𑍍𑌮 𑌅𑌭𑌿𑌵𑌿𑌶𑍍𑌰𑌾𑌮𑍍𑌯𑌨𑍍𑌤𑍇। 𑌅𑌷𑍍𑌟𑌾𑌭𑌿𑌶𑍍𑌚 𑌤𑍂𑌰𑍍𑌯𑌶𑌤𑌸𑌹𑌸𑍍𑌰𑍈𑌃 𑌪𑍍𑌰𑌘𑍁𑌷𑍍𑌯𑌮𑌾𑌣𑍈𑌰𑍍𑌮𑌹𑌤𑌾 𑌚 𑌪𑍁𑌷𑍍𑌪𑌵𑌰𑍍𑌷𑍇𑌣𑌾𑌭𑌿𑌪𑍍𑌰𑌵𑌰𑍍𑌷𑌤𑌾 𑌵𑌿𑌤𑌰𑍍𑌦𑌿𑌨𑌿𑌰𑍍𑌯𑍂𑌹𑌤𑍋𑌰𑌣𑌗𑌵𑌾𑌕𑍍𑌷𑌹𑌰𑍍𑌮𑍍𑌯𑌕𑍂𑌟𑌾𑌗𑌾𑌰𑌪𑍍𑌰𑌾𑌸𑌾𑌦𑌤𑌲𑍇𑌷𑍁 𑌕𑌨𑍍𑌯𑌾𑌶𑌤𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿 𑌸𑌰𑍍𑌵𑌾𑌲𑌂𑌕𑌾𑌰𑌭𑍂𑌷𑌿𑌤𑌾𑌃 𑌸𑍍𑌥𑌿𑌤𑌾 𑌅𑌭𑍂𑌵𑌨𑍍। 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌂 𑌪𑍍𑌰𑍇𑌕𑍍𑌷𑌮𑌾𑌣𑌾𑌃 𑌕𑍁𑌸𑍁𑌮𑌾𑌨𑌿 𑌚 𑌕𑍍𑌷𑌿𑌪𑌨𑍍𑌤𑌿 𑌸𑍍𑌮। 𑌅𑌷𑍍𑌟𑍗 𑌚 𑌮𑌰𑍁𑌤𑍍𑌕𑌨𑍍𑌯𑌾𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿 𑌵𑌿𑌗𑌲𑌿𑌤𑌾𑌲𑌂𑌕𑌾𑌰𑌾𑌭𑌰𑌣𑌾𑌲𑌂𑌕𑍃𑌤𑌾𑌨𑌿 𑌰𑌤𑍍𑌨𑌭𑌦𑍍𑌰𑌂𑌕𑌰𑍇𑌣 𑌗𑍃𑌹𑍀𑌤𑌾𑌨𑌿 𑌮𑌾𑌰𑍍𑌗𑌂 𑌶𑍋𑌧𑌯𑌨𑍍𑌤𑍍𑌯𑍋 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌸𑍍𑌯 𑌪𑍁𑌰𑌤𑍋 𑌗𑌚𑍍𑌛𑌨𑍍𑌤𑌿 𑌸𑍍𑌮। 𑌦𑍇𑌵𑌨𑌾𑌗𑌯𑌕𑍍𑌷𑌗𑌨𑍍𑌧𑌰𑍍𑌵𑌾𑌸𑍁𑌰𑌗𑌰𑍁𑌡𑌕𑌿𑌨𑍍𑌨𑌰𑌮𑌹𑍋𑌰𑌗𑌾𑌶𑍍𑌚𑌾𑌰𑍍𑌧𑌕𑌾𑌯𑌿𑌕𑌾 𑌗𑌗𑌨𑌤𑌲𑌾𑌤𑍍𑌪𑍁𑌷𑍍𑌪𑌪𑌟𑍍𑌟𑌦𑌾𑌮𑌾𑌨𑍍𑌯𑌭𑌿𑌪𑍍𑌰𑌲𑌮𑍍𑌬𑌯𑌨𑍍𑌤𑌿 𑌸𑍍𑌮। 𑌸𑌰𑍍𑌵𑍇 𑌚 𑌶𑌾𑌕𑍍𑌯𑌗𑌣𑌾𑌃 𑌶𑍁𑌦𑍍𑌧𑍋𑌦𑌨𑌂 𑌰𑌾𑌜𑌾𑌨𑌂 𑌪𑍁𑌰𑌸𑍍𑌕𑍃𑌤𑍍𑌯 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌸𑍍𑌯 𑌪𑍁𑌰𑌤𑍋 𑌗𑌚𑍍𑌛𑌨𑍍𑌤𑌿 𑌸𑍍𑌮। 𑌅𑌨𑍇𑌨𑍈𑌵𑌂𑌰𑍂𑌪𑍇𑌣 𑌵𑍍𑌯𑍂𑌹𑍇𑌨 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍋 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍁𑌪𑌨𑍀𑌯𑌤𑍇 𑌸𑍍𑌮॥

𑌸𑌮𑌨𑌨𑍍𑌤𑌰𑌪𑍍𑌰𑌵𑍇𑌶𑌿𑌤𑌶𑍍𑌚 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍋 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍍। 𑌅𑌥 𑌵𑌿𑌶𑍍𑌵𑌾𑌮𑌿𑌤𑍍𑌰𑍋 𑌨𑌾𑌮 𑌦𑌾𑌰𑌕𑌾𑌚𑌾𑌰𑍍𑌯𑍋 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌸𑍍𑌯 𑌶𑍍𑌰𑌿𑌯𑌂 𑌤𑍇𑌜𑌶𑍍𑌚𑌾𑌸𑌹𑌮𑌾𑌨𑍋 𑌧𑌰𑌣𑌿𑌤𑌲𑍇 𑌨𑌿𑌵𑌿𑌷𑍍𑌟𑍋𑌽𑌧𑍋𑌮𑍁𑌖𑌃 𑌪𑍍𑌰𑌪𑌤𑌤𑌿 𑌸𑍍𑌮। 𑌤𑌂 𑌤𑌥𑌾 𑌪𑍍𑌰𑌪𑌤𑌿𑌤𑌂 𑌦𑍃𑌷𑍍𑌟𑍍𑌵𑌾 𑌶𑍁𑌭𑌾𑌙𑍍𑌗𑍋 𑌨𑌾𑌮 𑌤𑍁𑌷𑌿𑌤𑌕𑌾𑌯𑌿𑌕𑍋 𑌦𑍇𑌵𑌪𑍁𑌤𑍍𑌰𑍋 𑌦𑌕𑍍𑌷𑌿𑌣𑍇𑌨 𑌕𑌰𑌤𑌲𑍇𑌨 𑌪𑌰𑌿𑌗𑍃𑌹𑍍𑌯𑍋𑌤𑍍𑌥𑌾𑌪𑌯𑌤𑌿 𑌸𑍍𑌮। 𑌉𑌤𑍍𑌥𑌾𑌪𑍍𑌯 𑌚 𑌗𑌗𑌨𑌤𑌲𑌸𑍍𑌥𑍋 𑌰𑌾𑌜𑌾𑌨𑌂 𑌶𑍁𑌦𑍍𑌧𑍋𑌦𑌨𑌂 𑌤𑌂 𑌚 𑌮𑌹𑌾𑌨𑍍𑌤𑌂 𑌜𑌨𑌕𑌾𑌯𑌂 𑌗𑌾𑌥𑌾𑌭𑌿𑌰𑌭𑍍𑌯𑌭𑌾𑌷𑌤𑍍—

𑌶𑌾𑌸𑍍𑌤𑍍𑌰𑌾𑌣𑌿 𑌯𑌾𑌨𑌿 𑌪𑍍𑌰𑌚𑌲𑌨𑍍𑌤𑌿 𑌮𑌨𑍁𑌷𑍍𑌯𑌲𑍋𑌕𑍇
𑌸𑌂𑌖𑍍𑌯𑌾 𑌲𑌿𑌪𑌿𑌶𑍍𑌚 𑌗𑌣𑌨𑌾𑌪𑌿 𑌚 𑌧𑌾𑌤𑍁𑌤𑌨𑍍𑌤𑍍𑌰𑌮𑍍।
𑌯𑍇 𑌶𑌿𑌲𑍍𑌪𑌯𑍋𑌗 𑌪𑍃𑌥𑍁 𑌲𑍗𑌕𑌿𑌕 𑌅𑌪𑍍𑌰𑌮𑍇𑌯𑌾𑌃
𑌤𑍇𑌷𑍍𑌵𑍇𑌷𑍁 𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑍁 𑌪𑍁𑌰𑌾 𑌬𑌹𑍁𑌕𑌲𑍍𑌪𑌕𑍋𑌟𑍍𑌯𑌃॥௧॥

𑌕𑌿𑌂 𑌤𑍂 𑌜𑌨𑌸𑍍𑌯 𑌅𑌨𑍁𑌵𑌰𑍍𑌤𑌨𑌤𑌾𑌂 𑌕𑌰𑍋𑌤𑌿
𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌮𑌾𑌗𑌤𑍁 𑌸𑍁𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑍁 𑌶𑌿𑌷𑍍𑌯𑌣𑌾𑌰𑍍𑌥𑌮𑍍।
𑌪𑌰𑌿𑌪𑌾𑌚𑌨𑌾𑌰𑍍𑌥 𑌬𑌹𑍁𑌦𑌾𑌰𑌕 𑌅𑌗𑍍𑌰𑌯𑌾𑌨𑍇
𑌅𑌨𑍍𑌯𑌾𑌂𑌶𑍍𑌚 𑌸𑌤𑍍𑌤𑍍𑌵𑌨𑌯𑍁𑌤𑌾𑌨𑌮𑍃𑌤𑍇 𑌵𑌿𑌨𑍇𑌤𑍁𑌮𑍍॥௨॥

𑌲𑍋𑌕𑍋𑌤𑍍𑌤𑌰𑍇𑌷𑍁 𑌚𑌤𑍁𑌸𑌤𑍍𑌯𑌪𑌥𑍇 𑌵𑌿𑌧𑌿𑌜𑍍𑌞𑍋
𑌹𑍇𑌤𑍁𑌪𑍍𑌰𑌤𑍀𑌤𑍍𑌯𑌕𑍁𑌶𑌲𑍋 𑌯𑌥 𑌸𑌂𑌭𑌵𑌨𑍍𑌤𑌿।
𑌯𑌥 𑌚𑌾𑌨𑌿𑌰𑍋𑌧𑌕𑍍𑌷𑌯𑍁 𑌸𑌂𑌸𑍍𑌥𑌿𑌤𑍁 𑌶𑍀𑌤𑌿𑌭𑌾𑌵𑌃
𑌤𑌸𑍍𑌮𑌿𑌨𑍍𑌵𑌿𑌧𑌿𑌜𑍍𑌞 𑌕𑌿𑌮𑌥𑍋 𑌲𑌿𑌪𑌿𑌶𑌾𑌸𑍍𑌤𑍍𑌰𑌮𑌾𑌤𑍍𑌰𑍇॥௩॥

𑌨𑍇𑌤𑌸𑍍𑌯 𑌆𑌚𑌰𑌿𑌯 𑌉𑌤𑍍𑌤𑌰𑌿 𑌵𑌾 𑌤𑍍𑌰𑌿𑌲𑍋𑌕𑍇
𑌸𑌰𑍍𑌵𑍇𑌷𑍁 𑌦𑍇𑌵𑌮𑌨𑍁𑌜𑍇𑌷𑍍𑌵𑌯𑌮𑍇𑌵 𑌜𑍇𑌷𑍍𑌠𑌃।
𑌨𑌾𑌮𑌾𑌪𑌿 𑌤𑍇𑌷 𑌲𑌿𑌪𑌿𑌨𑌾𑌂 𑌨 𑌹𑌿 𑌵𑌿𑌤𑍍𑌥 𑌯𑍂𑌯𑌂
𑌯𑌤𑍍𑌰𑍇𑌷𑍁 𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑍁 𑌪𑍁𑌰𑌾 𑌬𑌹𑍁𑌕𑌲𑍍𑌪𑌕𑍋𑌟𑍍𑌯𑌃॥௪॥

𑌸𑍋 𑌚𑌿𑌤𑍍𑌤𑌧𑌾𑌰 𑌜𑌗𑌤𑌾𑌂 𑌵𑌿𑌵𑌿𑌧𑌾 𑌵𑌿𑌚𑌿𑌤𑍍𑌰𑌾
𑌏𑌕𑌕𑍍𑌷𑌣𑍇𑌨 𑌅𑌯𑍁 𑌜𑌾𑌨𑌤𑌿 𑌶𑍁𑌦𑍍𑌧𑌸𑌤𑍍𑌤𑍍𑌵𑌃।
𑌅𑌦𑍃𑌶𑍍𑌯𑌰𑍂𑌪𑌰𑌹𑌿𑌤𑌸𑍍𑌯 𑌗𑌤𑌿𑌂 𑌚 𑌵𑍇𑌤𑍍𑌤𑌿
𑌕𑌿𑌂 𑌵𑌾 𑌪𑍁𑌨𑍋𑌽𑌥 𑌲𑌿𑌪𑌿𑌨𑍋𑌽𑌕𑍍𑌷𑌰𑌦𑍃𑌶𑍍𑌯𑌰𑍂𑌪𑌾𑌮𑍍॥௫॥

𑌇𑌤𑍍𑌯𑍁𑌕𑍍𑌤𑍍𑌵𑌾 𑌸 𑌦𑍇𑌵𑌪𑍁𑌤𑍍𑌰𑍋 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌂 𑌦𑌿𑌵𑍍𑌯𑍈𑌃 𑌕𑍁𑌸𑍁𑌮𑍈𑌰𑌭𑍍𑌯𑌰𑍍𑌚𑍍𑌯 𑌤𑌤𑍍𑌰𑍈𑌵𑌾𑌨𑍍𑌤𑌰𑍍𑌦𑌧𑍇। 𑌤𑌤𑍍𑌰 𑌧𑌾𑌤𑍍𑌰𑍍𑌯𑌶𑍍𑌚 𑌚𑍇𑌟𑍀𑌵𑌰𑍍𑌗𑌾𑌶𑍍𑌚 𑌸𑍍𑌥𑌾𑌪𑌿𑌤𑌾 𑌅𑌭𑍂𑌵𑌨𑍍। 𑌪𑌰𑌿𑌶𑍇𑌷𑌾𑌃 𑌶𑌾𑌕𑍍𑌯𑌾𑌃 𑌶𑍁𑌦𑍍𑌧𑍋𑌦𑌨𑌪𑍍𑌰𑌮𑍁𑌖𑌾𑌃 𑌪𑍍𑌰𑌕𑍍𑌰𑌾𑌮𑌨𑍍𑌤𑌃॥

𑌅𑌥 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵 𑌉𑌰𑌗𑌸𑌾𑌰𑌚𑌨𑍍𑌦𑌨𑌮𑌯𑌂 𑌲𑌿𑌪𑌿𑌫𑌲𑌕𑌮𑌾𑌦𑌾𑌯 𑌦𑌿𑌵𑍍𑌯𑌾𑌰𑍍𑌷𑌸𑍁𑌵𑌰𑍍𑌣𑌤𑌿𑌰𑌕𑌂 𑌸𑌮𑌨𑍍𑌤𑌾𑌨𑍍𑌮𑌣𑌿𑌰𑌤𑍍𑌨𑌪𑍍𑌰𑌤𑍍𑌯𑍁𑌪𑍍𑌤𑌂 𑌵𑌿𑌶𑍍𑌵𑌾𑌮𑌿𑌤𑍍𑌰𑌮𑌾𑌚𑌾𑌰𑍍𑌯𑌮𑍇𑌵𑌮𑌾𑌹-𑌕𑌤𑌮𑌾𑌂 𑌮𑍇 𑌭𑍋 𑌉𑌪𑌾𑌧𑍍𑌯𑌾𑌯 𑌲𑌿𑌪𑌿𑌂 𑌶𑌿𑌕𑍍𑌷𑌾𑌪𑌯𑌸𑌿। 𑌬𑍍𑌰𑌾𑌹𑍍𑌮𑍀𑌖𑌰𑍋𑌷𑍍𑌟𑍀𑌪𑍁𑌷𑍍𑌕𑌰𑌸𑌾𑌰𑌿𑌂 𑌅𑌙𑍍𑌗𑌲𑌿𑌪𑌿𑌂 𑌵𑌙𑍍𑌗𑌲𑌿𑌪𑌿𑌂 𑌮𑌗𑌧𑌲𑌿𑌪𑌿𑌂 𑌮𑌙𑍍𑌗𑌲𑍍𑌯𑌲𑌿𑌪𑌿𑌂 𑌅𑌙𑍍𑌗𑍁𑌲𑍀𑌯𑌲𑌿𑌪𑌿𑌂 𑌶𑌕𑌾𑌰𑌿𑌲𑌿𑌪𑌿𑌂 𑌬𑍍𑌰𑌹𑍍𑌮𑌵𑌲𑌿𑌲𑌿𑌪𑌿𑌂 𑌪𑌾𑌰𑍁𑌷𑍍𑌯𑌲𑌿𑌪𑌿𑌂 𑌦𑍍𑌰𑌾𑌵𑌿𑌡𑌲𑌿𑌪𑌿𑌂 𑌕𑌿𑌰𑌾𑌤𑌲𑌿𑌪𑌿𑌂 𑌦𑌾𑌕𑍍𑌷𑌿𑌣𑍍𑌯𑌲𑌿𑌪𑌿𑌂 𑌉𑌗𑍍𑌰𑌲𑌿𑌪𑌿𑌂 𑌸𑌂𑌖𑍍𑌯𑌾𑌲𑌿𑌪𑌿𑌂 𑌅𑌨𑍁𑌲𑍋𑌮𑌲𑌿𑌪𑌿𑌂 𑌅𑌵𑌮𑍂𑌰𑍍𑌧𑌲𑌿𑌪𑌿𑌂 𑌦𑌰𑌦𑌲𑌿𑌪𑌿𑌂 𑌖𑌾𑌷𑍍𑌯𑌲𑌿𑌪𑌿𑌂 𑌚𑍀𑌨𑌲𑌿𑌪𑌿𑌂 𑌲𑍂𑌨𑌲𑌿𑌪𑌿𑌂 𑌹𑍂𑌣𑌲𑌿𑌪𑌿𑌂 𑌮𑌧𑍍𑌯𑌾𑌕𑍍𑌷𑌰𑌵𑌿𑌸𑍍𑌤𑌰𑌲𑌿𑌪𑌿𑌂 𑌪𑍁𑌷𑍍𑌪𑌲𑌿𑌪𑌿𑌂 𑌦𑍇𑌵𑌲𑌿𑌪𑌿𑌂 𑌨𑌾𑌗𑌲𑌿𑌪𑌿𑌂 𑌯𑌕𑍍𑌷𑌲𑌿𑌪𑌿𑌂 𑌗𑌨𑍍𑌧𑌰𑍍𑌵𑌲𑌿𑌪𑌿𑌂 𑌕𑌿𑌨𑍍𑌨𑌰𑌲𑌿𑌪𑌿𑌂 𑌮𑌹𑍋𑌰𑌗𑌲𑌿𑌪𑌿𑌂 𑌅𑌸𑍁𑌰𑌲𑌿𑌪𑌿𑌂 𑌗𑌰𑍁𑌡𑌲𑌿𑌪𑌿𑌂 𑌮𑍃𑌗𑌚𑌕𑍍𑌰𑌲𑌿𑌪𑌿𑌂 𑌵𑌾𑌯𑌸𑌰𑍁𑌤𑌲𑌿𑌪𑌿𑌂 𑌭𑍗𑌮𑌦𑍇𑌵𑌲𑌿𑌪𑌿𑌂 𑌅𑌨𑍍𑌤𑌰𑍀𑌕𑍍𑌷𑌦𑍇𑌵𑌲𑌿𑌪𑌿𑌂 𑌉𑌤𑍍𑌤𑌰𑌕𑍁𑌰𑍁𑌦𑍍𑌵𑍀𑌪𑌲𑌿𑌪𑌿𑌂 𑌅𑌪𑌰𑌗𑍋𑌡𑌾𑌨𑍀𑌲𑌿𑌪𑌿𑌂 𑌪𑍂𑌰𑍍𑌵𑌵𑌿𑌦𑍇𑌹𑌲𑌿𑌪𑌿𑌂 𑌉𑌤𑍍𑌕𑍍𑌷𑍇𑌪𑌲𑌿𑌪𑌿𑌂 𑌨𑌿𑌕𑍍𑌷𑍇𑌪𑌲𑌿𑌪𑌿𑌂 𑌵𑌿𑌕𑍍𑌷𑍇𑌪𑌲𑌿𑌪𑌿𑌂 𑌪𑍍𑌰𑌕𑍍𑌷𑍇𑌪𑌲𑌿𑌪𑌿𑌂 𑌸𑌾𑌗𑌰𑌲𑌿𑌪𑌿𑌂 𑌵𑌜𑍍𑌰𑌲𑌿𑌪𑌿𑌂 𑌲𑍇𑌖𑌪𑍍𑌰𑌤𑌿𑌲𑍇𑌖𑌲𑌿𑌪𑌿𑌂 𑌅𑌨𑍁𑌦𑍍𑌰𑍁𑌤𑌲𑌿𑌪𑌿𑌂 𑌶𑌾𑌸𑍍𑌤𑍍𑌰𑌾𑌵𑌰𑍍𑌤𑌾𑌂 𑌗𑌣𑌨𑌾𑌵𑌰𑍍𑌤𑌲𑌿𑌪𑌿𑌂 𑌉𑌤𑍍𑌕𑍍𑌷𑍇𑌪𑌾𑌵𑌰𑍍𑌤𑌲𑌿𑌪𑌿𑌂 𑌨𑌿𑌕𑍍𑌷𑍇𑌪𑌾𑌵𑌰𑍍𑌤𑌲𑌿𑌪𑌿𑌂 𑌪𑌾𑌦𑌲𑌿𑌖𑌿𑌤𑌲𑌿𑌪𑌿𑌂 𑌦𑍍𑌵𑌿𑌰𑍁𑌤𑍍𑌤𑌰𑌪𑌦𑌸𑌂𑌧𑌿𑌲𑌿𑌪𑌿𑌂 𑌯𑌾𑌵𑌦𑍍𑌦𑌶𑍋𑌤𑍍𑌤𑌰𑌪𑌦𑌸𑌂𑌧𑌿𑌲𑌿𑌪𑌿𑌂 𑌮𑌧𑍍𑌯𑌾𑌹𑌾𑌰𑌿𑌣𑍀𑌲𑌿𑌪𑌿𑌂 𑌸𑌰𑍍𑌵𑌰𑍁𑌤𑌸𑌂𑌗𑍍𑌰𑌹𑌣𑍀𑌲𑌿𑌪𑌿𑌂 𑌵𑌿𑌦𑍍𑌯𑌾𑌨𑍁𑌲𑍋𑌮𑌾𑌵𑌿𑌮𑌿𑌶𑍍𑌰𑌿𑌤𑌲𑌿𑌪𑌿𑌂 𑌋𑌷𑌿𑌤𑌪𑌸𑍍𑌤𑌪𑍍𑌤𑌾𑌂 𑌰𑍋𑌚𑌮𑌾𑌨𑌾𑌂 𑌧𑌰𑌣𑍀𑌪𑍍𑌰𑍇𑌕𑍍𑌷𑌿𑌣𑍀𑌲𑌿𑌪𑌿𑌂 𑌗𑌗𑌨𑌪𑍍𑌰𑍇𑌕𑍍𑌷𑌿𑌣𑍀𑌲𑌿𑌪𑌿𑌂 𑌸𑌰𑍍𑌵𑍗𑌷𑌧𑌿𑌨𑌿𑌷𑍍𑌯𑌨𑍍𑌦𑌾𑌂 𑌸𑌰𑍍𑌵𑌸𑌾𑌰𑌸𑌂𑌗𑍍𑌰𑌹𑌣𑍀𑌂 𑌸𑌰𑍍𑌵𑌭𑍂𑌤𑌰𑍁𑌤𑌗𑍍𑌰𑌹𑌣𑍀𑌮𑍍। 𑌆𑌸𑌾𑌂 𑌭𑍋 𑌉𑌪𑌾𑌧𑍍𑌯𑌾𑌯 𑌚𑌤𑍁𑌷𑍍𑌷𑌷𑍍𑌟𑍀𑌲𑌿𑌪𑍀𑌨𑌾𑌂 𑌕𑌤𑌮𑌾𑌂 𑌤𑍍𑌵𑌂 𑌶𑌿𑌷𑍍𑌯𑌾𑌪𑌯𑌿𑌷𑍍𑌯𑌸𑌿?

𑌅𑌥 𑌵𑌿𑌶𑍍𑌵𑌾𑌮𑌿𑌤𑍍𑌰𑍋 𑌦𑌾𑌰𑌕𑌾𑌚𑌾𑌰𑍍𑌯𑍋 𑌵𑌿𑌸𑍍𑌮𑌿𑌤𑌃 𑌪𑍍𑌰𑌹𑌸𑌿𑌤𑌵𑌦𑌨𑍋 𑌨𑌿𑌹𑌤𑌮𑌾𑌨𑌮𑌦𑌦𑌰𑍍𑌪 𑌇𑌮𑌾𑌂 𑌗𑌾𑌥𑌾𑌮𑌭𑌾𑌷𑌤—

𑌆𑌶𑍍𑌚𑌰𑍍𑌯𑌂 𑌶𑍁𑌦𑍍𑌧𑌸𑌤𑍍𑌤𑍍𑌵𑌸𑍍𑌯 𑌲𑍋𑌕𑍇 𑌲𑍋𑌕𑌾𑌨𑍁𑌵𑌰𑍍𑌤𑌿𑌨𑍋।
𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑌃 𑌸𑌰𑍍𑌵𑌶𑌾𑌸𑍍𑌤𑍍𑌰𑍇𑌷𑍁 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍁𑌪𑌾𑌗𑌤𑌃॥௬॥

𑌯𑍇𑌷𑌾𑌮𑌹𑌂 𑌨𑌾𑌮𑌧𑍇𑌯𑌂 𑌲𑌿𑌪𑍀𑌨𑌾𑌂 𑌨 𑌪𑍍𑌰𑌜𑌾𑌨𑌮𑌿।
𑌤𑌤𑍍𑌰𑍈𑌷 𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑌃 𑌸𑌨𑍍𑌤𑍋 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍁𑌪𑌾𑌗𑌤𑌃॥௭॥

𑌵𑌕𑍍𑌤𑍍𑌰𑌂 𑌚𑌾𑌸𑍍𑌯 𑌨 𑌪𑌶𑍍𑌯𑌾𑌮𑌿 𑌮𑍂𑌰𑍍𑌧𑌾𑌨𑌂 𑌤𑌸𑍍𑌯 𑌨𑍈𑌵 𑌚।
𑌶𑌿𑌷𑍍𑌯𑌯𑌿𑌷𑍍𑌯𑍇 𑌕𑌥𑌂 𑌹𑍍𑌯𑍇𑌨𑌂 𑌲𑌿𑌪𑌿𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌯 𑌪𑌾𑌰𑌗𑌮𑍍॥௮॥

𑌦𑍇𑌵𑌦𑍇𑌵𑍋 𑌹𑍍𑌯𑌤𑌿𑌦𑍇𑌵𑌃 𑌸𑌰𑍍𑌵𑌦𑍇𑌵𑍋𑌤𑍍𑌤𑌮𑍋 𑌵𑌿𑌭𑍁𑌃।
𑌅𑌸𑌮𑌶𑍍𑌚 𑌵𑌿𑌶𑌿𑌷𑍍𑌟𑌶𑍍𑌚 𑌲𑍋𑌕𑍇𑌷𑍍𑌵𑌪𑍍𑌰𑌤𑌿𑌪𑍁𑌦𑍍𑌗𑌲𑌃॥௯॥

𑌅𑌸𑍍𑌯𑍈𑌵 𑌤𑍍𑌵𑌨𑍁𑌭𑌾𑌵𑍇𑌨 𑌪𑍍𑌰𑌜𑍍𑌞𑍋𑌪𑌾𑌯𑍇 𑌵𑌿𑌶𑍇𑌷𑌤𑌃।
𑌶𑌿𑌕𑍍𑌷𑌿𑌤𑌂 𑌶𑌿𑌷𑍍𑌯𑌯𑌿𑌷𑍍𑌯𑌾𑌮𑌿 𑌸𑌰𑍍𑌵𑌲𑍋𑌕𑌪𑌰𑌾𑌯𑌣𑌮𑍍॥௧௦॥

𑌇𑌤𑌿 𑌹𑌿 𑌭𑌿𑌕𑍍𑌷𑌵𑍋 𑌦𑌶 𑌦𑌾𑌰𑌕𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍇𑌨 𑌸𑌾𑌰𑍍𑌧𑌂 𑌲𑌿𑌪𑌿𑌂 𑌶𑌿𑌷𑍍𑌯𑌨𑍍𑌤𑍇 𑌸𑍍𑌮। 𑌤𑌤𑍍𑌰 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌾𑌧𑌿𑌸𑍍𑌥𑌾𑌨𑍇𑌨 𑌤𑍇𑌷𑌾𑌂 𑌦𑌾𑌰𑌕𑌾𑌣𑌾𑌂 𑌮𑌾𑌤𑍃𑌕𑌾𑌂 𑌵𑌾𑌚𑌯𑌤𑌾𑌂 𑌯𑌦𑌾 𑌅𑌕𑌾𑌰𑌂 𑌪𑌰𑌿𑌕𑍀𑌰𑍍𑌤𑌯𑌨𑍍𑌤𑌿 𑌸𑍍𑌮, 𑌤𑌦𑌾 𑌅𑌨𑌿𑌤𑍍𑌯𑌃 𑌸𑌰𑍍𑌵𑌸𑌂𑌸𑍍𑌕𑌾𑌰𑌶𑌬𑍍𑌦𑍋 𑌨𑌿𑌶𑍍𑌚𑌰𑌤𑌿 𑌸𑍍𑌮। 𑌆𑌕𑌾𑌰𑍇 𑌪𑌰𑌿𑌕𑍀𑌰𑍍𑌤𑍍𑌯𑌮𑌾𑌨𑍇 𑌆𑌤𑍍𑌮𑌪𑌰𑌹𑌿𑌤𑌶𑌬𑍍𑌦𑍋 𑌨𑌿𑌶𑍍𑌚𑌰𑌤𑌿 𑌸𑍍𑌮। 𑌇𑌕𑌾𑌰𑍇 𑌇𑌨𑍍𑌦𑍍𑌰𑌿𑌯𑌵𑍈𑌕𑌲𑍍𑌯𑌶𑌬𑍍𑌦𑌃। 𑌈𑌕𑌾𑌰𑍇 𑌈𑌤𑌿𑌬𑌹𑍁𑌲𑌂 𑌜𑌗𑌦𑌿𑌤𑌿। 𑌉𑌕𑌾𑌰𑍇 𑌉𑌪𑌦𑍍𑌰𑌵𑌬𑌹𑍁𑌲𑌂 𑌜𑌗𑌦𑌿𑌤𑌿। 𑌊𑌕𑌾𑌰𑍇 𑌊𑌨𑌸𑌤𑍍𑌤𑍍𑌵𑌂 𑌜𑌗𑌦𑌿𑌤𑌿। 𑌏𑌕𑌾𑌰𑍇 𑌏𑌷𑌣𑌾𑌸𑌮𑍁𑌤𑍍𑌥𑌾𑌨𑌦𑍋𑌷𑌶𑌬𑍍𑌦𑌃। 𑌐𑌕𑌾𑌰𑍇 𑌐𑌰𑍍𑌯𑌾𑌪𑌥𑌃 𑌶𑍍𑌰𑍇𑌯𑌾𑌨𑌿𑌤𑌿। 𑌓𑌕𑌾𑌰𑍇 𑌓𑌘𑍋𑌤𑍍𑌤𑌰𑌶𑌬𑍍𑌦𑌃। 𑌔𑌕𑌾𑌰𑍇 𑌔𑌪𑌪𑌾𑌦𑍁𑌕𑌶𑌬𑍍𑌦𑌃। 𑌅𑌂𑌕𑌾𑌰𑍇 𑌅𑌮𑍋𑌘𑍋𑌤𑍍𑌪𑌤𑍍𑌤𑌿𑌶𑌬𑍍𑌦𑌃। 𑌅𑌃𑌕𑌾𑌰𑍇 𑌅𑌸𑍍𑌤𑌂𑌗𑌮𑌨𑌶𑌬𑍍𑌦𑍋 𑌨𑌿𑌶𑍍𑌚𑌰𑌤𑌿 𑌸𑍍𑌮। 𑌕𑌕𑌾𑌰𑍇 𑌕𑌰𑍍𑌮𑌵𑌿𑌪𑌾𑌕𑌾𑌵𑌤𑌾𑌰𑌶𑌬𑍍𑌦𑌃। 𑌖𑌕𑌾𑌰𑍇 𑌖𑌸𑌮𑌸𑌰𑍍𑌵𑌧𑌰𑍍𑌮𑌶𑌬𑍍𑌦𑌃। 𑌗𑌕𑌾𑌰𑍇 𑌗𑌮𑍍𑌭𑍀𑌰𑌧𑌰𑍍𑌮𑌪𑍍𑌰𑌤𑍀𑌤𑍍𑌯𑌸𑌮𑍁𑌤𑍍𑌪𑌾𑌦𑌾𑌵𑌤𑌾𑌰𑌶𑌬𑍍𑌦𑌃। 𑌘𑌕𑌾𑌰𑍇 𑌘𑌨𑌪𑌟𑌲𑌾𑌵𑌿𑌦𑍍𑌯𑌾𑌮𑍋𑌹𑌾𑌨𑍍𑌧𑌕𑌾𑌰𑌵𑌿𑌧𑌮𑌨𑌶𑌬𑍍𑌦𑌃। 𑌙𑌕𑌾𑌰𑍇𑌽𑌙𑍍𑌗𑌵𑌿𑌶𑍁𑌦𑍍𑌧𑌿𑌶𑌬𑍍𑌦𑌃। 𑌚𑌕𑌾𑌰𑍇 𑌚𑌤𑍁𑌰𑌾𑌰𑍍𑌯𑌸𑌤𑍍𑌯𑌶𑌬𑍍𑌦𑌃। 𑌛𑌕𑌾𑌰𑍇 𑌛𑌨𑍍𑌦𑌰𑌾𑌗𑌪𑍍𑌰𑌹𑌾𑌣𑌶𑌬𑍍𑌦𑌃। 𑌜𑌕𑌾𑌰𑍇 𑌜𑌰𑌾𑌮𑌰𑌣𑌸𑌮𑌤𑌿𑌕𑍍𑌰𑌮𑌣𑌶𑌬𑍍𑌦𑌃। 𑌝𑌕𑌾𑌰𑍇 𑌝𑌷𑌧𑍍𑌵𑌜𑌬𑌲𑌨𑌿𑌗𑍍𑌰𑌹𑌣𑌶𑌬𑍍𑌦𑌃। 𑌞𑌕𑌾𑌰𑍇 𑌜𑍍𑌞𑌾𑌪𑌨𑌶𑌬𑍍𑌦𑌃। 𑌟𑌕𑌾𑌰𑍇 𑌪𑌟𑍋𑌪𑌚𑍍𑌛𑍇𑌦𑌨𑌶𑌬𑍍𑌦𑌃। 𑌠𑌕𑌾𑌰𑍇 𑌠𑌪𑌨𑍀𑌯𑌪𑍍𑌰𑌶𑍍𑌨𑌶𑌬𑍍𑌦𑌃। 𑌡𑌕𑌾𑌰𑍇 𑌡𑌮𑌰𑌮𑌾𑌰𑌨𑌿𑌗𑍍𑌰𑌹𑌣𑌶𑌬𑍍𑌦𑌃। 𑌢𑌕𑌾𑌰𑍇 𑌮𑍀𑌢𑌵𑌿𑌷𑌯𑌾 𑌇𑌤𑌿। 𑌣𑌕𑌾𑌰𑍇 𑌰𑍇𑌣𑍁𑌕𑍍𑌲𑍇𑌶𑌾 𑌇𑌤𑌿। 𑌤𑌕𑌾𑌰𑍇 𑌤𑌥𑌤𑌾𑌸𑌂𑌭𑍇𑌦𑌶𑌬𑍍𑌦𑌃। 𑌥𑌕𑌾𑌰𑍇 𑌥𑌾𑌮𑌬𑌲𑌵𑍇𑌗𑌵𑍈𑌶𑌾𑌰𑌦𑍍𑌯𑌶𑌬𑍍𑌦𑌃। 𑌦𑌕𑌾𑌰𑍇 𑌦𑌾𑌨𑌦𑌮𑌸𑌂𑌯𑌮𑌸𑍗𑌰𑌭𑍍𑌯𑌶𑌬𑍍𑌦𑌃। 𑌧𑌕𑌾𑌰𑍇 𑌧𑌨𑌮𑌾𑌰𑍍𑌯𑌾𑌣𑌾𑌂 𑌸𑌪𑍍𑌤𑌵𑌿𑌧𑌮𑌿𑌤𑌿। 𑌨𑌕𑌾𑌰𑍇 𑌨𑌾𑌮𑌰𑍂𑌪𑌪𑌰𑌿𑌜𑍍𑌞𑌾𑌶𑌬𑍍𑌦𑌃। 𑌪𑌕𑌾𑌰𑍇 𑌪𑌰𑌮𑌾𑌰𑍍𑌥𑌶𑌬𑍍𑌦𑌃। 𑌫𑌕𑌾𑌰𑍇 𑌫𑌲𑌪𑍍𑌰𑌾𑌪𑍍𑌤𑌿𑌸𑌾𑌕𑍍𑌷𑌾𑌤𑍍𑌕𑍍𑌰𑌿𑌯𑌾𑌶𑌬𑍍𑌦𑌃। 𑌬𑌕𑌾𑌰𑍇 𑌬𑌨𑍍𑌧𑌨𑌮𑍋𑌕𑍍𑌷𑌶𑌬𑍍𑌦𑌃। 𑌭𑌕𑌾𑌰𑍇 𑌭𑌵𑌵𑌿𑌭𑌵𑌶𑌬𑍍𑌦𑌃। 𑌮𑌕𑌾𑌰𑍇 𑌮𑌦𑌮𑌾𑌨𑍋𑌪𑌶𑌮𑌨𑌶𑌬𑍍𑌦𑌃। 𑌯𑌕𑌾𑌰𑍇 𑌯𑌥𑌾𑌵𑌦𑍍𑌧𑌰𑍍𑌮𑌪𑍍𑌰𑌤𑌿𑌵𑍇𑌧𑌶𑌬𑍍𑌦𑌃। 𑌰𑌕𑌾𑌰𑍇 𑌰𑌤𑍍𑌯𑌰𑌤𑌿𑌪𑌰𑌮𑌾𑌰𑍍𑌥𑌰𑌤𑌿𑌶𑌬𑍍𑌦𑌃। 𑌲𑌕𑌾𑌰𑍇 𑌲𑌤𑌾𑌛𑍇𑌦𑌨𑌶𑌬𑍍𑌦𑌃। 𑌵𑌕𑌾𑌰𑍇 𑌵𑌰𑌯𑌾𑌨𑌶𑌬𑍍𑌦𑌃। 𑌶𑌕𑌾𑌰𑍇 𑌶𑌮𑌥𑌵𑌿𑌪𑌶𑍍𑌯𑌨𑌾𑌶𑌬𑍍𑌦𑌃। 𑌷𑌕𑌾𑌰𑍇 𑌷𑌡𑌾𑌯𑌤𑌨𑌨𑌿𑌗𑍍𑌰𑌹𑌣𑌾𑌭𑌿𑌜𑍍𑌞𑌜𑍍𑌞𑌾𑌨𑌾𑌵𑌾𑌪𑍍𑌤𑌿𑌶𑌬𑍍𑌦𑌃। 𑌸𑌕𑌾𑌰𑍇 𑌸𑌰𑍍𑌵𑌜𑍍𑌞𑌜𑍍𑌞𑌾𑌨𑌾𑌭𑌿𑌸𑌂𑌬𑍋𑌧𑌨𑌶𑌬𑍍𑌦𑌃। 𑌹𑌕𑌾𑌰𑍇 𑌹𑌤𑌕𑍍𑌲𑍇𑌶𑌵𑌿𑌰𑌾𑌗𑌶𑌬𑍍𑌦𑌃। 𑌕𑍍𑌷𑌕𑌾𑌰𑍇 𑌪𑌰𑌿𑌕𑍀𑌰𑍍𑌤𑍍𑌯𑌮𑌾𑌨𑍇 𑌕𑍍𑌷𑌣𑌪𑌰𑍍𑌯𑌨𑍍𑌤𑌾𑌭𑌿𑌲𑌾𑌪𑍍𑌯𑌸𑌰𑍍𑌵𑌧𑌰𑍍𑌮𑌶𑌬𑍍𑌦𑍋 𑌨𑌿𑌶𑍍𑌚𑌰𑌤𑌿 𑌸𑍍𑌮॥

𑌇𑌤𑌿 𑌹𑌿 𑌭𑌿𑌕𑍍𑌷𑌵𑌸𑍍𑌤𑍇𑌷𑌾𑌂 𑌦𑌾𑌰𑌕𑌾𑌣𑌾𑌂 𑌮𑌾𑌤𑍃𑌕𑌾𑌂 𑌵𑌾𑌚𑌯𑌤𑌾𑌂 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌾𑌨𑍁𑌭𑌾𑌵𑍇𑌨𑍈𑌵 𑌪𑍍𑌰𑌮𑍁𑌖𑌾𑌨𑍍𑌯𑌸𑌂𑌖𑍍𑌯𑍇𑌯𑌾𑌨𑌿 𑌧𑌰𑍍𑌮𑌮𑍁𑌖𑌶𑌤𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿 𑌨𑌿𑌶𑍍𑌚𑌰𑌨𑍍𑌤𑌿 𑌸𑍍𑌮॥

𑌤𑌦𑌾𑌨𑍁𑌪𑍂𑌰𑍍𑌵𑍇𑌣 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍇𑌨 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌸𑍍𑌥𑌿𑌤𑍇𑌨 𑌦𑍍𑌵𑌾𑌤𑍍𑌰𑌿𑌂𑌶𑌦𑍍𑌦𑌾𑌰𑌕𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿 𑌪𑌰𑌿𑌪𑌾𑌚𑌿𑌤𑌾𑌨𑍍𑌯𑌭𑍂𑌵𑌨𑍍। 𑌅𑌨𑍁𑌤𑍍𑌤𑌰𑌾𑌯𑌾𑌂 𑌸𑌮𑍍𑌯𑌕𑍍𑌸𑌂𑌬𑍋𑌧𑍗 𑌚𑌿𑌤𑍍𑌤𑌾𑌨𑍍𑌯𑍁𑌤𑍍𑌪𑌾𑌦𑌿𑌤𑌾𑌨𑌿 𑌦𑍍𑌵𑌾𑌤𑍍𑌰𑌿𑌂𑌶𑌦𑍍𑌦𑌾𑌰𑌿𑌕𑌾𑌸𑌹𑌸𑍍𑌰𑌾𑌣𑌿। 𑌅𑌯𑌂 𑌹𑍇𑌤𑍁𑌰𑌯𑌂 𑌪𑍍𑌰𑌤𑍍𑌯𑌯𑍋 𑌯𑌚𑍍𑌛𑌿𑌕𑍍𑌷𑌿𑌤𑍋𑌽𑌪𑌿 𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍋 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌮𑍁𑌪𑌾𑌗𑌚𑍍𑌛𑌤𑌿 𑌸𑍍𑌮॥

॥ 𑌇𑌤𑌿 𑌶𑍍𑌰𑍀𑌲𑌲𑌿𑌤𑌵𑌿𑌸𑍍𑌤𑌰𑍇 𑌲𑌿𑌪𑌿𑌶𑌾𑌲𑌾𑌸𑌂𑌦𑌰𑍍𑌶𑌨𑌪𑌰𑌿𑌵𑌰𑍍𑌤𑍋 𑌨𑌾𑌮 𑌦𑌶𑌮𑍋𑌽𑌧𑍍𑌯𑌾𑌯𑌃॥
