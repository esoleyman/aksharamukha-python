𑋚𑋚𑋡𑋍𑋛𑋡𑋝𑋪𑋍𑋙𑋞𑋪

𑋑𑋗𑋞𑋪 𑋝𑋙𑋪𑋛𑋔𑋣𑋏𑋪𑋐𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋥𑋖𑋪𑋘𑋞𑋪।

॥ 𑊸𑋟 𑋑𑋗𑋧 𑋏𑋜𑋏𑋡𑊼𑋑𑋑𑋪𑋍𑋠𑋒𑋙𑋪𑋘𑋑𑋪𑋍𑋚𑋧𑊺𑋐𑋠𑋍𑋣𑋒𑋪𑋙𑋍𑋡𑋜𑋩𑋪𑋇𑋡𑋍𑋝𑋙𑋪𑋛𑋔𑋣𑋏𑋪𑋐𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋙𑋪𑋘𑋜𑋪𑋙𑋠𑋛𑊺𑋒𑋪𑋙𑋍𑋪𑋘𑋥𑊺𑋔𑋣𑋏𑋪𑋐𑋥𑋖𑋪𑋘𑋧ऽ𑋍𑋢𑋍𑋠𑋑𑋠𑊼𑋍𑋒𑋪𑋙𑋍𑋪𑋘𑋣𑋍𑋪𑋒𑋑𑋪𑋑𑋥𑋖𑋪𑋘𑋞𑋪॥

𑋱𑋰 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋝𑋟𑋏𑋙𑋪𑋜𑋑𑋒𑋙𑋡𑋛𑋙𑋪𑋍𑋧 𑋏𑋜𑋗𑋞𑋪।

𑊲𑋍𑋡 𑋞𑋡 𑋖𑋡𑊺𑋪𑋜𑋩𑋛𑋞𑋪 𑋝𑋟𑋛𑋪𑋙𑋡𑋏𑋪𑋐𑋞𑋪 𑊺𑋣𑋗𑋠𑋙𑋞𑋪। 𑋍𑋏𑋠 𑋗𑋠𑊿𑋪𑊼𑋚𑋪𑋘𑋜𑋍𑋝𑋞𑋝𑋪𑋙𑋦𑋞𑋪 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋣𑋒𑋑𑋢𑋘𑋍𑋥 𑋝𑋪𑋗 𑋏𑋜𑋖𑋡𑋙𑋪𑋏𑋠𑋙𑊺𑋝𑋞𑋝𑋪𑋙𑋦𑋞𑋪 𑋒𑋙𑋡𑋛𑋪𑋙𑋡𑋍𑋞𑋪 𑋒𑋣𑋙𑋝𑋪𑊺𑋪𑋙𑋡𑋍𑋞𑋪, 𑋏𑋜𑋖𑋡𑋜𑋪𑋀 𑋙𑋎𑋝𑋞𑋝𑋪𑋙𑋦𑋞𑋪 𑊻𑋠𑋏𑋑𑋢𑋘𑋖𑋧𑋂𑋑𑋢𑋘𑋝𑋪𑋛𑋠𑋏𑋑𑋢𑋘𑋒𑋙𑋡𑋒𑋤𑋙𑋪𑋌𑋦𑋙𑋪𑋞𑋡𑋙𑋌𑋪𑋘𑋝𑋣𑋛𑋙𑋪𑋌𑋒𑋙𑋡𑋒𑋤𑋙𑋪𑋌𑋦𑋜𑋪𑋀। 𑋘𑋥𑋑 𑊺𑋒𑋡𑋚𑋛𑋝𑋪𑋍𑋣𑋑𑋡 𑋗𑋞𑋠𑋑𑊼𑋙𑋥 𑋛𑋢𑋎𑋡𑋀𑋍𑋪𑋛𑋙𑋙𑋎𑋪𑋘𑋠𑋑𑋪𑋍𑋙𑋠𑋒𑋌𑋗𑋣𑊻𑋥𑋜𑋩𑋪𑋛𑋖𑋪𑋘𑋛𑊺𑋢𑋙𑋪𑋘𑋍𑋥 𑋝𑋪𑋗 𑊰𑋖𑋡𑋛𑋡𑋜𑋪𑋙𑋠𑋗𑋪𑋘𑋑𑋪𑋍𑋥। 𑊰𑋜𑋩𑋪𑋆𑋠𑋖𑋡𑋜𑋪𑋀 𑋍𑋤𑋙𑋪𑋘𑋜𑋍𑋝𑋞𑋝𑋪𑋙𑋦𑋞𑋪 𑋒𑋪𑋙𑊾𑋣𑋜𑋩𑋪𑋘𑋗𑋠𑋌𑋦𑋙𑋪𑋗𑋞𑋍𑋠 𑋀 𑋒𑋣𑋜𑋩𑋪𑋒𑋛𑋙𑋪𑋜𑋩𑋥𑋌𑋠𑋖𑋡𑋒𑋪𑋙𑋛𑋙𑋪𑋜𑋩𑋍𑋠 𑋛𑋡𑋍𑋙𑋪𑋏𑋡𑋑𑋡𑋙𑋪𑋘𑋤𑋞𑋍𑋧𑋙𑋌𑊼𑋛𑋠𑊺𑋪𑋜𑋩𑋞𑋙𑋪𑋗𑋪𑋘𑊺𑋤𑋆𑋠𑊼𑋠𑋙𑋒𑋪𑋙𑋠𑋝𑋠𑋏𑋍𑋚𑋥𑋜𑋩𑋣 𑊺𑋑𑋪𑋘𑋠𑋜𑋍𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡 𑋝𑋙𑋪𑋛𑋠𑋚𑋟𑊺𑋠𑋙𑋖𑋤𑋜𑋩𑋡𑋍𑋠𑋞𑋪 𑋝𑋪𑋎𑋡𑋍𑋠 𑊰𑋖𑋤𑋛𑋑𑋪। 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋟 𑋒𑋪𑋙𑋥𑊺𑋪𑋜𑋩𑋗𑋠𑋌𑋠𑋞𑋪 𑊺𑋣𑋝𑋣𑋗𑋠𑋑𑋡 𑋀 𑊺𑋪𑋜𑋩𑋡𑋒𑋑𑋪𑋍𑋡 𑋝𑋪𑋗। 𑊰𑋜𑋩𑋪𑋆𑋨 𑋀 𑋗𑋙𑋣𑋍𑋪𑊺𑋑𑋪𑋘𑋠𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡 𑋛𑋡𑊼𑋚𑋡𑋍𑋠𑋚𑋟𑊺𑋠𑋙𑋠𑋖𑋙𑋌𑋠𑋚𑋟𑊺𑋪𑋙𑋡𑋍𑋠𑋑𑋡 𑋙𑋍𑋪𑋑𑋖𑋏𑋪𑋙𑋟𑊺𑋙𑋥𑋌 𑊼𑋪𑋙𑋡𑋞𑋢𑋍𑋠𑋑𑋡 𑋗𑋠𑋙𑋪𑊼𑋟 𑋜𑋧𑋐𑋘𑋑𑋪𑋍𑋪𑋘𑋧 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋝𑋪𑋘 𑋒𑋣𑋙𑋍𑋧 𑊼𑋀𑋪𑋁𑋑𑋪𑋍𑋡 𑋝𑋪𑋗। 𑋏𑋥𑋛𑋑𑋠𑊼𑋘𑊺𑋪𑋜𑋩𑊼𑋑𑋪𑋐𑋙𑋪𑋛𑋠𑋝𑋣𑋙𑊼𑋙𑋣𑋈𑊺𑋡𑋑𑋪𑋑𑋙𑋗𑋞𑋧𑋙𑊼𑋠𑋜𑋪𑋀𑋠𑋙𑋪𑋐𑊺𑋠𑋘𑋡𑊺𑋠 𑊼𑊼𑋑𑋍𑋚𑋠𑋍𑋪𑋒𑋣𑋜𑋩𑋪𑋒𑋒𑋆𑋪𑋆𑋏𑋠𑋗𑋠𑋑𑋪𑋘𑋖𑋡𑋒𑋪𑋙𑋚𑋗𑋪𑋔𑋘𑋑𑋪𑋍𑋡 𑋝𑋪𑋗। 𑋝𑋙𑋪𑋛𑋥 𑋀 𑋜𑋠𑊺𑋪𑋘𑊼𑋌𑋠𑋞𑋪 𑋜𑋣𑋏𑋪𑋐𑋧𑋏𑋑𑋟 𑋙𑋠𑋂𑋠𑋑𑋟 𑋒𑋣𑋙𑋝𑋪𑊺𑋪𑋙𑋡𑋍𑋪𑋘 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋝𑋪𑋘 𑋒𑋣𑋙𑋍𑋧 𑊼𑋀𑋪𑋁𑋑𑋪𑋍𑋡 𑋝𑋪𑋗। 𑊰𑋑𑋥𑋑𑋦𑋛𑋟𑋙𑋤𑋒𑋥𑋌 𑋛𑋪𑋘𑋤𑋞𑋥𑋑 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋧 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋣𑋒𑋑𑋢𑋘𑋍𑋥 𑋝𑋪𑋗॥

𑋝𑋗𑋑𑋑𑋪𑋍𑋙𑋒𑋪𑋙𑋛𑋥𑋜𑋡𑋍𑋜𑋪𑋀 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋧 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋪। 𑊰𑋎 𑋛𑋡𑋜𑋪𑋛𑋠𑋗𑋡𑋍𑋪𑋙𑋧 𑋑𑋠𑋗 𑋏𑋠𑋙𑊺𑋠𑋀𑋠𑋙𑋪𑋘𑋧 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋝𑋪𑋘 𑋜𑋪𑋙𑋡𑋘𑋟 𑋍𑋥𑋂𑋜𑋪𑋀𑋠𑋝𑋞𑋗𑋠𑋑𑋧 𑋐𑋙𑋌𑋡𑋍𑋚𑋥 𑋑𑋡𑋛𑋡𑋜𑋩𑋪𑋆𑋧ऽ𑋐𑋧𑋗𑋣𑊻𑋞𑋪 𑋒𑋪𑋙𑋒𑋍𑋍𑋡 𑋝𑋪𑋗। 𑋍𑋟 𑋍𑋎𑋠 𑋒𑋪𑋙𑋒𑋍𑋡𑋍𑋟 𑋏𑋪𑋙𑋡𑋜𑋩𑋪𑋆𑋪𑋛𑋠 𑋜𑋣𑋖𑋠𑊿𑋪𑊼𑋧 𑋑𑋠𑋗 𑋍𑋣𑋜𑋩𑋡𑋍𑊺𑋠𑋘𑋡𑊺𑋧 𑋏𑋥𑋛𑋒𑋣𑋍𑋪𑋙𑋧 𑋏𑊺𑋪𑋜𑋩𑋡𑋌𑋥𑋑 𑊺𑋙𑋍𑋚𑋥𑋑 𑋒𑋙𑋡𑊼𑋪𑋙𑋡𑋞𑋪𑋘𑋧𑋍𑋪𑋎𑋠𑋒𑋘𑋍𑋡 𑋝𑋪𑋗। 𑊴𑋍𑋪𑋎𑋠𑋒𑋪𑋘 𑋀 𑊼𑊼𑋑𑋍𑋚𑋝𑋪𑋎𑋧 𑋙𑋠𑋂𑋠𑋑𑋟 𑋜𑋣𑋏𑋪𑋐𑋧𑋏𑋑𑋟 𑋍𑋟 𑋀 𑋗𑋞𑋠𑋑𑋪𑋍𑋟 𑋂𑋑𑊺𑋠𑋘𑋟 𑊼𑋠𑋎𑋠𑋖𑋡𑋙𑋖𑋪𑋘𑋖𑋠𑋜𑋩𑋍𑋪—

𑋜𑋠𑋝𑋪𑋍𑋪𑋙𑋠𑋌𑋡 𑋘𑋠𑋑𑋡 𑋒𑋪𑋙𑋀𑋚𑋑𑋪𑋍𑋡 𑋗𑋑𑋣𑋜𑋩𑋪𑋘𑋚𑋧𑊺𑋥
𑋝𑋟𑊻𑋪𑋘𑋠 𑋚𑋡𑋒𑋡𑋜𑋪𑋀 𑊼𑋌𑋑𑋠𑋒𑋡 𑋀 𑋐𑋠𑋍𑋣𑋍𑋑𑋪𑋍𑋪𑋙𑋗𑋪।
𑋘𑋥 𑋜𑋡𑋚𑋪𑋒𑋘𑋧𑊼 𑋒𑋪𑋙𑋡𑋎𑋣 𑋚𑋨𑊺𑋡𑊺 𑊰𑋒𑋪𑋙𑋗𑋥𑋘𑋠𑋞𑋪
𑋍𑋥𑋜𑋩𑋪𑋛𑋥𑋜𑋩𑋣 𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋣 𑋒𑋣𑋙𑋠 𑋔𑋞𑋣𑊺𑋚𑋪𑋒𑊺𑋧𑋆𑋪𑋘𑋞𑋪॥𑋱॥

𑊺𑋡𑋟 𑋍𑋤 𑋂𑋑𑋝𑋪𑋘 𑊰𑋑𑋣𑋛𑋙𑋪𑋍𑋑𑋍𑋠𑋟 𑊺𑋙𑋧𑋍𑋡
𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋗𑋠𑊼𑋍𑋣 𑋝𑋣𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋣 𑋜𑋡𑋜𑋩𑋪𑋘𑋌𑋠𑋙𑋪𑋎𑋗𑋪।
𑋒𑋙𑋡𑋒𑋠𑋀𑋑𑋠𑋙𑋪𑋎 𑋔𑋞𑋣𑋏𑋠𑋙𑊺 𑊰𑊼𑋪𑋙𑋘𑋠𑋑𑋥
𑊰𑋑𑋪𑋘𑋠𑋟𑋜𑋪𑋀 𑋝𑋍𑋪𑋍𑋪𑋛𑋑𑋘𑋣𑋍𑋠𑋑𑋗𑋪𑋙𑋡𑋍𑋥 𑋛𑋡𑋑𑋥𑋍𑋣𑋗𑋪॥𑋲॥

𑋚𑋧𑊺𑋧𑋍𑋪𑋍𑋙𑋥𑋜𑋩𑋣 𑋀𑋍𑋣𑋝𑋍𑋪𑋘𑋒𑋎𑋥 𑋛𑋡𑋐𑋡𑋂𑋪𑋅𑋧
𑋞𑋥𑋍𑋣𑋒𑋪𑋙𑋍𑋢𑋍𑋪𑋘𑊺𑋣𑋜𑋚𑋧 𑋘𑋎 𑋝𑋟𑋖𑋛𑋑𑋪𑋍𑋡।
𑋘𑋎 𑋀𑋠𑋑𑋡𑋙𑋧𑋐𑊺𑋪𑋜𑋩𑋘𑋣 𑋝𑋟𑋝𑋪𑋎𑋡𑋍𑋣 𑋜𑋢𑋍𑋡𑋖𑋠𑋛𑋞𑋪
𑋍𑋝𑋪𑋗𑋡𑋑𑋪𑋛𑋡𑋐𑋡𑋂𑋪𑋅 𑊺𑋡𑋗𑋎𑋧 𑋚𑋡𑋒𑋡𑋜𑋠𑋝𑋪𑋍𑋪𑋙𑋗𑋠𑋍𑋪𑋙𑋥॥𑋳॥

𑋑𑋥𑋍𑋝𑋪𑋘 𑊱𑋀𑋙𑋡𑋘 𑊴𑋍𑋪𑋍𑋙𑋡 𑋛𑋠 𑋍𑋪𑋙𑋡𑋚𑋧𑊺𑋥
𑋝𑋙𑋪𑋛𑋥𑋜𑋩𑋣 𑋏𑋥𑋛𑋗𑋑𑋣𑋂𑋥𑋜𑋩𑋪𑋛𑋘𑋗𑋥𑋛 𑋂𑋥𑋜𑋩𑋪𑋇𑋞𑋪।
𑋑𑋠𑋗𑋠𑋒𑋡 𑋍𑋥𑋜𑋩 𑋚𑋡𑋒𑋡𑋑𑋠𑋟 𑋑 𑋞𑋡 𑋛𑋡𑋍𑋪𑋎 𑋘𑋤𑋘𑋟
𑋘𑋍𑋪𑋙𑋥𑋜𑋩𑋣 𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋣 𑋒𑋣𑋙𑋠 𑋔𑋞𑋣𑊺𑋚𑋪𑋒𑊺𑋧𑋆𑋪𑋘𑋞𑋪॥𑋴॥

𑋝𑋧 𑋀𑋡𑋍𑋪𑋍𑋐𑋠𑋙 𑋂𑊼𑋍𑋠𑋟 𑋛𑋡𑋛𑋡𑋐𑋠 𑋛𑋡𑋀𑋡𑋍𑋪𑋙𑋠
𑊶𑊺𑊺𑋪𑋜𑋩𑋌𑋥𑋑 𑊰𑋘𑋣 𑋂𑋠𑋑𑋍𑋡 𑋜𑋣𑋏𑋪𑋐𑋝𑋍𑋪𑋍𑋪𑋛𑋞𑋪।
𑊰𑋏𑋪𑋙𑋡𑋜𑋪𑋘𑋙𑋤𑋒𑋙𑋞𑋡𑋍𑋝𑋪𑋘 𑊼𑋍𑋡𑋟 𑋀 𑋛𑋥𑋍𑋪𑋍𑋡
𑊺𑋡𑋟 𑋛𑋠 𑋒𑋣𑋑𑋧ऽ𑋎 𑋚𑋡𑋒𑋡𑋑𑋧ऽ𑊺𑋪𑋜𑋩𑋙𑋏𑋪𑋙𑋡𑋜𑋪𑋘𑋙𑋤𑋒𑋠𑋗𑋪॥𑋵॥

𑊲𑋍𑋪𑋘𑋣𑊺𑋪𑋍𑋪𑋛𑋠 𑋝 𑋏𑋥𑋛𑋒𑋣𑋍𑋪𑋙𑋧 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋟 𑋏𑋡𑋛𑋪𑋘𑋦𑋞𑋪 𑊺𑋣𑋝𑋣𑋗𑋦𑋙𑋖𑋪𑋘𑋙𑋪𑋀𑋪𑋘 𑋍𑋍𑋪𑋙𑋦𑋛𑋠𑋑𑋪𑋍𑋙𑋪𑋏𑋐𑋥। 𑋍𑋍𑋪𑋙 𑋐𑋠𑋍𑋪𑋙𑋪𑋘𑋜𑋪𑋀 𑋀𑋥𑋆𑋢𑋛𑋙𑋪𑊼𑋠𑋜𑋪𑋀 𑋝𑋪𑋎𑋠𑋒𑋡𑋍𑋠 𑊰𑋖𑋤𑋛𑋑𑋪। 𑋒𑋙𑋡𑋜𑋥𑋜𑋩𑋠𑋞𑋪 𑋜𑋠𑊺𑋪𑋘𑋠𑋞𑋪 𑋜𑋣𑋏𑋪𑋐𑋧𑋏𑋑𑋒𑋪𑋙𑋗𑋣𑊻𑋠𑋞𑋪 𑋒𑋪𑋙𑊺𑋪𑋙𑋠𑋗𑋑𑋪𑋍𑋞𑋪॥

𑊰𑋎 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛 𑊴𑋙𑊼𑋝𑋠𑋙𑋀𑋑𑋪𑋏𑋑𑋗𑋘𑋟 𑋚𑋡𑋒𑋡𑋓𑋚𑊺𑋗𑋠𑋏𑋠𑋘 𑋏𑋡𑋛𑋪𑋘𑋠𑋙𑋪𑋜𑋩𑋝𑋣𑋛𑋙𑋪𑋌𑋍𑋡𑋙𑊺𑋟 𑋝𑋗𑋑𑋪𑋍𑋠𑋑𑋪𑋗𑋌𑋡𑋙𑋍𑋪𑋑𑋒𑋪𑋙𑋍𑋪𑋘𑋣𑋒𑋪𑋍𑋟 𑋛𑋡𑋜𑋪𑋛𑋠𑋗𑋡𑋍𑋪𑋙𑋗𑋠𑋀𑋠𑋙𑋪𑋘𑋗𑋥𑋛𑋗𑋠𑋞-𑊺𑋍𑋗𑋠𑋟 𑋗𑋥 𑋖𑋧 𑊴𑋒𑋠𑋐𑋪𑋘𑋠𑋘 𑋚𑋡𑋒𑋡𑋟 𑋜𑋡𑊺𑋪𑋜𑋩𑋠𑋒𑋘𑋝𑋡। 𑋔𑋪𑋙𑋠𑋞𑋪𑋗𑋢𑊻𑋙𑋧𑋜𑋩𑋪𑋆𑋢𑋒𑋣𑋜𑋩𑋪𑊺𑋙𑋝𑋠𑋙𑋡𑋟 𑊰𑊿𑋪𑊼𑋚𑋡𑋒𑋡𑋟 𑋛𑊿𑋪𑊼𑋚𑋡𑋒𑋡𑋟 𑋗𑊼𑋐𑋚𑋡𑋒𑋡𑋟 𑋗𑊿𑋪𑊼𑋚𑋪𑋘𑋚𑋡𑋒𑋡𑋟 𑊰𑊿𑋪𑊼𑋣𑋚𑋢𑋘𑋚𑋡𑋒𑋡𑋟 𑋜𑊺𑋠𑋙𑋡𑋚𑋡𑋒𑋡𑋟 𑋔𑋪𑋙𑋞𑋪𑋗𑋛𑋚𑋡𑋚𑋡𑋒𑋡𑋟 𑋒𑋠𑋙𑋣𑋜𑋩𑋪𑋘𑋚𑋡𑋒𑋡𑋟 𑋏𑋪𑋙𑋠𑋛𑋡𑋈𑋚𑋡𑋒𑋡𑋟 𑊺𑋡𑋙𑋠𑋍𑋚𑋡𑋒𑋡𑋟 𑋏𑋠𑊺𑋪𑋜𑋩𑋡𑋌𑋪𑋘𑋚𑋡𑋒𑋡𑋟 𑊴𑊼𑋪𑋙𑋚𑋡𑋒𑋡𑋟 𑋝𑋟𑊻𑋪𑋘𑋠𑋚𑋡𑋒𑋡𑋟 𑊰𑋑𑋣𑋚𑋧𑋗𑋚𑋡𑋒𑋡𑋟 𑊰𑋛𑋗𑋤𑋙𑋪𑋐𑋚𑋡𑋒𑋡𑋟 𑋏𑋙𑋏𑋚𑋡𑋒𑋡𑋟 𑊻𑋠𑋜𑋩𑋪𑋘𑋚𑋡𑋒𑋡𑋟 𑋀𑋢𑋑𑋚𑋡𑋒𑋡𑋟 𑋚𑋤𑋑𑋚𑋡𑋒𑋡𑋟 𑋞𑋤𑋌𑋚𑋡𑋒𑋡𑋟 𑋗𑋐𑋪𑋘𑋠𑊺𑋪𑋜𑋩𑋙𑋛𑋡𑋝𑋪𑋍𑋙𑋚𑋡𑋒𑋡𑋟 𑋒𑋣𑋜𑋩𑋪𑋒𑋚𑋡𑋒𑋡𑋟 𑋏𑋥𑋛𑋚𑋡𑋒𑋡𑋟 𑋑𑋠𑊼𑋚𑋡𑋒𑋡𑋟 𑋘𑊺𑋪𑋜𑋩𑋚𑋡𑋒𑋡𑋟 𑊼𑋑𑋪𑋐𑋙𑋪𑋛𑋚𑋡𑋒𑋡𑋟 𑊺𑋡𑋑𑋪𑋑𑋙𑋚𑋡𑋒𑋡𑋟 𑋗𑋞𑋧𑋙𑊼𑋚𑋡𑋒𑋡𑋟 𑊰𑋝𑋣𑋙𑋚𑋡𑋒𑋡𑋟 𑊼𑋙𑋣𑋈𑋚𑋡𑋒𑋡𑋟 𑋗𑋪𑋙𑋡𑊼𑋀𑊺𑋪𑋙𑋚𑋡𑋒𑋡𑋟 𑋛𑋠𑋘𑋝𑋙𑋣𑋍𑋚𑋡𑋒𑋡𑋟 𑋖𑋨𑋗𑋏𑋥𑋛𑋚𑋡𑋒𑋡𑋟 𑊰𑋑𑋪𑋍𑋙𑋢𑊺𑋪𑋜𑋩𑋏𑋥𑋛𑋚𑋡𑋒𑋡𑋟 𑊴𑋍𑋪𑋍𑋙𑊺𑋣𑋙𑋣𑋏𑋪𑋛𑋢𑋒𑋚𑋡𑋒𑋡𑋟 𑊰𑋒𑋙𑊼𑋧𑋈𑋠𑋑𑋢𑋚𑋡𑋒𑋡𑋟 𑋒𑋤𑋙𑋪𑋛𑋛𑋡𑋏𑋥𑋞𑋚𑋡𑋒𑋡𑋟 𑊴𑋍𑋪𑊺𑋪𑋜𑋩𑋥𑋒𑋚𑋡𑋒𑋡𑋟 𑋑𑋡𑊺𑋪𑋜𑋩𑋥𑋒𑋚𑋡𑋒𑋡𑋟 𑋛𑋡𑊺𑋪𑋜𑋩𑋥𑋒𑋚𑋡𑋒𑋡𑋟 𑋒𑋪𑋙𑊺𑋪𑋜𑋩𑋥𑋒𑋚𑋡𑋒𑋡𑋟 𑋝𑋠𑊼𑋙𑋚𑋡𑋒𑋡𑋟 𑋛𑋂𑋪𑋙𑋚𑋡𑋒𑋡𑋟 𑋚𑋥𑊻𑋒𑋪𑋙𑋍𑋡𑋚𑋥𑊻𑋚𑋡𑋒𑋡𑋟 𑊰𑋑𑋣𑋏𑋪𑋙𑋣𑋍𑋚𑋡𑋒𑋡𑋟 𑋜𑋠𑋝𑋪𑋍𑋪𑋙𑋠𑋛𑋙𑋪𑋍𑋠𑋟 𑊼𑋌𑋑𑋠𑋛𑋙𑋪𑋍𑋚𑋡𑋒𑋡𑋟 𑊴𑋍𑋪𑊺𑋪𑋜𑋩𑋥𑋒𑋠𑋛𑋙𑋪𑋍𑋚𑋡𑋒𑋡𑋟 𑋑𑋡𑊺𑋪𑋜𑋩𑋥𑋒𑋠𑋛𑋙𑋪𑋍𑋚𑋡𑋒𑋡𑋟 𑋒𑋠𑋏𑋚𑋡𑊻𑋡𑋍𑋚𑋡𑋒𑋡𑋟 𑋏𑋪𑋛𑋡𑋙𑋣𑋍𑋪𑋍𑋙𑋒𑋏𑋝𑋟𑋐𑋡𑋚𑋡𑋒𑋡𑋟 𑋘𑋠𑋛𑋏𑋪𑋏𑋜𑋧𑋍𑋪𑋍𑋙𑋒𑋏𑋝𑋟𑋐𑋡𑋚𑋡𑋒𑋡𑋟 𑋗𑋐𑋪𑋘𑋠𑋞𑋠𑋙𑋡𑋌𑋢𑋚𑋡𑋒𑋡𑋟 𑋝𑋙𑋪𑋛𑋙𑋣𑋍𑋝𑋟𑊼𑋪𑋙𑋞𑋌𑋢𑋚𑋡𑋒𑋡𑋟 𑋛𑋡𑋏𑋪𑋘𑋠𑋑𑋣𑋚𑋧𑋗𑋠𑋛𑋡𑋗𑋡𑋜𑋪𑋙𑋡𑋍𑋚𑋡𑋒𑋡𑋟 𑋙𑋡𑋜𑋩𑋡𑋍𑋒𑋝𑋪𑋍𑋒𑋪𑋍𑋠𑋟 𑋙𑋧𑋀𑋗𑋠𑋑𑋠𑋟 𑋐𑋙𑋌𑋢𑋒𑋪𑋙𑋥𑊺𑋪𑋜𑋩𑋡𑋌𑋢𑋚𑋡𑋒𑋡𑋟 𑊼𑊼𑋑𑋒𑋪𑋙𑋥𑊺𑋪𑋜𑋩𑋡𑋌𑋢𑋚𑋡𑋒𑋡𑋟 𑋝𑋙𑋪𑋛𑋨𑋜𑋩𑋐𑋡𑋑𑋡𑋜𑋩𑋪𑋘𑋑𑋪𑋏𑋠𑋟 𑋝𑋙𑋪𑋛𑋝𑋠𑋙𑋝𑋟𑊼𑋪𑋙𑋞𑋌𑋢𑋟 𑋝𑋙𑋪𑋛𑋖𑋤𑋍𑋙𑋣𑋍𑊼𑋪𑋙𑋞𑋌𑋢𑋗𑋪। 𑊱𑋝𑋠𑋟 𑋖𑋧 𑊴𑋒𑋠𑋐𑋪𑋘𑋠𑋘 𑋀𑋍𑋣𑋜𑋩𑋪𑋜𑋩𑋜𑋩𑋪𑋆𑋢𑋚𑋡𑋒𑋢𑋑𑋠𑋟 𑊺𑋍𑋗𑋠𑋟 𑋍𑋪𑋛𑋟 𑋜𑋡𑋜𑋩𑋪𑋘𑋠𑋒𑋘𑋡𑋜𑋩𑋪𑋘𑋝𑋡?

𑊰𑋎 𑋛𑋡𑋜𑋪𑋛𑋠𑋗𑋡𑋍𑋪𑋙𑋧 𑋏𑋠𑋙𑊺𑋠𑋀𑋠𑋙𑋪𑋘𑋧 𑋛𑋡𑋝𑋪𑋗𑋡𑋍𑋞𑋪 𑋒𑋪𑋙𑋞𑋝𑋡𑋍𑋛𑋏𑋑𑋧 𑋑𑋡𑋞𑋍𑋗𑋠𑋑𑋗𑋏𑋏𑋙𑋪𑋒 𑊲𑋗𑋠𑋟 𑊼𑋠𑋎𑋠𑋗𑋖𑋠𑋜𑋩𑋍—

𑊱𑋜𑋪𑋀𑋙𑋪𑋘𑋟 𑋜𑋣𑋏𑋪𑋐𑋝𑋍𑋪𑋍𑋪𑋛𑋝𑋪𑋘 𑋚𑋧𑊺𑋥 𑋚𑋧𑊺𑋠𑋑𑋣𑋛𑋙𑋪𑋍𑋡𑋑𑋧।
𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋞𑋪 𑋝𑋙𑋪𑋛𑋜𑋠𑋝𑋪𑋍𑋪𑋙𑋥𑋜𑋩𑋣 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋣𑋒𑋠𑊼𑋍𑋞𑋪॥𑋶॥

𑋘𑋥𑋜𑋩𑋠𑋗𑋞𑋟 𑋑𑋠𑋗𑋐𑋥𑋘𑋟 𑋚𑋡𑋒𑋢𑋑𑋠𑋟 𑋑 𑋒𑋪𑋙𑋂𑋠𑋑𑋗𑋡।
𑋍𑋍𑋪𑋙𑋦𑋜𑋩 𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋞𑋪 𑋝𑋑𑋪𑋍𑋧 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋣𑋒𑋠𑊼𑋍𑋞𑋪॥𑋷॥

𑋛𑊺𑋪𑋍𑋪𑋙𑋟 𑋀𑋠𑋝𑋪𑋘 𑋑 𑋒𑋜𑋪𑋘𑋠𑋗𑋡 𑋗𑋤𑋙𑋪𑋐𑋠𑋑𑋟 𑋍𑋝𑋪𑋘 𑋑𑋦𑋛 𑋀।
𑋜𑋡𑋜𑋩𑋪𑋘𑋘𑋡𑋜𑋩𑋪𑋘𑋥 𑊺𑋎𑋟 𑋞𑋪𑋘𑋥𑋑𑋟 𑋚𑋡𑋒𑋡𑋒𑋪𑋙𑋂𑋪𑋅𑋠𑋘 𑋒𑋠𑋙𑊼𑋗𑋪॥𑋸॥

𑋏𑋥𑋛𑋏𑋥𑋛𑋧 𑋞𑋪𑋘𑋍𑋡𑋏𑋥𑋛𑋞𑋪 𑋝𑋙𑋪𑋛𑋏𑋥𑋛𑋧𑋍𑋪𑋍𑋗𑋧 𑋛𑋡𑋖𑋣𑋞𑋪।
𑊰𑋝𑋗𑋜𑋪𑋀 𑋛𑋡𑋜𑋡𑋜𑋩𑋪𑋆𑋜𑋪𑋀 𑋚𑋧𑊺𑋥𑋜𑋩𑋪𑋛𑋒𑋪𑋙𑋍𑋡𑋒𑋣𑋏𑋪𑊼𑋚𑋞𑋪॥𑋹॥

𑊰𑋝𑋪𑋘𑋦𑋛 𑋍𑋪𑋛𑋑𑋣𑋖𑋠𑋛𑋥𑋑 𑋒𑋪𑋙𑋂𑋪𑋅𑋧𑋒𑋠𑋘𑋥 𑋛𑋡𑋜𑋥𑋜𑋩𑋍𑋞𑋪।
𑋜𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋟 𑋜𑋡𑋜𑋩𑋪𑋘𑋘𑋡𑋜𑋩𑋪𑋘𑋠𑋗𑋡 𑋝𑋙𑋪𑋛𑋚𑋧𑊺𑋒𑋙𑋠𑋘𑋌𑋗𑋪॥𑋱𑋰॥

𑊲𑋍𑋡 𑋞𑋡 𑋖𑋡𑊺𑋪𑋜𑋩𑋛𑋧 𑋏𑋜 𑋏𑋠𑋙𑊺𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋥𑋑 𑋝𑋠𑋙𑋪𑋐𑋟 𑋚𑋡𑋒𑋡𑋟 𑋜𑋡𑋜𑋩𑋪𑋘𑋑𑋪𑋍𑋥 𑋝𑋪𑋗। 𑋍𑋍𑋪𑋙 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋐𑋡𑋝𑋪𑋎𑋠𑋑𑋥𑋑 𑋍𑋥𑋜𑋩𑋠𑋟 𑋏𑋠𑋙𑊺𑋠𑋌𑋠𑋟 𑋗𑋠𑋍𑋪𑋙𑋡𑊺𑋠𑋟 𑋛𑋠𑋀𑋘𑋍𑋠𑋟 𑋘𑋏𑋠 𑊰𑊺𑋠𑋙𑋟 𑋒𑋙𑋡𑊺𑋢𑋙𑋪𑋍𑋘𑋑𑋪𑋍𑋡 𑋝𑋪𑋗, 𑋍𑋏𑋠 𑊰𑋑𑋡𑋍𑋪𑋘𑋞𑋪 𑋝𑋙𑋪𑋛𑋝𑋟𑋝𑋪𑊺𑋠𑋙𑋜𑋔𑋪𑋏𑋧 𑋑𑋡𑋜𑋪𑋀𑋙𑋍𑋡 𑋝𑋪𑋗। 𑊱𑊺𑋠𑋙𑋥 𑋒𑋙𑋡𑊺𑋢𑋙𑋪𑋍𑋪𑋘𑋗𑋠𑋑𑋥 𑊱𑋍𑋪𑋗𑋒𑋙𑋞𑋡𑋍𑋜𑋔𑋪𑋏𑋧 𑋑𑋡𑋜𑋪𑋀𑋙𑋍𑋡 𑋝𑋪𑋗। 𑊲𑊺𑋠𑋙𑋥 𑊲𑋑𑋪𑋏𑋪𑋙𑋡𑋘𑋛𑋦𑊺𑋚𑋪𑋘𑋜𑋔𑋪𑋏𑋞𑋪। 𑊳𑊺𑋠𑋙𑋥 𑊳𑋍𑋡𑋔𑋞𑋣𑋚𑋟 𑋂𑊼𑋏𑋡𑋍𑋡। 𑊴𑊺𑋠𑋙𑋥 𑊴𑋒𑋏𑋪𑋙𑋛𑋔𑋞𑋣𑋚𑋟 𑋂𑊼𑋏𑋡𑋍𑋡। 𑊵𑊺𑋠𑋙𑋥 𑊵𑋑𑋝𑋍𑋪𑋍𑋪𑋛𑋟 𑋂𑊼𑋏𑋡𑋍𑋡। 𑊶𑊺𑋠𑋙𑋥 𑊶𑋜𑋩𑋌𑋠𑋝𑋗𑋣𑋍𑋪𑋎𑋠𑋑𑋏𑋧𑋜𑋩𑋜𑋔𑋪𑋏𑋞𑋪। 𑊷𑊺𑋠𑋙𑋥 𑊷𑋙𑋪𑋘𑋠𑋒𑋎𑋞𑋪 𑋜𑋪𑋙𑋥𑋘𑋠𑋑𑋡𑋍𑋡। 𑊸𑊺𑋠𑋙𑋥 𑊸𑊾𑋧𑋍𑋪𑋍𑋙𑋜𑋔𑋪𑋏𑋞𑋪। 𑊹𑊺𑋠𑋙𑋥 𑊹𑋒𑋒𑋠𑋏𑋣𑊺𑋜𑋔𑋪𑋏𑋞𑋪। 𑊰𑋟𑊺𑋠𑋙𑋥 𑊰𑋗𑋧𑊾𑋧𑋍𑋪𑋒𑋍𑋪𑋍𑋡𑋜𑋔𑋪𑋏𑋞𑋪। 𑊰𑋞𑋪𑊺𑋠𑋙𑋥 𑊰𑋝𑋪𑋍𑋟𑊼𑋗𑋑𑋜𑋔𑋪𑋏𑋧 𑋑𑋡𑋜𑋪𑋀𑋙𑋍𑋡 𑋝𑋪𑋗। 𑊺𑊺𑋠𑋙𑋥 𑊺𑋙𑋪𑋗𑋛𑋡𑋒𑋠𑊺𑋠𑋛𑋍𑋠𑋙𑋜𑋔𑋪𑋏𑋞𑋪। 𑊻𑊺𑋠𑋙𑋥 𑊻𑋝𑋗𑋝𑋙𑋪𑋛𑋐𑋙𑋪𑋗𑋜𑋔𑋪𑋏𑋞𑋪। 𑊼𑊺𑋠𑋙𑋥 𑊼𑋗𑋪𑋖𑋢𑋙𑋐𑋙𑋪𑋗𑋒𑋪𑋙𑋍𑋢𑋍𑋪𑋘𑋝𑋗𑋣𑋍𑋪𑋒𑋠𑋏𑋠𑋛𑋍𑋠𑋙𑋜𑋔𑋪𑋏𑋞𑋪। 𑊾𑊺𑋠𑋙𑋥 𑊾𑋑𑋒𑋆𑋚𑋠𑋛𑋡𑋏𑋪𑋘𑋠𑋗𑋧𑋞𑋠𑋑𑋪𑋐𑊺𑋠𑋙𑋛𑋡𑋐𑋗𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑊿𑊺𑋠𑋙𑋥ऽ𑊿𑋪𑊼𑋛𑋡𑋜𑋣𑋏𑋪𑋐𑋡𑋜𑋔𑋪𑋏𑋞𑋪। 𑋀𑊺𑋠𑋙𑋥 𑋀𑋍𑋣𑋙𑋠𑋙𑋪𑋘𑋝𑋍𑋪𑋘𑋜𑋔𑋪𑋏𑋞𑋪। 𑋁𑊺𑋠𑋙𑋥 𑋁𑋑𑋪𑋏𑋙𑋠𑊼𑋒𑋪𑋙𑋞𑋠𑋌𑋜𑋔𑋪𑋏𑋞𑋪। 𑋂𑊺𑋠𑋙𑋥 𑋂𑋙𑋠𑋗𑋙𑋌𑋝𑋗𑋍𑋡𑊺𑋪𑋙𑋗𑋌𑋜𑋔𑋪𑋏𑋞𑋪। 𑋄𑊺𑋠𑋙𑋥 𑋄𑋜𑋩𑋐𑋪𑋛𑋂𑋔𑋚𑋑𑋡𑊼𑋪𑋙𑋞𑋌𑋜𑋔𑋪𑋏𑋞𑋪। 𑋅𑊺𑋠𑋙𑋥 𑋂𑋪𑋅𑋠𑋒𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋆𑊺𑋠𑋙𑋥 𑋒𑋆𑋧𑋒𑋀𑋪𑋁𑋥𑋏𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋇𑊺𑋠𑋙𑋥 𑋇𑋒𑋑𑋢𑋘𑋒𑋪𑋙𑋜𑋪𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋈𑊺𑋠𑋙𑋥 𑋈𑋗𑋙𑋗𑋠𑋙𑋑𑋡𑊼𑋪𑋙𑋞𑋌𑋜𑋔𑋪𑋏𑋞𑋪। 𑋋𑊺𑋠𑋙𑋥 𑋗𑋢𑋋𑋛𑋡𑋜𑋩𑋘𑋠 𑊲𑋍𑋡। 𑋌𑊺𑋠𑋙𑋥 𑋙𑋥𑋌𑋣𑊺𑋪𑋚𑋥𑋜𑋠 𑊲𑋍𑋡। 𑋍𑊺𑋠𑋙𑋥 𑋍𑋎𑋍𑋠𑋝𑋟𑋖𑋥𑋏𑋜𑋔𑋪𑋏𑋞𑋪। 𑋎𑊺𑋠𑋙𑋥 𑋎𑋠𑋗𑋔𑋚𑋛𑋥𑊼𑋛𑋦𑋜𑋠𑋙𑋏𑋪𑋘𑋜𑋔𑋪𑋏𑋞𑋪। 𑋏𑊺𑋠𑋙𑋥 𑋏𑋠𑋑𑋏𑋗𑋝𑋟𑋘𑋗𑋝𑋨𑋙𑋖𑋪𑋘𑋜𑋔𑋪𑋏𑋞𑋪। 𑋐𑊺𑋠𑋙𑋥 𑋐𑋑𑋗𑋠𑋙𑋪𑋘𑋠𑋌𑋠𑋟 𑋝𑋒𑋪𑋍𑋛𑋡𑋐𑋗𑋡𑋍𑋡। 𑋑𑊺𑋠𑋙𑋥 𑋑𑋠𑋗𑋙𑋤𑋒𑋒𑋙𑋡𑋂𑋪𑋅𑋠𑋜𑋔𑋪𑋏𑋞𑋪। 𑋒𑊺𑋠𑋙𑋥 𑋒𑋙𑋗𑋠𑋙𑋪𑋎𑋜𑋔𑋪𑋏𑋞𑋪। 𑋓𑊺𑋠𑋙𑋥 𑋓𑋚𑋒𑋪𑋙𑋠𑋒𑋪𑋍𑋡𑋝𑋠𑊺𑋪𑋜𑋩𑋠𑋍𑋪𑊺𑋪𑋙𑋡𑋘𑋠𑋜𑋔𑋪𑋏𑋞𑋪। 𑋔𑊺𑋠𑋙𑋥 𑋔𑋑𑋪𑋐𑋑𑋗𑋧𑊺𑋪𑋜𑋩𑋜𑋔𑋪𑋏𑋞𑋪। 𑋖𑊺𑋠𑋙𑋥 𑋖𑋛𑋛𑋡𑋖𑋛𑋜𑋔𑋪𑋏𑋞𑋪। 𑋗𑊺𑋠𑋙𑋥 𑋗𑋏𑋗𑋠𑋑𑋧𑋒𑋜𑋗𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋘𑊺𑋠𑋙𑋥 𑋘𑋎𑋠𑋛𑋏𑋪𑋐𑋙𑋪𑋗𑋒𑋪𑋙𑋍𑋡𑋛𑋥𑋐𑋜𑋔𑋪𑋏𑋞𑋪। 𑋙𑊺𑋠𑋙𑋥 𑋙𑋍𑋪𑋘𑋙𑋍𑋡𑋒𑋙𑋗𑋠𑋙𑋪𑋎𑋙𑋍𑋡𑋜𑋔𑋪𑋏𑋞𑋪। 𑋚𑊺𑋠𑋙𑋥 𑋚𑋍𑋠𑋁𑋥𑋏𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋛𑊺𑋠𑋙𑋥 𑋛𑋙𑋘𑋠𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋜𑊺𑋠𑋙𑋥 𑋜𑋗𑋎𑋛𑋡𑋒𑋜𑋪𑋘𑋑𑋠𑋜𑋔𑋪𑋏𑋞𑋪। 𑋜𑋩𑊺𑋠𑋙𑋥 𑋜𑋩𑋈𑋠𑋘𑋍𑋑𑋑𑋡𑊼𑋪𑋙𑋞𑋌𑋠𑋖𑋡𑋂𑋪𑋅𑋂𑋪𑋅𑋠𑋑𑋠𑋛𑋠𑋒𑋪𑋍𑋡𑋜𑋔𑋪𑋏𑋞𑋪। 𑋝𑊺𑋠𑋙𑋥 𑋝𑋙𑋪𑋛𑋂𑋪𑋅𑋂𑋪𑋅𑋠𑋑𑋠𑋖𑋡𑋝𑋟𑋔𑋧𑋐𑋑𑋜𑋔𑋪𑋏𑋞𑋪। 𑋞𑊺𑋠𑋙𑋥 𑋞𑋍𑊺𑋪𑋚𑋥𑋜𑋛𑋡𑋙𑋠𑊼𑋜𑋔𑋪𑋏𑋞𑋪। 𑊺𑋪𑋜𑋩𑊺𑋠𑋙𑋥 𑋒𑋙𑋡𑊺𑋢𑋙𑋪𑋍𑋪𑋘𑋗𑋠𑋑𑋥 𑊺𑋪𑋜𑋩𑋌𑋒𑋙𑋪𑋘𑋑𑋪𑋍𑋠𑋖𑋡𑋚𑋠𑋒𑋪𑋘𑋝𑋙𑋪𑋛𑋐𑋙𑋪𑋗𑋜𑋔𑋪𑋏𑋧 𑋑𑋡𑋜𑋪𑋀𑋙𑋍𑋡 𑋝𑋪𑋗॥

𑊲𑋍𑋡 𑋞𑋡 𑋖𑋡𑊺𑋪𑋜𑋩𑋛𑋝𑋪𑋍𑋥𑋜𑋩𑋠𑋟 𑋏𑋠𑋙𑊺𑋠𑋌𑋠𑋟 𑋗𑋠𑋍𑋪𑋙𑋡𑊺𑋠𑋟 𑋛𑋠𑋀𑋘𑋍𑋠𑋟 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋑𑋣𑋖𑋠𑋛𑋥𑋑𑋦𑋛 𑋒𑋪𑋙𑋗𑋣𑊻𑋠𑋑𑋪𑋘𑋝𑋟𑊻𑋪𑋘𑋥𑋘𑋠𑋑𑋡 𑋐𑋙𑋪𑋗𑋗𑋣𑊻𑋜𑋍𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡 𑋑𑋡𑋜𑋪𑋀𑋙𑋑𑋪𑋍𑋡 𑋝𑋪𑋗॥

𑋍𑋏𑋠𑋑𑋣𑋒𑋤𑋙𑋪𑋛𑋥𑋌 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋥𑋑 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋝𑋪𑋎𑋡𑋍𑋥𑋑 𑋏𑋪𑋛𑋠𑋍𑋪𑋙𑋡𑋟𑋜𑋏𑋪𑋏𑋠𑋙𑊺𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡 𑋒𑋙𑋡𑋒𑋠𑋀𑋡𑋍𑋠𑋑𑋪𑋘𑋖𑋤𑋛𑋑𑋪। 𑊰𑋑𑋣𑋍𑋪𑋍𑋙𑋠𑋘𑋠𑋟 𑋝𑋗𑋪𑋘𑊺𑋪𑋝𑋟𑋔𑋧𑋐𑋨 𑋀𑋡𑋍𑋪𑋍𑋠𑋑𑋪𑋘𑋣𑋍𑋪𑋒𑋠𑋏𑋡𑋍𑋠𑋑𑋡 𑋏𑋪𑋛𑋠𑋍𑋪𑋙𑋡𑋟𑋜𑋏𑋪𑋏𑋠𑋙𑋡𑊺𑋠𑋝𑋞𑋝𑋪𑋙𑋠𑋌𑋡। 𑊰𑋘𑋟 𑋞𑋥𑋍𑋣𑋙𑋘𑋟 𑋒𑋪𑋙𑋍𑋪𑋘𑋘𑋧 𑋘𑋀𑋪𑋁𑋡𑊺𑋪𑋜𑋩𑋡𑋍𑋧ऽ𑋒𑋡 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋧 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋗𑋣𑋒𑋠𑊼𑋀𑋪𑋁𑋍𑋡 𑋝𑋪𑋗॥

॥ 𑊲𑋍𑋡 𑋜𑋪𑋙𑋢𑋚𑋚𑋡𑋍𑋛𑋡𑋝𑋪𑋍𑋙𑋥 𑋚𑋡𑋒𑋡𑋜𑋠𑋚𑋠𑋝𑋟𑋏𑋙𑋪𑋜𑋑𑋒𑋙𑋡𑋛𑋙𑋪𑋍𑋧 𑋑𑋠𑋗 𑋏𑋜𑋗𑋧ऽ𑋐𑋪𑋘𑋠𑋘𑋞𑋪॥
