𑈟𑈵𑈦𑈐𑈵𑈓𑈬𑈟𑈬𑈦𑈤𑈭𑈙𑈬𑈪𑈵𑈦𑈭𑈛𑈥𑈩𑈯𑈙𑈵𑈦𑈤𑈵𑈸

[𑈩𑈴𑈈𑈵𑈩𑈶𑈭𑈟𑈵𑈙𑈤𑈬𑈙𑈵𑈦𑈭𑈈𑈬]

𑈹𑈺𑈞𑈤𑈪𑈵𑈺𑈩𑈦𑈵𑈨𑈐𑈵𑈓𑈬𑈥𑈺𑈹

𑈹𑈺𑈆𑈴𑈺𑈞𑈤𑈲𑈺𑈣𑈊𑈨𑈙𑈵𑈥𑈱𑈺𑈁𑈦𑈵𑈥𑈟𑈵𑈦𑈐𑈵𑈓𑈬𑈟𑈬𑈦𑈤𑈭𑈙𑈬𑈥𑈱𑈺𑈹

𑈁𑈦𑈵𑈥𑈬𑈨𑈧𑈲𑈈𑈭𑈙𑈰𑈩𑈶𑈵𑈨𑈦𑈡𑈲𑈝𑈭𑈩𑈙𑈷𑈵𑈨𑈲𑈺𑈊𑈤𑈵𑈣𑈮𑈦𑈬𑈥𑈬𑈴𑈺𑈟𑈵𑈦𑈐𑈵𑈓𑈬𑈟𑈬𑈦𑈤𑈭𑈙𑈬𑈥𑈬𑈴𑈺𑈎𑈦𑈵𑈥𑈬𑈴𑈺𑈎𑈦𑈤𑈬𑈘𑈲𑈺𑈨𑈵𑈥𑈨𑈧𑈲𑈈𑈥𑈙𑈭𑈺𑈩𑈵𑈤𑈸𑈺𑈟𑈓𑈵𑈎𑈺𑈩𑈵𑈈𑈞𑈵𑈝𑈬𑈪𑈵,𑈺𑈙𑈬𑈴𑈩𑈶𑈵𑈎𑈺𑈩𑈵𑈨𑈣𑈬𑈨𑈩𑈶𑈯𑈞𑈵𑈥𑈬𑈞𑈵𑈺𑈟𑈩𑈶𑈵𑈥𑈙𑈭𑈺𑈩𑈵𑈤𑈹

𑈂𑈪𑈺𑈩𑈶𑈬𑈦𑈭𑈟𑈯𑈙𑈵𑈦𑈺𑈦𑈯𑈟𑈴𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈬,𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈱𑈨𑈺𑈦𑈯𑈟𑈤𑈵𑈸𑈺𑈦𑈯𑈟𑈬𑈞𑈷𑈺𑈟𑈵𑈦𑈭𑈚𑈈𑈵𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈬,𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈬𑈥𑈬𑈺𑈞𑈺𑈟𑈵𑈦𑈭𑈚𑈊𑈵𑈺𑈦𑈯𑈟𑈤𑈵𑈸𑈺𑈥𑈛𑈵𑈦𑈯𑈟𑈴𑈺𑈩𑈬𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈬,𑈺𑈥𑈬𑈺𑈩𑈶𑈯𑈞𑈵𑈥𑈙𑈬𑈺𑈙𑈛𑈵𑈦𑈯𑈟𑈤𑈵𑈹

𑈄𑈨𑈤𑈰𑈨𑈺𑈨𑈰𑈛𑈞𑈬𑈩𑈴𑈐𑈵𑈓𑈬𑈩𑈴𑈩𑈵𑈈𑈬𑈦𑈨𑈭𑈐𑈵𑈓𑈬𑈞𑈬𑈞𑈭𑈹

𑈞𑈺𑈨𑈭𑈛𑈵𑈥𑈬𑈺𑈞𑈬𑈨𑈭𑈛𑈵𑈥𑈬𑈺𑈞𑈺𑈨𑈭𑈛𑈵𑈥𑈬𑈈𑈵𑈩𑈶𑈥𑈲𑈺𑈞𑈬𑈨𑈭𑈛𑈵𑈥𑈬𑈈𑈵𑈩𑈶𑈥𑈲𑈺𑈥𑈬𑈨𑈞𑈷𑈺𑈐𑈦𑈬𑈤𑈦𑈘𑈴𑈺𑈞𑈺𑈐𑈦𑈬𑈤𑈦𑈘𑈈𑈵𑈩𑈶𑈥𑈲𑈺𑈞𑈺𑈛𑈯𑈪𑈵𑈉𑈩𑈤𑈯𑈛𑈥𑈞𑈭𑈦𑈲𑈝𑈤𑈬𑈦𑈵𑈊𑈬𑈺𑈞𑈺𑈐𑈵𑈓𑈬𑈞𑈴𑈺𑈞𑈺𑈟𑈵𑈦𑈬𑈟𑈵𑈙𑈭𑈙𑈵𑈨𑈤𑈵𑈹
