𑃢𑃝𑃜𑃑𑃝𑃤𑃝𑃑𑃙𑃢𑃙𑃥𑃐𑃖𑃝𑃤𑃑𑃤𑃐𑃥𑃑𑃝𑃖

(𑃟𑃠) 𑃐𑃖𑃐𑃟𑃝𑃤𑃑𑃛𑃢𑃑𑃨𑃠𑃞

𑃒𑃠𑃢𑃝𑃑𑃤𑃜𑃒𑃠𑃢𑃐𑃨𑃢𑃜𑃢𑃖-𑃢𑃝𑃜𑃑𑃝𑃤𑃝𑃑𑃙𑃢𑃙𑃥𑃐𑃖𑃝𑃤𑃑𑃤𑃐𑃥𑃑𑃝𑃖

𑃒𑃠𑃦𑃑𑃨𑃒𑃠𑃢𑃐𑃨𑃢𑃜𑃢𑃖-𑃛𑃠𑃕𑃐-𑃛𑃢-𑃔𑃨𑃟𑃦𑃙-𑃖𑃓𑃓𑃠𑃦𑃕-𑃕𑃐𑃥𑃖-𑃠𑃣𑃐-𑃐𑃥-𑃔𑃝𑃙-𑃛𑃢-𑃖𑃔𑃨𑃦

𑃙𑃖𑃞 𑃐𑃝𑃚𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃣𑃒𑃠𑃜𑃞

(𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃢𑃙𑃥𑃐𑃖𑃝𑃤𑃑𑃤𑃞)

𑃤𑃑𑃜𑃛𑃤 𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃦 𑃒𑃠𑃕𑃚𑃢𑃖𑃐𑃑𑃑𑃠𑃢𑃕𑃑𑃦'𑃝𑃞𑃙 𑃐𑃖𑃜𑃟𑃐𑃖𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃦 𑃚𑃤𑃔𑃨𑃜𑃢𑃓𑃝𑃙𑃨𑃐𑃖𑃛𑃙𑃙𑃞 𑃐𑃥𑃕𑃑𑃦 𑃘𑃦𑃟𑃚𑃤𑃔𑃨𑃙𑃥𑃑𑃑𑃝𑃞 𑃛𑃥𑃝𑃥𑃐𑃨𑃔𑃨𑃖𑃜𑃐𑃢𑃝𑃑𑃠𑃤𑃞 𑃐𑃠𑃢𑃐𑃑𑃢 𑃔𑃨𑃣𑃚𑃖𑃙𑃥𑃐𑃨𑃜𑃢𑃙𑃨𑃢𑃖 𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃦 𑃒𑃠𑃕𑃚𑃢𑃙𑃤𑃑𑃤.

𑃙𑃤𑃐𑃨𑃜𑃙𑃔𑃨𑃞 𑃐 𑃑𑃑𑃠𑃢𑃕𑃑𑃞 𑃛𑃥𑃙𑃨𑃜𑃢𑃙𑃢𑃖, 𑃦𑃨𑃚𑃤𑃛𑃝𑃙𑃨𑃢𑃐𑃠𑃞 𑃟𑃥𑃐𑃠𑃘𑃖𑃥𑃘𑃢𑃙𑃢𑃖, 𑃦𑃨𑃘𑃗𑃟𑃝𑃤𑃑𑃞 𑃟𑃐𑃨𑃢𑃙𑃑𑃜𑃢, 𑃢𑃘𑃜𑃞 𑃛𑃥𑃙𑃨𑃜𑃙𑃤𑃔𑃨𑃠𑃢𑃙𑃢𑃙𑃢𑃖, 𑃓𑃤𑃑𑃝𑃤𑃑𑃦'𑃙𑃥𑃚𑃜𑃡𑃠𑃙𑃧𑃞 𑃟𑃥𑃐𑃥𑃖𑃤𑃑𑃦 𑃘𑃟𑃐𑃨𑃙𑃨𑃧𑃞, 𑃛𑃝𑃑𑃤𑃝𑃥𑃛𑃦 𑃕𑃦𑃓𑃝𑃣𑃙𑃨, 𑃦𑃨𑃛𑃝𑃑𑃤𑃟𑃥𑃘𑃦 𑃔𑃨𑃝𑃐𑃠𑃙𑃣𑃙, 𑃦𑃨𑃒𑃠𑃤𑃝𑃑𑃤𑃞 𑃐𑃠𑃝𑃔𑃨𑃔𑃨𑃠𑃢(𑃐𑃠𑃥𑃔𑃨𑃔𑃨𑃠𑃢)𑃔𑃨𑃠𑃤𑃖𑃥𑃟𑃑𑃢𑃙𑃢𑃖, 𑃦𑃨𑃙𑃒𑃠𑃤𑃒𑃠𑃥𑃑𑃞 𑃛𑃝𑃠𑃡𑃜𑃢, 𑃦𑃨𑃙𑃚𑃖𑃝𑃔𑃨𑃙𑃤𑃜𑃦 𑃒𑃘𑃧𑃞, 𑃐𑃠𑃢𑃐𑃑𑃢 𑃐𑃝𑃚𑃐𑃑𑃑𑃚𑃢𑃙𑃢𑃖, 𑃛𑃤𑃑𑃢 𑃒𑃦𑃔𑃨𑃠𑃤𑃐𑃑𑃑𑃚𑃢𑃙𑃢𑃖, 𑃝𑃢𑃠𑃢 𑃢𑃝𑃜𑃛𑃥𑃔𑃨𑃕𑃘𑃢𑃙𑃢𑃖, 𑃐𑃢𑃝𑃑𑃠𑃚𑃢𑃞𑃞 𑃙𑃤𑃝𑃚𑃢𑃙𑃨𑃙𑃕𑃝𑃐𑃖𑃛𑃝𑃐𑃑𑃠𑃤𑃑𑃢𑃙𑃢𑃖, 𑃦𑃨𑃛𑃝𑃖𑃣𑃜𑃦 𑃠𑃡𑃢𑃙𑃣𑃙, 𑃦𑃨𑃓𑃤𑃙𑃑𑃜𑃞 𑃛𑃝𑃑𑃤𑃒𑃠𑃢𑃙𑃣𑃙, 𑃚𑃤𑃐𑃠𑃥𑃔𑃨𑃔𑃨𑃠𑃞 𑃐𑃚𑃝𑃣𑃙𑃨, 𑃢𑃐𑃚𑃔𑃨𑃙𑃤𑃜𑃦 𑃕𑃠𑃦𑃐𑃨𑃣𑃙𑃨, 𑃦𑃨𑃐𑃣𑃓𑃙𑃟𑃦 𑃝𑃥𑃛𑃣𑃙𑃨, 𑃦𑃨𑃛𑃝𑃑𑃤𑃐𑃖𑃞 𑃟𑃢𑃜𑃣𑃙, 𑃦𑃨𑃘𑃤𑃛𑃑𑃞 𑃟𑃢𑃖𑃧𑃞, 𑃦𑃨𑃙𑃥𑃛𑃘𑃤𑃛𑃑𑃦 𑃝𑃥𑃛𑃧𑃞 𑃦𑃨𑃐𑃖𑃐𑃝𑃤𑃐𑃨𑃑𑃨 𑃢𑃝𑃥𑃛𑃜𑃧𑃞, 𑃚𑃤𑃛𑃝𑃖𑃥𑃟𑃑𑃞 𑃐𑃟𑃙𑃔𑃨𑃠𑃣𑃒𑃠𑃜𑃞, 𑃚𑃤𑃐𑃖𑃛𑃝𑃜𑃥𑃟𑃑𑃦 𑃔𑃨𑃠𑃢𑃑𑃥𑃒𑃠𑃤𑃞, 𑃐𑃖𑃚𑃝𑃤𑃑 𑃢𑃜𑃑𑃙𑃧𑃞, 𑃛𑃝𑃓𑃓𑃠𑃤𑃙𑃙𑃦 𑃕𑃝𑃙𑃑𑃠𑃧𑃞, 𑃚𑃤𑃖𑃥𑃟𑃑𑃞 𑃛𑃝𑃤𑃔𑃨𑃢𑃕𑃠𑃙𑃧𑃞, 𑃛𑃝𑃤𑃖𑃥𑃟𑃑𑃐𑃑𑃝𑃤𑃐𑃨𑃙𑃨𑃜𑃢, 𑃦𑃕𑃠𑃢𑃔𑃨𑃥𑃑𑃑𑃤𑃝𑃙𑃨𑃞 𑃛𑃝𑃤𑃛𑃥𑃝𑃙𑃨𑃦 𑃠𑃡𑃢𑃙𑃣𑃙, 𑃛𑃝𑃑𑃤𑃐𑃨𑃑𑃨𑃠𑃤𑃑𑃦'𑃑𑃤𑃑𑃢𑃙𑃢𑃕𑃑𑃛𑃝𑃑𑃜𑃥𑃑𑃛𑃙𑃙𑃢𑃙𑃢𑃖 𑃒𑃥𑃔𑃨𑃔𑃨𑃠𑃢𑃙𑃢𑃖 𑃒𑃠𑃕𑃚𑃑𑃢𑃖 𑃠𑃡𑃢𑃙𑃣, 𑃦𑃨𑃛𑃝𑃑𑃤𑃐𑃨𑃑𑃨𑃠𑃤𑃑𑃦 𑃙𑃤𑃝𑃚𑃢𑃙𑃨𑃣, 𑃐𑃑𑃠𑃤𑃑𑃦 𑃒𑃠𑃥𑃑𑃟𑃦𑃑𑃨𑃜𑃢𑃖, 𑃐𑃑𑃠𑃤𑃑𑃞 𑃐𑃝𑃚𑃐𑃑𑃑𑃚𑃢𑃘𑃦𑃟𑃙𑃤𑃜𑃢𑃜𑃢𑃖 𑃒𑃠𑃥𑃖𑃦𑃥, 𑃐𑃝𑃚 𑃤𑃖𑃣 𑃑𑃑𑃠𑃢𑃕𑃑𑃢𑃙𑃢𑃖 𑃚𑃤𑃐𑃠𑃣𑃐𑃨𑃑𑃞 𑃐𑃖𑃜𑃕 𑃕𑃥𑃙𑃨𑃢𑃞.

(𑃔𑃨𑃠𑃝𑃖𑃢𑃙𑃥𑃐𑃖𑃝𑃤𑃑𑃤𑃞)

𑃐𑃔𑃨𑃔𑃨𑃠𑃝𑃖𑃐𑃑𑃥 𑃢𑃔𑃨𑃦𑃥 𑃟𑃘𑃜𑃢𑃙𑃨𑃞, 𑃖𑃔𑃨𑃠𑃜𑃣 𑃟𑃘𑃜𑃢𑃙𑃨𑃞, 𑃛𑃝𑃜𑃚𑃐𑃢𑃙𑃣 𑃟𑃘𑃜𑃢𑃙𑃨𑃞, 𑃐𑃚𑃝𑃑𑃠𑃞 𑃐𑃥𑃚𑃜𑃡𑃠𑃙𑃞, 𑃟𑃣𑃚𑃘𑃞, 𑃛𑃝𑃤𑃛𑃥𑃝𑃙𑃨𑃞, 𑃛𑃝𑃤𑃐𑃠𑃥𑃔𑃨𑃔𑃨𑃠𑃞, 𑃛𑃝𑃜𑃚𑃔𑃨𑃢𑃑𑃞, 𑃐𑃚𑃢𑃟𑃠𑃜𑃢𑃑𑃞 𑃒𑃠𑃕𑃚𑃑𑃦 𑃔𑃨𑃠𑃝𑃖𑃞, 𑃐𑃢𑃙𑃔𑃨𑃝𑃤𑃐𑃨𑃑𑃨𑃤𑃟𑃞, 𑃙𑃤𑃝𑃠𑃚𑃝𑃞, 𑃢𑃟𑃢𑃘𑃤𑃟𑃞, 𑃦𑃥𑃛𑃙𑃢𑃜𑃤𑃟𑃞, 𑃧𑃞𑃤𑃛𑃐𑃠𑃜𑃤𑃟𑃞, 𑃛𑃝𑃑𑃜𑃢𑃑𑃖𑃚𑃣𑃔𑃨𑃙𑃤𑃜𑃦 𑃚𑃤𑃠𑃡𑃧𑃞, 𑃐𑃚𑃢𑃟𑃠𑃜𑃢𑃑𑃦 𑃒𑃠𑃕𑃚𑃑𑃦 𑃔𑃨𑃠𑃝𑃖𑃚𑃤𑃙𑃜𑃞 𑃐𑃥𑃛𑃝𑃚𑃣𑃔𑃨𑃤𑃑𑃞 𑃙𑃧𑃝𑃜𑃢𑃙𑃨𑃤𑃟𑃞, 𑃐𑃖𑃒𑃦𑃔𑃨𑃠𑃤𑃕𑃢𑃖𑃤, 𑃦𑃨𑃒𑃠𑃤𑃙𑃙𑃞 𑃐𑃖𑃐𑃑𑃥𑃛𑃞, 𑃐𑃛𑃝𑃑𑃤𑃐𑃠𑃝𑃙𑃨𑃞, 𑃓𑃠𑃤𑃙𑃙𑃛𑃘𑃦𑃑𑃤𑃟𑃞.

(𑃐𑃖𑃕𑃠𑃢𑃙𑃥𑃐𑃖𑃝𑃤𑃑𑃤𑃞)

𑃐𑃥𑃛𑃝𑃑𑃤𑃛𑃙𑃙𑃦 𑃒𑃠𑃕𑃚𑃑 𑃢𑃝𑃜𑃐𑃖𑃕𑃠𑃞, 𑃙𑃜𑃢𑃜𑃛𑃝𑃑𑃤𑃛𑃙𑃙𑃞, 𑃝𑃤𑃠𑃥𑃛𑃝𑃑𑃤𑃛𑃙𑃙𑃞, 𑃐𑃢𑃖𑃤𑃓𑃤𑃛𑃝𑃑𑃤𑃛𑃙𑃙𑃞, 𑃦𑃨𑃡𑃠𑃘𑃤𑃟𑃝𑃙𑃨𑃤𑃜𑃞, 𑃐𑃢𑃖𑃤𑃓𑃤𑃟𑃝𑃙𑃨𑃤𑃜𑃞, 𑃛𑃥𑃙𑃨𑃜𑃐𑃠𑃝𑃤𑃟𑃐𑃨𑃣𑃑𑃝𑃞, 𑃖𑃞𑃢𑃔𑃨𑃟𑃐𑃨𑃤𑃙𑃨𑃢𑃛𑃝𑃤𑃐𑃠𑃦𑃔𑃨𑃠𑃟𑃞, 𑃛𑃝𑃢𑃞𑃚𑃙𑃤𑃜𑃞, 𑃢𑃞𑃚𑃙𑃤𑃜𑃞.
