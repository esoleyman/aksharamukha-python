𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣙𑣜𑣃𑣔𑣅𑣞𑣇𑣕𑣜𑣖.

[𑣞𑣀𑣌𑣞𑣂𑣘𑣕𑣖𑣁𑣕𑣜𑣃𑣌𑣁]

.. 𑣓𑣖𑣄 𑣞𑣜𑣟𑣎𑣍𑣁𑣅 ..

.. 𑣿 𑣓𑣖𑣉 𑣗𑣋𑣟𑣕𑣅𑣆 𑣁𑣙𑣜𑣅𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣅𑣆 ..

𑣁𑣙𑣜𑣅𑣁𑣟𑣚𑣉𑣌𑣂𑣕𑣈𑣞𑣟𑣜𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣉 𑣋𑣖𑣗𑣆𑣜𑣁𑣅𑣁𑣀 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣅𑣁𑣀 𑣏𑣜𑣅𑣁𑣀 𑣏𑣜𑣖𑣁𑣐𑣉 𑣟𑣅𑣟𑣚𑣉𑣌𑣅𑣕𑣂 𑣞𑣖. 𑣘𑣍𑣏 𑣞𑣌𑣓𑣔𑣁𑣄, 𑣕𑣁𑣀𑣞𑣏 𑣞𑣟𑣗𑣁𑣟𑣞𑣇𑣓𑣅𑣁𑣓 𑣘𑣞𑣅𑣕𑣂 𑣞𑣖..

𑣂𑣙 𑣞𑣁𑣜𑣂𑣘𑣃𑣕𑣜 𑣜𑣇𑣘𑣀 𑣞𑣇𑣓𑣅𑣕𑣁, 𑣞𑣇𑣓𑣅𑣕𑣆𑣟 𑣜𑣇𑣘𑣖. 𑣜𑣇𑣘𑣁𑣓𑣓 𑣘𑣜𑣃𑣕𑣌 𑣞𑣇𑣓𑣅𑣕𑣁, 𑣞𑣇𑣓𑣅𑣕𑣁𑣅𑣁 𑣓 𑣘𑣜𑣃𑣕𑣋 𑣜𑣇𑣘𑣖. 𑣅𑣔𑣜𑣇𑣘𑣀 𑣞𑣁 𑣞𑣇𑣓𑣅𑣕𑣁, 𑣅𑣁 𑣞𑣇𑣓𑣅𑣕𑣁 𑣕𑣔𑣜𑣇𑣘𑣖..

𑣈𑣟𑣖𑣈𑣟 𑣟𑣈𑣔𑣓𑣁𑣞𑣀𑣎𑣍𑣁𑣞𑣀𑣞𑣌𑣁𑣜𑣟𑣂𑣎𑣍𑣁𑣓𑣁𑣓𑣂..

𑣂𑣙𑣀 𑣞𑣁𑣜𑣂𑣘𑣃𑣕𑣜 𑣞𑣜𑣟𑣔𑣜𑣖𑣁𑣄 𑣞𑣇𑣓𑣅𑣕𑣁𑣚𑣌𑣞𑣐𑣁 𑣁𑣓𑣃𑣕𑣘𑣓𑣓𑣁 𑣁𑣓𑣂𑣜𑣃𑣔𑣔𑣁 𑣁𑣖𑣚𑣁 𑣓 𑣟𑣂𑣖𑣚𑣁 𑣓𑣉𑣓𑣁 𑣓 𑣘𑣜𑣂𑣘𑣇𑣜𑣐𑣁𑣄. 𑣕𑣞𑣖𑣁𑣏𑣏𑣁𑣜𑣂𑣘𑣃𑣕𑣜 𑣞𑣇𑣓𑣅𑣕𑣁𑣅𑣁𑣀 𑣓 𑣜𑣇𑣘𑣖, 𑣓 𑣟𑣈𑣔𑣓𑣁, 𑣓 𑣞𑣀𑣎𑣍𑣁, 𑣓 𑣞𑣀𑣞𑣌𑣁𑣜𑣁𑣄, 𑣓 𑣟𑣂𑣎𑣍𑣁𑣓𑣁𑣓𑣂. 𑣓 𑣏𑣌𑣞𑣃𑣄𑣞𑣜𑣉𑣕𑣜𑣋𑣜𑣁𑣐𑣎𑣂𑣙𑣟𑣁𑣌𑣁𑣅𑣖𑣓𑣁𑣀𑣞𑣂, 𑣓 𑣜𑣇𑣘𑣞𑣗𑣔𑣋𑣓𑣔𑣜𑣞𑣞𑣘𑣜𑣞𑣒𑣟𑣅𑣔𑣜𑣖𑣁𑣄. 𑣓 𑣏𑣌𑣞𑣃𑣜𑣔𑣁𑣕𑣃𑣜𑣅𑣁𑣟𑣓𑣓 𑣖𑣓𑣉𑣔𑣁𑣕𑣃𑣄..

𑣓 𑣟𑣂𑣔𑣅𑣁 𑣓𑣁𑣟𑣂𑣔𑣅𑣁 𑣓 𑣟𑣂𑣔𑣅𑣁𑣌𑣞𑣅𑣉 𑣓𑣁𑣟𑣂𑣔𑣅𑣁𑣌𑣞𑣅𑣉 𑣅𑣁𑣟𑣓𑣓 𑣎𑣜𑣁𑣖𑣜𑣐𑣀 𑣓 𑣎𑣜𑣁𑣖𑣜𑣐𑣌𑣞𑣅𑣉 𑣓 𑣔𑣃𑣄𑣌𑣞𑣖𑣃𑣔𑣅𑣓𑣂𑣜𑣉𑣔𑣖𑣁𑣜𑣋𑣁 𑣓 𑣎𑣍𑣁𑣓𑣀 𑣓 𑣘𑣜𑣁𑣘𑣕𑣂𑣕𑣟𑣖..

𑣗𑣉𑣔𑣂𑣞𑣕𑣕𑣟𑣞𑣅 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣖𑣁𑣞𑣜𑣂𑣕𑣅 𑣟𑣂𑣙𑣜𑣕𑣂 𑣏𑣂𑣕𑣕𑣁𑣟𑣜𑣐𑣄. 𑣏𑣂𑣕𑣕𑣁𑣟𑣜𑣐𑣓𑣁𑣞𑣕𑣂𑣕𑣟𑣁𑣔𑣕𑣜𑣞𑣕𑣉 𑣟𑣂𑣘𑣜𑣅𑣁𑣞𑣁𑣕𑣂𑣌𑣜𑣁𑣓𑣕𑣉 𑣓𑣂𑣞𑣒𑣓𑣂𑣜𑣟𑣁𑣐𑣄. 𑣕𑣜𑣅𑣔𑣟𑣟𑣅𑣟𑣞𑣕𑣂𑣕𑣁𑣄 𑣞𑣜𑣟𑣗𑣃𑣔𑣔𑣁𑣄 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣖𑣁𑣞𑣜𑣂𑣕𑣅 𑣁𑣓𑣃𑣕𑣕𑣜𑣁𑣀 𑣞𑣖𑣅𑣌𑣞𑣀𑣗𑣉𑣔𑣂𑣖𑣗𑣂𑣞𑣀𑣗𑣃𑣔𑣔𑣁𑣄..

𑣕𑣞𑣖𑣁𑣎𑣎𑣍𑣁𑣕𑣟𑣅𑣄 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣖𑣙𑣁𑣖𑣓𑣕𑣜𑣉 𑣖𑣙𑣁𑣟𑣂𑣔𑣅𑣁𑣖𑣓𑣕𑣜𑣉'𑣓𑣃𑣕𑣕𑣜𑣖𑣓𑣕𑣜𑣉'𑣞𑣖𑣞𑣖𑣖𑣓𑣕𑣜𑣄 𑣞𑣜𑣟𑣔𑣃𑣄𑣌𑣘𑣜𑣞𑣖𑣓𑣄 𑣞𑣕𑣅𑣖𑣖𑣂𑣕𑣅𑣕𑣟𑣁𑣕 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣅𑣁𑣖𑣃𑣌𑣕𑣉 𑣖𑣓𑣕𑣜𑣄.

𑣕𑣔𑣅𑣕𑣁- 𑣋𑣕𑣈 𑣋𑣕𑣈 𑣘𑣁𑣜𑣋𑣕𑣈 𑣘𑣁𑣜𑣞𑣀𑣋𑣕𑣈 𑣗𑣉𑣔𑣂 𑣞𑣟𑣁𑣙𑣁..

𑣂𑣕𑣂 𑣘𑣜𑣎𑣍𑣁𑣘𑣁𑣜𑣖𑣂𑣕𑣁𑣙𑣜𑣃𑣔𑣅𑣞𑣇𑣕𑣜𑣀 𑣞𑣖𑣁𑣘𑣕𑣖..
