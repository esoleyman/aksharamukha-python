𑈁𑈦𑈵𑈥𑈙𑈵𑈦𑈭𑈦𑈙𑈵𑈞𑈬𑈞𑈯𑈩𑈵𑈤𑈵𑈦𑈭𑈙𑈭𑈩𑈯𑈙𑈵𑈦𑈤𑈵

(𑈉)𑈺𑈩𑈴𑈩𑈵𑈈𑈵𑈦𑈭𑈙𑈟𑈬𑈕𑈪𑈵

𑈣𑈬𑈦𑈙𑈮𑈥𑈣𑈬𑈩𑈶𑈬𑈥𑈬𑈤𑈵-𑈁𑈦𑈵𑈥𑈙𑈵𑈦𑈭𑈦𑈙𑈵𑈞𑈬𑈞𑈯𑈩𑈵𑈤𑈵𑈦𑈭𑈙𑈭𑈩𑈯𑈙𑈵𑈦𑈤𑈵

𑈣𑈲𑈔𑈣𑈬𑈩𑈶𑈬𑈥𑈬𑈤𑈵-𑈠𑈊𑈵𑈩𑈵-𑈟𑈬-𑈛𑈈𑈲𑈞𑈵-𑈤𑈵𑈏𑈷𑈲𑈊𑈵-𑈊𑈵𑈩𑈯𑈤𑈵-𑈐𑈰𑈩𑈵-𑈩𑈯-𑈖𑈵𑈦𑈞𑈵-𑈟𑈬-𑈤𑈵𑈛𑈲

𑈞𑈤𑈪𑈵𑈺𑈩𑈦𑈵𑈨𑈡𑈯𑈝𑈷𑈡𑈲𑈝𑈭𑈩𑈙𑈷𑈵𑈨𑈰𑈣𑈵𑈥𑈪𑈵

(𑈡𑈯𑈝𑈷𑈬𑈞𑈯𑈩𑈵𑈤𑈵𑈦𑈭𑈙𑈭𑈪𑈵)

𑈂𑈙𑈵𑈥𑈟𑈭𑈺𑈡𑈯𑈝𑈷𑈲𑈺𑈣𑈊𑈨𑈬𑈴𑈩𑈵𑈙𑈚𑈬𑈊𑈙𑈲𑈦𑈵𑈪𑈞𑈵𑈺𑈩𑈤𑈵𑈥𑈈𑈵𑈩𑈴𑈡𑈯𑈝𑈷𑈲𑈺𑈨𑈭𑈛𑈵𑈥𑈬𑈎𑈦𑈘𑈩𑈤𑈵𑈟𑈞𑈷𑈪𑈵𑈺𑈩𑈯𑈊𑈙𑈲𑈺𑈧𑈲𑈈𑈨𑈭𑈛𑈞𑈯𑈙𑈷𑈦𑈪𑈵𑈺𑈟𑈯𑈦𑈯𑈩𑈶𑈛𑈤𑈵𑈥𑈩𑈬𑈦𑈚𑈭𑈪𑈵𑈺𑈩𑈶𑈬𑈩𑈵𑈙𑈬𑈺𑈛𑈰𑈨𑈤𑈞𑈯𑈩𑈶𑈵𑈥𑈬𑈘𑈬𑈴𑈺𑈡𑈯𑈝𑈷𑈲𑈺𑈣𑈊𑈨𑈬𑈞𑈭𑈙𑈭𑈸

𑈞𑈭𑈩𑈶𑈵𑈥𑈞𑈵𑈛𑈪𑈵𑈺𑈩𑈺𑈙𑈚𑈬𑈊𑈙𑈪𑈵𑈺𑈟𑈯𑈘𑈵𑈥𑈬𑈞𑈬𑈤𑈵,𑈺𑈀𑈨𑈭𑈟𑈵𑈦𑈘𑈬𑈩𑈶𑈪𑈵𑈺𑈈𑈯𑈩𑈶𑈧𑈤𑈯𑈧𑈬𑈞𑈬𑈤𑈵,𑈺𑈀𑈧𑈍𑈵𑈈𑈵𑈦𑈭𑈙𑈪𑈵𑈺𑈈𑈵𑈩𑈶𑈬𑈞𑈵𑈙𑈵𑈥𑈬,𑈺𑈁𑈧𑈥𑈪𑈵𑈺𑈟𑈯𑈘𑈵𑈥𑈞𑈭𑈝𑈬𑈞𑈬𑈞𑈬𑈤𑈵,𑈺𑈎𑈭𑈙𑈵𑈦𑈭𑈙𑈲𑈞𑈯𑈨𑈵𑈥𑈓𑈵𑈐𑈞𑈱𑈪𑈵𑈺𑈈𑈯𑈩𑈯𑈤𑈭𑈙𑈲𑈺𑈧𑈈𑈵𑈩𑈶𑈘𑈱𑈪𑈵,𑈺𑈟𑈵𑈦𑈙𑈭𑈦𑈯𑈟𑈲𑈺𑈊𑈲𑈎𑈦𑈰𑈘,𑈺𑈀𑈟𑈵𑈦𑈙𑈭𑈈𑈯𑈧𑈲𑈺𑈛𑈦𑈵𑈩𑈶𑈞𑈰𑈞,𑈺𑈀𑈣𑈭𑈦𑈙𑈭𑈪𑈵𑈺𑈩𑈶𑈵𑈦𑈝𑈷𑈬(𑈩𑈶𑈯𑈝𑈷𑈬)𑈝𑈭𑈤𑈯𑈈𑈵𑈙𑈬𑈞𑈬𑈤𑈵,𑈺𑈀𑈞𑈣𑈭𑈣𑈯𑈙𑈪𑈵𑈺𑈟𑈵𑈦𑈐𑈵𑈓𑈥𑈬,𑈺𑈀𑈞𑈨𑈤𑈦𑈵𑈛𑈞𑈮𑈥𑈲𑈺𑈡𑈧𑈱𑈪𑈵,𑈺𑈩𑈶𑈬𑈩𑈵𑈙𑈬𑈺𑈩𑈦𑈵𑈨𑈩𑈙𑈷𑈵𑈨𑈬𑈞𑈬𑈤𑈵,𑈺𑈟𑈭𑈙𑈬𑈺𑈡𑈲𑈝𑈭𑈩𑈙𑈷𑈵𑈨𑈬𑈞𑈬𑈤𑈵,𑈺𑈦𑈬𑈐𑈬𑈺𑈁𑈦𑈵𑈥𑈟𑈯𑈛𑈵𑈊𑈧𑈬𑈞𑈬𑈤𑈵,𑈺𑈩𑈬𑈦𑈵𑈚𑈨𑈬𑈪𑈪𑈵𑈺𑈞𑈭𑈦𑈵𑈨𑈬𑈘𑈞𑈊𑈦𑈩𑈤𑈵𑈟𑈵𑈦𑈩𑈵𑈚𑈭𑈙𑈬𑈞𑈬𑈤𑈵,𑈺𑈀𑈟𑈵𑈦𑈤𑈰𑈥𑈲𑈺𑈐𑈵𑈓𑈬𑈞𑈰𑈞,𑈺𑈀𑈎𑈭𑈞𑈵𑈙𑈵𑈥𑈪𑈵𑈺𑈟𑈵𑈦𑈙𑈭𑈣𑈬𑈞𑈰𑈞,𑈺𑈨𑈭𑈩𑈶𑈯𑈝𑈷𑈪𑈵𑈺𑈩𑈵𑈨𑈦𑈰𑈘,𑈺𑈁𑈩𑈵𑈨𑈛𑈞𑈮𑈥𑈲𑈺𑈌𑈲𑈩𑈶𑈰𑈘,𑈺𑈀𑈩𑈰𑈎𑈞𑈈𑈲𑈺𑈦𑈯𑈟𑈰𑈘,𑈺𑈀𑈟𑈵𑈦𑈙𑈭𑈩𑈤𑈪𑈵𑈺𑈈𑈬𑈥𑈰𑈞,𑈺𑈀𑈧𑈭𑈟𑈵𑈙𑈪𑈵𑈺𑈈𑈬𑈤𑈱𑈪𑈵,𑈺𑈀𑈞𑈯𑈟𑈧𑈭𑈟𑈵𑈙𑈲𑈺𑈦𑈯𑈟𑈱𑈪𑈵𑈺𑈀𑈩𑈴𑈩𑈵𑈦𑈭𑈩𑈶𑈵𑈔𑈺𑈁𑈦𑈯𑈟𑈵𑈥𑈱𑈪𑈵,𑈺𑈨𑈭𑈟𑈵𑈦𑈤𑈯𑈈𑈵𑈙𑈪𑈵𑈺𑈩𑈵𑈈𑈞𑈵𑈝𑈰𑈣𑈵𑈥𑈪𑈵,𑈺𑈨𑈭𑈩𑈤𑈵𑈟𑈵𑈦𑈥𑈯𑈈𑈵𑈙𑈲𑈺𑈝𑈬𑈙𑈯𑈣𑈭𑈪𑈵,𑈺𑈩𑈴𑈨𑈵𑈦𑈭𑈙𑈺𑈁𑈥𑈙𑈞𑈱𑈪𑈵,𑈺𑈟𑈵𑈦𑈏𑈷𑈭𑈞𑈷𑈲𑈺𑈊𑈵𑈦𑈞𑈵𑈚𑈱𑈪𑈵,𑈺𑈨𑈭𑈤𑈯𑈈𑈵𑈙𑈪𑈵𑈺𑈟𑈦𑈭𑈛𑈬𑈌𑈵𑈞𑈱𑈪𑈵,𑈺𑈟𑈦𑈭𑈤𑈯𑈈𑈵𑈙𑈩𑈵𑈙𑈵𑈦𑈭𑈩𑈶𑈵𑈘𑈥𑈬,𑈺𑈆𑈌𑈬𑈛𑈯𑈙𑈷𑈮𑈦𑈵𑈘𑈪𑈵𑈺𑈟𑈦𑈭𑈟𑈯𑈦𑈵𑈘𑈲𑈺𑈐𑈵𑈓𑈬𑈞𑈰𑈞,𑈺𑈟𑈵𑈦𑈙𑈭𑈩𑈶𑈵𑈕𑈭𑈙𑈲𑈙𑈮𑈙𑈬𑈞𑈬𑈊𑈙𑈟𑈵𑈦𑈙𑈵𑈥𑈯𑈙𑈵𑈟𑈞𑈷𑈬𑈞𑈬𑈴𑈺𑈡𑈯𑈝𑈷𑈬𑈞𑈬𑈴𑈺𑈣𑈊𑈨𑈙𑈬𑈴𑈺𑈐𑈵𑈓𑈬𑈞𑈰,𑈺𑈀𑈟𑈵𑈦𑈙𑈭𑈩𑈶𑈵𑈕𑈭𑈙𑈲𑈺𑈞𑈭𑈦𑈵𑈨𑈬𑈘𑈰,𑈺𑈩𑈵𑈚𑈭𑈙𑈲𑈺𑈣𑈯𑈙𑈈𑈲𑈔𑈵𑈥𑈬𑈤𑈵,𑈺𑈩𑈵𑈚𑈭𑈙𑈪𑈵𑈺𑈩𑈦𑈵𑈨𑈩𑈙𑈷𑈵𑈨𑈬𑈧𑈲𑈈𑈞𑈮𑈥𑈬𑈥𑈬𑈴𑈺𑈣𑈯𑈤𑈳,𑈺𑈩𑈦𑈵𑈨𑈺𑈂𑈤𑈰𑈺𑈙𑈚𑈬𑈊𑈙𑈬𑈞𑈬𑈴𑈺𑈨𑈭𑈩𑈶𑈰𑈩𑈶𑈙𑈪𑈵𑈺𑈩𑈤𑈵𑈥𑈊𑈵𑈺𑈊𑈯𑈘𑈬𑈪𑈵𑈸

(𑈝𑈦𑈵𑈤𑈬𑈞𑈯𑈩𑈵𑈤𑈵𑈦𑈭𑈙𑈭𑈪𑈵)

𑈩𑈝𑈷𑈦𑈵𑈤𑈩𑈵𑈙𑈯𑈺𑈁𑈛𑈳𑈺𑈈𑈧𑈵𑈥𑈬𑈘𑈪𑈵,𑈺𑈤𑈝𑈵𑈥𑈰𑈺𑈈𑈧𑈵𑈥𑈬𑈘𑈪𑈵,𑈺𑈟𑈦𑈵𑈥𑈨𑈩𑈬𑈞𑈰𑈺𑈈𑈧𑈵𑈥𑈬𑈘𑈪𑈵,𑈺𑈩𑈵𑈨𑈦𑈵𑈚𑈪𑈵𑈺𑈩𑈯𑈨𑈵𑈥𑈓𑈵𑈐𑈞𑈪𑈵,𑈺𑈈𑈰𑈨𑈧𑈪𑈵,𑈺𑈟𑈦𑈭𑈟𑈯𑈦𑈵𑈘𑈪𑈵,𑈺𑈟𑈦𑈭𑈩𑈶𑈯𑈝𑈷𑈪𑈵,𑈺𑈟𑈦𑈵𑈥𑈨𑈛𑈬𑈙𑈪𑈵,𑈺𑈩𑈵𑈨𑈬𑈉𑈵𑈥𑈬𑈙𑈪𑈵𑈺𑈣𑈊𑈨𑈙𑈲𑈺𑈝𑈦𑈵𑈤𑈪𑈵,𑈺𑈩𑈬𑈞𑈵𑈛𑈵𑈦𑈭𑈩𑈶𑈵𑈔𑈭𑈈𑈪𑈵,𑈺𑈞𑈭𑈦𑈵𑈐𑈵𑈨𑈦𑈪𑈵,𑈺𑈁𑈈𑈬𑈧𑈭𑈈𑈪𑈵,𑈺𑈇𑈟𑈞𑈬𑈥𑈭𑈈𑈪𑈵,𑈺𑈅𑈪𑈭𑈟𑈩𑈶𑈵𑈥𑈭𑈈𑈪𑈵,𑈺𑈟𑈵𑈦𑈙𑈵𑈥𑈬𑈙𑈵𑈤𑈨𑈰𑈛𑈞𑈮𑈥𑈲𑈺𑈨𑈭𑈐𑈵𑈓𑈱𑈪𑈵,𑈺𑈩𑈵𑈨𑈬𑈉𑈵𑈥𑈬𑈙𑈲𑈺𑈣𑈊𑈨𑈙𑈲𑈺𑈝𑈦𑈵𑈤𑈨𑈭𑈞𑈥𑈪𑈵𑈺𑈩𑈯𑈟𑈵𑈦𑈨𑈰𑈛𑈭𑈙𑈪𑈵𑈺𑈞𑈱𑈦𑈵𑈥𑈬𑈘𑈭𑈈𑈪𑈵,𑈺𑈩𑈴𑈡𑈲𑈝𑈭𑈊𑈬𑈤𑈮,𑈺𑈀𑈣𑈭𑈞𑈷𑈪𑈵𑈺𑈩𑈴𑈩𑈵𑈙𑈯𑈟𑈪𑈵,𑈺𑈩𑈟𑈵𑈦𑈙𑈭𑈩𑈶𑈦𑈘𑈪𑈵,𑈺𑈏𑈭𑈞𑈷𑈟𑈵𑈧𑈲𑈙𑈭𑈈𑈪𑈵𑈸

(𑈩𑈴𑈌𑈬𑈞𑈯𑈩𑈵𑈤𑈵𑈦𑈭𑈙𑈭𑈪𑈵)

𑈩𑈯𑈟𑈵𑈦𑈙𑈭𑈟𑈞𑈷𑈲𑈺𑈣𑈊𑈨𑈙𑈺𑈁𑈦𑈵𑈥𑈩𑈴𑈌𑈪𑈵,𑈺𑈞𑈵𑈥𑈬𑈥𑈟𑈵𑈦𑈙𑈭𑈟𑈞𑈷𑈪𑈵,𑈺𑈦𑈭𑈐𑈯𑈟𑈵𑈦𑈙𑈭𑈟𑈞𑈷𑈪𑈵,𑈺𑈩𑈬𑈤𑈮𑈎𑈮𑈟𑈵𑈦𑈙𑈭𑈟𑈞𑈷𑈪𑈵,𑈺𑈀𑈓𑈵𑈐𑈧𑈮𑈈𑈦𑈘𑈮𑈥𑈪𑈵,𑈺𑈩𑈬𑈤𑈮𑈎𑈮𑈈𑈦𑈘𑈮𑈥𑈪𑈵,𑈺𑈟𑈯𑈘𑈵𑈥𑈩𑈶𑈵𑈦𑈮𑈈𑈵𑈩𑈶𑈰𑈙𑈵𑈦𑈪𑈵,𑈺𑈤𑈪𑈬𑈛𑈈𑈵𑈩𑈶𑈭𑈘𑈬𑈟𑈦𑈭𑈩𑈶𑈲𑈝𑈈𑈪𑈵,𑈺𑈟𑈵𑈦𑈬𑈪𑈨𑈞𑈮𑈥𑈪𑈵,𑈺𑈁𑈪𑈨𑈞𑈮𑈥𑈪𑈵𑈸
