𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌹𑍃𑌦𑌯𑌸𑍂𑌤𑍍𑌰𑌮𑍍।

[𑌸𑌂𑌕𑍍𑌷𑌿𑌪𑍍𑌤𑌮𑌾𑌤𑍃𑌕𑌾]

॥ 𑌨𑌮𑌃 𑌸𑌰𑍍𑌵𑌜𑍍𑌞𑌾𑌯 ॥

॥ 𑍐 𑌨𑌮𑍋 𑌭𑌗𑌵𑌤𑍍𑌯𑍈 𑌆𑌰𑍍𑌯𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌯𑍈 ॥

𑌆𑌰𑍍𑌯𑌾𑌵𑌲𑍋𑌕𑌿𑌤𑍇𑌶𑍍𑌵𑌰𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑍋 𑌗𑌮𑍍𑌭𑍀𑌰𑌾𑌯𑌾𑌂 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌯𑌾𑌂 𑌚𑌰𑍍𑌯𑌾𑌂 𑌚𑌰𑌮𑌾𑌣𑍋 𑌵𑍍𑌯𑌵𑌲𑍋𑌕𑌯𑌤𑌿 𑌸𑍍𑌮। 𑌪𑌞𑍍𑌚 𑌸𑍍𑌕𑌨𑍍𑌧𑌾𑌃, 𑌤𑌾𑌂𑌶𑍍𑌚 𑌸𑍍𑌵𑌭𑌾𑌵𑌶𑍂𑌨𑍍𑌯𑌾𑌨𑍍 𑌪𑌶𑍍𑌯𑌤𑌿 𑌸𑍍𑌮॥

𑌇𑌹 𑌶𑌾𑌰𑌿𑌪𑍁𑌤𑍍𑌰 𑌰𑍂𑌪𑌂 𑌶𑍂𑌨𑍍𑌯𑌤𑌾, 𑌶𑍂𑌨𑍍𑌯𑌤𑍈𑌵 𑌰𑍂𑌪𑌮𑍍। 𑌰𑍂𑌪𑌾𑌨𑍍𑌨 𑌪𑍃𑌥𑌕𑍍 𑌶𑍂𑌨𑍍𑌯𑌤𑌾, 𑌶𑍂𑌨𑍍𑌯𑌤𑌾𑌯𑌾 𑌨 𑌪𑍃𑌥𑌗𑍍 𑌰𑍂𑌪𑌮𑍍। 𑌯𑌦𑍍𑌰𑍂𑌪𑌂 𑌸𑌾 𑌶𑍂𑌨𑍍𑌯𑌤𑌾, 𑌯𑌾 𑌶𑍂𑌨𑍍𑌯𑌤𑌾 𑌤𑌦𑍍𑌰𑍂𑌪𑌮𑍍॥

𑌏𑌵𑌮𑍇𑌵 𑌵𑍇𑌦𑌨𑌾𑌸𑌂𑌜𑍍𑌞𑌾𑌸𑌂𑌸𑍍𑌕𑌾𑌰𑌵𑌿𑌜𑍍𑌞𑌾𑌨𑌾𑌨𑌿॥

𑌇𑌹𑌂 𑌶𑌾𑌰𑌿𑌪𑍁𑌤𑍍𑌰 𑌸𑌰𑍍𑌵𑌧𑌰𑍍𑌮𑌾𑌃 𑌶𑍂𑌨𑍍𑌯𑌤𑌾𑌲𑌕𑍍𑌷𑌣𑌾 𑌅𑌨𑍁𑌤𑍍𑌪𑌨𑍍𑌨𑌾 𑌅𑌨𑌿𑌰𑍁𑌦𑍍𑌧𑌾 𑌅𑌮𑌲𑌾 𑌨 𑌵𑌿𑌮𑌲𑌾 𑌨𑍋𑌨𑌾 𑌨 𑌪𑌰𑌿𑌪𑍂𑌰𑍍𑌣𑌾𑌃। 𑌤𑌸𑍍𑌮𑌾𑌚𑍍𑌛𑌾𑌰𑌿𑌪𑍁𑌤𑍍𑌰 𑌶𑍂𑌨𑍍𑌯𑌤𑌾𑌯𑌾𑌂 𑌨 𑌰𑍂𑌪𑌮𑍍, 𑌨 𑌵𑍇𑌦𑌨𑌾, 𑌨 𑌸𑌂𑌜𑍍𑌞𑌾, 𑌨 𑌸𑌂𑌸𑍍𑌕𑌾𑌰𑌾𑌃, 𑌨 𑌵𑌿𑌜𑍍𑌞𑌾𑌨𑌾𑌨𑌿। 𑌨 𑌚𑌕𑍍𑌷𑍁𑌃𑌶𑍍𑌰𑍋𑌤𑍍𑌰𑌘𑍍𑌰𑌾𑌣𑌜𑌿𑌹𑍍𑌵𑌾𑌕𑌾𑌯𑌮𑌨𑌾𑌂𑌸𑌿, 𑌨 𑌰𑍂𑌪𑌶𑌬𑍍𑌦𑌗𑌨𑍍𑌧𑌰𑌸𑌸𑍍𑌪𑍍𑌰𑌷𑍍𑌟𑌵𑍍𑌯𑌧𑌰𑍍𑌮𑌾𑌃। 𑌨 𑌚𑌕𑍍𑌷𑍁𑌰𑍍𑌧𑌾𑌤𑍁𑌰𑍍𑌯𑌾𑌵𑌨𑍍𑌨 𑌮𑌨𑍋𑌧𑌾𑌤𑍁𑌃॥

𑌨 𑌵𑌿𑌦𑍍𑌯𑌾 𑌨𑌾𑌵𑌿𑌦𑍍𑌯𑌾 𑌨 𑌵𑌿𑌦𑍍𑌯𑌾𑌕𑍍𑌷𑌯𑍋 𑌨𑌾𑌵𑌿𑌦𑍍𑌯𑌾𑌕𑍍𑌷𑌯𑍋 𑌯𑌾𑌵𑌨𑍍𑌨 𑌜𑌰𑌾𑌮𑌰𑌣𑌂 𑌨 𑌜𑌰𑌾𑌮𑌰𑌣𑌕𑍍𑌷𑌯𑍋 𑌨 𑌦𑍁𑌃𑌖𑌸𑌮𑍁𑌦𑌯𑌨𑌿𑌰𑍋𑌧𑌮𑌾𑌰𑍍𑌗𑌾 𑌨 𑌜𑍍𑌞𑌾𑌨𑌂 𑌨 𑌪𑍍𑌰𑌾𑌪𑍍𑌤𑌿𑌤𑍍𑌵𑌮𑍍॥

𑌬𑍋𑌧𑌿𑌸𑌤𑍍𑌤𑍍𑌵𑌸𑍍𑌯 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌮𑌾𑌶𑍍𑌰𑌿𑌤𑍍𑌯 𑌵𑌿𑌹𑌰𑌤𑌿 𑌚𑌿𑌤𑍍𑌤𑌾𑌵𑌰𑌣𑌃। 𑌚𑌿𑌤𑍍𑌤𑌾𑌵𑌰𑌣𑌨𑌾𑌸𑍍𑌤𑌿𑌤𑍍𑌵𑌾𑌦𑌤𑍍𑌰𑌸𑍍𑌤𑍋 𑌵𑌿𑌪𑌰𑍍𑌯𑌾𑌸𑌾𑌤𑌿𑌕𑍍𑌰𑌾𑌨𑍍𑌤𑍋 𑌨𑌿𑌷𑍍𑌠𑌨𑌿𑌰𑍍𑌵𑌾𑌣𑌃। 𑌤𑍍𑌰𑍍𑌯𑌧𑍍𑌵𑌵𑍍𑌯𑌵𑌸𑍍𑌥𑌿𑌤𑌾𑌃 𑌸𑌰𑍍𑌵𑌬𑍁𑌦𑍍𑌧𑌾𑌃 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌮𑌾𑌶𑍍𑌰𑌿𑌤𑍍𑌯 𑌅𑌨𑍁𑌤𑍍𑌤𑌰𑌾𑌂 𑌸𑌮𑍍𑌯𑌕𑍍𑌸𑌂𑌬𑍋𑌧𑌿𑌮𑌭𑌿𑌸𑌂𑌬𑍁𑌦𑍍𑌧𑌾𑌃॥

𑌤𑌸𑍍𑌮𑌾𑌜𑍍𑌜𑍍𑌞𑌾𑌤𑌵𑍍𑌯𑌃 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌮𑌹𑌾𑌮𑌨𑍍𑌤𑍍𑌰𑍋 𑌮𑌹𑌾𑌵𑌿𑌦𑍍𑌯𑌾𑌮𑌨𑍍𑌤𑍍𑌰𑍋𑌽𑌨𑍁𑌤𑍍𑌤𑌰𑌮𑌨𑍍𑌤𑍍𑌰𑍋𑌽𑌸𑌮𑌸𑌮𑌮𑌨𑍍𑌤𑍍𑌰𑌃 𑌸𑌰𑍍𑌵𑌦𑍁𑌃𑌖𑌪𑍍𑌰𑌶𑌮𑌨𑌃 𑌸𑌤𑍍𑌯𑌮𑌮𑌿𑌥𑍍𑌯𑌤𑍍𑌵𑌾𑌤𑍍 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌯𑌾𑌮𑍁𑌕𑍍𑌤𑍋 𑌮𑌨𑍍𑌤𑍍𑌰𑌃।

𑌤𑌦𑍍𑌯𑌥𑌾- 𑌗𑌤𑍇 𑌗𑌤𑍇 𑌪𑌾𑌰𑌗𑌤𑍇 𑌪𑌾𑌰𑌸𑌂𑌗𑌤𑍇 𑌬𑍋𑌧𑌿 𑌸𑍍𑌵𑌾𑌹𑌾॥

𑌇𑌤𑌿 𑌪𑍍𑌰𑌜𑍍𑌞𑌾𑌪𑌾𑌰𑌮𑌿𑌤𑌾𑌹𑍃𑌦𑌯𑌸𑍂𑌤𑍍𑌰𑌂 𑌸𑌮𑌾𑌪𑍍𑌤𑌮𑍍॥
