𑊱𑋙𑋪𑋘𑋍𑋪𑋙𑋡𑋙𑋍𑋪𑋑𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋝𑋤𑋍𑋪𑋙𑋗𑋪

(𑊻) 𑋝𑋟𑋝𑋪𑊺𑋪𑋙𑋡𑋍𑋒𑋠𑋇𑋞𑋪

𑋖𑋠𑋙𑋍𑋢𑋘𑋖𑋠𑋜𑋩𑋠𑋘𑋠𑋗𑋪-𑊱𑋙𑋪𑋘𑋍𑋪𑋙𑋡𑋙𑋍𑋪𑋑𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋝𑋤𑋍𑋪𑋙𑋗𑋪

𑋖𑋧𑋆𑋖𑋠𑋜𑋩𑋠𑋘𑋠𑋗𑋪-𑋓𑊼𑋪𑋝𑋪-𑋒𑋠-𑋏𑊺𑋧𑋑𑋪-𑋗𑋪𑋀𑋪𑋁𑋧𑊼𑋪-𑊼𑋪𑋝𑋣𑋗𑋪-𑋂𑋥𑋝𑋪-𑋝𑋣-𑋈𑋪𑋙𑋑𑋪-𑋒𑋠-𑋗𑋪𑋏𑋧

𑋑𑋗𑋞𑋪 𑋝𑋙𑋪𑋛𑋔𑋣𑋏𑋪𑋐𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋥𑋖𑋪𑋘𑋞𑋪

(𑋔𑋣𑋏𑋪𑋐𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋞𑋪)

𑊲𑋍𑋪𑋘𑋒𑋡 𑋔𑋣𑋏𑋪𑋐𑋧 𑋖𑊼𑋛𑋠𑋟𑋝𑋪𑋍𑋎𑋠𑊼𑋍𑋧ऽ𑋙𑋪𑋞𑋑𑋪 𑋝𑋗𑋪𑋘𑊺𑋪𑋝𑋟𑋔𑋣𑋏𑋪𑋐𑋧 𑋛𑋡𑋏𑋪𑋘𑋠𑋀𑋙𑋌𑋝𑋗𑋪𑋒𑋑𑋪𑋑𑋞𑋪 𑋝𑋣𑊼𑋍𑋧 𑋚𑋧𑊺𑋛𑋡𑋏𑋑𑋣𑋍𑋪𑋍𑋙𑋞𑋪 𑋒𑋣𑋙𑋣𑋜𑋩𑋏𑋗𑋪𑋘𑋝𑋠𑋙𑋎𑋡𑋞𑋪 𑋜𑋠𑋝𑋪𑋍𑋠 𑋏𑋥𑋛𑋗𑋑𑋣𑋜𑋩𑋪𑋘𑋠𑋌𑋠𑋟 𑋔𑋣𑋏𑋪𑋐𑋧 𑋖𑊼𑋛𑋠𑋑𑋡𑋍𑋡।

𑋑𑋡𑋜𑋩𑋪𑋘𑋑𑋪𑋏𑋞𑋪 𑋝 𑋍𑋎𑋠𑊼𑋍𑋞𑋪 𑋒𑋣𑋌𑋪𑋘𑋠𑋑𑋠𑋗𑋪, 𑊰𑋛𑋡𑋒𑋪𑋙𑋌𑋠𑋜𑋞𑋪 𑊺𑋣𑋜𑋚𑋗𑋤𑋚𑋠𑋑𑋠𑋗𑋪, 𑊰𑋚𑊿𑋪𑊺𑋪𑋙𑋡𑋍𑋞𑋪 𑊺𑋪𑋜𑋩𑋠𑋑𑋪𑋍𑋪𑋘𑋠, 𑊱𑋚𑋘𑋞𑋪 𑋒𑋣𑋌𑋪𑋘𑋑𑋡𑋐𑋠𑋑𑋠𑋑𑋠𑋗𑋪, 𑋀𑋡𑋍𑋪𑋙𑋡𑋍𑋧ऽ𑋑𑋣𑋛𑋪𑋘𑋅𑋪𑋂𑋑𑋦𑋞𑋪 𑊺𑋣𑋝𑋣𑋗𑋡𑋍𑋧 𑋚𑊺𑋪𑋜𑋩𑋌𑋦𑋞𑋪, 𑋒𑋪𑋙𑋍𑋡𑋙𑋤𑋒𑋧 𑊼𑋧𑋀𑋙𑋥𑋌, 𑊰𑋒𑋪𑋙𑋍𑋡𑊺𑋤𑋚𑋧 𑋏𑋙𑋪𑋜𑋑𑋥𑋑, 𑊰𑋖𑋡𑋙𑋍𑋡𑋞𑋪 𑋜𑋪𑋙𑋏𑋪𑋐𑋠(𑋜𑋣𑋏𑋪𑋐𑋠)𑋐𑋡𑋗𑋣𑊺𑋪𑋍𑋠𑋑𑋠𑋗𑋪, 𑊰𑋑𑋖𑋡𑋖𑋤𑋍𑋞𑋪 𑋒𑋪𑋙𑋂𑋪𑋅𑋘𑋠, 𑊰𑋑𑋛𑋗𑋙𑋪𑋏𑋑𑋢𑋘𑋧 𑋔𑋚𑋦𑋞𑋪, 𑋜𑋠𑋝𑋪𑋍𑋠 𑋝𑋙𑋪𑋛𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋑𑋠𑋗𑋪, 𑋒𑋡𑋍𑋠 𑋔𑋧𑋐𑋡𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋑𑋠𑋗𑋪, 𑋙𑋠𑋂𑋠 𑊱𑋙𑋪𑋘𑋒𑋣𑋏𑋪𑊼𑋚𑋠𑋑𑋠𑋗𑋪, 𑋝𑋠𑋙𑋪𑋎𑋛𑋠𑋞𑋞𑋪 𑋑𑋡𑋙𑋪𑋛𑋠𑋌𑋑𑊼𑋙𑋝𑋗𑋪𑋒𑋪𑋙𑋝𑋪𑋎𑋡𑋍𑋠𑋑𑋠𑋗𑋪, 𑊰𑋒𑋪𑋙𑋗𑋥𑋘𑋧 𑋂𑋪𑋅𑋠𑋑𑋥𑋑, 𑊰𑋀𑋡𑋑𑋪𑋍𑋪𑋘𑋞𑋪 𑋒𑋪𑋙𑋍𑋡𑋖𑋠𑋑𑋥𑋑, 𑋛𑋡𑋜𑋣𑋏𑋪𑋐𑋞𑋪 𑋝𑋪𑋛𑋙𑋥𑋌, 𑊱𑋝𑋪𑋛𑋏𑋑𑋢𑋘𑋧 𑊾𑋧𑋜𑋩𑋥𑋌, 𑊰𑋝𑋥𑋀𑋑𑊺𑋧 𑋙𑋤𑋒𑋥𑋌, 𑊰𑋒𑋪𑋙𑋍𑋡𑋝𑋗𑋞𑋪 𑊺𑋠𑋘𑋥𑋑, 𑊰𑋚𑋡𑋒𑋪𑋍𑋞𑋪 𑊺𑋠𑋗𑋦𑋞𑋪, 𑊰𑋑𑋣𑋒𑋚𑋡𑋒𑋪𑋍𑋧 𑋙𑋤𑋒𑋦𑋞𑋪 𑊰𑋝𑋟𑋝𑋪𑋙𑋡𑋜𑋩𑋪𑋆 𑊱𑋙𑋤𑋒𑋪𑋘𑋦𑋞𑋪, 𑋛𑋡𑋒𑋪𑋙𑋗𑋣𑊺𑋪𑋍𑋞𑋪 𑋝𑋪𑊺𑋑𑋪𑋐𑋥𑋖𑋪𑋘𑋞𑋪, 𑋛𑋡𑋝𑋗𑋪𑋒𑋪𑋙𑋘𑋣𑊺𑋪𑋍𑋧 𑋐𑋠𑋍𑋣𑋖𑋡𑋞𑋪, 𑋝𑋟𑋛𑋪𑋙𑋡𑋍 𑊱𑋘𑋍𑋑𑋦𑋞𑋪, 𑋒𑋪𑋙𑋀𑋪𑋁𑋡𑋑𑋪𑋑𑋧 𑊼𑋪𑋙𑋑𑋪𑋎𑋦𑋞𑋪, 𑋛𑋡𑋗𑋣𑊺𑋪𑋍𑋞𑋪 𑋒𑋙𑋡𑋏𑋠𑊾𑋪𑋑𑋦𑋞𑋪, 𑋒𑋙𑋡𑋗𑋣𑊺𑋪𑋍𑋝𑋪𑋍𑋪𑋙𑋡𑋜𑋩𑋪𑋌𑋘𑋠, 𑊸𑊾𑋠𑋏𑋣𑋍𑋪𑋍𑋢𑋙𑋪𑋌𑋞𑋪 𑋒𑋙𑋡𑋒𑋤𑋙𑋪𑋌𑋧 𑋂𑋪𑋅𑋠𑋑𑋥𑋑, 𑋒𑋪𑋙𑋍𑋡𑋜𑋩𑋪𑋇𑋡𑋍𑋧ऽ𑋍𑋢𑋍𑋠𑋑𑋠𑊼𑋍𑋒𑋪𑋙𑋍𑋪𑋘𑋣𑋍𑋪𑋒𑋑𑋪𑋑𑋠𑋑𑋠𑋟 𑋔𑋣𑋏𑋪𑋐𑋠𑋑𑋠𑋟 𑋖𑊼𑋛𑋍𑋠𑋟 𑋂𑋪𑋅𑋠𑋑𑋥, 𑊰𑋒𑋪𑋙𑋍𑋡𑋜𑋩𑋪𑋇𑋡𑋍𑋧 𑋑𑋡𑋙𑋪𑋛𑋠𑋌𑋥, 𑋝𑋪𑋎𑋡𑋍𑋧 𑋖𑋤𑋍𑊺𑋧𑋆𑋪𑋘𑋠𑋗𑋪, 𑋝𑋪𑋎𑋡𑋍𑋞𑋪 𑋝𑋙𑋪𑋛𑋝𑋍𑋪𑋍𑋪𑋛𑋠𑋚𑋧𑊺𑋑𑋢𑋘𑋠𑋘𑋠𑋟 𑋖𑋤𑋗𑋨, 𑋝𑋙𑋪𑋛 𑊲𑋗𑋥 𑋍𑋎𑋠𑊼𑋍𑋠𑋑𑋠𑋟 𑋛𑋡𑋜𑋥𑋜𑋩𑋍𑋞𑋪 𑋝𑋗𑋪𑋘𑊼𑋪 𑊼𑋣𑋌𑋠𑋞𑋪।

(𑋐𑋙𑋪𑋗𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋞𑋪)

𑋝𑋏𑋪𑋐𑋙𑋪𑋗𑋝𑋪𑋍𑋣 𑊱𑋏𑋨 𑊺𑋚𑋪𑋘𑋠𑋌𑋞𑋪, 𑋗𑋐𑋪𑋘𑋥 𑊺𑋚𑋪𑋘𑋠𑋌𑋞𑋪, 𑋒𑋙𑋪𑋘𑋛𑋝𑋠𑋑𑋥 𑊺𑋚𑋪𑋘𑋠𑋌𑋞𑋪, 𑋝𑋪𑋛𑋙𑋪𑋎𑋞𑋪 𑋝𑋣𑋛𑋪𑋘𑋅𑋪𑋂𑋑𑋞𑋪, 𑊺𑋥𑋛𑋚𑋞𑋪, 𑋒𑋙𑋡𑋒𑋤𑋙𑋪𑋌𑋞𑋪, 𑋒𑋙𑋡𑋜𑋣𑋏𑋪𑋐𑋞𑋪, 𑋒𑋙𑋪𑋘𑋛𑋏𑋠𑋍𑋞𑋪, 𑋝𑋪𑋛𑋠𑊻𑋪𑋘𑋠𑋍𑋞𑋪 𑋖𑊼𑋛𑋍𑋧 𑋐𑋙𑋪𑋗𑋞𑋪, 𑋝𑋠𑋑𑋪𑋏𑋪𑋙𑋡𑋜𑋩𑋪𑋆𑋡𑊺𑋞𑋪, 𑋑𑋡𑋙𑋪𑋂𑋪𑋛𑋙𑋞𑋪, 𑊱𑊺𑋠𑋚𑋡𑊺𑋞𑋪, 𑊹𑋒𑋑𑋠𑋘𑋡𑊺𑋞𑋪, 𑊷𑋞𑋡𑋒𑋜𑋪𑋘𑋡𑊺𑋞𑋪, 𑋒𑋪𑋙𑋍𑋪𑋘𑋠𑋍𑋪𑋗𑋛𑋥𑋏𑋑𑋢𑋘𑋧 𑋛𑋡𑋂𑋪𑋅𑋦𑋞𑋪, 𑋝𑋪𑋛𑋠𑊻𑋪𑋘𑋠𑋍𑋧 𑋖𑊼𑋛𑋍𑋧 𑋐𑋙𑋪𑋗𑋛𑋡𑋑𑋘𑋞𑋪 𑋝𑋣𑋒𑋪𑋙𑋛𑋥𑋏𑋡𑋍𑋞𑋪 𑋑𑋦𑋙𑋪𑋘𑋠𑋌𑋡𑊺𑋞𑋪, 𑋝𑋟𑋔𑋧𑋐𑋡𑊼𑋠𑋗𑋢, 𑊰𑋖𑋡𑋑𑋪𑋑𑋞𑋪 𑋝𑋟𑋝𑋪𑋍𑋤𑋒𑋞𑋪, 𑋝𑋒𑋪𑋙𑋍𑋡𑋜𑋙𑋌𑋞𑋪, 𑋁𑋡𑋑𑋪𑋑𑋒𑋪𑋚𑋧𑋍𑋡𑊺𑋞𑋪।

(𑋝𑋟𑊾𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋞𑋪)

𑋝𑋣𑋒𑋪𑋙𑋍𑋡𑋒𑋑𑋪𑋑𑋧 𑋖𑊼𑋛𑋍 𑊱𑋙𑋪𑋘𑋝𑋟𑊾𑋞𑋪, 𑋑𑋪𑋘𑋠𑋘𑋒𑋪𑋙𑋍𑋡𑋒𑋑𑋪𑋑𑋞𑋪, 𑋙𑋡𑋂𑋣𑋒𑋪𑋙𑋍𑋡𑋒𑋑𑋪𑋑𑋞𑋪, 𑋝𑋠𑋗𑋢𑋀𑋢𑋒𑋪𑋙𑋍𑋡𑋒𑋑𑋪𑋑𑋞𑋪, 𑊰𑋅𑋪𑋂𑋚𑋢𑊺𑋙𑋌𑋢𑋘𑋞𑋪, 𑋝𑋠𑋗𑋢𑋀𑋢𑊺𑋙𑋌𑋢𑋘𑋞𑋪, 𑋒𑋣𑋌𑋪𑋘𑋜𑋪𑋙𑋢𑊺𑋪𑋜𑋩𑋥𑋍𑋪𑋙𑋞𑋪, 𑋗𑋞𑋠𑋏𑊺𑋪𑋜𑋩𑋡𑋌𑋠𑋒𑋙𑋡𑋜𑋧𑋐𑊺𑋞𑋪, 𑋒𑋪𑋙𑋠𑋞𑋛𑋑𑋢𑋘𑋞𑋪, 𑊱𑋞𑋛𑋑𑋢𑋘𑋞𑋪।

॥ 𑊱𑋙𑋪𑋘𑋍𑋪𑋙𑋡𑋙𑋍𑋪𑋑𑋠𑋑𑋣𑋝𑋪𑋗𑋪𑋙𑋡𑋍𑋡𑋝𑋤𑋍𑋪𑋙𑋟 𑋝𑋗𑋠𑋒𑋪𑋍𑋗𑋪॥
