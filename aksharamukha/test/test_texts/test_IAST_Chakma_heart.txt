𑄃𑄢𑄳𑄠𑄧𑄖𑄳𑄢𑄨𑄢𑄧𑄖𑄳𑄚𑄚𑄪𑄥𑄴𑄟𑄳𑄢𑄨𑄖𑄨𑄥𑄫𑄖𑄳𑄢𑄧𑄟𑄴

(𑄈𑄧) 𑄥𑄧𑄁𑄥𑄴𑄇𑄳𑄢𑄨𑄖𑄧𑄛𑄒𑄧𑄂

𑄞𑄢𑄧𑄖𑄩𑄠𑄧𑄞𑄥𑄠𑄟𑄴-𑄃𑄢𑄳𑄠𑄧𑄖𑄳𑄢𑄨𑄢𑄧𑄖𑄳𑄚𑄚𑄪𑄥𑄴𑄟𑄳𑄢𑄨𑄖𑄨𑄥𑄫𑄖𑄳𑄢𑄧𑄟𑄴

𑄞𑄮𑄑𑄧𑄞𑄥𑄠𑄟𑄴-𑄜𑄧𑄉𑄴𑄥𑄴-𑄛-𑄘𑄧𑄇𑄮𑄚𑄴-𑄟𑄴𑄌𑄴𑄍𑄮𑄉𑄴-𑄉𑄴𑄥𑄪𑄟𑄴-𑄎𑄬𑄥𑄴-𑄥𑄪-𑄓𑄳𑄢𑄧𑄚𑄴-𑄛-𑄟𑄴𑄘𑄮

𑄚𑄧𑄟𑄧𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄝𑄪𑄘𑄴𑄙𑄧𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄬𑄞𑄳𑄠𑄧𑄂

(𑄝𑄪𑄘𑄴𑄙𑄚𑄪𑄥𑄴𑄟𑄳𑄢𑄨𑄖𑄨𑄂)

𑄃𑄨𑄖𑄳𑄠𑄧𑄛𑄨 𑄝𑄪𑄘𑄴𑄙𑄮 𑄞𑄧𑄉𑄧𑄤𑄁𑄥𑄴𑄖𑄧𑄗𑄉𑄧𑄖𑄮'𑄢𑄴𑄦𑄧𑄚𑄴 𑄥𑄧𑄟𑄳𑄠𑄧𑄇𑄴𑄥𑄧𑄁𑄝𑄪𑄘𑄴𑄙𑄮 𑄤𑄨𑄘𑄳𑄠𑄌𑄧𑄢𑄧𑄕𑄧𑄥𑄧𑄟𑄴𑄛𑄧𑄚𑄴𑄧𑄂 𑄥𑄪𑄉𑄧𑄖𑄮 𑄣𑄮𑄇𑄧𑄤𑄨𑄘𑄧𑄚𑄪𑄖𑄴𑄧𑄢𑄧𑄂 𑄛𑄪𑄢𑄪𑄥𑄧𑄘𑄧𑄟𑄳𑄠𑄧𑄥𑄢𑄧𑄗𑄨𑄂 𑄥𑄥𑄴𑄖 𑄘𑄬𑄤𑄧𑄟𑄧𑄚𑄪𑄥𑄳𑄠𑄕𑄁 𑄝𑄪𑄘𑄴𑄙𑄮 𑄞𑄧𑄉𑄧𑄤𑄚𑄨𑄖𑄨𑅁

𑄚𑄨𑄥𑄳𑄠𑄧𑄚𑄴𑄘𑄧𑄂 𑄥𑄧 𑄖𑄧𑄗𑄉𑄧𑄖𑄧𑄂 𑄛𑄪𑄕𑄳𑄠𑄚𑄟𑄴, 𑄃𑄧𑄤𑄨𑄛𑄳𑄢𑄧𑄕𑄥𑄧𑄂 𑄇𑄪𑄥𑄧𑄣𑄧𑄟𑄫𑄣𑄚𑄟𑄴, 𑄃𑄧𑄣𑄧𑄋𑄴𑄇𑄳𑄢𑄨𑄖𑄧𑄂 𑄇𑄴𑄥𑄚𑄴𑄖𑄳𑄠, 𑄃𑄣𑄧𑄠𑄧𑄂 𑄛𑄪𑄕𑄳𑄠𑄧𑄚𑄨𑄙𑄚𑄚𑄟𑄴, 𑄌𑄨𑄖𑄳𑄢𑄨𑄖𑄮'𑄚𑄪𑄤𑄳𑄠𑄧𑄐𑄴𑄎𑄧𑄚𑄰𑄂 𑄇𑄪𑄥𑄪𑄟𑄨𑄖𑄮 𑄣𑄧𑄇𑄴𑄥𑄧𑄕𑄰𑄂, 𑄛𑄳𑄢𑄧𑄖𑄨𑄢𑄫𑄛𑄮 𑄉𑄮𑄌𑄧𑄢𑄬𑄕𑄧, 𑄃𑄧𑄛𑄳𑄢𑄧𑄖𑄨𑄇𑄫𑄣𑄮 𑄘𑄧𑄢𑄴𑄥𑄧𑄚𑄬𑄚𑄧, 𑄃𑄧𑄞𑄨𑄢𑄧𑄖𑄨𑄂 𑄥𑄳𑄢𑄧𑄘𑄴𑄙(𑄥𑄪𑄘𑄴𑄙)𑄙𑄨𑄟𑄪𑄇𑄴𑄖𑄚𑄟𑄴, 𑄃𑄧𑄚𑄧𑄞𑄨𑄞𑄫𑄖𑄧𑄂 𑄛𑄳𑄢𑄧𑄎𑄴𑄐𑄧𑄠, 𑄃𑄧𑄚𑄧𑄤𑄧𑄟𑄧𑄢𑄴𑄘𑄧𑄚𑄩𑄠𑄮 𑄝𑄧𑄣𑄰𑄂, 𑄥𑄥𑄴𑄖 𑄥𑄧𑄢𑄳𑄤𑄧𑄥𑄧𑄖𑄴𑄳𑄤𑄚𑄟𑄴, 𑄛𑄨𑄖 𑄝𑄮𑄙𑄨𑄥𑄧𑄖𑄴𑄳𑄤𑄚𑄟𑄴, 𑄢𑄎 𑄃𑄢𑄳𑄠𑄧𑄛𑄪𑄘𑄴𑄉𑄧𑄣𑄚𑄟𑄴, 𑄥𑄢𑄴𑄗𑄧𑄤𑄦𑄧𑄂 𑄚𑄨𑄢𑄳𑄤𑄕𑄧𑄚𑄧𑄉𑄧𑄢𑄧𑄥𑄧𑄟𑄴𑄛𑄳𑄢𑄧𑄥𑄴𑄗𑄨𑄖𑄚𑄟𑄴, 𑄃𑄧𑄛𑄳𑄢𑄧𑄟𑄬𑄠𑄮 𑄎𑄴𑄐𑄚𑄬𑄚𑄧, 𑄃𑄧𑄌𑄨𑄚𑄴𑄖𑄳𑄠𑄧𑄂 𑄛𑄳𑄢𑄧𑄖𑄨𑄞𑄚𑄬𑄚𑄧, 𑄤𑄨𑄥𑄪𑄘𑄴𑄙𑄧𑄂 𑄥𑄳𑄤𑄧𑄢𑄬𑄕𑄧, 𑄃𑄥𑄳𑄤𑄧𑄘𑄧𑄚𑄩𑄠𑄮 𑄊𑄮𑄥𑄬𑄕𑄧, 𑄃𑄧𑄥𑄬𑄌𑄧𑄚𑄧𑄇𑄮 𑄢𑄫𑄛𑄬𑄕𑄧, 𑄃𑄧𑄛𑄳𑄢𑄧𑄖𑄨𑄥𑄧𑄟𑄧𑄂 𑄇𑄠𑄬𑄚𑄧, 𑄃𑄧𑄣𑄨𑄛𑄴𑄖𑄧𑄂 𑄇𑄟𑄰𑄂, 𑄃𑄧𑄚𑄪𑄛𑄧𑄣𑄨𑄛𑄴𑄖𑄮 𑄢𑄫𑄛𑄰𑄂 𑄃𑄧𑄥𑄧𑄁𑄥𑄳𑄢𑄨𑄥𑄴𑄑𑄧 𑄃𑄢𑄫𑄛𑄳𑄠𑄰𑄂, 𑄤𑄨𑄛𑄳𑄢𑄧𑄟𑄪𑄇𑄴𑄖𑄧𑄂 𑄥𑄴𑄇𑄧𑄚𑄴𑄙𑄬𑄞𑄳𑄠𑄧𑄂, 𑄤𑄨𑄥𑄧𑄟𑄴𑄛𑄳𑄢𑄧𑄠𑄪𑄇𑄴𑄖𑄮 𑄙𑄖𑄪𑄞𑄨𑄂, 𑄥𑄧𑄁𑄤𑄳𑄢𑄨𑄖𑄧 𑄃𑄠𑄧𑄖𑄧𑄚𑄰𑄂, 𑄛𑄳𑄢𑄧𑄌𑄴𑄍𑄨𑄚𑄴𑄮 𑄉𑄳𑄢𑄧𑄚𑄴𑄗𑄰𑄂, 𑄤𑄨𑄟𑄪𑄇𑄴𑄖𑄧𑄂 𑄛𑄧𑄢𑄨𑄘𑄊𑄳𑄚𑄰𑄂, 𑄛𑄧𑄢𑄨𑄟𑄪𑄇𑄴𑄖𑄧𑄥𑄴𑄖𑄳𑄢𑄨𑄥𑄴𑄕𑄧𑄠, 𑄃𑄮𑄊𑄘𑄪𑄖𑄴𑄩𑄢𑄴𑄕𑄧𑄂 𑄛𑄧𑄢𑄨𑄛𑄫𑄢𑄴𑄕𑄮 𑄎𑄴𑄐𑄚𑄬𑄚𑄧, 𑄛𑄳𑄢𑄧𑄖𑄨𑄥𑄴𑄒𑄨𑄖𑄮'𑄖𑄩𑄖𑄚𑄉𑄧𑄖𑄧𑄛𑄳𑄢𑄧𑄖𑄳𑄠𑄪𑄖𑄴𑄛𑄧𑄚𑄳𑄚𑄚𑄁 𑄝𑄪𑄘𑄴𑄙𑄚𑄁 𑄞𑄧𑄉𑄧𑄤𑄧𑄖𑄁 𑄎𑄴𑄐𑄚𑄬, 𑄃𑄧𑄛𑄳𑄢𑄧𑄖𑄨𑄥𑄴𑄒𑄨𑄖𑄮 𑄚𑄨𑄢𑄳𑄤𑄕𑄬, 𑄥𑄴𑄗𑄨𑄖𑄮 𑄞𑄫𑄖𑄧𑄇𑄮𑄑𑄳𑄠𑄟𑄴, 𑄥𑄴𑄗𑄨𑄖𑄧𑄂 𑄥𑄧𑄢𑄳𑄤𑄧𑄥𑄧𑄖𑄴𑄳𑄤𑄣𑄮𑄇𑄧𑄚𑄩𑄠𑄠𑄁 𑄞𑄫𑄟𑄯, 𑄥𑄧𑄢𑄳𑄤𑄧 𑄃𑄨𑄟𑄬 𑄖𑄧𑄗𑄉𑄧𑄖𑄚𑄁 𑄤𑄨𑄥𑄬𑄥𑄧𑄖𑄧𑄂 𑄥𑄧𑄟𑄳𑄠𑄧𑄉𑄴 𑄉𑄪𑄕𑄂𑅁

(𑄙𑄧𑄢𑄴𑄟𑄚𑄪𑄥𑄴𑄟𑄳𑄢𑄨𑄖𑄨𑄂)

𑄥𑄧𑄘𑄴𑄙𑄧𑄢𑄴𑄟𑄧𑄥𑄴𑄖𑄪 𑄃𑄘𑄯 𑄇𑄧𑄣𑄳𑄠𑄕𑄧𑄂, 𑄟𑄧𑄙𑄳𑄠𑄬 𑄇𑄧𑄣𑄳𑄠𑄕𑄧𑄂, 𑄛𑄧𑄢𑄳𑄠𑄧𑄤𑄧𑄥𑄚𑄬 𑄇𑄧𑄣𑄳𑄠𑄕𑄧𑄂, 𑄥𑄳𑄤𑄧𑄢𑄴𑄗𑄧𑄂 𑄥𑄪𑄤𑄳𑄠𑄧𑄐𑄴𑄎𑄧𑄚𑄧𑄂, 𑄇𑄬𑄤𑄧𑄣𑄧𑄂, 𑄛𑄧𑄢𑄨𑄛𑄫𑄢𑄴𑄕𑄧𑄂, 𑄛𑄧𑄢𑄨𑄥𑄪𑄘𑄴𑄙𑄧𑄂, 𑄛𑄧𑄢𑄳𑄠𑄧𑄤𑄧𑄘𑄖𑄧𑄂, 𑄥𑄳𑄤𑄈𑄳𑄠𑄖𑄧𑄂 𑄞𑄧𑄉𑄧𑄤𑄧𑄖𑄮 𑄙𑄧𑄢𑄴𑄟𑄧𑄂, 𑄥𑄚𑄴𑄘𑄳𑄢𑄨𑄥𑄴𑄑𑄨𑄇𑄧𑄂, 𑄚𑄨𑄢𑄴𑄎𑄳𑄤𑄧𑄢𑄧𑄂, 𑄃𑄇𑄣𑄨𑄇𑄧𑄂, 𑄃𑄯𑄛𑄧𑄚𑄠𑄨𑄇𑄧𑄂, 𑄃𑄰𑄦𑄨𑄛𑄧𑄥𑄳𑄠𑄨𑄇𑄧𑄂, 𑄛𑄳𑄢𑄧𑄖𑄳𑄠𑄖𑄴𑄟𑄧𑄤𑄬𑄘𑄧𑄚𑄩𑄠𑄮 𑄤𑄨𑄎𑄴𑄐𑄰𑄂, 𑄥𑄳𑄤𑄈𑄳𑄠𑄖𑄮 𑄞𑄧𑄉𑄧𑄤𑄧𑄖𑄮 𑄙𑄧𑄢𑄴𑄟𑄧𑄤𑄨𑄚𑄧𑄠𑄧𑄂 𑄥𑄪𑄛𑄳𑄢𑄧𑄤𑄬𑄘𑄨𑄖𑄧𑄂 𑄚𑄰𑄢𑄳𑄠𑄕𑄨𑄇𑄧𑄂, 𑄥𑄧𑄁𑄝𑄮𑄙𑄨𑄉𑄟𑄩, 𑄃𑄧𑄞𑄨𑄚𑄴𑄧𑄂 𑄥𑄧𑄁𑄥𑄴𑄖𑄫𑄛𑄧𑄂, 𑄥𑄧𑄛𑄳𑄢𑄧𑄖𑄨𑄥𑄧𑄢𑄧𑄕𑄧𑄂, 𑄍𑄨𑄚𑄴𑄧𑄛𑄳𑄣𑄮𑄖𑄨𑄇𑄧𑄂𑅁

(𑄥𑄧𑄁𑄊𑄚𑄪𑄥𑄴𑄟𑄳𑄢𑄨𑄖𑄨𑄂)

𑄥𑄪𑄛𑄳𑄢𑄧𑄖𑄨𑄛𑄧𑄚𑄴𑄮 𑄞𑄧𑄉𑄧𑄤𑄧𑄖𑄧 𑄃𑄢𑄳𑄠𑄧𑄥𑄧𑄁𑄊𑄧𑄂, 𑄚𑄳𑄠𑄠𑄧𑄛𑄳𑄢𑄧𑄖𑄨𑄛𑄧𑄚𑄴𑄧𑄂, 𑄢𑄨𑄎𑄪𑄛𑄳𑄢𑄧𑄖𑄨𑄛𑄧𑄚𑄴𑄧𑄂, 𑄥𑄟𑄩𑄌𑄩𑄛𑄳𑄢𑄧𑄖𑄨𑄛𑄧𑄚𑄴𑄧𑄂, 𑄃𑄧𑄐𑄴𑄎𑄧𑄣𑄩𑄇𑄧𑄢𑄧𑄕𑄩𑄠𑄧𑄂, 𑄥𑄟𑄩𑄌𑄩𑄇𑄧𑄢𑄧𑄕𑄩𑄠𑄧𑄂, 𑄛𑄪𑄕𑄳𑄠𑄧𑄥𑄳𑄢𑄩𑄇𑄴𑄥𑄬𑄖𑄳𑄢𑄧𑄂, 𑄟𑄧𑄦𑄘𑄧𑄇𑄴𑄥𑄨𑄕𑄛𑄧𑄢𑄨𑄥𑄮𑄙𑄧𑄇𑄧𑄂, 𑄛𑄳𑄢𑄦𑄧𑄤𑄧𑄚𑄩𑄠𑄧𑄂, 𑄃𑄦𑄧𑄤𑄧𑄚𑄩𑄠𑄧𑄂𑅁
