𑵵𑵵𑶋𑵳𑵭𑶋𑶉𑶗𑵳𑶈𑶖

𑵺𑵰𑶖 𑶉𑶈𑶗𑵭𑵮𑶍𑵸𑶗𑵹𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶐𑵯𑶗𑵬𑶖.

.. 𑶘 𑵺𑵰𑶓 𑵸𑶉𑵸𑶋𑵶𑵺𑵺𑶗𑵳𑶊𑶅𑶈𑶗𑵬𑵺𑶗𑵳𑵵𑶓𑵱𑵹𑶊𑵳𑶍𑶅𑶗𑶈𑵳𑶋𑶉𑶗𑵾𑶋𑵳𑶉𑶈𑶗𑵭𑵮𑶍𑵸𑶗𑵹𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶊𑶈𑶗𑵬𑶉𑶗𑶈𑶊𑵭𑵱𑶅𑶗𑶈𑵳𑶗𑵬𑶐𑵱𑵮𑶍𑵸𑶗𑵹𑶐𑵯𑶗𑵬𑶓𑵳𑶌𑵳𑶊𑵺𑶊𑵶𑵳𑶅𑶗𑶈𑵳𑶗𑵬𑶍𑵳𑶗𑶅𑵺𑶗𑵺𑶐𑵯𑶗𑵬𑶖..

𑶡𑶠 𑵵𑶋𑶅𑶋𑶉𑶊𑵵𑶊𑶉𑶕𑵸𑶈𑶗𑶉𑵺𑶅𑶈𑶋𑵭𑶈𑶗𑵳𑶓 𑵸𑶉𑵰𑶖.

𑵢𑵳𑶋 𑶇𑶋 𑵯𑶋𑵱𑶗𑶉𑵭𑶖 𑶉𑶕𑵭𑶗𑶈𑶍𑵸𑶗𑵹𑶖 𑵱𑶍𑵰𑶊𑶈𑶖. 𑵳𑵸𑶊 𑵰𑶊𑶄𑶗𑵶𑵵𑶗𑵬𑶉𑵳𑶉𑶇𑶉𑶗𑶈𑶑𑶖 𑵵𑶋𑶅𑶋𑶉𑶊𑵵𑶊𑵰𑶍𑶅𑵺𑶌𑵬𑵳𑶐 𑶉𑶗𑵰 𑵸𑶉𑵯𑶋𑶈𑶗𑵸𑶊𑶈𑵱𑶉𑶇𑶉𑶗𑶈𑶑𑶖 𑶅𑶈𑶋𑵭𑶗𑶈𑶍𑵳𑶖 𑶅𑶍𑶈𑶉𑶗𑵱𑶗𑶈𑶍𑵳𑶖, 𑵸𑶉𑵯𑶋𑶉𑶗𑵻 𑶈𑵴𑶉𑶇𑶉𑶗𑶈𑶑𑶖 𑵲𑶊𑵸𑵺𑶌𑵬𑵯𑶓𑶀𑵺𑶌𑵬𑶉𑶗𑵭𑶊𑵸𑵺𑶌𑵬𑶅𑶈𑶋𑶅𑶎𑶈𑶗𑵺𑶑𑶈𑶗𑶇𑶋𑶈𑵺𑶗𑵬𑶉𑶍𑵭𑶈𑶗𑵺𑶅𑶈𑶋𑶅𑶎𑶈𑶗𑵺𑶑𑶉𑶗𑵻. 𑵬𑶐𑵺 𑵱𑶅𑶋𑵵𑵭𑶉𑶗𑵳𑶍𑵺𑶋 𑵰𑶇𑶊𑵺𑵶𑶈𑶐 𑵭𑶌𑵴𑶋𑵻𑵳𑶗𑵭𑶈𑶈𑵴𑶗𑵬𑶊𑵺𑶗𑵳𑶈𑶊𑶅𑵺𑵰𑶍𑵲𑶐𑶉𑶗𑵭𑵯𑶗𑵬𑵭𑵱𑶌𑶈𑶗𑵬𑵳𑶐 𑶉𑶗𑵰 𑵠𑵯𑶋𑵭𑶋𑶉𑶗𑶈𑶊𑵰𑶗𑵬𑵺𑶗𑵳𑶐. 𑵠𑶉𑶗𑵽𑶊𑵯𑶋𑶉𑶗𑵻 𑵳𑶎𑶈𑶗𑵬𑶉𑵳𑶉𑶇𑶉𑶗𑶈𑶑𑶖 𑶅𑶗𑶈𑵷𑶍𑶉𑶗𑵬𑵰𑶊𑵺𑶑𑶈𑶗𑵰𑶇𑵳𑶊 𑵻 𑶅𑶍𑶉𑶗𑶅𑵭𑶈𑶗𑶉𑶐𑵺𑶊𑵯𑶋𑶅𑶗𑶈𑵭𑶈𑶗𑶉𑵳𑶊 𑵭𑶋𑵳𑶈𑶗𑵸𑶋𑵺𑶋𑶈𑶗𑵬𑶎𑶇𑵳𑶓𑶈𑵺𑵶𑵭𑶊𑵱𑶗𑶉𑶇𑶈𑶗𑵰𑶗𑵬𑵱𑶎𑵽𑶊𑵶𑶊𑶈𑶅𑶗𑶈𑶊𑶉𑶊𑵸𑵳𑵵𑶐𑶉𑶍 𑵱𑵺𑶗𑵬𑶊𑶉𑵳𑶉𑶇𑶉𑶗𑶈𑶊𑵺𑶋 𑶉𑶈𑶗𑵭𑶊𑵵𑶕𑵱𑶊𑶈𑵯𑶎𑶉𑶋𑵳𑶊𑶖 𑶉𑶗𑵴𑶋𑵳𑶊 𑵠𑵯𑶎𑵭𑵺. 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶕 𑶅𑶗𑶈𑶐𑵱𑶗𑶉𑵰𑶊𑵺𑶊𑶖 𑵱𑶍𑶉𑶍𑵰𑶊𑵺𑶋 𑵻 𑵱𑶗𑶉𑶋𑶅𑵺𑶗𑵳𑶋 𑶉𑶗𑵰. 𑵠𑶉𑶗𑵽𑶔 𑵻 𑵰𑶈𑶍𑵳𑶗𑵱𑵺𑶗𑵬𑶊𑶉𑶇𑶉𑶗𑶈𑶊𑵺𑶋 𑵭𑶋𑵶𑵵𑶋𑵳𑶊𑵵𑶕𑵱𑶊𑶈𑶊𑵯𑶈𑵺𑶊𑵵𑶕𑵱𑶗𑶈𑶍𑵳𑶊𑵺𑶋 𑶈𑵳𑶗𑵺𑵯𑵸𑶗𑶈𑶕𑵱𑶈𑶐𑵺 𑵶𑶗𑶈𑶍𑶇𑶌𑵳𑶊𑵺𑶋 𑵰𑶊𑶈𑶗𑵶𑶕 𑶉𑶓𑵹𑵬𑵺𑶗𑵳𑶗𑵬𑶓 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶉𑶗𑵬 𑶅𑶍𑶈𑵳𑶓 𑵶𑵻𑶗𑵼𑵺𑶗𑵳𑶋 𑶉𑶗𑵰. 𑵸𑶐𑵭𑵺𑶊𑵶𑵬𑵱𑶗𑶉𑵶𑵺𑶗𑵹𑶈𑶗𑵭𑶊𑶉𑶍𑶈𑵶𑶈𑶍𑶂𑵱𑶋𑵺𑶗𑵺𑶈𑵰𑶇𑶓𑶈𑵶𑶊𑶉𑶗𑵻𑶊𑶈𑶗𑵹𑵱𑶊𑵬𑶋𑵱𑶊 𑵶𑵶𑵺𑵳𑵵𑶊𑵳𑶗𑶅𑶍𑶉𑶗𑶅𑶅𑵽𑶗𑵽𑵸𑶊𑵰𑶊𑵺𑶗𑵬𑵯𑶋𑶅𑶗𑶈𑵵𑵰𑶗𑵮𑵬𑵺𑶗𑵳𑶋 𑶉𑶗𑵰. 𑶉𑶈𑶗𑵭𑶐 𑵻 𑶉𑶊𑵱𑶗𑵬𑵶𑵺𑶊𑶖 𑶉𑶍𑵸𑶗𑵹𑶓𑵸𑵺𑶕 𑶈𑶊𑶀𑶊𑵺𑶕 𑶅𑶍𑶈𑶉𑶗𑵱𑶗𑶈𑶍𑵳𑶗𑵬 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶉𑶗𑵬 𑶅𑶍𑶈𑵳𑶓 𑵶𑵻𑶗𑵼𑵺𑶗𑵳𑶋 𑶉𑶗𑵰. 𑵠𑵺𑶐𑵺𑶑𑵭𑶕𑶈𑶎𑶅𑶐𑵺 𑵭𑶗𑵬𑶎𑶇𑶐𑵺 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶓 𑵵𑶋𑶅𑶋𑶉𑶊𑵵𑶊𑵰𑶍𑶅𑵺𑶌𑵬𑵳𑶐 𑶉𑶗𑵰..

𑶉𑵰𑵺𑵺𑶗𑵳𑶈𑶅𑶗𑶈𑵭𑶐𑶉𑶋𑵳𑶉𑶗𑵻 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶓 𑵵𑶋𑶅𑶋𑶉𑶊𑵵𑶊𑵰. 𑵠𑵴 𑵭𑶋𑶉𑶗𑵭𑶊𑵰𑶋𑵳𑶗𑶈𑶓 𑵺𑶊𑵰 𑵸𑶊𑶈𑵱𑶊𑵻𑶊𑶈𑶗𑵬𑶓 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶉𑶗𑵬 𑶉𑶗𑶈𑶋𑵬𑶕 𑵳𑶐𑶀𑶉𑶗𑵻𑶊𑶉𑶇𑵰𑶊𑵺𑶓 𑵹𑶈𑵺𑶋𑵳𑵵𑶐 𑵺𑶋𑵭𑶋𑶉𑶗𑵽𑶓𑵹𑶓𑵰𑶍𑵲𑶖 𑶅𑶗𑶈𑶅𑵳𑵳𑶋 𑶉𑶗𑵰. 𑵳𑶕 𑵳𑵴𑶊 𑶅𑶗𑶈𑶅𑵳𑶋𑵳𑶕 𑵸𑶗𑶈𑶍𑶉𑶗𑵽𑶗𑵭𑶊 𑶉𑶍𑵯𑶊𑶄𑶗𑵶𑶓 𑵺𑶊𑵰 𑵳𑶍𑶉𑶋𑵳𑵱𑶊𑵬𑶋𑵱𑶓 𑵸𑶐𑵭𑶅𑶍𑵳𑶗𑶈𑶓 𑵸𑵱𑶗𑶉𑶋𑵺𑶐𑵺 𑵱𑶈𑵳𑵵𑶐𑵺 𑶅𑶈𑶋𑵶𑶗𑶈𑶍𑶇𑶗𑵬𑶓𑵳𑶗𑵴𑶊𑶅𑵬𑵳𑶋 𑶉𑶗𑵰. 𑵤𑵳𑶗𑵴𑶊𑶅𑶗𑵬 𑵻 𑵶𑵶𑵺𑵳𑵵𑶉𑶗𑵴𑶓 𑶈𑶊𑶀𑶊𑵺𑶕 𑶉𑶍𑵸𑶗𑵹𑶓𑵸𑵺𑶕 𑵳𑶕 𑵻 𑵰𑶇𑶊𑵺𑶗𑵳𑶕 𑶀𑵺𑵱𑶊𑵬𑶕 𑵶𑶊𑵴𑶊𑵯𑶋𑶈𑵯𑶗𑵬𑵯𑶊𑶉𑵳—

𑶉𑶊𑶉𑶗𑵳𑶗𑶈𑶊𑵺𑶋 𑵬𑶊𑵺𑶋 𑶅𑶗𑶈𑵻𑵵𑵺𑶗𑵳𑶋 𑵰𑵺𑶍𑶉𑶗𑵬𑵵𑶓𑵱𑶐
𑶉𑶕𑵲𑶗𑵬𑶊 𑵵𑶋𑶅𑶋𑶉𑶗𑵻 𑵶𑵺𑵺𑶊𑶅𑶋 𑵻 𑵹𑶊𑵳𑶍𑵳𑵺𑶗𑵳𑶗𑶈𑵰.
𑵬𑶐 𑶉𑶋𑵵𑶗𑶅𑵬𑶓𑵶 𑶅𑶗𑶈𑶍𑵴𑶍 𑵵𑶔𑵱𑶋𑵱 𑵠𑶅𑶗𑶈𑵰𑶐𑵬𑶊𑶖
𑵳𑶐𑶉𑶗𑵭𑶐𑶉𑶍 𑶉𑶋𑵱𑶗𑶉𑶋𑵳𑶍 𑶅𑶍𑶈𑶊 𑵮𑶇𑶍𑵱𑵵𑶗𑶅𑵱𑶓𑵽𑶗𑵬𑶖..𑶡..

𑵱𑶋𑶕 𑵳𑶎 𑶀𑵺𑶉𑶗𑵬 𑵠𑵺𑶍𑵭𑶈𑶗𑵳𑵺𑵳𑶊𑶕 𑵱𑶈𑶓𑵳𑶋
𑵵𑶋𑶅𑶋𑶉𑶊𑵵𑵰𑶊𑵶𑵳𑶍 𑶉𑶍𑶉𑶋𑵱𑶗𑶉𑶋𑵳𑶍 𑶉𑶋𑶉𑶗𑵬𑵺𑶊𑶈𑶗𑵴𑵰.
𑶅𑶈𑶋𑶅𑶊𑵻𑵺𑶊𑶈𑶗𑵴 𑵮𑶇𑶍𑵸𑶊𑶈𑵱 𑵠𑵶𑶗𑶈𑵬𑶊𑵺𑶐
𑵠𑵺𑶗𑵬𑶊𑶕𑶉𑶗𑵻 𑶉𑵳𑶗𑵳𑶗𑵭𑵺𑵬𑶍𑵳𑶊𑵺𑵰𑶗𑶈𑶍𑵳𑶐 𑵭𑶋𑵺𑶐𑵳𑶍𑵰..𑶢..

𑵵𑶓𑵱𑶓𑵳𑶗𑵳𑶈𑶐𑶉𑶍 𑵻𑵳𑶍𑶉𑵳𑶗𑵬𑶅𑵴𑶐 𑵭𑶋𑵹𑶋𑶀𑶗𑵺𑶓
𑶇𑶐𑵳𑶍𑶅𑶗𑶈𑵳𑶌𑵳𑶗𑵬𑵱𑶍𑶉𑵵𑶓 𑵬𑵴 𑶉𑶕𑵯𑵭𑵺𑶗𑵳𑶋.
𑵬𑵴 𑵻𑶊𑵺𑶋𑶈𑶓𑵹𑵱𑶗𑶉𑵬𑶍 𑶉𑶕𑶉𑶗𑵴𑶋𑵳𑶍 𑶉𑶌𑵳𑶋𑵯𑶊𑵭𑶖
𑵳𑶉𑶗𑵰𑶋𑵺𑶗𑵭𑶋𑵹𑶋𑶀𑶗𑵺 𑵱𑶋𑵰𑵴𑶓 𑵵𑶋𑶅𑶋𑶉𑶊𑶉𑶗𑵳𑶗𑶈𑵰𑶊𑵳𑶗𑶈𑶐..𑶣..

𑵺𑶐𑵳𑶉𑶗𑵬 𑵡𑵻𑶈𑶋𑵬 𑵤𑵳𑶗𑵳𑶈𑶋 𑵭𑶊 𑵳𑶗𑶈𑶋𑵵𑶓𑵱𑶐
𑶉𑶈𑶗𑵭𑶐𑶉𑶍 𑵸𑶐𑵭𑵰𑵺𑶍𑶀𑶐𑶉𑶗𑵭𑵬𑵰𑶐𑵭 𑶀𑶐𑶉𑶗𑵾𑶖.
𑵺𑶊𑵰𑶊𑶅𑶋 𑵳𑶐𑶉 𑵵𑶋𑶅𑶋𑵺𑶊𑶕 𑵺 𑶇𑶋 𑵭𑶋𑵳𑶗𑵴 𑵬𑶎𑵬𑶕
𑵬𑵳𑶗𑶈𑶐𑶉𑶍 𑶉𑶋𑵱𑶗𑶉𑶋𑵳𑶍 𑶅𑶍𑶈𑶊 𑵮𑶇𑶍𑵱𑵵𑶗𑶅𑵱𑶓𑵽𑶗𑵬𑶖..𑶤..

𑵢𑵳𑶗𑵬𑶍𑵱𑶗𑵳𑶗𑵭𑶊 𑶉 𑵸𑶐𑵭𑶅𑶍𑵳𑶗𑶈𑶓 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭𑶕 𑵸𑶋𑵭𑶗𑵬𑶑𑶖 𑵱𑶍𑶉𑶍𑵰𑶑𑶈𑵯𑶗𑵬𑶈𑶗𑵻𑶗𑵬 𑵳𑵳𑶗𑶈𑶑𑵭𑶊𑵺𑶗𑵳𑶈𑶗𑵸𑵹𑶐. 𑵳𑵳𑶗𑶈 𑵹𑶊𑵳𑶗𑶈𑶗𑵬𑶉𑶗𑵻 𑵻𑶐𑵽𑶌𑵭𑶈𑶗𑵶𑶊𑶉𑶗𑵻 𑶉𑶗𑵴𑶊𑶅𑶋𑵳𑶊 𑵠𑵯𑶎𑵭𑵺. 𑶅𑶈𑶋𑶉𑶐𑶉𑶊𑶖 𑶉𑶊𑵱𑶗𑵬𑶊𑶖 𑶉𑶍𑵸𑶗𑵹𑶓𑵸𑵺𑶅𑶗𑶈𑵰𑶍𑵲𑶊𑶖 𑶅𑶗𑶈𑵱𑶗𑶈𑶊𑵰𑵺𑶗𑵳𑶖..

𑵠𑵴 𑵮𑶓𑵹𑶋𑶉𑵳𑶗𑵳𑶗𑵭 𑵤𑶈𑵶𑶉𑶊𑶈𑵻𑵺𑶗𑵸𑵺𑵰𑵬𑶕 𑵵𑶋𑶅𑶋𑶆𑵵𑵱𑵰𑶊𑵸𑶊𑵬 𑵸𑶋𑵭𑶗𑵬𑶊𑶈𑶗𑶉𑶉𑶍𑵭𑶈𑶗𑵺𑵳𑶋𑶈𑵱𑶕 𑶉𑵰𑵺𑶗𑵳𑶊𑵺𑶗𑵰𑵺𑶋𑶈𑵳𑶗𑵺𑶅𑶗𑶈𑵳𑶗𑵬𑶍𑶅𑶗𑵳𑶕 𑵭𑶋𑶉𑶗𑵭𑶊𑵰𑶋𑵳𑶗𑶈𑵰𑶊𑵻𑶊𑶈𑶗𑵬𑵰𑶐𑵭𑵰𑶊𑶇-𑵱𑵳𑵰𑶊𑶕 𑵰𑶐 𑵯𑶓 𑵤𑶅𑶊𑵹𑶗𑵬𑶊𑵬 𑵵𑶋𑶅𑶋𑶕 𑶉𑶋𑵱𑶗𑶉𑶊𑶅𑵬𑶉𑶋. 𑵮𑶗𑶈𑶊𑶇𑶗𑵰𑶌𑵲𑶈𑶓𑶉𑶗𑵽𑶌𑶅𑶍𑶉𑶗𑵱𑶈𑶉𑶊𑶈𑶋𑶕 𑵠𑶄𑶗𑵶𑵵𑶋𑶅𑶋𑶕 𑵭𑶄𑶗𑵶𑵵𑶋𑶅𑶋𑶕 𑵰𑵶𑵹𑵵𑶋𑶅𑶋𑶕 𑵰𑶄𑶗𑵶𑵵𑶗𑵬𑵵𑶋𑶅𑶋𑶕 𑵠𑶄𑶗𑵶𑶍𑵵𑶌𑵬𑵵𑶋𑶅𑶋𑶕 𑶉𑵱𑶊𑶈𑶋𑵵𑶋𑶅𑶋𑶕 𑵮𑶗𑶈𑶇𑶗𑵰𑵭𑵵𑶋𑵵𑶋𑶅𑶋𑶕 𑶅𑶊𑶈𑶍𑶉𑶗𑵬𑵵𑶋𑶅𑶋𑶕 𑵸𑶗𑶈𑶊𑵭𑶋𑶂𑵵𑶋𑶅𑶋𑶕 𑵱𑶋𑶈𑶊𑵳𑵵𑶋𑶅𑶋𑶕 𑵸𑶊𑵱𑶗𑶉𑶋𑵺𑶗𑵬𑵵𑶋𑶅𑶋𑶕 𑵤𑵶𑶗𑶈𑵵𑶋𑶅𑶋𑶕 𑶉𑶕𑵲𑶗𑵬𑶊𑵵𑶋𑶅𑶋𑶕 𑵠𑵺𑶍𑵵𑶓𑵰𑵵𑶋𑶅𑶋𑶕 𑵠𑵭𑵰𑶎𑶈𑶗𑵹𑵵𑶋𑶅𑶋𑶕 𑵸𑶈𑵸𑵵𑶋𑶅𑶋𑶕 𑵲𑶊𑶉𑶗𑵬𑵵𑶋𑶅𑶋𑶕 𑵻𑶌𑵺𑵵𑶋𑶅𑶋𑶕 𑵵𑶎𑵺𑵵𑶋𑶅𑶋𑶕 𑶇𑶎𑵺𑵵𑶋𑶅𑶋𑶕 𑵰𑵹𑶗𑵬𑶊𑵱𑶗𑶉𑶈𑵭𑶋𑶉𑶗𑵳𑶈𑵵𑶋𑶅𑶋𑶕 𑶅𑶍𑶉𑶗𑶅𑵵𑶋𑶅𑶋𑶕 𑵸𑶐𑵭𑵵𑶋𑶅𑶋𑶕 𑵺𑶊𑵶𑵵𑶋𑶅𑶋𑶕 𑵬𑵱𑶗𑶉𑵵𑶋𑶅𑶋𑶕 𑵶𑵺𑶗𑵹𑶈𑶗𑵭𑵵𑶋𑶅𑶋𑶕 𑵱𑶋𑵺𑶗𑵺𑶈𑵵𑶋𑶅𑶋𑶕 𑵰𑶇𑶓𑶈𑵶𑵵𑶋𑶅𑶋𑶕 𑵠𑶉𑶍𑶈𑵵𑶋𑶅𑶋𑶕 𑵶𑶈𑶍𑶂𑵵𑶋𑶅𑶋𑶕 𑵰𑶗𑶈𑶍𑵶𑵻𑵱𑶗𑶈𑵵𑶋𑶅𑶋𑶕 𑵭𑶊𑵬𑶉𑶈𑶍𑵳𑵵𑶋𑶅𑶋𑶕 𑵯𑶔𑵰𑵸𑶐𑵭𑵵𑶋𑶅𑶋𑶕 𑵠𑵺𑶗𑵳𑶈𑶌𑵱𑶗𑶉𑵸𑶐𑵭𑵵𑶋𑶅𑶋𑶕 𑵤𑵳𑶗𑵳𑶈𑵱𑶍𑶈𑶍𑵸𑶗𑵭𑶌𑶅𑵵𑶋𑶅𑶋𑶕 𑵠𑶅𑶈𑵶𑶓𑶂𑶊𑵺𑶌𑵵𑶋𑶅𑶋𑶕 𑶅𑶎𑶈𑶗𑵭𑵭𑶋𑵸𑶐𑶇𑵵𑶋𑶅𑶋𑶕 𑵤𑵳𑶗𑵱𑶗𑶉𑶐𑶅𑵵𑶋𑶅𑶋𑶕 𑵺𑶋𑵱𑶗𑶉𑶐𑶅𑵵𑶋𑶅𑶋𑶕 𑵭𑶋𑵱𑶗𑶉𑶐𑶅𑵵𑶋𑶅𑶋𑶕 𑶅𑶗𑶈𑵱𑶗𑶉𑶐𑶅𑵵𑶋𑶅𑶋𑶕 𑶉𑶊𑵶𑶈𑵵𑶋𑶅𑶋𑶕 𑵭𑶀𑶗𑶈𑵵𑶋𑶅𑶋𑶕 𑵵𑶐𑵲𑶅𑶗𑶈𑵳𑶋𑵵𑶐𑵲𑵵𑶋𑶅𑶋𑶕 𑵠𑵺𑶍𑵸𑶗𑶈𑶍𑵳𑵵𑶋𑶅𑶋𑶕 𑶉𑶊𑶉𑶗𑵳𑶗𑶈𑶊𑵭𑶈𑶗𑵳𑶊𑶕 𑵶𑵺𑵺𑶊𑵭𑶈𑶗𑵳𑵵𑶋𑶅𑶋𑶕 𑵤𑵳𑶗𑵱𑶗𑶉𑶐𑶅𑶊𑵭𑶈𑶗𑵳𑵵𑶋𑶅𑶋𑶕 𑵺𑶋𑵱𑶗𑶉𑶐𑶅𑶊𑵭𑶈𑶗𑵳𑵵𑶋𑶅𑶋𑶕 𑶅𑶊𑵸𑵵𑶋𑵲𑶋𑵳𑵵𑶋𑶅𑶋𑶕 𑵸𑶗𑵭𑶋𑶈𑶍𑵳𑶗𑵳𑶈𑶅𑵸𑶉𑶕𑵹𑶋𑵵𑶋𑶅𑶋𑶕 𑵬𑶊𑵭𑵸𑶗𑵸𑶉𑶓𑵳𑶗𑵳𑶈𑶅𑵸𑶉𑶕𑵹𑶋𑵵𑶋𑶅𑶋𑶕 𑵰𑵹𑶗𑵬𑶊𑶇𑶊𑶈𑶋𑵺𑶌𑵵𑶋𑶅𑶋𑶕 𑶉𑶈𑶗𑵭𑶈𑶍𑵳𑶉𑶕𑵶𑶗𑶈𑶇𑵺𑶌𑵵𑶋𑶅𑶋𑶕 𑵭𑶋𑵸𑶗𑵬𑶊𑵺𑶍𑵵𑶓𑵰𑶊𑵭𑶋𑵰𑶋𑶉𑶗𑶈𑶋𑵳𑵵𑶋𑶅𑶋𑶕 𑶈𑶍𑶉𑶋𑵳𑶅𑶉𑶗𑵳𑶅𑶗𑵳𑶊𑶕 𑶈𑶓𑵻𑵰𑶊𑵺𑶊𑶕 𑵹𑶈𑵺𑶌𑶅𑶗𑶈𑶐𑵱𑶗𑶉𑶋𑵺𑶌𑵵𑶋𑶅𑶋𑶕 𑵶𑵶𑵺𑶅𑶗𑶈𑶐𑵱𑶗𑶉𑶋𑵺𑶌𑵵𑶋𑶅𑶋𑶕 𑶉𑶈𑶗𑵭𑶔𑶉𑵹𑶋𑵺𑶋𑶉𑶗𑵬𑵺𑶗𑵸𑶊𑶕 𑶉𑶈𑶗𑵭𑶉𑶊𑶈𑶉𑶕𑵶𑶗𑶈𑶇𑵺𑶌𑶕 𑶉𑶈𑶗𑵭𑵯𑶎𑵳𑶈𑶍𑵳𑵶𑶗𑶈𑶇𑵺𑶌𑵰. 𑵡𑶉𑶊𑶕 𑵯𑶓 𑵤𑶅𑶊𑵹𑶗𑵬𑶊𑵬 𑵻𑵳𑶍𑶉𑶗𑶉𑶉𑶗𑵽𑶌𑵵𑶋𑶅𑶌𑵺𑶊𑶕 𑵱𑵳𑵰𑶊𑶕 𑵳𑶗𑵭𑶕 𑶉𑶋𑶉𑶗𑵬𑶊𑶅𑵬𑶋𑶉𑶗𑵬𑶉𑶋?
