𑀆𑀭𑁆𑀬𑀢𑁆𑀭𑀺𑀭𑀢𑁆𑀦𑀸𑀦𑀼𑀲𑁆𑀫𑁆𑀭𑀼𑀢𑀺𑀲𑀽𑀢𑁆𑀭𑀫𑁆

(𑀓) 𑀲𑀫𑁆𑀲𑁆𑀓𑁆𑀭𑀼𑀢𑀧𑀸𑀝𑀂

𑀧𑀸𑀭𑀢𑀻𑀬𑀧𑀸𑀱𑀸𑀬𑀸𑀫𑁆-𑀆𑀭𑁆𑀬𑀢𑁆𑀭𑀺𑀭𑀢𑁆𑀦𑀸𑀦𑀼𑀲𑁆𑀫𑁆𑀭𑀼𑀢𑀺𑀲𑀽𑀢𑁆𑀭𑀫𑁆

𑀧𑁄𑀝𑀧𑀸𑀱𑀸𑀬𑀸𑀫𑁆-𑀧𑀓𑁆𑀲𑁆-𑀧𑀸-𑀢𑀓𑁄𑀦𑁆-𑀫𑁆𑀘𑁆𑀘𑁄𑀓𑁆-𑀓𑁆𑀲𑀼𑀫𑁆-𑀚𑁂𑀲𑁆-𑀲𑀼-𑀝𑁆𑀭𑀦𑁆-𑀧𑀸-𑀫𑁆𑀢𑁄

𑀦𑀫𑀂 𑀲𑀭𑁆𑀯𑀧𑀼𑀢𑁆𑀥𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑁂𑀧𑁆𑀬𑀂

(𑀧𑀼𑀢𑁆𑀥𑀸𑀦𑀼𑀲𑁆𑀫𑁆𑀭𑀼𑀢𑀺𑀂)

𑀇𑀢𑁆𑀬𑀧𑀺 𑀧𑀼𑀢𑁆𑀥𑁄 𑀧𑀓𑀯𑀸𑀫𑁆𑀲𑁆𑀢𑀢𑀸𑀓𑀢𑁄'𑀭𑁆𑀳𑀦𑁆 𑀲𑀫𑁆𑀬𑀓𑁆𑀲𑀫𑁆𑀧𑀼𑀢𑁆𑀥𑁄 𑀯𑀺𑀢𑁆𑀬𑀸𑀘𑀭𑀡𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀂 𑀲𑀼𑀓𑀢𑁄 𑀮𑁄𑀓𑀯𑀺𑀢𑀦𑀼𑀢𑁆𑀢𑀭𑀂 𑀧𑀼𑀭𑀼𑀱𑀢𑀫𑁆𑀬𑀲𑀸𑀭𑀢𑀺𑀂 𑀰𑀸𑀲𑁆𑀢𑀸 𑀢𑁂𑀯𑀫𑀦𑀼𑀱𑁆𑀬𑀸𑀡𑀸𑀫𑁆 𑀧𑀼𑀢𑁆𑀥𑁄 𑀧𑀓𑀯𑀸𑀦𑀺𑀢𑀺.

𑀦𑀺𑀱𑁆𑀬𑀦𑁆𑀢𑀂 𑀲 𑀢𑀢𑀸𑀓𑀢𑀂 𑀧𑀼𑀡𑁆𑀬𑀸𑀦𑀸𑀫𑁆, 𑀅𑀯𑀺𑀧𑁆𑀭𑀡𑀸𑀰𑀂 𑀓𑀼𑀰𑀮𑀫𑀽𑀮𑀸𑀦𑀸𑀫𑁆, 𑀅𑀮𑀗𑁆𑀓𑁆𑀭𑀼𑀢𑀂 𑀓𑁆𑀱𑀸𑀦𑁆𑀢𑁆𑀬𑀸, 𑀆𑀮𑀬𑀂 𑀧𑀼𑀡𑁆𑀬𑀦𑀺𑀥𑀸𑀦𑀸𑀦𑀸𑀫𑁆, 𑀘𑀺𑀢𑁆𑀭𑀺𑀢𑁄'𑀦𑀼𑀯𑁆𑀬𑀜𑁆𑀚𑀦𑁃𑀂 𑀓𑀼𑀲𑀼𑀫𑀺𑀢𑁄 𑀮𑀓𑁆𑀱𑀡𑁃𑀂, 𑀧𑁆𑀭𑀢𑀺𑀭𑀽𑀧𑁄 𑀓𑁄𑀘𑀭𑁂𑀡, 𑀅𑀧𑁆𑀭𑀢𑀺𑀓𑀽𑀮𑁄 𑀢𑀭𑁆𑀰𑀦𑁂𑀦, 𑀅𑀧𑀺𑀭𑀢𑀺𑀂 𑀰𑁆𑀭𑀢𑁆𑀥𑀸(𑀰𑀼𑀢𑁆𑀥𑀸)𑀥𑀺𑀫𑀼𑀓𑁆𑀢𑀸𑀦𑀸𑀫𑁆, 𑀅𑀦𑀧𑀺𑀧𑀽𑀢𑀂 𑀧𑁆𑀭𑀚𑁆𑀜𑀬𑀸, 𑀅𑀦𑀯𑀫𑀭𑁆𑀢𑀦𑀻𑀬𑁄 𑀧𑀮𑁃𑀂, 𑀰𑀸𑀲𑁆𑀢𑀸 𑀲𑀭𑁆𑀯𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀦𑀸𑀫𑁆, 𑀧𑀺𑀢𑀸 𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀦𑀸𑀫𑁆, 𑀭𑀸𑀚𑀸 𑀆𑀭𑁆𑀬𑀧𑀼𑀢𑁆𑀓𑀮𑀸𑀦𑀸𑀫𑁆, 𑀲𑀸𑀭𑁆𑀢𑀯𑀸𑀳𑀂 𑀦𑀺𑀭𑁆𑀯𑀸𑀡𑀦𑀓𑀭𑀲𑀫𑁆𑀧𑁆𑀭𑀲𑁆𑀢𑀺𑀢𑀸𑀦𑀸𑀫𑁆, 𑀅𑀧𑁆𑀭𑀫𑁂𑀬𑁄 𑀚𑁆𑀜𑀸𑀦𑁂𑀦, 𑀅𑀘𑀺𑀦𑁆𑀢𑁆𑀬𑀂 𑀧𑁆𑀭𑀢𑀺𑀧𑀸𑀦𑁂𑀦, 𑀯𑀺𑀰𑀼𑀢𑁆𑀥𑀂 𑀲𑁆𑀯𑀭𑁂𑀡, 𑀆𑀲𑁆𑀯𑀢𑀦𑀻𑀬𑁄 𑀓𑁄𑀱𑁂𑀡, 𑀅𑀲𑁂𑀘𑀦𑀓𑁄 𑀭𑀽𑀧𑁂𑀡, 𑀅𑀧𑁆𑀭𑀢𑀺𑀲𑀫𑀂 𑀓𑀸𑀬𑁂𑀦, 𑀅𑀮𑀺𑀧𑁆𑀢𑀂 𑀓𑀸𑀫𑁃𑀂, 𑀅𑀦𑀼𑀧𑀮𑀺𑀧𑁆𑀢𑁄 𑀭𑀽𑀧𑁃𑀂 𑀅𑀲𑀫𑁆𑀲𑁆𑀭𑀼𑀱𑁆𑀝 𑀆𑀭𑀽𑀧𑁆𑀬𑁃𑀂, 𑀯𑀺𑀧𑁆𑀭𑀫𑀼𑀓𑁆𑀢𑀂 𑀲𑁆𑀓𑀦𑁆𑀥𑁂𑀧𑁆𑀬𑀂, 𑀯𑀺𑀲𑀫𑁆𑀧𑁆𑀭𑀬𑀼𑀓𑁆𑀢𑁄 𑀥𑀸𑀢𑀼𑀧𑀺𑀂, 𑀲𑀫𑁆𑀯𑁆𑀭𑀼𑀢 𑀆𑀬𑀢𑀦𑁃𑀂, 𑀧𑁆𑀭𑀘𑁆𑀘𑀺𑀦𑁆𑀦𑁄 𑀓𑁆𑀭𑀦𑁆𑀢𑁃𑀂, 𑀯𑀺𑀫𑀼𑀓𑁆𑀢𑀂 𑀧𑀭𑀺𑀢𑀸𑀓𑁆𑀦𑁃𑀂, 𑀧𑀭𑀺𑀫𑀼𑀓𑁆𑀢𑀲𑁆𑀢𑁆𑀭𑀼𑀱𑁆𑀡𑀬𑀸, 𑀑𑀓𑀸𑀢𑀼𑀢𑁆𑀢𑀻𑀭𑁆𑀡𑀂 𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑁄 𑀚𑁆𑀜𑀸𑀦𑁂𑀦, 𑀧𑁆𑀭𑀢𑀺𑀱𑁆𑀝𑀺𑀢𑁄'𑀢𑀻𑀢𑀸𑀦𑀸𑀓𑀢𑀧𑁆𑀭𑀢𑁆𑀬𑀼𑀢𑁆𑀧𑀦𑁆𑀦𑀸𑀦𑀸𑀫𑁆 𑀧𑀼𑀢𑁆𑀥𑀸𑀦𑀸𑀫𑁆 𑀧𑀓𑀯𑀢𑀸𑀫𑁆 𑀚𑁆𑀜𑀸𑀦𑁂, 𑀅𑀧𑁆𑀭𑀢𑀺𑀱𑁆𑀝𑀺𑀢𑁄 𑀦𑀺𑀭𑁆𑀯𑀸𑀡𑁂, 𑀲𑁆𑀢𑀺𑀢𑁄 𑀧𑀽𑀢𑀓𑁄𑀝𑁆𑀬𑀸𑀫𑁆, 𑀲𑁆𑀢𑀺𑀢𑀂 𑀲𑀭𑁆𑀯𑀲𑀢𑁆𑀢𑁆𑀯𑀸𑀮𑁄𑀓𑀦𑀻𑀬𑀸𑀬𑀸𑀫𑁆 𑀧𑀽𑀫𑁅, 𑀲𑀭𑁆𑀯 𑀇𑀫𑁂 𑀢𑀢𑀸𑀓𑀢𑀸𑀦𑀸𑀫𑁆 𑀯𑀺𑀰𑁂𑀱𑀢𑀂 𑀲𑀫𑁆𑀬𑀓𑁆 𑀓𑀼𑀡𑀸𑀂.

(𑀥𑀭𑁆𑀫𑀸𑀦𑀼𑀲𑁆𑀫𑁆𑀭𑀼𑀢𑀺𑀂)

𑀲𑀢𑁆𑀥𑀭𑁆𑀫𑀲𑁆𑀢𑀼 𑀆𑀢𑁅 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀫𑀥𑁆𑀬𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀧𑀭𑁆𑀬𑀯𑀲𑀸𑀦𑁂 𑀓𑀮𑁆𑀬𑀸𑀡𑀂, 𑀲𑁆𑀯𑀭𑁆𑀢𑀂 𑀲𑀼𑀯𑁆𑀬𑀜𑁆𑀚𑀦𑀂, 𑀓𑁂𑀯𑀮𑀂, 𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑀂, 𑀧𑀭𑀺𑀰𑀼𑀢𑁆𑀥𑀂, 𑀧𑀭𑁆𑀬𑀯𑀢𑀸𑀢𑀂, 𑀲𑁆𑀯𑀸𑀓𑁆𑀬𑀸𑀢𑀂 𑀧𑀓𑀯𑀢𑁄 𑀥𑀭𑁆𑀫𑀂, 𑀲𑀸𑀦𑁆𑀢𑁆𑀭𑀼𑀱𑁆𑀝𑀺𑀓𑀂, 𑀦𑀺𑀭𑁆𑀚𑁆𑀯𑀭𑀂, 𑀆𑀓𑀸𑀮𑀺𑀓𑀂, 𑀒𑀧𑀦𑀸𑀬𑀺𑀓𑀂, 𑀐𑀳𑀺𑀧𑀰𑁆𑀬𑀺𑀓𑀂, 𑀧𑁆𑀭𑀢𑁆𑀬𑀸𑀢𑁆𑀫𑀯𑁂𑀢𑀦𑀻𑀬𑁄 𑀯𑀺𑀚𑁆𑀜𑁃𑀂, 𑀲𑁆𑀯𑀸𑀓𑁆𑀬𑀸𑀢𑁄 𑀧𑀓𑀯𑀢𑁄 𑀥𑀭𑁆𑀫𑀯𑀺𑀦𑀬𑀂 𑀲𑀼𑀧𑁆𑀭𑀯𑁂𑀢𑀺𑀢𑀂 𑀦𑁃𑀭𑁆𑀬𑀸𑀡𑀺𑀓𑀂, 𑀲𑀫𑁆𑀧𑁄𑀥𑀺𑀓𑀸𑀫𑀻, 𑀅𑀧𑀺𑀦𑁆𑀦𑀂 𑀲𑀫𑁆𑀲𑁆𑀢𑀽𑀧𑀂, 𑀲𑀧𑁆𑀭𑀢𑀺𑀰𑀭𑀡𑀂, 𑀘𑀺𑀦𑁆𑀦𑀧𑁆𑀮𑁄𑀢𑀺𑀓𑀂.

(𑀲𑀫𑁆𑀓𑀸𑀦𑀼𑀲𑁆𑀫𑁆𑀭𑀼𑀢𑀺𑀂)

𑀲𑀼𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑁄 𑀧𑀓𑀯𑀢 𑀆𑀭𑁆𑀬𑀲𑀫𑁆𑀓𑀂, 𑀦𑁆𑀬𑀸𑀬𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀭𑀼𑀚𑀼𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀲𑀸𑀫𑀻𑀘𑀻𑀧𑁆𑀭𑀢𑀺𑀧𑀦𑁆𑀦𑀂, 𑀅𑀜𑁆𑀚𑀮𑀻𑀓𑀭𑀡𑀻𑀬𑀂, 𑀲𑀸𑀫𑀻𑀘𑀻𑀓𑀭𑀡𑀻𑀬𑀂, 𑀧𑀼𑀡𑁆𑀬𑀰𑁆𑀭𑀻𑀓𑁆𑀱𑁂𑀢𑁆𑀭𑀂, 𑀫𑀳𑀸𑀢𑀓𑁆𑀱𑀺𑀡𑀸𑀧𑀭𑀺𑀰𑁄𑀥𑀓𑀂, 𑀧𑁆𑀭𑀸𑀳𑀯𑀦𑀻𑀬𑀂, 𑀆𑀳𑀯𑀦𑀻𑀬𑀂.
