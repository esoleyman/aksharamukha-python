𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀳𑁆𑀭𑀼𑀢𑀬𑀲𑀽𑀢𑁆𑀭𑀫𑁆.

[𑀲𑀫𑁆𑀓𑁆𑀱𑀺𑀧𑁆𑀢𑀫𑀸𑀢𑁆𑀭𑀼𑀓𑀸]

.. 𑀦𑀫𑀂 𑀲𑀭𑁆𑀯𑀚𑁆𑀜𑀸𑀬 ..

.. 𑀑𑀫𑁆 𑀦𑀫𑁄 𑀧𑀓𑀯𑀢𑁆𑀬𑁃 𑀆𑀭𑁆𑀬𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀬𑁃 ..

𑀆𑀭𑁆𑀬𑀸𑀯𑀮𑁄𑀓𑀺𑀢𑁂𑀰𑁆𑀯𑀭𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑁄 𑀓𑀫𑁆𑀧𑀻𑀭𑀸𑀬𑀸𑀫𑁆 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀬𑀸𑀫𑁆 𑀘𑀭𑁆𑀬𑀸𑀫𑁆 𑀘𑀭𑀫𑀸𑀡𑁄 𑀯𑁆𑀬𑀯𑀮𑁄𑀓𑀬𑀢𑀺 𑀲𑁆𑀫. 𑀧𑀜𑁆𑀘 𑀲𑁆𑀓𑀦𑁆𑀥𑀸𑀂, 𑀢𑀸𑀫𑁆𑀰𑁆𑀘 𑀲𑁆𑀯𑀧𑀸𑀯𑀰𑀽𑀦𑁆𑀬𑀸𑀦𑁆 𑀧𑀰𑁆𑀬𑀢𑀺 𑀲𑁆𑀫..

𑀇𑀳 𑀰𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀭 𑀭𑀽𑀧𑀫𑁆 𑀰𑀽𑀦𑁆𑀬𑀢𑀸, 𑀰𑀽𑀦𑁆𑀬𑀢𑁃𑀯 𑀭𑀽𑀧𑀫𑁆. 𑀭𑀽𑀧𑀸𑀦𑁆𑀦 𑀧𑁆𑀭𑀼𑀢𑀓𑁆 𑀰𑀽𑀦𑁆𑀬𑀢𑀸, 𑀰𑀽𑀦𑁆𑀬𑀢𑀸𑀬𑀸 𑀦 𑀧𑁆𑀭𑀼𑀢𑀓𑁆 𑀭𑀽𑀧𑀫𑁆. 𑀬𑀢𑁆𑀭𑀽𑀧𑀫𑁆 𑀲𑀸 𑀰𑀽𑀦𑁆𑀬𑀢𑀸, 𑀬𑀸 𑀰𑀽𑀦𑁆𑀬𑀢𑀸 𑀢𑀢𑁆𑀭𑀽𑀧𑀫𑁆..

𑀏𑀯𑀫𑁂𑀯 𑀯𑁂𑀢𑀦𑀸𑀲𑀫𑁆𑀚𑁆𑀜𑀸𑀲𑀫𑁆𑀲𑁆𑀓𑀸𑀭𑀯𑀺𑀚𑁆𑀜𑀸𑀦𑀸𑀦𑀺..

𑀇𑀳𑀫𑁆 𑀰𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀭 𑀲𑀭𑁆𑀯𑀥𑀭𑁆𑀫𑀸𑀂 𑀰𑀽𑀦𑁆𑀬𑀢𑀸𑀮𑀓𑁆𑀱𑀡𑀸 𑀅𑀦𑀼𑀢𑁆𑀧𑀦𑁆𑀦𑀸 𑀅𑀦𑀺𑀭𑀼𑀢𑁆𑀥𑀸 𑀅𑀫𑀮𑀸 𑀦 𑀯𑀺𑀫𑀮𑀸 𑀦𑁄𑀦𑀸 𑀦 𑀧𑀭𑀺𑀧𑀽𑀭𑁆𑀡𑀸𑀂. 𑀢𑀲𑁆𑀫𑀸𑀘𑁆𑀘𑀸𑀭𑀺𑀧𑀼𑀢𑁆𑀭 𑀰𑀽𑀦𑁆𑀬𑀢𑀸𑀬𑀸𑀫𑁆 𑀦 𑀭𑀽𑀧𑀫𑁆, 𑀦 𑀯𑁂𑀢𑀦𑀸, 𑀦 𑀲𑀫𑁆𑀚𑁆𑀜𑀸, 𑀦 𑀲𑀫𑁆𑀲𑁆𑀓𑀸𑀭𑀸𑀂, 𑀦 𑀯𑀺𑀚𑁆𑀜𑀸𑀦𑀸𑀦𑀺. 𑀦 𑀘𑀓𑁆𑀱𑀼𑀂𑀰𑁆𑀭𑁄𑀢𑁆𑀭𑀓𑁆𑀭𑀸𑀡𑀚𑀺𑀳𑁆𑀯𑀸𑀓𑀸𑀬𑀫𑀦𑀸𑀫𑁆𑀲𑀺, 𑀦 𑀭𑀽𑀧𑀰𑀧𑁆𑀢𑀓𑀦𑁆𑀥𑀭𑀲𑀲𑁆𑀧𑁆𑀭𑀱𑁆𑀝𑀯𑁆𑀬𑀥𑀭𑁆𑀫𑀸𑀂. 𑀦 𑀘𑀓𑁆𑀱𑀼𑀭𑁆𑀥𑀸𑀢𑀼𑀭𑁆𑀬𑀸𑀯𑀦𑁆𑀦 𑀫𑀦𑁄𑀥𑀸𑀢𑀼𑀂..

𑀦 𑀯𑀺𑀢𑁆𑀬𑀸 𑀦𑀸𑀯𑀺𑀢𑁆𑀬𑀸 𑀦 𑀯𑀺𑀢𑁆𑀬𑀸𑀓𑁆𑀱𑀬𑁄 𑀦𑀸𑀯𑀺𑀢𑁆𑀬𑀸𑀓𑁆𑀱𑀬𑁄 𑀬𑀸𑀯𑀦𑁆𑀦 𑀚𑀭𑀸𑀫𑀭𑀡𑀫𑁆 𑀦 𑀚𑀭𑀸𑀫𑀭𑀡𑀓𑁆𑀱𑀬𑁄 𑀦 𑀢𑀼𑀂𑀓𑀲𑀫𑀼𑀢𑀬𑀦𑀺𑀭𑁄𑀥𑀫𑀸𑀭𑁆𑀓𑀸 𑀦 𑀚𑁆𑀜𑀸𑀦𑀫𑁆 𑀦 𑀧𑁆𑀭𑀸𑀧𑁆𑀢𑀺𑀢𑁆𑀯𑀫𑁆..

𑀧𑁄𑀥𑀺𑀲𑀢𑁆𑀢𑁆𑀯𑀲𑁆𑀬 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀫𑀸𑀰𑁆𑀭𑀺𑀢𑁆𑀬 𑀯𑀺𑀳𑀭𑀢𑀺 𑀘𑀺𑀢𑁆𑀢𑀸𑀯𑀭𑀡𑀂. 𑀘𑀺𑀢𑁆𑀢𑀸𑀯𑀭𑀡𑀦𑀸𑀲𑁆𑀢𑀺𑀢𑁆𑀯𑀸𑀢𑀢𑁆𑀭𑀲𑁆𑀢𑁄 𑀯𑀺𑀧𑀭𑁆𑀬𑀸𑀲𑀸𑀢𑀺𑀓𑁆𑀭𑀸𑀦𑁆𑀢𑁄 𑀦𑀺𑀱𑁆𑀝𑀦𑀺𑀭𑁆𑀯𑀸𑀡𑀂. 𑀢𑁆𑀭𑁆𑀬𑀥𑁆𑀯𑀯𑁆𑀬𑀯𑀲𑁆𑀢𑀺𑀢𑀸𑀂 𑀲𑀭𑁆𑀯𑀧𑀼𑀢𑁆𑀥𑀸𑀂 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀫𑀸𑀰𑁆𑀭𑀺𑀢𑁆𑀬 𑀅𑀦𑀼𑀢𑁆𑀢𑀭𑀸𑀫𑁆 𑀲𑀫𑁆𑀬𑀓𑁆𑀲𑀫𑁆𑀧𑁄𑀥𑀺𑀫𑀧𑀺𑀲𑀫𑁆𑀧𑀼𑀢𑁆𑀥𑀸𑀂..

𑀢𑀲𑁆𑀫𑀸𑀚𑁆𑀚𑁆𑀜𑀸𑀢𑀯𑁆𑀬𑀂 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀫𑀳𑀸𑀫𑀦𑁆𑀢𑁆𑀭𑁄 𑀫𑀳𑀸𑀯𑀺𑀢𑁆𑀬𑀸𑀫𑀦𑁆𑀢𑁆𑀭𑁄'𑀦𑀼𑀢𑁆𑀢𑀭𑀫𑀦𑁆𑀢𑁆𑀭𑁄'𑀲𑀫𑀲𑀫𑀫𑀦𑁆𑀢𑁆𑀭𑀂 𑀲𑀭𑁆𑀯𑀢𑀼𑀂𑀓𑀧𑁆𑀭𑀰𑀫𑀦𑀂 𑀲𑀢𑁆𑀬𑀫𑀫𑀺𑀢𑁆𑀬𑀢𑁆𑀯𑀸𑀢𑁆 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀬𑀸𑀫𑀼𑀓𑁆𑀢𑁄 𑀫𑀦𑁆𑀢𑁆𑀭𑀂.

𑀢𑀢𑁆𑀬𑀢𑀸- 𑀓𑀢𑁂 𑀓𑀢𑁂 𑀧𑀸𑀭𑀓𑀢𑁂 𑀧𑀸𑀭𑀲𑀫𑁆𑀓𑀢𑁂 𑀧𑁄𑀥𑀺 𑀲𑁆𑀯𑀸𑀳𑀸..

𑀇𑀢𑀺 𑀧𑁆𑀭𑀚𑁆𑀜𑀸𑀧𑀸𑀭𑀫𑀺𑀢𑀸𑀳𑁆𑀭𑀼𑀢𑀬𑀲𑀽𑀢𑁆𑀭𑀫𑁆 𑀲𑀫𑀸𑀧𑁆𑀢𑀫𑁆..
