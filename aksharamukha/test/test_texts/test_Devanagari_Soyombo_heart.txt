𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑪂𑩙𑩭𑩻𑪁𑩒𑩛𑩫𑪙𑩼𑩴𑪛

[𑪁𑪖𑪃𑩑𑩰𑪙𑩫𑩴𑩛𑩫𑩙𑩜𑩛]

𑪜 𑩯𑩴𑪗 𑪁𑩼𑪙𑩾𑩷𑪙𑩥𑩛𑩻 𑪜

𑪜 𑩐𑩖𑪖 𑩯𑩴𑩖 𑩳𑩞𑩾𑩫𑪙𑩻𑩗 𑩐𑩛𑩼𑪙𑩻𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩻𑩗 𑪜

𑩐𑩛𑩼𑪙𑩻𑩛𑩾𑩽𑩖𑩜𑩑𑩫𑩔𑪈𑩾𑩼𑩲𑩖𑩮𑩑𑪁𑩫𑪘𑪙𑩾𑩖 𑩞𑩴𑪙𑩳𑩑𑩛𑩼𑩛𑩻𑩛𑪖 𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩻𑩛𑪖 𑩵𑩼𑪙𑩻𑩛𑪖 𑩵𑩼𑩴𑩛𑩪𑩖 𑩾𑪙𑩻𑩾𑩽𑩖𑩜𑩻𑩫𑩑 𑪁𑪙𑩴𑪛 𑩰𑩥𑪙𑩵 𑪁𑪙𑩜𑩯𑪙𑩮𑩛𑪗, 𑩫𑩛𑪖𑪈𑩵 𑪁𑪙𑩾𑩳𑩛𑩾𑩿𑩒𑩛𑩯𑪙𑩻𑩛𑩯 𑩰𑪈𑩻𑩫𑩑 𑪁𑪙𑩴𑪜

𑩐𑩑𑪂 𑩿𑩛𑩼𑩑𑩰𑩒𑩫𑪙𑩼 𑩼𑩒𑩛𑩰𑪖 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛, 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩗𑩾 𑩼𑩒𑩛𑩰𑩴𑪛 𑩼𑩒𑩛𑩰𑩛𑩯𑪘 𑩰𑩙𑩬𑩜 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛, 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛𑩻𑩛 𑩯 𑩰𑩙𑩬𑩞 𑩼𑩒𑩛𑩰𑩴𑪛 𑩻𑩭𑪙𑩼𑩒𑩛𑩰𑪖 𑪁𑩛 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛, 𑩻𑩛 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛 𑩫𑩭𑪙𑩼𑩒𑩛𑩰𑩴𑪜

𑩐𑩔𑩾𑩴𑩔𑩾 𑩾𑩔𑩭𑩯𑩛𑪁𑪖𑩷𑪙𑩥𑩛𑪁𑪖𑪁𑪙𑩜𑩛𑩼𑩾𑩑𑩷𑪙𑩥𑩛𑩯𑩛𑩯𑩑𑪜

𑩐𑩑𑪂𑪖 𑩿𑩛𑩼𑩑𑩰𑩒𑩫𑪙𑩼 𑪁𑩼𑪙𑩾𑩮𑩼𑪙𑩴𑩛𑪗 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛𑩽𑪃𑩪𑩛 𑩐𑩯𑩒𑩫𑪙𑩰𑩯𑪘𑩛 𑩐𑩯𑩑𑩼𑩒𑩭𑪙𑩮𑩛 𑩐𑩴𑩽𑩛 𑩯 𑩾𑩑𑩴𑩽𑩛 𑩯𑩖𑩯𑩛 𑩯 𑩰𑩼𑩑𑩰𑩒𑩛𑩼𑪙𑩪𑩛𑪗𑪛 𑩫𑪁𑪙𑩴𑩛𑩵𑪙𑩶𑩛𑩼𑩑𑩰𑩒𑩫𑪙𑩼 𑩿𑩒𑩛𑩯𑪙𑩻𑩫𑩛𑩻𑩛𑪖 𑩯 𑩼𑩒𑩛𑩰𑩴, 𑩯 𑩾𑩔𑩭𑩯𑩛, 𑩯 𑪁𑪖𑩷𑪙𑩥𑩛, 𑩯 𑪁𑪖𑪁𑪙𑩜𑩛𑩼𑩛𑪗, 𑩯 𑩾𑩑𑩷𑪙𑩥𑩛𑩯𑩛𑩯𑩑𑪛 𑩯 𑩵𑪃𑩒𑪗𑪈𑩼𑩖𑩫𑪙𑩼𑩟𑪙𑩼𑩛𑩪𑩷𑩑𑪂𑪙𑩾𑩛𑩜𑩛𑩻𑩴𑩯𑩛𑪖𑪁𑩑, 𑩯 𑩼𑩒𑩛𑩰𑩿𑩲𑪙𑩭𑩞𑩯𑪙𑩮𑩼𑪁𑪁𑪙𑩰𑪙𑩼𑪀𑪙𑩦𑩾𑪙𑩻𑩮𑩼𑪙𑩴𑩛𑪗𑪛 𑩯 𑩵𑪃𑩒𑩼𑪙𑩮𑩛𑩫𑩒𑩼𑪙𑩻𑩛𑩾𑩯𑪘 𑩴𑩯𑩖𑩮𑩛𑩫𑩒𑪗𑪜

𑩯 𑩾𑩑𑩭𑪙𑩻𑩛 𑩯𑩛𑩾𑩑𑩭𑪙𑩻𑩛 𑩯 𑩾𑩑𑩭𑪙𑩻𑩛𑪃𑩻𑩖 𑩯𑩛𑩾𑩑𑩭𑪙𑩻𑩛𑪃𑩻𑩖 𑩻𑩛𑩾𑩯𑪘 𑩷𑩼𑩛𑩴𑩼𑩪𑪖 𑩯 𑩷𑩼𑩛𑩴𑩼𑩪𑪃𑩻𑩖 𑩯 𑩭𑩒𑪗𑩝𑪁𑩴𑩒𑩭𑩻𑩯𑩑𑩼𑩖𑩮𑩴𑩛𑩼𑪙𑩞𑩛 𑩯 𑩷𑪙𑩥𑩛𑩯𑪖 𑩯 𑩰𑪙𑩼𑩛𑩰𑪙𑩫𑩑𑩫𑪙𑩾𑩴𑪜

𑩲𑩖𑩮𑩑𑪁𑩫𑪘𑪙𑩾𑪁𑪙𑩻 𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩴𑩛𑪈𑩼𑩑𑩫𑪙𑩻 𑩾𑩑𑪂𑩼𑩫𑩑 𑩵𑩑𑩫𑪘𑩛𑩾𑩼𑩪𑪗𑪛 𑩵𑩑𑩫𑪘𑩛𑩾𑩼𑩪𑩯𑩛𑪁𑪙𑩫𑩑𑩫𑪙𑩾𑩛𑩭𑩫𑪙𑩼𑪁𑪙𑩫𑩖 𑩾𑩑𑩰𑩼𑪙𑩻𑩛𑪁𑩛𑩫𑩑𑩜𑪙𑩼𑩛𑩯𑪙𑩫𑩖 𑩯𑩑𑪀𑪙𑩧𑩯𑩑𑩼𑪙𑩾𑩛𑩪𑪗𑪛 𑩫𑪙𑩼𑪙𑩻𑩮𑪙𑩾𑩾𑪙𑩻𑩾𑪁𑪙𑩬𑩑𑩫𑩛𑪗 𑪁𑩼𑪙𑩾𑩲𑩒𑩭𑪙𑩮𑩛𑪗 𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩴𑩛𑪈𑩼𑩑𑩫𑪙𑩻 𑩐𑩯𑩒𑩫𑪘𑩼𑩛𑪖 𑪁𑩴𑪙𑩻𑩜𑪙𑪁𑪖𑩲𑩖𑩮𑩑𑩴𑩳𑩑𑪁𑪖𑩲𑩒𑩭𑪙𑩮𑩛𑪗𑪜

𑩫𑪁𑪙𑩴𑩛𑩷𑪘𑪙𑩥𑩛𑩫𑩾𑪙𑩻𑪗 𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩴𑪂𑩛𑩴𑩯𑪙𑩫𑪙𑩼𑩖 𑩴𑪂𑩛𑩾𑩑𑩭𑪙𑩻𑩛𑩴𑩯𑪙𑩫𑪙𑩼𑩖𑩯𑩒𑩫𑪘𑩼𑩴𑩯𑪙𑩫𑪙𑩼𑩖𑪁𑩴𑪁𑩴𑩴𑩯𑪙𑩫𑪙𑩼𑪗 𑪁𑩼𑪙𑩾𑩭𑩒𑪗𑩝𑩰𑪙𑩼𑩿𑩴𑩯𑪗 𑪁𑩫𑪙𑩻𑩴𑩴𑩑𑩬𑪙𑩻𑩫𑪙𑩾𑩛𑩫 𑩰𑪙𑩼𑩷𑪙𑩥𑩛𑩰𑩛𑩼𑩴𑩑𑩫𑩛𑩻𑩛𑩴𑩒𑩜𑪙𑩫𑩖 𑩴𑩯𑪙𑩫𑪙𑩼𑪗𑪛
